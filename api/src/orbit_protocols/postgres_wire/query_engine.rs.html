13/// Query result types
14#[derive(Debug, Clone)]
15pub enum QueryResult {
16    Select {
17        columns: Vec<String>,
18        rows: Vec<Vec<Option<String>>>,
19    },
20    Insert {
21        count: usize,
22    },
23    Update {
24        count: usize,
25    },
26    Delete {
27        count: usize,
28    },
29}
30
31/// Parsed SQL statement
32#[derive(Debug, Clone)]
33enum Statement {
34    Select {
35        columns: Vec<String>,
36        table: String,
37        where_clause: Option<WhereClause>,
38    },
39    Insert {
40        table: String,
41        columns: Vec<String>,
42        values: Vec<String>,
43    },
44    Update {
45        table: String,
46        set_clauses: Vec<(String, String)>,
47        where_clause: Option<WhereClause>,
48    },
49    Delete {
50        table: String,
51        where_clause: Option<WhereClause>,
52    },
53}
54
55#[derive(Debug, Clone)]
56struct WhereClause {
57    conditions: Vec<Condition>,
58}
59
60#[derive(Debug, Clone)]
61struct Condition {
62    column: String,
63    operator: String,
64    value: String,
65}

109    /// Execute a SQL query
110    pub async fn execute_query(&self, sql: &str) -> ProtocolResult<QueryResult> {
111        let sql_upper = sql.trim().to_uppercase();
112
113        // Check if this is a GraphRAG function query
114        if self.is_graphrag_query(&sql_upper) {
115            if let Some(ref graphrag_engine) = self.graphrag_engine {
116                return graphrag_engine.execute_graphrag_query(sql).await;
117            } else {
118                return Err(ProtocolError::PostgresError(
119                    "GraphRAG support not enabled. Use new_with_vector_support() to enable GraphRAG functions.".to_string()
120                ));
121            }
122        }
123
124        // Check if this is a vector-related query
125        if let Some(ref vector_engine) = self.vector_engine {
126            if self.is_vector_query(&sql_upper) {
127                return vector_engine.execute_vector_query(sql).await;
128            }
129        }
130
131        // Handle regular SQL queries
132        let statement = self.parse_sql(sql)?;
133
134        match statement {
135            Statement::Select {
136                columns,
137                table,
138                where_clause,
139            } => self.execute_select(columns, &table, where_clause).await,
140            Statement::Insert {
141                table,
142                columns,
143                values,
144            } => self.execute_insert(&table, columns, values).await,
145            Statement::Update {
146                table,
147                set_clauses,
148                where_clause,
149            } => self.execute_update(&table, set_clauses, where_clause).await,
150            Statement::Delete {
151                table,
152                where_clause,
153            } => self.execute_delete(&table, where_clause).await,
154        }
155    }

157    /// Check if a query is vector-related
158    fn is_vector_query(&self, sql: &str) -> bool {
159        sql.contains("CREATE EXTENSION VECTOR")
160            || sql.contains("VECTOR(")
161            || sql.contains("HALFVEC(")
162            || sql.contains("<->")
163            || sql.contains("<#>")
164            || sql.contains("<=>")
165            || sql.contains("VECTOR_DIMS")
166            || sql.contains("VECTOR_NORM")
167            || (sql.contains("CREATE INDEX")
168                && (sql.contains("USING IVFFLAT") || sql.contains("USING HNSW")))
169    }

174            || sql.contains("GRAPHRAG_QUERY(")

178            || sql.contains("GRAPHRAG_ENTITIES(")
179            || sql.contains("GRAPHRAG_SIMILAR(")

182    /// Parse SQL statement
183    fn parse_sql(&self, sql: &str) -> ProtocolResult<Statement> {
184        let sql = sql.trim().to_uppercase();
185        let original_sql = sql.clone();
186
187        if sql.starts_with("SELECT") {
188            self.parse_select(&original_sql)
189        } else if sql.starts_with("INSERT") {
190            self.parse_insert(&original_sql)
191        } else if sql.starts_with("UPDATE") {
192            self.parse_update(&original_sql)
193        } else if sql.starts_with("DELETE") {
194            self.parse_delete(&original_sql)
195        } else {
196            Err(ProtocolError::PostgresError(format!(
197                "Unsupported SQL statement: {}",
198                sql
199            )))
200        }
201    }

203    /// Parse SELECT statement
204    fn parse_select(&self, sql: &str) -> ProtocolResult<Statement> {
205        // Simple parser: SELECT columns FROM table [WHERE condition]
206        let parts: Vec<&str> = sql.split_whitespace().collect();
207
208        if parts.len() < 4 || parts[0].to_uppercase() != "SELECT" {
209            return Err(ProtocolError::PostgresError(
210                "Invalid SELECT syntax".to_string(),
211            ));
212        }
213
214        // Find FROM
215        let from_idx = parts
216            .iter()
217            .position(|&p| p.to_uppercase() == "FROM")
218            .ok_or_else(|| ProtocolError::PostgresError("Missing FROM clause".to_string()))?;
219
220        // Parse columns
221        let columns_str = parts[1..from_idx].join(" ");
222        let columns: Vec<String> = if columns_str == "*" {
223            vec!["*".to_string()]
224        } else {
225            columns_str
226                .split(',')
227                .map(|s| s.trim().to_string())
228                .collect()
229        };
230
231        // Parse table - convert to uppercase and trim semicolon
232        let table = parts[from_idx + 1].trim_end_matches(';').to_uppercase();
233
234        // Parse WHERE clause if present
235        let where_idx = parts.iter().position(|&p| p.to_uppercase() == "WHERE");
236        let where_clause = if let Some(idx) = where_idx {
237            Some(self.parse_where_clause(&parts[idx + 1..])?)
238        } else {
239            None
240        };
241
242        Ok(Statement::Select {
243            columns,
244            table,
245            where_clause,
246        })
247    }

249    /// Parse INSERT statement
250    fn parse_insert(&self, sql: &str) -> ProtocolResult<Statement> {
251        // Simple parser: INSERT INTO table (columns) VALUES (values)
252        let sql_upper = sql.to_uppercase();
253
254        if !sql_upper.contains("INSERT INTO") || !sql_upper.contains("VALUES") {
255            return Err(ProtocolError::PostgresError(
256                "Invalid INSERT syntax".to_string(),
257            ));
258        }
259
260        // Find positions using uppercase version
261        let table_start = sql_upper.find("INTO").unwrap() + 4;
262        let table_end = sql_upper[table_start..].find('(').unwrap() + table_start;
263        let col_start = table_end + 1;
264        let col_end = sql_upper[col_start..].find(')').unwrap() + col_start;
265        let val_keyword_pos = sql_upper.find("VALUES").unwrap() + 6;
266        let val_start = sql_upper[val_keyword_pos..].find('(').unwrap() + val_keyword_pos + 1;
267        let val_end = sql_upper[val_start..].rfind(')').unwrap() + val_start;
268
269        // Extract data using original SQL to preserve case
270        let table = sql[table_start..table_end].trim().to_uppercase();
271        let columns: Vec<String> = sql_upper[col_start..col_end]
272            .split(',')
273            .map(|s| s.trim().to_string())
274            .collect();
275        let values = self.parse_csv_values(&sql[val_start..val_end]);
276
277        Ok(Statement::Insert {
278            table,
279            columns,
280            values,
281        })
282    }

284    /// Parse UPDATE statement
285    fn parse_update(&self, sql: &str) -> ProtocolResult<Statement> {
286        // Simple parser: UPDATE table SET col=val [WHERE condition]
287        let parts: Vec<&str> = sql.split_whitespace().collect();
288
289        if parts.len() < 4 || parts[0].to_uppercase() != "UPDATE" {
290            return Err(ProtocolError::PostgresError(
291                "Invalid UPDATE syntax".to_string(),
292            ));
293        }
294
295        let table = parts[1].trim_end_matches(';').to_uppercase();
296
297        // Find SET
298        let set_idx = parts
299            .iter()
300            .position(|&p| p.to_uppercase() == "SET")
301            .ok_or_else(|| ProtocolError::PostgresError("Missing SET clause".to_string()))?;
302
303        // Find WHERE or end
304        let where_idx = parts.iter().position(|&p| p.to_uppercase() == "WHERE");
305        let set_end = where_idx.unwrap_or(parts.len());
306
307        // Parse SET clauses safely with JSON support
308        let set_str = parts[set_idx + 1..set_end].join(" ");
309        let set_clauses = self.parse_set_clauses(&set_str);
310
311        // Parse WHERE clause
312        let where_clause = if let Some(idx) = where_idx {
313            Some(self.parse_where_clause(&parts[idx + 1..])?)
314        } else {
315            None
316        };
317
318        Ok(Statement::Update {
319            table,
320            set_clauses,
321            where_clause,
322        })
323    }

331            return Err(ProtocolError::PostgresError(

337        let from_idx = parts

339            .position(|&p| p.to_uppercase() == "FROM")

343
344        // Parse WHERE clause
345        let where_idx = parts.iter().position(|&p| p.to_uppercase() == "WHERE");
346        let where_clause = if let Some(idx) = where_idx {

348        } else {
349            None
350        };
351
352        Ok(Statement::Delete {
353            table,
354            where_clause,

358    /// Parse SET clauses respecting quotes and JSON braces
359    fn parse_set_clauses(&self, set_str: &str) -> Vec<(String, String)> {
360        let mut clauses = Vec::new();
361        let mut current_clause = String::new();
362        let mut in_quotes = false;
363        let mut quote_char = '\0';
364        let mut brace_depth = 0;
365        let chars: Vec<char> = set_str.chars().collect();
366        let mut i = 0;
367
368        while i < chars.len() {
369            let ch = chars[i];
370
371            match ch {
372                '\'' | '"' if !in_quotes => {
373                    in_quotes = true;
374                    quote_char = ch;
375                    current_clause.push(ch);
376                }
377                c if in_quotes && c == quote_char => {
378                    in_quotes = false;
379                    current_clause.push(ch);
380                }
381                '{' if !in_quotes => {
382                    brace_depth += 1;
383                    current_clause.push(ch);
384                }
385                '}' if !in_quotes => {
386                    brace_depth -= 1;
387                    current_clause.push(ch);
388                }
389                ',' if !in_quotes && brace_depth == 0 => {
390                    // Found a separator - parse the current clause
391                    if let Some(parsed) = self.parse_single_set_clause(&current_clause) {
392                        clauses.push(parsed);
393                    }
394                    current_clause.clear();
395                }
396                _ => {
397                    current_clause.push(ch);
398                }
399            }
400            i += 1;
401        }
402
403        // Add the last clause
404        if !current_clause.is_empty() {
405            if let Some(parsed) = self.parse_single_set_clause(&current_clause) {
406                clauses.push(parsed);
407            }
408        }
409
410        clauses
411    }

413    /// Parse a single SET clause (key = value)
414    fn parse_single_set_clause(&self, clause: &str) -> Option<(String, String)> {
415        let eq_pos = clause.find('=')?;
416        let key = clause[..eq_pos].trim().to_string();
417        let value = clause[eq_pos + 1..]
418            .trim()
419            .trim_matches('\'')
420            .trim_matches('"')
421            .to_string();
422        Some((key, value))
423    }

429        let mut in_quotes = false;
430        let mut quote_char = '\0';
431        let mut brace_depth = 0;
432        let chars: Vec<char> = values_str.chars().collect();
433        let mut i = 0;
434
435        while i < chars.len() {

437
438            match ch {
439                '\'' | '"' if !in_quotes => {

441                    quote_char = ch;
442                    current_value.push(ch);
443                }
444                c if in_quotes && c == quote_char => {
445                    in_quotes = false;
446                    current_value.push(ch);
447                }
448                '{' if !in_quotes => {
449                    brace_depth += 1;
450                    current_value.push(ch);
451                }
452                '}' if !in_quotes => {
453                    brace_depth -= 1;
454                    current_value.push(ch);

470            }
471            i += 1;
472        }
473
474        // Add the last value
475        if !current_value.is_empty() {
476            values.push(
477                current_value
478                    .trim()
479                    .trim_matches('\'')
480                    .trim_matches('"')
481                    .to_string(),
482            );
483        }
484

488    /// Parse WHERE clause
489    fn parse_where_clause(&self, parts: &[&str]) -> ProtocolResult<WhereClause> {
490        // Simple parser: column operator value
491        if parts.len() < 3 {
492            return Err(ProtocolError::PostgresError(
493                "Invalid WHERE clause".to_string(),
494            ));
495        }
496
497        let column = parts[0].to_string();
498        let operator = parts[1].to_string();
499        // Parse the value more carefully - it might span multiple parts if it contains spaces
500        let value_part = parts[2..].join(" ");
501        let value = value_part
502            .trim_end_matches(';') // Remove trailing semicolon first
503            .trim_matches('\'')
504            .trim_matches('"')
505            .to_string();
506
507        Ok(WhereClause {
508            conditions: vec![Condition {
509                column,
510                operator,
511                value,
512            }],
513        })
514    }

516    /// Execute SELECT query
517    async fn execute_select(
518        &self,
519        columns: Vec<String>,
520        table: &str,
521        where_clause: Option<WhereClause>,
522    ) -> ProtocolResult<QueryResult> {
523        if table.to_uppercase() != "ACTORS" {
524            return Err(ProtocolError::PostgresError(format!(
525                "Unknown table: {}",
526                table
527            )));
528        }
529
530        let actors = self.actors.read().await;
531        let mut rows = Vec::new();
532
533        for actor in actors.values() {
534            // Apply WHERE filter
535            if let Some(ref wc) = where_clause {
536                if !self.matches_where(actor, wc) {
537                    continue;
538                }
539            }
540
541            // Build row
542            let mut row = Vec::new();
543            if columns.len() == 1 && columns[0] == "*" {
544                // For SELECT *, add all columns in the expected order
545                row.push(Some(actor.actor_id.clone()));
546                row.push(Some(actor.actor_type.clone()));
547                row.push(Some(actor.state.to_string()));
548            } else {
549                // For specific columns
550                for col in &columns {
551                    let value = match col.to_uppercase().as_str() {
552                        "ACTOR_ID" => Some(actor.actor_id.clone()),
553                        "ACTOR_TYPE" => Some(actor.actor_type.clone()),
554                        "STATE" => Some(actor.state.to_string()),
555                        _ => None,
556                    };
557                    row.push(value);
558                }
559            }
560            rows.push(row);
561        }
562
563        // Determine columns
564        let result_columns = if columns.len() == 1 && columns[0] == "*" {
565            vec![
566                "actor_id".to_string(),
567                "actor_type".to_string(),
568                "state".to_string(),
569            ]
570        } else {
571            columns
572        };
573
574        Ok(QueryResult::Select {
575            columns: result_columns,
576            rows,
577        })
578    }

580    /// Execute INSERT query
581    async fn execute_insert(
582        &self,
583        table: &str,
584        columns: Vec<String>,
585        values: Vec<String>,
586    ) -> ProtocolResult<QueryResult> {
587        if table.to_uppercase() != "ACTORS" {
588            return Err(ProtocolError::PostgresError(format!(
589                "Unknown table: {}",
590                table
591            )));
592        }
593
594        if columns.len() != values.len() {
595            return Err(ProtocolError::PostgresError(
596                "Column count doesn't match value count".to_string(),
597            ));
598        }
599
600        let mut actor_id = None;
601        let mut actor_type = None;
602        let mut state = JsonValue::Object(serde_json::Map::new());
603
604        for (col, val) in columns.iter().zip(values.iter()) {
605            match col.to_uppercase().as_str() {
606                "ACTOR_ID" => actor_id = Some(val.clone()),
607                "ACTOR_TYPE" => actor_type = Some(val.clone()),
608                "STATE" => {
609                    state = serde_json::from_str(val)
610                        .unwrap_or_else(|_| JsonValue::String(val.clone()));
611                }
612                _ => {}
613            }
614        }
615
616        let actor_id =
617            actor_id.ok_or_else(|| ProtocolError::PostgresError("Missing actor_id".to_string()))?;
618        let actor_type = actor_type
619            .ok_or_else(|| ProtocolError::PostgresError("Missing actor_type".to_string()))?;
620
621        let record = ActorRecord {
622            actor_id: actor_id.clone(),
623            actor_type,
624            state,
625        };
626
627        let mut actors = self.actors.write().await;
628        actors.insert(actor_id, record);
629
630        Ok(QueryResult::Insert { count: 1 })
631    }

633    /// Execute UPDATE query
634    async fn execute_update(
635        &self,
636        table: &str,
637        set_clauses: Vec<(String, String)>,
638        where_clause: Option<WhereClause>,
639    ) -> ProtocolResult<QueryResult> {
640        if table.to_uppercase() != "ACTORS" {
641            return Err(ProtocolError::PostgresError(format!(
642                "Unknown table: {}",
643                table
644            )));
645        }
646
647        let mut actors = self.actors.write().await;
648        let mut count = 0;
649
650        for actor in actors.values_mut() {
651            // Apply WHERE filter
652            if let Some(ref wc) = where_clause {
653                if !self.matches_where(actor, wc) {
654                    continue;
655                }
656            }
657
658            // Apply updates
659            for (col, val) in &set_clauses {
660                match col.to_uppercase().as_str() {
661                    "STATE" => {
662                        actor.state = serde_json::from_str(val)
663                            .unwrap_or_else(|_| JsonValue::String(val.clone()));
664                    }
665                    "ACTOR_TYPE" => {
666                        actor.actor_type = val.clone();
667                    }
668                    _ => {}
669                }
670            }
671            count += 1;
672        }
673
674        Ok(QueryResult::Update { count })
675    }

677    /// Execute DELETE query
678    async fn execute_delete(
679        &self,
680        table: &str,
681        where_clause: Option<WhereClause>,
682    ) -> ProtocolResult<QueryResult> {
683        if table.to_uppercase() != "ACTORS" {
684            return Err(ProtocolError::PostgresError(format!(
685                "Unknown table: {}",
686                table
687            )));
688        }
689
690        let mut actors = self.actors.write().await;
691        let mut to_delete = Vec::new();
692
693        for (id, actor) in actors.iter() {
694            // Apply WHERE filter
695            if let Some(ref wc) = where_clause {
696                if !self.matches_where(actor, wc) {
697                    continue;
698                }
699            }
700            to_delete.push(id.clone());
701        }
702
703        let count = to_delete.len();
704        for id in to_delete {
705            actors.remove(&id);
706        }
707
708        Ok(QueryResult::Delete { count })
709    }

711    /// Check if actor matches WHERE clause
712    fn matches_where(&self, actor: &ActorRecord, where_clause: &WhereClause) -> bool {
713        for condition in &where_clause.conditions {
714            let value = match condition.column.to_uppercase().as_str() {
715                "ACTOR_ID" => &actor.actor_id,
716                "ACTOR_TYPE" => &actor.actor_type,
717                _ => return false,
718            };
719
720            let matches = match condition.operator.as_str() {
721                "=" => value == &condition.value,
722                "!=" | "<>" => value != &condition.value,
723                _ => false,
724            };
725
726            if !matches {
727                return false;
728            }
729        }
730        true
731    }
