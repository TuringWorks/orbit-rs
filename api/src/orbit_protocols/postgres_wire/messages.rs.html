52/// Frontend (client) messages
53#[derive(Debug, Clone)]
54pub enum FrontendMessage {
55    /// Startup message (no type byte, length-prefixed)
56    Startup {
57        protocol_version: i32,
58        parameters: HashMap<String, String>,
59    },
60    /// Simple query
61    Query { query: String },
62    /// Parse (prepared statement)
63    Parse {
64        statement_name: String,
65        query: String,
66        param_types: Vec<i32>,
67    },
68    /// Bind parameters to prepared statement
69    Bind {
70        portal: String,
71        statement: String,
72        param_formats: Vec<i16>,
73        params: Vec<Option<Bytes>>,
74        result_formats: Vec<i16>,
75    },
76    /// Execute portal
77    Execute { portal: String, max_rows: i32 },
78    /// Describe prepared statement or portal
79    Describe {
80        target: DescribeTarget,
81        name: String,
82    },
83    /// Close prepared statement or portal
84    Close { target: CloseTarget, name: String },
85    /// Flush output
86    Flush,
87    /// Sync (end of extended query)
88    Sync,
89    /// Terminate connection
90    Terminate,
91    /// Password message
92    Password { password: String },
93    /// SSL request
94    SSLRequest,
95}

109/// Backend (server) messages
110#[derive(Debug, Clone)]
111pub enum BackendMessage {

177impl FrontendMessage {
178    /// Parse a frontend message from bytes
179    pub fn parse(buf: &mut BytesMut) -> ProtocolResult<Option<Self>> {
180        if buf.len() < 5 {
181            return Ok(None); // Need at least type byte + length
182        }
183
184        // Check for startup message (no type byte, starts with length)
185        if buf.len() >= 4 {
186            let mut peek = Cursor::new(&buf[..]);
187            let len = peek.get_i32() as usize;
188
189            if len >= 8 && buf.len() >= len {
190                // Check if this looks like a startup message
191                let protocol_version = (&buf[4..8]).get_i32();
192                if protocol_version == 196608 || protocol_version == 80877103 {
193                    // Valid startup or SSL request
194                    return Self::parse_startup(buf);
195                }
196            }
197        }
198
199        let msg_type = buf[0];
200
201        // Check if we have enough data for the length
202        if buf.len() < 5 {
203            return Ok(None);
204        }
205
206        let len = {
207            let mut cursor = Cursor::new(&buf[1..5]);
208            cursor.get_i32() as usize
209        };
210
211        if buf.len() < 1 + len {
212            return Ok(None); // Not enough data yet
213        }
214
215        // Copy the message data before advancing buffer
216        let msg_data = buf[5..1 + len].to_vec();
217
218        // Remove message from buffer
219        buf.advance(1 + len);
220
221        // Parse the message data
222        let mut cursor = Cursor::new(&msg_data[..]);
223        let message = match msg_type {
224            b'Q' => Self::parse_query(&mut cursor)?,
225            b'P' => Self::parse_parse(&mut cursor)?,
226            b'B' => Self::parse_bind(&mut cursor)?,
227            b'E' => Self::parse_execute(&mut cursor)?,
228            b'D' => Self::parse_describe(&mut cursor)?,
229            b'C' => Self::parse_close(&mut cursor)?,
230            b'H' => FrontendMessage::Flush,
231            b'S' => FrontendMessage::Sync,
232            b'X' => FrontendMessage::Terminate,
233            b'p' => Self::parse_password(&mut cursor)?,
234            _ => {
235                return Err(ProtocolError::PostgresError(format!(
236                    "Unknown message type: {}",
237                    msg_type as char
238                )))
239            }
240        };
241
242        Ok(Some(message))
243    }
244
245    fn parse_startup(buf: &mut BytesMut) -> ProtocolResult<Option<Self>> {
246        let mut cursor = Cursor::new(&buf[..]);
247        let len = cursor.get_i32() as usize;
248
249        if buf.len() < len {
250            return Ok(None);
251        }
252
253        let protocol_version = cursor.get_i32();
254
255        // SSL request
256        if protocol_version == 80877103 {
257            buf.advance(len);
258            // Return SSL request message
259            return Ok(Some(FrontendMessage::SSLRequest));
260        }
261
262        let mut parameters = HashMap::new();
263        while cursor.position() < len as u64 - 1 {
264            let key = read_cstring(&mut cursor)?;
265            if key.is_empty() {
266                break;
267            }
268            let value = read_cstring(&mut cursor)?;
269            parameters.insert(key, value);
270        }
271
272        buf.advance(len);
273
274        Ok(Some(FrontendMessage::Startup {
275            protocol_version,
276            parameters,
277        }))
278    }
279
280    fn parse_query(cursor: &mut Cursor<&[u8]>) -> ProtocolResult<Self> {
281        let query = read_cstring(cursor)?;
282        Ok(FrontendMessage::Query { query })
283    }
284
285    fn parse_parse(cursor: &mut Cursor<&[u8]>) -> ProtocolResult<Self> {
286        let statement_name = read_cstring(cursor)?;
287        let query = read_cstring(cursor)?;
288        let param_count = cursor.get_i16() as usize;
289        let mut param_types = Vec::with_capacity(param_count);
290        for _ in 0..param_count {
291            param_types.push(cursor.get_i32());
292        }
293        Ok(FrontendMessage::Parse {
294            statement_name,
295            query,
296            param_types,
297        })
298    }
299
300    fn parse_bind(cursor: &mut Cursor<&[u8]>) -> ProtocolResult<Self> {
301        let portal = read_cstring(cursor)?;
302        let statement = read_cstring(cursor)?;
303
304        let format_count = cursor.get_i16() as usize;
305        let mut param_formats = Vec::with_capacity(format_count);
306        for _ in 0..format_count {
307            param_formats.push(cursor.get_i16());
308        }
309
310        let param_count = cursor.get_i16() as usize;
311        let mut params = Vec::with_capacity(param_count);
312        for _ in 0..param_count {
313            let len = cursor.get_i32();
314            if len == -1 {
315                params.push(None);
316            } else {
317                let mut data = vec![0u8; len as usize];
318                cursor.copy_to_slice(&mut data);
319                params.push(Some(Bytes::from(data)));
320            }
321        }
322
323        let result_format_count = cursor.get_i16() as usize;
324        let mut result_formats = Vec::with_capacity(result_format_count);
325        for _ in 0..result_format_count {
326            result_formats.push(cursor.get_i16());
327        }
328
329        Ok(FrontendMessage::Bind {
330            portal,
331            statement,
332            param_formats,
333            params,
334            result_formats,
335        })
336    }
337
338    fn parse_execute(cursor: &mut Cursor<&[u8]>) -> ProtocolResult<Self> {
339        let portal = read_cstring(cursor)?;
340        let max_rows = cursor.get_i32();
341        Ok(FrontendMessage::Execute { portal, max_rows })
342    }
343
344    fn parse_describe(cursor: &mut Cursor<&[u8]>) -> ProtocolResult<Self> {
345        let target_type = cursor.get_u8();
346        let target = match target_type {
347            b'S' => DescribeTarget::Statement,
348            b'P' => DescribeTarget::Portal,
349            _ => {
350                return Err(ProtocolError::PostgresError(format!(
351                    "Invalid describe target: {}",
352                    target_type as char
353                )))
354            }
355        };
356        let name = read_cstring(cursor)?;
357        Ok(FrontendMessage::Describe { target, name })
358    }
359
360    fn parse_close(cursor: &mut Cursor<&[u8]>) -> ProtocolResult<Self> {
361        let target_type = cursor.get_u8();
362        let target = match target_type {
363            b'S' => CloseTarget::Statement,
364            b'P' => CloseTarget::Portal,
365            _ => {
366                return Err(ProtocolError::PostgresError(format!(
367                    "Invalid close target: {}",
368                    target_type as char
369                )))
370            }
371        };
372        let name = read_cstring(cursor)?;
373        Ok(FrontendMessage::Close { target, name })
374    }
375
376    fn parse_password(cursor: &mut Cursor<&[u8]>) -> ProtocolResult<Self> {
377        let password = read_cstring(cursor)?;
378        Ok(FrontendMessage::Password { password })
379    }
380}

382impl BackendMessage {
383    /// Encode a backend message to bytes
384    pub fn encode(&self, buf: &mut BytesMut) {
385        match self {
386            BackendMessage::Authentication(auth) => {
387                buf.put_u8(b'R');
388                let pos = buf.len();
389                buf.put_i32(0); // Placeholder for length
390
391                match auth {
392                    AuthenticationResponse::Ok => buf.put_i32(0),
393                    AuthenticationResponse::CleartextPassword => buf.put_i32(3),
394                    AuthenticationResponse::MD5Password { salt } => {
395                        buf.put_i32(5);
396                        buf.put_slice(salt);
397                    }
398                    _ => buf.put_i32(0), // TODO: Implement other auth types
399                }
400
401                let len = buf.len() - pos;
402                buf[pos..pos + 4].copy_from_slice(&(len as i32).to_be_bytes());
403            }
404            BackendMessage::ParameterStatus { name, value } => {
405                buf.put_u8(b'S');
406                let pos = buf.len();
407                buf.put_i32(0); // Placeholder
408
409                write_cstring(buf, name);
410                write_cstring(buf, value);
411
412                let len = buf.len() - pos;
413                buf[pos..pos + 4].copy_from_slice(&(len as i32).to_be_bytes());
414            }
415            BackendMessage::ReadyForQuery { status } => {
416                buf.put_u8(b'Z');
417                buf.put_i32(5); // Length
418                buf.put_u8(*status as u8);
419            }
420            BackendMessage::RowDescription { fields } => {
421                buf.put_u8(b'T');
422                let pos = buf.len();
423                buf.put_i32(0); // Placeholder
424
425                buf.put_i16(fields.len() as i16);
426                for field in fields {
427                    write_cstring(buf, &field.name);
428                    buf.put_i32(field.table_oid);
429                    buf.put_i16(field.column_id);
430                    buf.put_i32(field.type_oid);
431                    buf.put_i16(field.type_size);
432                    buf.put_i32(field.type_modifier);
433                    buf.put_i16(field.format);
434                }
435
436                let len = buf.len() - pos;
437                buf[pos..pos + 4].copy_from_slice(&(len as i32).to_be_bytes());
438            }
439            BackendMessage::DataRow { values } => {
440                buf.put_u8(b'D');
441                let pos = buf.len();
442                buf.put_i32(0); // Placeholder
443
444                buf.put_i16(values.len() as i16);
445                for value in values {
446                    match value {
447                        None => buf.put_i32(-1),
448                        Some(data) => {
449                            buf.put_i32(data.len() as i32);
450                            buf.put_slice(data);
451                        }
452                    }
453                }
454
455                let len = buf.len() - pos;
456                buf[pos..pos + 4].copy_from_slice(&(len as i32).to_be_bytes());
457            }
458            BackendMessage::CommandComplete { tag } => {
459                buf.put_u8(b'C');
460                let pos = buf.len();
461                buf.put_i32(0); // Placeholder
462
463                write_cstring(buf, tag);
464
465                let len = buf.len() - pos;
466                buf[pos..pos + 4].copy_from_slice(&(len as i32).to_be_bytes());
467            }
468            BackendMessage::EmptyQueryResponse => {
469                buf.put_u8(b'I');
470                buf.put_i32(4); // Length
471            }
472            BackendMessage::ErrorResponse { fields }
473            | BackendMessage::NoticeResponse { fields } => {
474                let msg_type = if matches!(self, BackendMessage::ErrorResponse { .. }) {
475                    b'E'
476                } else {
477                    b'N'
478                };
479                buf.put_u8(msg_type);
480                let pos = buf.len();
481                buf.put_i32(0); // Placeholder
482
483                for (field_type, value) in fields {
484                    buf.put_u8(*field_type);
485                    write_cstring(buf, value);
486                }
487                buf.put_u8(0); // Terminator
488
489                let len = buf.len() - pos;
490                buf[pos..pos + 4].copy_from_slice(&(len as i32).to_be_bytes());
491            }
492            BackendMessage::ParseComplete => {
493                buf.put_u8(b'1');
494                buf.put_i32(4);
495            }
496            BackendMessage::BindComplete => {
497                buf.put_u8(b'2');
498                buf.put_i32(4);
499            }
500            BackendMessage::CloseComplete => {
501                buf.put_u8(b'3');
502                buf.put_i32(4);
503            }
504            BackendMessage::NoData => {
505                buf.put_u8(b'n');
506                buf.put_i32(4);
507            }
508            BackendMessage::BackendKeyData {
509                process_id,
510                secret_key,
511            } => {
512                buf.put_u8(b'K');
513                buf.put_i32(12);
514                buf.put_i32(*process_id);
515                buf.put_i32(*secret_key);
516            }
517            BackendMessage::ParameterDescription { param_types } => {
518                buf.put_u8(b't');
519                let pos = buf.len();
520                buf.put_i32(0); // Placeholder
521
522                buf.put_i16(param_types.len() as i16);
523                for oid in param_types {
524                    buf.put_i32(*oid);
525                }
526
527                let len = buf.len() - pos;
528                buf[pos..pos + 4].copy_from_slice(&(len as i32).to_be_bytes());
529            }
530        }
531    }
532}

534/// Read a null-terminated string from cursor
535fn read_cstring(cursor: &mut Cursor<&[u8]>) -> ProtocolResult<String> {
536    let start = cursor.position() as usize;
537    let buf = cursor.get_ref();
538
539    let end = buf[start..]
540        .iter()
541        .position(|&b| b == 0)
542        .ok_or_else(|| ProtocolError::PostgresError("Unterminated string".to_string()))?;
543
544    let s = String::from_utf8(buf[start..start + end].to_vec())
545        .map_err(|e| ProtocolError::PostgresError(format!("Invalid UTF-8: {e}")))?;
546
547    cursor.set_position((start + end + 1) as u64);
548    Ok(s)
549}
550
551/// Write a null-terminated string to buffer
552fn write_cstring(buf: &mut BytesMut, s: &str) {
553    buf.put_slice(s.as_bytes());
554    buf.put_u8(0);
555}
