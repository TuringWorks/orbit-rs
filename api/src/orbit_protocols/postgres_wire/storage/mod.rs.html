20/// Configuration for storage backend selection
21#[derive(Debug, Clone, Serialize, Deserialize, Default)]
22pub enum StorageBackendConfig {
23    /// Pure in-memory storage (default, non-durable)
24    #[default]
25    Memory,
26    /// LSM-Tree based durable storage
27    Lsm {
28        data_dir: String,
29        memtable_size_limit: usize,
30        max_memtables: usize,
31        bloom_filter_fp_rate: f64,
32        enable_compaction: bool,
33        compaction_threshold: usize,
34    },
35    /// Distributed cluster storage
36    Cluster {
37        cluster_id: String,
38        node_id: String,
39        data_dir: String,
40        replication_factor: usize,
41    },
42}

279/// Factory for creating storage backends
280pub struct StorageBackendFactory;
281
282impl StorageBackendFactory {
283    /// Create a new storage backend based on configuration
284    pub async fn create_backend(
285        config: &StorageBackendConfig,
286    ) -> ProtocolResult<Arc<dyn TableStorage>> {
287        match config {
288            StorageBackendConfig::Memory => Ok(Arc::new(memory::MemoryTableStorage::new())),
289            StorageBackendConfig::Lsm { .. } => {
290                // TODO: Re-enable LSM storage once dependencies are resolved
291                Err(ProtocolError::Other(
292                    "LSM storage backend temporarily disabled".to_string(),
293                ))
294            }
295            StorageBackendConfig::Cluster { .. } => {
296                // For now, return an error as cluster storage isn't implemented yet
297                Err(ProtocolError::Other(
298                    "Cluster storage backend not yet implemented".to_string(),
299                ))
300            }
301        }
302    }
303}

305/// Helper trait for serializing storage data
306pub trait StorageSerializable {
307    fn to_storage_bytes(&self) -> ProtocolResult<Vec<u8>>;
308    fn from_storage_bytes(bytes: &[u8]) -> ProtocolResult<Self>
309    where
310        Self: Sized;
311}

313// Implement for common types
314impl StorageSerializable for TableSchema {
315    fn to_storage_bytes(&self) -> ProtocolResult<Vec<u8>> {
316        serde_json::to_vec(self)
317            .map_err(|e| ProtocolError::SerializationError(format!("Serialization failed: {e}")))
318    }
319
320    fn from_storage_bytes(bytes: &[u8]) -> ProtocolResult<Self> {
321        serde_json::from_slice(bytes)
322            .map_err(|e| ProtocolError::SerializationError(format!("Deserialization failed: {e}")))
323    }
324}

326impl StorageSerializable for HashMap<String, SqlValue> {
327    fn to_storage_bytes(&self) -> ProtocolResult<Vec<u8>> {
328        // Convert SqlValue::Jsonb to a compact binary representation (CBOR) wrapped in an object
329        // so that deserialization can distinguish binary-encoded jsonb from normal json string.
330        use crate::postgres_wire::jsonb;
331        use serde_json::Value as JsonValue;
332
333        let mut map = serde_json::Map::new();
334        for (k, v) in self.iter() {
335            match v {
336                SqlValue::Jsonb(jv) => {
337                    // Encode as base64 of CBOR bytes and mark as __jsonb_b64
338                    let bytes = jsonb::encode_jsonb(jv)?;
339                    let b64 =
340                        base64::prelude::Engine::encode(&base64::prelude::BASE64_STANDARD, &bytes);
341                    let wrapper = JsonValue::Object(serde_json::Map::from_iter(vec![(
342                        "__jsonb_b64".to_string(),
343                        JsonValue::String(b64),
344                    )]));
345                    map.insert(k.clone(), wrapper);
346                }
347                SqlValue::Json(jv) => {
348                    map.insert(k.clone(), jv.clone());
349                }
350                _ => {
351                    // Fallback to serde_json conversion for other SqlValue types
352                    let s = v.to_postgres_string();
353                    map.insert(k.clone(), JsonValue::String(s));
354                }
355            }
356        }
357
358        serde_json::to_vec(&JsonValue::Object(map))
359            .map_err(|e| ProtocolError::SerializationError(format!("Serialization failed: {e}")))
360    }
361
362    fn from_storage_bytes(bytes: &[u8]) -> ProtocolResult<Self> {
363        use crate::postgres_wire::jsonb;
364        use serde_json::Value as JsonValue;
365
366        let v: JsonValue = serde_json::from_slice(bytes).map_err(|e| {
367            ProtocolError::SerializationError(format!("Deserialization failed: {e}"))
368        })?;
369
370        let mut result = HashMap::new();
371        if let JsonValue::Object(map) = v {
372            for (k, val) in map.into_iter() {
373                if let JsonValue::Object(ref inner) = val {
374                    if let Some(JsonValue::String(b64)) = inner.get("__jsonb_b64") {
375                        let bytes =
376                            base64::prelude::Engine::decode(&base64::prelude::BASE64_STANDARD, b64)
377                                .map_err(|e| {
378                                    ProtocolError::SerializationError(format!(
379                                        "base64 decode failed: {e}"
380                                    ))
381                                })?;
382                        let jv = jsonb::decode_jsonb(&bytes)?;
383                        result.insert(k, SqlValue::Jsonb(jv));
384                        continue;
385                    }
386                }
387
388                // Otherwise treat as string value stored via to_postgres_string
389                match val {
390                    JsonValue::String(s) => result.insert(k, SqlValue::Text(s)),
391                    JsonValue::Null => result.insert(k, SqlValue::Null),
392                    other => result.insert(k, SqlValue::Text(other.to_string())),
393                };
394            }
395        }
396
397        Ok(result)
398    }
399}
