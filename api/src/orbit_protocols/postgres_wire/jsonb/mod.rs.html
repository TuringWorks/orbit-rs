116/// Core JSONB value type supporting all PostgreSQL JSON types
117#[derive(Debug, Clone, PartialEq, Serialize, Deserialize, Default)]
118pub enum JsonbValue {
119    /// Null value
120    #[default]
121    Null,
122    /// Boolean value
123    Bool(bool),
124    /// 64-bit integer
125    Number(f64),
126    /// UTF-8 string
127    String(String),
128    /// Array of JSONB values
129    Array(Vec<JsonbValue>),
130    /// Object with string keys and JSONB values
131    Object(HashMap<String, JsonbValue>),
132}
133
134impl JsonbValue {
135    /// Create a JSONB value from a JSON string
136    pub fn from_json_str(json: &str) -> Result<Self, JsonbError> {
137        let value: serde_json::Value = serde_json::from_str(json)?;
138        Ok(Self::from_serde_value(value))
139    }
140
141    /// Convert from serde_json::Value
142    pub fn from_serde_value(value: serde_json::Value) -> Self {
143        match value {
144            serde_json::Value::Null => JsonbValue::Null,
145            serde_json::Value::Bool(b) => JsonbValue::Bool(b),
146            serde_json::Value::Number(n) => JsonbValue::Number(n.as_f64().unwrap_or(0.0)),
147            serde_json::Value::String(s) => JsonbValue::String(s),
148            serde_json::Value::Array(arr) => {
149                JsonbValue::Array(arr.into_iter().map(Self::from_serde_value).collect())
150            }
151            serde_json::Value::Object(obj) => JsonbValue::Object(
152                obj.into_iter()
153                    .map(|(k, v)| (k, Self::from_serde_value(v)))
154                    .collect(),
155            ),
156        }
157    }
158
159    /// Convert to serde_json::Value
160    pub fn to_serde_value(&self) -> serde_json::Value {
161        match self {
162            JsonbValue::Null => serde_json::Value::Null,
163            JsonbValue::Bool(b) => serde_json::Value::Bool(*b),
164            JsonbValue::Number(n) => serde_json::Number::from_f64(*n)
165                .map(serde_json::Value::Number)
166                .unwrap_or(serde_json::Value::Null),
167            JsonbValue::String(s) => serde_json::Value::String(s.clone()),
168            JsonbValue::Array(arr) => {
169                serde_json::Value::Array(arr.iter().map(|v| v.to_serde_value()).collect())
170            }
171            JsonbValue::Object(obj) => serde_json::Value::Object(
172                obj.iter()
173                    .map(|(k, v)| (k.clone(), v.to_serde_value()))
174                    .collect(),
175            ),
176        }
177    }
178
179    /// Get the JSON type as a string
180    pub fn json_type(&self) -> &'static str {
181        match self {
182            JsonbValue::Null => "null",
183            JsonbValue::Bool(_) => "boolean",
184            JsonbValue::Number(_) => "number",
185            JsonbValue::String(_) => "string",
186            JsonbValue::Array(_) => "array",
187            JsonbValue::Object(_) => "object",
188        }
189    }
190
191    /// Check if this value is a JSON object
192    pub fn is_object(&self) -> bool {
193        matches!(self, JsonbValue::Object(_))
194    }
195
196    /// Check if this value is a JSON array
197    pub fn is_array(&self) -> bool {
198        matches!(self, JsonbValue::Array(_))
199    }
200
201    /// Get the size of the JSON value (number of elements for objects/arrays)
202    pub fn size(&self) -> usize {
203        match self {
204            JsonbValue::Array(arr) => arr.len(),
205            JsonbValue::Object(obj) => obj.len(),
206            _ => 0,
207        }
208    }
209
210    /// Convert to pretty-printed JSON string
211    pub fn to_json_string(&self) -> String {
212        serde_json::to_string_pretty(&self.to_serde_value()).unwrap_or_else(|_| "null".to_string())
213    }
214
215    /// Convert to compact JSON string
216    pub fn to_json_compact(&self) -> String {
217        serde_json::to_string(&self.to_serde_value()).unwrap_or_else(|_| "null".to_string())
218    }
219}
220
221impl fmt::Display for JsonbValue {
222    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
223        write!(f, "{}", self.to_json_compact())
224    }
225}

243    IndexOutOfBounds { index: usize },

245    #[error("Invalid operation: {0}")]
246    InvalidOperation(String),

248    #[error("Schema validation failed: {0}")]

257/// Encode a serde_json::Value into CBOR bytes for compact storage.
258pub fn encode_jsonb(value: &serde_json::Value) -> crate::error::ProtocolResult<Vec<u8>> {
259    use crate::error::ProtocolError;
260    serde_cbor::to_vec(value)
261        .map_err(|e| ProtocolError::SerializationError(format!("CBOR encode failed: {e}")))
262}
263
264/// Decode CBOR bytes back into serde_json::Value.
265pub fn decode_jsonb(bytes: &[u8]) -> crate::error::ProtocolResult<serde_json::Value> {
266    use crate::error::ProtocolError;
267    serde_cbor::from_slice(bytes)
268        .map_err(|e| ProtocolError::SerializationError(format!("CBOR decode failed: {e}")))
269}

275    #[test]
276    fn test_jsonb_value_creation() {
277        let json_str =
278            r#"{"name": "Alice", "age": 30, "active": true, "tags": ["rust", "developer"]}"#;
279        let jsonb = JsonbValue::from_json_str(json_str).unwrap();
280
281        assert!(jsonb.is_object());
282        assert_eq!(jsonb.size(), 4);
283
284        if let JsonbValue::Object(obj) = &jsonb {
285            assert_eq!(
286                obj.get("name"),
287                Some(&JsonbValue::String("Alice".to_string()))
288            );
289            assert_eq!(obj.get("age"), Some(&JsonbValue::Number(30.0)));
290            assert_eq!(obj.get("active"), Some(&JsonbValue::Bool(true)));
291        }
292    }
293
294    #[test]
295    fn test_jsonb_array_creation() {
296        let json_str = r#"[1, 2, {"key": "value"}, [3, 4]]"#;
297        let jsonb = JsonbValue::from_json_str(json_str).unwrap();
298
299        assert!(jsonb.is_array());
300        assert_eq!(jsonb.size(), 4);
301    }

307        assert_eq!(JsonbValue::Number(42.0).json_type(), "number");

313    #[test]
314    fn test_jsonb_serialization() {
315        let original = r#"{"name":"Alice","age":30}"#;
316        let jsonb = JsonbValue::from_json_str(original).unwrap();
317        let serialized = jsonb.to_json_compact();
318
319        // Parse the serialized JSON back to JsonbValue to ensure round-trip works
320        let deserialized = JsonbValue::from_json_str(&serialized).unwrap();
321
322        // Compare the JsonbValue structures (which handle f64 numbers consistently)
323        assert_eq!(jsonb, deserialized);
324
325        // Also verify that the serialized form contains the expected data
326        assert!(serialized.contains("Alice"));
327        assert!(serialized.contains("name"));
