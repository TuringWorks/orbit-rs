17/// Connection state
18#[derive(Debug, Clone, Copy, PartialEq, Eq)]
19pub enum ConnectionState {
20    Initial,
21    Authenticating,
22    Authenticated,
23    Ready,
24    InTransaction,
25    Closed,
26}

72    /// Handle a client connection
73    pub async fn handle_connection(&mut self, mut stream: TcpStream) -> ProtocolResult<()> {
74        info!("New PostgreSQL client connection");
75
76        let mut read_buf = BytesMut::with_capacity(8192);
77        let mut write_buf = BytesMut::with_capacity(8192);
78
79        loop {
80            // Read data from client
81            let n = stream.read_buf(&mut read_buf).await?;
82
83            if n == 0 {
84                info!("Client disconnected");
85                break;
86            }
87
88            // Process messages
89            while let Some(msg) = FrontendMessage::parse(&mut read_buf)? {
90                debug!("Received message: {:?}", msg);
91
92                match self.handle_message(msg, &mut write_buf).await {
93                    Ok(should_continue) => {
94                        if !should_continue {
95                            info!("Client terminated connection");
96                            return Ok(());
97                        }
98                    }
99                    Err(e) => {
100                        error!("Error handling message: {}", e);
101                        self.send_error(&mut write_buf, &e.to_string());
102                    }
103                }
104
105                // Flush write buffer
106                if !write_buf.is_empty() {
107                    stream.write_all(&write_buf).await?;
108                    write_buf.clear();
109                }
110            }
111        }
112
113        Ok(())
114    }

116    /// Handle a single frontend message
117    async fn handle_message(
118        &mut self,
119        msg: FrontendMessage,
120        buf: &mut BytesMut,
121    ) -> ProtocolResult<bool> {
122        match msg {
123            FrontendMessage::Startup {
124                protocol_version,
125                parameters,
126            } => {
127                self.handle_startup(protocol_version, parameters, buf)
128                    .await?;
129            }
130            FrontendMessage::Password { password } => {
131                self.handle_password(&password, buf).await?;
132            }
133            FrontendMessage::Query { query } => {
134                self.handle_query(&query, buf).await?;
135            }
136            FrontendMessage::Parse {
137                statement_name,
138                query,
139                param_types,
140            } => {
141                self.handle_parse(&statement_name, &query, param_types, buf)?;
142            }
143            FrontendMessage::Bind {
144                portal,
145                statement,
146                param_formats,
147                params,
148                result_formats,
149            } => {
150                self.handle_bind(
151                    &portal,
152                    &statement,
153                    param_formats,
154                    params,
155                    result_formats,
156                    buf,
157                )?;
158            }
159            FrontendMessage::Execute { portal, max_rows } => {
160                self.handle_execute(&portal, max_rows, buf).await?;
161            }
162            FrontendMessage::Describe { target, name } => {
163                self.handle_describe(target, &name, buf)?;
164            }
165            FrontendMessage::Close { target, name } => {
166                self.handle_close(target, &name, buf)?;
167            }
168            FrontendMessage::Sync => {
169                BackendMessage::ReadyForQuery {
170                    status: TransactionStatus::Idle,
171                }
172                .encode(buf);
173            }
174            FrontendMessage::Flush => {
175                // Nothing to do, data is flushed after each message
176            }
177            FrontendMessage::Terminate => {
178                return Ok(false);
179            }
180            FrontendMessage::SSLRequest => {
181                self.handle_ssl_request(buf).await?;
182            }
183        }
184
185        Ok(true)
186    }

188    /// Handle startup message
189    async fn handle_startup(
190        &mut self,
191        protocol_version: i32,
192        parameters: HashMap<String, String>,
193        buf: &mut BytesMut,
194    ) -> ProtocolResult<()> {
195        info!(
196            "Startup: protocol_version={}, parameters={:?}",
197            protocol_version, parameters
198        );
199
200        self.username = parameters.get("user").cloned();
201        self.database = parameters.get("database").cloned();
202        self.parameters = parameters;
203        self.state = ConnectionState::Authenticating;
204
205        // For simplicity, use trust authentication (no password required)
206        // In production, should use MD5 or SCRAM-SHA-256
207        BackendMessage::Authentication(AuthenticationResponse::Ok).encode(buf);
208
209        // Send parameter status
210        BackendMessage::ParameterStatus {
211            name: "server_version".to_string(),
212            value: "14.0 (Orbit-RS Protocol Adapter)".to_string(),
213        }
214        .encode(buf);
215
216        BackendMessage::ParameterStatus {
217            name: "server_encoding".to_string(),
218            value: "UTF8".to_string(),
219        }
220        .encode(buf);
221
222        BackendMessage::ParameterStatus {
223            name: "client_encoding".to_string(),
224            value: "UTF8".to_string(),
225        }
226        .encode(buf);
227
228        // Send backend key data
229        BackendMessage::BackendKeyData {
230            process_id: self.process_id,
231            secret_key: self.secret_key,
232        }
233        .encode(buf);
234
235        // Ready for query
236        BackendMessage::ReadyForQuery {
237            status: TransactionStatus::Idle,
238        }
239        .encode(buf);
240
241        self.state = ConnectionState::Ready;
242
243        Ok(())
244    }

254    /// Handle simple query
255    async fn handle_query(&mut self, query: &str, buf: &mut BytesMut) -> ProtocolResult<()> {
256        info!("Query: {}", query);
257
258        if query.trim().is_empty() {
259            BackendMessage::EmptyQueryResponse.encode(buf);
260            BackendMessage::ReadyForQuery {
261                status: TransactionStatus::Idle,
262            }
263            .encode(buf);
264            return Ok(());
265        }
266
267        match self.query_engine.execute_query(query).await {
268            Ok(result) => {
269                self.send_query_result(&result, buf);
270                BackendMessage::ReadyForQuery {
271                    status: TransactionStatus::Idle,
272                }
273                .encode(buf);
274            }
275            Err(e) => {
276                self.send_error(buf, &e.to_string());
277                BackendMessage::ReadyForQuery {
278                    status: TransactionStatus::Idle,
279                }
280                .encode(buf);
281            }
282        }
283
284        Ok(())
285    }

305    fn handle_bind(
306        &mut self,
307        portal: &str,
308        statement: &str,
309        _param_formats: Vec<i16>,
310        params: Vec<Option<bytes::Bytes>>,
311        _result_formats: Vec<i16>,

314        debug!("Bind: portal={}, statement={}", portal, statement);
315
316        self.portals
317            .insert(portal.to_string(), (statement.to_string(), params));
318
319        BackendMessage::BindComplete.encode(buf);
320        Ok(())

323    /// Handle execute message
324    async fn handle_execute(
325        &mut self,
326        portal: &str,
327        _max_rows: i32,
328        buf: &mut BytesMut,
329    ) -> ProtocolResult<()> {
330        debug!("Execute: portal={}", portal);
331
332        let (statement_name, params) = self
333            .portals
334            .get(portal)
335            .ok_or_else(|| ProtocolError::PostgresError(format!("Portal not found: {}", portal)))?;
336
337        let query = self
338            .prepared_statements
339            .get(statement_name)
340            .ok_or_else(|| {
341                ProtocolError::PostgresError(format!("Statement not found: {}", statement_name))
342            })?;
343
344        // For now, ignore parameters and execute the query
345        // TODO: Implement parameter substitution
346        let _ = params;
347
348        match self.query_engine.execute_query(query).await {
349            Ok(result) => {
350                self.send_query_result(&result, buf);
351            }
352            Err(e) => {
353                self.send_error(buf, &e.to_string());
354            }
355        }
356
357        Ok(())
358    }

375    /// Handle close message
376    fn handle_close(
377        &mut self,
378        target: super::messages::CloseTarget,
379        name: &str,
380        buf: &mut BytesMut,
381    ) -> ProtocolResult<()> {
382        debug!("Close: target={:?}, name={}", target, name);
383
384        match target {
385            super::messages::CloseTarget::Statement => {
386                self.prepared_statements.remove(name);
387            }
388            super::messages::CloseTarget::Portal => {
389                self.portals.remove(name);
390            }
391        }
392
393        BackendMessage::CloseComplete.encode(buf);
394        Ok(())
395    }

397    /// Send query result
398    fn send_query_result(&self, result: &QueryResult, buf: &mut BytesMut) {
399        match result {
400            QueryResult::Select { columns, rows } => {
401                // Send row description
402                let fields: Vec<FieldDescription> = columns
403                    .iter()
404                    .map(|col| FieldDescription {
405                        name: col.clone(),
406                        table_oid: 0,
407                        column_id: 0,
408                        type_oid: type_oids::TEXT, // Default to TEXT
409                        type_size: -1,
410                        type_modifier: -1,
411                        format: 0,
412                    })
413                    .collect();
414
415                BackendMessage::RowDescription { fields }.encode(buf);
416
417                // Send data rows
418                for row in rows {
419                    let values: Vec<Option<bytes::Bytes>> = row
420                        .iter()
421                        .map(|v| v.as_ref().map(|s| bytes::Bytes::from(s.clone())))
422                        .collect();
423                    BackendMessage::DataRow { values }.encode(buf);
424                }
425
426                // Send command complete
427                BackendMessage::CommandComplete {
428                    tag: format!("SELECT {}", rows.len()),
429                }
430                .encode(buf);
431            }
432            QueryResult::Insert { count } => {
433                BackendMessage::CommandComplete {
434                    tag: format!("INSERT 0 {}", count),
435                }
436                .encode(buf);
437            }
438            QueryResult::Update { count } => {
439                BackendMessage::CommandComplete {
440                    tag: format!("UPDATE {}", count),
441                }
442                .encode(buf);
443            }
444            QueryResult::Delete { count } => {
445                BackendMessage::CommandComplete {
446                    tag: format!("DELETE {}", count),
447                }
448                .encode(buf);
449            }
450        }
451    }

453    /// Handle SSL request
454    async fn handle_ssl_request(&mut self, buf: &mut BytesMut) -> ProtocolResult<()> {
455        // Reject SSL request - send 'N' to indicate SSL not supported
456        buf.put_u8(b'N');
457        Ok(())
458    }
459
460    /// Send error response
461    fn send_error(&self, buf: &mut BytesMut, message: &str) {
462        let mut fields = HashMap::new();
463        fields.insert(b'S', "ERROR".to_string());
464        fields.insert(b'C', "XX000".to_string()); // Internal error
465        fields.insert(b'M', message.to_string());
466
467        BackendMessage::ErrorResponse { fields }.encode(buf);
468    }
