258    /// Create a new lexer with SQL input
259    pub fn new(input: &str) -> Self {
260        let chars: Vec<char> = input.chars().collect();
261        let current_char = chars.first().copied();
262
263        let mut lexer = Self {
264            input: chars,
265            position: 0,
266            current_char,
267            keywords: HashMap::new(),
268        };

460    /// Advance to the next character
461    fn advance(&mut self) {
462        self.position += 1;
463        self.current_char = self.input.get(self.position).copied();
464    }
465
466    /// Peek at the next character without advancing
467    fn peek(&self) -> Option<char> {
468        self.input.get(self.position + 1).copied()
469    }
470
471    /// Peek at character at offset without advancing
472    #[allow(dead_code)]
473    fn peek_offset(&self, offset: usize) -> Option<char> {
474        self.input.get(self.position + offset).copied()
475    }

477    /// Skip whitespace characters
478    fn skip_whitespace(&mut self) {
479        while let Some(ch) = self.current_char {
480            if ch.is_whitespace() {
481                self.advance();
482            } else {
483                break;
484            }
485        }
486    }

488    /// Read a single-line comment
489    fn read_line_comment(&mut self) -> Token {
490        let mut comment = String::new();
491
492        // Skip the "--"
493        self.advance();
494        self.advance();
495
496        while let Some(ch) = self.current_char {
497            if ch == '\n' || ch == '\r' {
498                break;
499            }
500            comment.push(ch);
501            self.advance();
502        }
503
504        Token::Comment(comment)
505    }
506
507    /// Read a multi-line comment
508    fn read_multiline_comment(&mut self) -> Token {
509        let mut comment = String::new();
510
511        // Skip the "/*"
512        self.advance();
513        self.advance();
514
515        while let Some(ch) = self.current_char {
516            if ch == '*' && self.peek() == Some('/') {
517                // Skip "*/"
518                self.advance();
519                self.advance();
520                break;
521            }
522            comment.push(ch);
523            self.advance();
524        }
525
526        Token::Comment(comment)
527    }

529    /// Read a string literal
530    fn read_string_literal(&mut self) -> Token {
531        let quote_char = self.current_char.unwrap();
532        let mut string = String::new();
533        self.advance(); // Skip opening quote
534
535        while let Some(ch) = self.current_char {
536            if ch == quote_char {
537                // Check for escaped quote
538                if self.peek() == Some(quote_char) {
539                    string.push(ch);
540                    self.advance();
541                    self.advance();
542                } else {
543                    // End of string
544                    self.advance();
545                    break;
546                }
547            } else if ch == '\\' {
548                // Handle escape sequences
549                self.advance();
550                if let Some(escaped) = self.current_char {
551                    match escaped {
552                        'n' => string.push('\n'),
553                        't' => string.push('\t'),
554                        'r' => string.push('\r'),
555                        '\\' => string.push('\\'),
556                        '\'' => string.push('\''),
557                        '"' => string.push('"'),
558                        _ => {
559                            string.push('\\');
560                            string.push(escaped);
561                        }
562                    }
563                    self.advance();
564                }
565            } else {
566                string.push(ch);
567                self.advance();
568            }
569        }
570
571        Token::StringLiteral(string)
572    }

574    /// Read a quoted identifier
575    fn read_quoted_identifier(&mut self) -> Token {
576        let mut identifier = String::new();
577        self.advance(); // Skip opening quote
578
579        while let Some(ch) = self.current_char {
580            if ch == '"' {
581                if self.peek() == Some('"') {
582                    // Escaped quote
583                    identifier.push('"');
584                    self.advance();
585                    self.advance();
586                } else {
587                    // End of identifier
588                    self.advance();
589                    break;
590                }
591            } else {
592                identifier.push(ch);
593                self.advance();
594            }
595        }
596
597        Token::QuotedIdentifier(identifier)
598    }

600    /// Read a numeric literal
601    fn read_numeric_literal(&mut self) -> Token {
602        let mut number = String::new();
603
604        // Read integer part
605        while let Some(ch) = self.current_char {
606            if ch.is_ascii_digit() {
607                number.push(ch);
608                self.advance();
609            } else {
610                break;
611            }
612        }
613
614        // Check for decimal point
615        if self.current_char == Some('.') && self.peek().is_some_and(|c| c.is_ascii_digit()) {
616            number.push('.');
617            self.advance();
618
619            // Read fractional part
620            while let Some(ch) = self.current_char {
621                if ch.is_ascii_digit() {
622                    number.push(ch);
623                    self.advance();
624                } else {
625                    break;
626                }
627            }
628        }
629
630        // Check for scientific notation
631        if let Some(ch) = self.current_char {
632            if ch == 'e' || ch == 'E' {
633                number.push(ch);
634                self.advance();
635
636                // Check for sign
637                if let Some(sign) = self.current_char {
638                    if sign == '+' || sign == '-' {
639                        number.push(sign);
640                        self.advance();
641                    }
642                }
643
644                // Read exponent
645                while let Some(ch) = self.current_char {
646                    if ch.is_ascii_digit() {
647                        number.push(ch);
648                        self.advance();
649                    } else {
650                        break;
651                    }
652                }
653            }
654        }
655
656        Token::NumericLiteral(number)
657    }

659    /// Read an identifier or keyword
660    fn read_identifier(&mut self) -> Token {
661        let mut identifier = String::new();
662
663        while let Some(ch) = self.current_char {
664            if ch.is_alphanumeric() || ch == '_' || ch == '$' {
665                identifier.push(ch);
666                self.advance();
667            } else {
668                break;
669            }
670        }
671
672        // Check if it's a keyword
673        if let Some(keyword) = self.keywords.get(&identifier.to_uppercase()) {
674            keyword.clone()
675        } else {
676            Token::Identifier(identifier)
677        }
678    }
679
680    /// Read a parameter placeholder ($1, $2, etc.)
681    fn read_parameter(&mut self) -> Token {
682        self.advance(); // Skip '$'
683        let mut number = String::new();
684
685        while let Some(ch) = self.current_char {
686            if ch.is_ascii_digit() {
687                number.push(ch);
688                self.advance();
689            } else {
690                break;
691            }
692        }
693
694        if let Ok(param_num) = number.parse::<u32>() {
695            Token::Parameter(param_num)
696        } else {
697            // If parsing fails, treat as identifier
698            Token::Identifier(format!("${number}"))
699        }
700    }

702    /// Get the next token
703    pub fn next_token(&mut self) -> Token {
704        loop {
705            match self.current_char {
706                None => return Token::Eof,
707                Some(ch) => {
708                    if ch.is_whitespace() {
709                        self.skip_whitespace();
710                        continue;
711                    }
712
713                    match ch {
714                        // Comments
715                        '-' if self.peek() == Some('-') => return self.read_line_comment(),
716                        '/' if self.peek() == Some('*') => return self.read_multiline_comment(),
717
718                        // String literals
719                        '\'' | '"' if ch == '\'' => return self.read_string_literal(),
720
721                        // Quoted identifiers
722                        '"' => return self.read_quoted_identifier(),
723
724                        // Numeric literals
725                        c if c.is_ascii_digit() => return self.read_numeric_literal(),
726                        '.' if self.peek().is_some_and(|c| c.is_ascii_digit()) => {
727                            return self.read_numeric_literal()
728                        }
729
730                        // Parameters
731                        '$' => return self.read_parameter(),
732
733                        // Identifiers and keywords
734                        c if c.is_alphabetic() || c == '_' => return self.read_identifier(),
735
736                        // Operators and punctuation
737                        '+' => {
738                            self.advance();
739                            return Token::Plus;
740                        }
741                        '-' => {
742                            self.advance();
743                            return Token::Minus;
744                        }
745                        '*' => {
746                            self.advance();
747                            return Token::Multiply;
748                        }
749                        '/' => {
750                            self.advance();
751                            return Token::Divide;
752                        }
753                        '%' => {
754                            self.advance();
755                            return Token::Modulo;
756                        }
757                        '^' => {
758                            self.advance();
759                            return Token::Power;
760                        }
761
762                        '=' => {
763                            self.advance();
764                            return Token::Equal;
765                        }
766                        '!' if self.peek() == Some('=') => {
767                            self.advance();
768                            self.advance();
769                            return Token::NotEqual;
770                        }
771                        '<' => {
772                            self.advance();
773                            return match self.current_char {
774                                Some('=') => {
775                                    if self.peek() == Some('>') {
776                                        // <=> vector cosine distance
777                                        self.advance();
778                                        self.advance();
779                                        Token::VectorCosineDistance
780                                    } else {
781                                        self.advance();
782                                        Token::LessThanOrEqual
783                                    }
784                                }
785                                Some('>') => {
786                                    self.advance();
787                                    Token::NotEqual
788                                }
789                                Some('-') if self.peek() == Some('>') => {
790                                    self.advance();
791                                    self.advance();
792                                    Token::VectorDistance
793                                }
794                                Some('#') if self.peek() == Some('>') => {
795                                    self.advance();
796                                    self.advance();
797                                    Token::VectorInnerProduct
798                                }
799                                Some('<') => {
800                                    self.advance();
801                                    Token::LeftShift
802                                }
803                                _ => Token::LessThan,
804                            };
805                        }
806                        '>' => {
807                            self.advance();
808                            return match self.current_char {
809                                Some('=') => {
810                                    self.advance();
811                                    Token::GreaterThanOrEqual
812                                }
813                                Some('>') => {
814                                    self.advance();
815                                    Token::RightShift
816                                }
817                                _ => Token::GreaterThan,
818                            };
819                        }
820
821                        '|' if self.peek() == Some('|') => {
822                            self.advance();
823                            self.advance();
824                            return Token::Concat;
825                        }
826                        '|' => {
827                            self.advance();
828                            return Token::BitwiseOr;
829                        }
830                        '&' => {
831                            self.advance();
832                            return Token::BitwiseAnd;
833                        }
834                        '~' => {
835                            self.advance();
836                            return Token::BitwiseNot;
837                        }
838
839                        '(' => {
840                            self.advance();
841                            return Token::LeftParen;
842                        }
843                        ')' => {
844                            self.advance();
845                            return Token::RightParen;
846                        }
847                        '[' => {
848                            self.advance();
849                            return Token::LeftBracket;
850                        }
851                        ']' => {
852                            self.advance();
853                            return Token::RightBracket;
854                        }
855                        '{' => {
856                            self.advance();
857                            return Token::LeftBrace;
858                        }
859                        '}' => {
860                            self.advance();
861                            return Token::RightBrace;
862                        }
863                        ',' => {
864                            self.advance();
865                            return Token::Comma;
866                        }
867                        ';' => {
868                            self.advance();
869                            return Token::Semicolon;
870                        }
871                        '.' => {
872                            self.advance();
873                            return Token::Dot;
874                        }
875                        ':' => {
876                            self.advance();
877                            return Token::Colon;
878                        }
879
880                        _ => {
881                            // Unknown character, skip it
882                            self.advance();
883                            continue;
884                        }
885                    }
886                }
887            }
888        }
889    }
890
891    /// Tokenize the entire input
892    pub fn tokenize(&mut self) -> Vec<Token> {
893        let mut tokens = Vec::new();
894
895        loop {
896            let token = self.next_token();
897            let is_eof = matches!(token, Token::Eof);
898            tokens.push(token);
899
900            if is_eof {
901                break;
902            }
903        }
904
905        tokens
906    }
907}
