20/// Represents the estimated cost of a query operation
21#[derive(Debug, Clone, PartialEq)]
22pub struct QueryCost {
23    /// CPU cost (operations)
24    pub cpu_cost: f64,
25    /// I/O cost (page reads)
26    pub io_cost: f64,
27    /// Memory cost (bytes)
28    pub memory_cost: f64,
29    /// Network cost (for distributed queries)
30    pub network_cost: f64,
31    /// Estimated result cardinality
32    pub cardinality: usize,
33}
34
35impl QueryCost {
36    /// Create a new query cost
37    pub fn new(
38        cpu_cost: f64,
39        io_cost: f64,
40        memory_cost: f64,
41        network_cost: f64,
42        cardinality: usize,
43    ) -> Self {
44        Self {
45            cpu_cost,
46            io_cost,
47            memory_cost,
48            network_cost,
49            cardinality,
50        }
51    }
52
53    /// Calculate total cost (weighted sum)
54    pub fn total_cost(&self) -> f64 {
55        // Default weights - could be configurable
56        const CPU_WEIGHT: f64 = 1.0;
57        const IO_WEIGHT: f64 = 10.0;
58        const MEMORY_WEIGHT: f64 = 0.1;
59        const NETWORK_WEIGHT: f64 = 100.0;
60
61        self.cpu_cost * CPU_WEIGHT
62            + self.io_cost * IO_WEIGHT
63            + self.memory_cost * MEMORY_WEIGHT
64            + self.network_cost * NETWORK_WEIGHT
65    }
66
67    /// Combine two costs (e.g., for joins)
68    pub fn combine(&self, other: &QueryCost) -> QueryCost {
69        QueryCost {
70            cpu_cost: self.cpu_cost + other.cpu_cost,
71            io_cost: self.io_cost + other.io_cost,
72            memory_cost: self.memory_cost.max(other.memory_cost), // Max for memory
73            network_cost: self.network_cost + other.network_cost,
74            cardinality: self.cardinality * other.cardinality, // Estimated join result
75        }
76    }
77}

87    /// Apply cost-based optimizations to a statement
88    pub fn optimize(
89        &mut self,
90        statement: Statement,
91        stats: &StatisticsCollector,
92    ) -> ProtocolResult<Statement> {
93        match statement {
94            Statement::Select(select) => {
95                let optimized_select = self.optimize_select(*select, stats)?;
96                Ok(Statement::Select(Box::new(optimized_select)))
97            }
98            // For other statement types, return as-is for now
99            other => Ok(other),
100        }
101    }

103    /// Apply cost-based optimizations with step tracking
104    pub fn optimize_with_steps(
105        &mut self,
106        statement: Statement,
107        stats: &StatisticsCollector,
108    ) -> ProtocolResult<(Statement, Vec<String>)> {
109        let steps = Vec::new();
110
111        match statement {
112            Statement::Select(select) => {
113                let (optimized_select, select_steps) =
114                    self.optimize_select_with_steps(*select, stats)?;
115                let steps = select_steps;
116                Ok((Statement::Select(Box::new(optimized_select)), steps))
117            }
118            other => Ok((other, steps)),
119        }
120    }

122    /// Optimize SELECT statement using cost-based techniques
123    fn optimize_select(
124        &mut self,
125        select: SelectStatement,
126        stats: &StatisticsCollector,
127    ) -> ProtocolResult<SelectStatement> {
128        let mut optimized = select;
129
130        // 1. Join reordering based on estimated costs
131        if self.config.enable_join_reorder {
132            optimized = self.optimize_join_order(optimized, stats)?;
133        }
134
135        // 2. Access method selection (table scan vs index scan)
136        optimized = self.optimize_access_methods(optimized, stats)?;
137
138        // 3. Join algorithm selection (nested loop, hash join, merge join)
139        optimized = self.optimize_join_algorithms(optimized, stats)?;
140
141        Ok(optimized)
142    }

144    /// Optimize SELECT with step tracking
145    fn optimize_select_with_steps(
146        &mut self,
147        select: SelectStatement,
148        stats: &StatisticsCollector,
149    ) -> ProtocolResult<(SelectStatement, Vec<String>)> {
150        let mut steps = Vec::new();
151        let mut optimized = select;
152
153        // 1. Join reordering
154        if self.config.enable_join_reorder {
155            let before = optimized.clone();
156            optimized = self.optimize_join_order(optimized, stats)?;
157            if format!("{before:?}") != format!("{optimized:?}") {
158                steps.push("Applied cost-based join reordering".to_string());
159            }
160        }
161
162        // 2. Access method optimization
163        let before = optimized.clone();
164        optimized = self.optimize_access_methods(optimized, stats)?;
165        if format!("{before:?}") != format!("{optimized:?}") {
166            steps.push("Optimized access methods".to_string());
167        }
168
169        Ok((optimized, steps))
170    }

172    /// Optimize join order based on cost estimates
173    fn optimize_join_order(
174        &self,
175        select: SelectStatement,
176        stats: &StatisticsCollector,
177    ) -> ProtocolResult<SelectStatement> {
178        // Extract all tables involved in the query
179        let tables = self.extract_tables(&select);
180
181        if tables.len() <= 1 {
182            // No joins to optimize
183            return Ok(select);
184        }
185
186        // For small number of tables (<=4), try all permutations
187        // For larger sets, use greedy heuristics
188        if tables.len() <= 4 {
189            self.optimize_join_order_exhaustive(select, &tables, stats)
190        } else {
191            self.optimize_join_order_greedy(select, &tables, stats)
192        }
193    }

221                // Values don't need optimization

235    ) -> ProtocolResult<SelectStatement> {
236        // Generate all permutations and pick the one with lowest cost
237        let mut best_order = tables.to_vec();
238        let mut best_cost = f64::MAX;

240        // For simplicity, evaluate current order and a few reorderings
241        // Full implementation would use dynamic programming
242        for order in self.generate_join_orders(tables) {
243            let cost = self.estimate_join_order_cost(&order, stats);
244            if cost < best_cost {
245                best_cost = cost;
246                best_order = order;
247            }
248        }
249
250        // If we found a better order, apply it
251        if best_order != tables {
252            // Note: Actual reordering of the FROM clause would require
253            // more sophisticated AST manipulation
254            // For now, just return the original select
255        }
256
257        Ok(select)
258    }

267            reversed.reverse();
268            orders.push(reversed);

273            let by_size = tables.to_vec();
274            // In real implementation, would sort by estimated size
275            orders.push(by_size);

281    /// Estimate cost of a specific join order
282    fn estimate_join_order_cost(&self, tables: &[String], stats: &StatisticsCollector) -> f64 {
283        let mut total_cost = 0.0;
284        let mut current_rows = 1;
285
286        for table in tables {
287            let table_cost = self.estimate_table_scan_cost(table, stats);
288            // Cost increases with current intermediate result size
289            total_cost += table_cost.total_cost() * current_rows as f64;
290            current_rows *= table_cost.cardinality;
291        }
292
293        total_cost
294    }

296    /// Optimize join order using greedy heuristics (for large number of tables)
297    fn optimize_join_order_greedy(
298        &self,
299        select: SelectStatement,
300        tables: &[String],
301        stats: &StatisticsCollector,
302    ) -> ProtocolResult<SelectStatement> {
303        // Greedy algorithm: Always pick the smallest table next
304        let mut ordered_tables = Vec::new();
305        let mut remaining_tables: Vec<_> = tables.to_vec();
306
307        while !remaining_tables.is_empty() {
308            // Find table with smallest estimated cardinality
309            let mut best_idx = 0;
310            let mut best_size = usize::MAX;
311
312            for (idx, table) in remaining_tables.iter().enumerate() {
313                let cost = self.estimate_table_scan_cost(table, stats);
314                if cost.cardinality < best_size {
315                    best_size = cost.cardinality;
316                    best_idx = idx;
317                }
318            }
319
320            ordered_tables.push(remaining_tables.remove(best_idx));
321        }
322
323        // Apply the reordering (simplified - actual implementation would
324        // reconstruct the FROM clause)
325        Ok(select)
326    }

350    /// Estimate the cost of a table scan
351    pub fn estimate_table_scan_cost(
352        &self,
353        table_name: &str,
354        stats: &StatisticsCollector,
355    ) -> QueryCost {
356        if let Some(table_stats) = stats.get_table_stats(table_name) {
357            let pages = (table_stats.row_count as f64 / table_stats.rows_per_page as f64).ceil();
358            QueryCost::new(
359                table_stats.row_count as f64, // CPU cost proportional to rows
360                pages,                        // I/O cost proportional to pages
361                0.0,                          // Memory cost minimal for scan
362                0.0,                          // No network cost for local scan
363                table_stats.row_count,
364            )
365        } else {
366            // Default estimates for unknown tables
367            QueryCost::new(1000.0, 100.0, 0.0, 0.0, 1000)
368        }
369    }

371    /// Estimate the cost of an index scan
372    pub fn estimate_index_scan_cost(
373        &self,
374        table_name: &str,
375        _index_name: &str,
376        selectivity: f64,
377        stats: &StatisticsCollector,
378    ) -> QueryCost {
379        if let Some(table_stats) = stats.get_table_stats(table_name) {
380            let estimated_rows = (table_stats.row_count as f64 * selectivity) as usize;
381            let index_pages = (estimated_rows as f64 / 100.0).ceil(); // Assume 100 rows per index page
382
383            QueryCost::new(
384                estimated_rows as f64 * 2.0, // Slightly higher CPU cost for index traversal
385                index_pages + (estimated_rows as f64 / table_stats.rows_per_page as f64), // Index pages + data pages
386                0.0,
387                0.0,
388                estimated_rows,
389            )
390        } else {
391            // Default estimates
392            let estimated_rows = (1000.0 * selectivity) as usize;
393            QueryCost::new(
394                estimated_rows as f64 * 2.0,
395                estimated_rows as f64 / 10.0,
396                0.0,
397                0.0,
398                estimated_rows,
399            )
400        }
401    }

403    /// Estimate the cost of a nested loop join
404    pub fn estimate_nested_loop_join_cost(
405        &self,
406        left_cost: &QueryCost,
407        right_cost: &QueryCost,
408    ) -> QueryCost {
409        QueryCost::new(
410            left_cost.cpu_cost + (left_cost.cardinality as f64 * right_cost.cpu_cost),
411            left_cost.io_cost + (left_cost.cardinality as f64 * right_cost.io_cost),
412            left_cost.memory_cost.max(right_cost.memory_cost),
413            left_cost.network_cost + right_cost.network_cost,
414            (left_cost.cardinality as f64 * right_cost.cardinality as f64 * 0.1) as usize, // Assume 10% join selectivity
415        )
416    }

418    /// Estimate the cost of a hash join
419    pub fn estimate_hash_join_cost(
420        &self,
421        left_cost: &QueryCost,
422        right_cost: &QueryCost,
423    ) -> QueryCost {
424        // Hash join: build hash table from smaller relation, probe with larger
425        let (build_cost, probe_cost) = if left_cost.cardinality <= right_cost.cardinality {
426            (left_cost, right_cost)
427        } else {
428            (right_cost, left_cost)
429        };
430
431        QueryCost::new(
432            build_cost.cpu_cost
433                + probe_cost.cpu_cost
434                + (build_cost.cardinality + probe_cost.cardinality) as f64,
435            build_cost.io_cost + probe_cost.io_cost,
436            build_cost.cardinality as f64 * 8.0, // Memory for hash table (8 bytes per row estimate)
437            build_cost.network_cost + probe_cost.network_cost,
438            (build_cost.cardinality as f64 * probe_cost.cardinality as f64 * 0.1) as usize,
439        )
440    }

460    /// Estimate selectivity of a predicate
461    #[allow(clippy::only_used_in_recursion)]
462    pub fn estimate_selectivity(&self, expr: &Expression, stats: &StatisticsCollector) -> f64 {
463        match expr {
464            Expression::Binary {
465                left,
466                operator,
467                right,
468            } => self.estimate_binary_selectivity(left, operator, right, stats),
469            Expression::Unary { operator, operand } => {
470                self.estimate_unary_selectivity(operator, operand, stats)
471            }
472            Expression::IsNull { negated, .. } => self.estimate_null_selectivity(*negated),
473            _ => 0.5, // Default selectivity
474        }
475    }

477    /// Estimate selectivity for binary expressions
478    fn estimate_binary_selectivity(
479        &self,
480        left: &Expression,
481        operator: &BinaryOperator,
482        right: &Expression,
483        stats: &StatisticsCollector,
484    ) -> f64 {
485        match operator {
486            BinaryOperator::Equal => self.estimate_equality_selectivity(left, stats),
487            BinaryOperator::NotEqual => 0.9,
488            BinaryOperator::LessThan
489            | BinaryOperator::LessThanOrEqual
490            | BinaryOperator::GreaterThan
491            | BinaryOperator::GreaterThanOrEqual => {
492                self.estimate_range_selectivity_for_expr(left, operator, right, stats)
493            }
494            BinaryOperator::And => self.estimate_and_selectivity(left, right, stats),
495            BinaryOperator::Or => self.estimate_or_selectivity(left, right, stats),
496            BinaryOperator::Like => 0.15,
497            BinaryOperator::In => 0.2,
498            _ => 0.5,
499        }
500    }

502    /// Estimate selectivity for equality operations
503    fn estimate_equality_selectivity(&self, left: &Expression, stats: &StatisticsCollector) -> f64 {
504        if let Expression::Column(col_ref) = left {
505            if let Some(table) = &col_ref.table {
506                return stats.estimate_equality_selectivity(table, &col_ref.name);
507            }
508        }
509        0.1 // Default equality selectivity
510    }

512    /// Estimate selectivity for range operations
513    fn estimate_range_selectivity_for_expr(
514        &self,
515        left: &Expression,
516        operator: &BinaryOperator,
517        right: &Expression,
518        stats: &StatisticsCollector,
519    ) -> f64 {
520        if let (Expression::Column(col_ref), Expression::Literal(val)) = (left, right) {
521            if let Some(table) = &col_ref.table {
522                return stats.estimate_range_selectivity(
523                    table,
524                    &col_ref.name,
525                    &format!("{operator:?}"),
526                    &format!("{val:?}"),
527                );
528            }
529        }
530        0.33 // Default range selectivity
531    }

533    /// Estimate selectivity for AND operations
534    fn estimate_and_selectivity(
535        &self,
536        left: &Expression,
537        right: &Expression,
538        stats: &StatisticsCollector,
539    ) -> f64 {
540        let left_sel = self.estimate_selectivity(left, stats);
541        let right_sel = self.estimate_selectivity(right, stats);
542        left_sel * right_sel
543    }
544
545    /// Estimate selectivity for OR operations
546    fn estimate_or_selectivity(
547        &self,
548        left: &Expression,
549        right: &Expression,
550        stats: &StatisticsCollector,
551    ) -> f64 {
552        let left_sel = self.estimate_selectivity(left, stats);
553        let right_sel = self.estimate_selectivity(right, stats);
554        left_sel + right_sel - (left_sel * right_sel)
555    }
556
557    /// Estimate selectivity for unary expressions
558    fn estimate_unary_selectivity(
559        &self,
560        operator: &UnaryOperator,
561        operand: &Expression,
562        stats: &StatisticsCollector,
563    ) -> f64 {
564        match operator {
565            UnaryOperator::Not => 1.0 - self.estimate_selectivity(operand, stats),
566            _ => 0.5,
567        }
568    }

570    /// Estimate selectivity for NULL checks
571    fn estimate_null_selectivity(&self, negated: bool) -> f64 {
572        if negated {
573            0.9 // IS NOT NULL is usually high selectivity
574        } else {
575            0.1 // IS NULL checks are usually selective
576        }
577    }

579    /// Estimate selectivity for complex predicates with correlation
580    pub fn estimate_complex_selectivity(
581        &self,
582        expressions: &[Expression],
583        stats: &StatisticsCollector,
584    ) -> f64 {
585        if expressions.is_empty() {
586            return 1.0;
587        }
588
589        // For multiple independent predicates, multiply selectivities
590        let mut combined_selectivity = 1.0;
591        for expr in expressions {
592            let selectivity = self.estimate_selectivity(expr, stats);
593            combined_selectivity *= selectivity;
594        }
595
596        // Apply correlation factor to avoid over-optimization
597        // Real-world predicates are often correlated
598        let correlation_factor = 0.8;
599        combined_selectivity * correlation_factor + (1.0 - correlation_factor) * 0.5
600    }
