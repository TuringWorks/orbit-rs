12/// A single time series data point
13#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
14pub struct Sample {
15    pub timestamp: u64, // Unix timestamp in milliseconds
16    pub value: f64,
17}

19impl Sample {
20    pub fn new(timestamp: u64, value: f64) -> Self {
21        Self { timestamp, value }
22    }
23
24    pub fn now(value: f64) -> Self {
25        let timestamp = SystemTime::now()
26            .duration_since(UNIX_EPOCH)
27            .unwrap()
28            .as_millis() as u64;
29        Self::new(timestamp, value)
30    }

33/// Time series aggregation functions
34#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
35pub enum AggregationType {
36    Avg,   // Average
37    Sum,   // Sum
38    Min,   // Minimum
39    Max,   // Maximum
40    Count, // Count
41    First, // First value
42    Last,  // Last value
43    Range, // Max - Min
44    Std,   // Standard deviation
45}
46
47impl AggregationType {
48    pub fn parse(s: &str) -> Option<Self> {
49        match s.to_uppercase().as_str() {
50            "AVG" | "AVERAGE" => Some(Self::Avg),
51            "SUM" => Some(Self::Sum),
52            "MIN" | "MINIMUM" => Some(Self::Min),
53            "MAX" | "MAXIMUM" => Some(Self::Max),
54            "COUNT" => Some(Self::Count),
55            "FIRST" => Some(Self::First),
56            "LAST" => Some(Self::Last),
57            "RANGE" => Some(Self::Range),
58            "STD" | "STDDEV" => Some(Self::Std),
59            _ => None,
60        }
61    }
62
63    pub fn as_str(&self) -> &'static str {
64        match self {
65            Self::Avg => "avg",
66            Self::Sum => "sum",
67            Self::Min => "min",
68            Self::Max => "max",
69            Self::Count => "count",
70            Self::First => "first",
71            Self::Last => "last",
72            Self::Range => "range",
73            Self::Std => "std",
74        }
75    }
76
77    /// Apply aggregation to a set of samples
78    pub fn aggregate(&self, samples: &[Sample]) -> f64 {
79        if samples.is_empty() {
80            return 0.0;
81        }
82
83        match self {
84            Self::Avg => {
85                let sum: f64 = samples.iter().map(|s| s.value).sum();
86                sum / samples.len() as f64
87            }
88            Self::Sum => samples.iter().map(|s| s.value).sum(),
89            Self::Min => samples
90                .iter()
91                .map(|s| s.value)
92                .fold(f64::INFINITY, f64::min),
93            Self::Max => samples
94                .iter()
95                .map(|s| s.value)
96                .fold(f64::NEG_INFINITY, f64::max),
97            Self::Count => samples.len() as f64,
98            Self::First => samples.first().map(|s| s.value).unwrap_or(0.0),
99            Self::Last => samples.last().map(|s| s.value).unwrap_or(0.0),
100            Self::Range => {
101                let min = samples
102                    .iter()
103                    .map(|s| s.value)
104                    .fold(f64::INFINITY, f64::min);
105                let max = samples
106                    .iter()
107                    .map(|s| s.value)
108                    .fold(f64::NEG_INFINITY, f64::max);
109                max - min
110            }
111            Self::Std => {
112                if samples.len() <= 1 {
113                    return 0.0;
114                }
115                let mean = samples.iter().map(|s| s.value).sum::<f64>() / samples.len() as f64;
116                let variance = samples
117                    .iter()
118                    .map(|s| (s.value - mean).powi(2))
119                    .sum::<f64>()
120                    / samples.len() as f64;
121                variance.sqrt()
122            }
123        }
124    }
125}

158/// Policy for handling duplicate timestamps

166    Sum,   // Sum values

175            "MIN" => Some(Self::Min),
176            "MAX" => Some(Self::Max),
177            "SUM" => Some(Self::Sum),

179        }
180    }

183        match self {
184            Self::Block => "block",
185            Self::First => "first",
186            Self::Last => "last",
187            Self::Min => "min",
188            Self::Max => "max",

218/// Time series actor for managing time-based data
219#[derive(Debug, Clone, Serialize, Deserialize)]
220pub struct TimeSeriesActor {
221    /// Time series configuration
222    pub config: TimeSeriesConfig,
223    /// Ordered samples (timestamp -> value)
224    pub samples: BTreeMap<u64, f64>,
225    /// Compaction rules for automatic downsampling
226    pub compaction_rules: Vec<CompactionRule>,

228    pub created_at: u64,
229    /// Last update timestamp

231}

239impl TimeSeriesActor {
240    /// Create a new empty time series
241    pub fn new() -> Self {
242        let now = SystemTime::now()
243            .duration_since(UNIX_EPOCH)
244            .unwrap()
245            .as_millis() as u64;
246
247        Self {
248            config: TimeSeriesConfig::default(),
249            samples: BTreeMap::new(),
250            compaction_rules: Vec::new(),
251            created_at: now,
252            updated_at: now,
253        }
254    }

256    /// Create time series with custom configuration
257    pub fn with_config(config: TimeSeriesConfig) -> Self {
258        let mut ts = Self::new();
259        ts.config = config;
260        ts
261    }

263    /// Add a sample to the time series
264    pub fn add_sample(&mut self, timestamp: u64, value: f64) -> Result<(), String> {
265        // Check for duplicate timestamps
266        if self.samples.contains_key(&timestamp) {
267            match self.config.duplicate_policy {
268                DuplicatePolicy::Block => {
269                    return Err("Duplicate timestamp not allowed".to_string());
270                }
271                DuplicatePolicy::First => {
272                    // Keep existing value
273                    return Ok(());
274                }
275                DuplicatePolicy::Last => {
276                    // Replace with new value
277                    self.samples.insert(timestamp, value);
278                }
279                DuplicatePolicy::Min => {
280                    let existing = self.samples.get(&timestamp).unwrap();
281                    self.samples.insert(timestamp, existing.min(value));
282                }
283                DuplicatePolicy::Max => {
284                    let existing = self.samples.get(&timestamp).unwrap();
285                    self.samples.insert(timestamp, existing.max(value));
286                }
287                DuplicatePolicy::Sum => {
288                    let existing = self.samples.get(&timestamp).unwrap();
289                    self.samples.insert(timestamp, existing + value);
290                }
291            }
292        } else {
293            self.samples.insert(timestamp, value);
294        }
295
296        self.updated_at = SystemTime::now()
297            .duration_since(UNIX_EPOCH)
298            .unwrap()
299            .as_millis() as u64;
300
301        // Apply retention policy
302        self.apply_retention();
303
304        // Apply compaction rules
305        self.apply_compaction_rules(timestamp, value);
306
307        Ok(())
308    }
309
310    /// Add multiple samples
311    pub fn add_samples(&mut self, samples: Vec<Sample>) -> Result<Vec<String>, String> {
312        let mut errors = Vec::new();
313
314        for sample in samples {
315            if let Err(e) = self.add_sample(sample.timestamp, sample.value) {
316                errors.push(format!("{}:{}: {}", sample.timestamp, sample.value, e));
317            }
318        }
319
320        Ok(errors)
321    }
322
323    /// Increment/decrement the latest sample or create new one
324    pub fn increment_by(&mut self, timestamp: u64, increment: f64) -> Result<f64, String> {
325        // If timestamp is provided and exists, increment that value
326        if let Some(existing_value) = self.samples.get(&timestamp) {
327            let new_value = existing_value + increment;
328            self.samples.insert(timestamp, new_value);
329            return Ok(new_value);
330        }
331
332        // Otherwise, find the latest sample
333        if let Some((&latest_ts, &latest_value)) = self.samples.iter().next_back() {
334            // If timestamp is newer than latest, create new sample
335            if timestamp > latest_ts {
336                let new_value = increment; // Start from increment for new timestamp
337                self.add_sample(timestamp, new_value)?;
338                Ok(new_value)
339            } else {
340                // Increment the latest sample
341                let new_value = latest_value + increment;
342                self.samples.insert(latest_ts, new_value);
343                Ok(new_value)
344            }
345        } else {
346            // No samples exist, create first sample
347            self.add_sample(timestamp, increment)?;
348            Ok(increment)
349        }
350    }

352    /// Get the latest sample
353    pub fn get_latest(&self) -> Option<Sample> {
354        self.samples
355            .iter()
356            .next_back()
357            .map(|(&timestamp, &value)| Sample::new(timestamp, value))
358    }

360    /// Get samples within a timestamp range
361    pub fn get_range(&self, from_ts: u64, to_ts: u64) -> Vec<Sample> {
362        self.samples
363            .range(from_ts..=to_ts)
364            .map(|(&timestamp, &value)| Sample::new(timestamp, value))
365            .collect()
366    }

368    /// Get samples within a range in reverse order
369    pub fn get_range_reverse(&self, from_ts: u64, to_ts: u64) -> Vec<Sample> {
370        self.samples
371            .range(from_ts..=to_ts)
372            .rev()
373            .map(|(&timestamp, &value)| Sample::new(timestamp, value))
374            .collect()
375    }

377    /// Get aggregated samples within a range
378    pub fn get_range_aggregated(
379        &self,
380        from_ts: u64,
381        to_ts: u64,
382        bucket_duration: u64,
383        aggregation: AggregationType,
384    ) -> Vec<Sample> {
385        let samples = self.get_range(from_ts, to_ts);
386        self.aggregate_samples(samples, bucket_duration, aggregation)
387    }

389    /// Aggregate samples into buckets
390    pub fn aggregate_samples(
391        &self,
392        samples: Vec<Sample>,
393        bucket_duration: u64,
394        aggregation: AggregationType,
395    ) -> Vec<Sample> {
396        if samples.is_empty() || bucket_duration == 0 {
397            return samples;
398        }
399
400        let mut buckets: BTreeMap<u64, Vec<Sample>> = BTreeMap::new();
401
402        // Group samples into buckets
403        for sample in samples {
404            let bucket_start = (sample.timestamp / bucket_duration) * bucket_duration;
405            buckets.entry(bucket_start).or_default().push(sample);
406        }
407
408        // Aggregate each bucket
409        buckets
410            .into_iter()
411            .map(|(bucket_start, bucket_samples)| {
412                let aggregated_value = aggregation.aggregate(&bucket_samples);
413                Sample::new(bucket_start, aggregated_value)
414            })
415            .collect()
416    }
417
418    /// Delete samples within a timestamp range
419    pub fn delete_range(&mut self, from_ts: u64, to_ts: u64) -> usize {
420        let keys_to_remove: Vec<u64> = self
421            .samples
422            .range(from_ts..=to_ts)
423            .map(|(&timestamp, _)| timestamp)
424            .collect();
425
426        let count = keys_to_remove.len();
427        for key in keys_to_remove {
428            self.samples.remove(&key);
429        }
430
431        if count > 0 {
432            self.updated_at = SystemTime::now()
433                .duration_since(UNIX_EPOCH)
434                .unwrap()
435                .as_millis() as u64;
436        }
437
438        count
439    }

441    /// Create a compaction rule
442    pub fn create_compaction_rule(&mut self, rule: CompactionRule) -> Result<(), String> {
443        // Check if rule already exists for the destination
444        if self
445            .compaction_rules
446            .iter()
447            .any(|r| r.dest_key == rule.dest_key)
448        {
449            return Err(format!(
450                "Compaction rule for '{}' already exists",
451                rule.dest_key
452            ));
453        }
454
455        self.compaction_rules.push(rule);
456        Ok(())
457    }
458
459    /// Delete a compaction rule
460    pub fn delete_compaction_rule(&mut self, dest_key: &str) -> bool {
461        let initial_len = self.compaction_rules.len();
462        self.compaction_rules
463            .retain(|rule| rule.dest_key != dest_key);
464        self.compaction_rules.len() < initial_len
465    }

467    /// Get series statistics
468    pub fn get_stats(&self) -> TimeSeriesStats {
469        let total_samples = self.samples.len();
470        let memory_usage = self.estimate_memory_usage();
471        let first_timestamp = self.samples.keys().next().copied();
472        let last_timestamp = self.samples.keys().next_back().copied();
473
474        TimeSeriesStats {
475            total_samples,
476            memory_usage,
477            first_timestamp,
478            last_timestamp,
479            retention_time: self.config.retention,
480            chunk_size: self.config.chunk_size,
481            duplicate_policy: self.config.duplicate_policy,
482            labels: self.config.labels.clone(),
483            rules: self.compaction_rules.len(),
484            source_key: None, // Will be set by caller if needed
485            created_at: self.created_at,
486            updated_at: self.updated_at,
487        }
488    }

490    /// Check if series matches label filters
491    pub fn matches_labels(&self, filters: &HashMap<String, String>) -> bool {
492        filters
493            .iter()
494            .all(|(key, value)| self.config.labels.get(key) == Some(value))
495    }

497    /// Apply retention policy
498    fn apply_retention(&mut self) {
499        if let Some(retention_ms) = self.config.retention {
500            let cutoff_time = SystemTime::now()
501                .duration_since(UNIX_EPOCH)
502                .unwrap()
503                .as_millis() as u64;
504
505            if cutoff_time > retention_ms {
506                let expire_before = cutoff_time - retention_ms;
507                let keys_to_remove: Vec<u64> = self
508                    .samples
509                    .range(..expire_before)
510                    .map(|(&timestamp, _)| timestamp)
511                    .collect();
512
513                for key in keys_to_remove {
514                    self.samples.remove(&key);
515                }
516            }
517        }
518    }

526    /// Estimate memory usage in bytes
527    fn estimate_memory_usage(&self) -> usize {
528        // Rough estimate: timestamp (8) + value (8) + overhead per sample
529        let sample_size = 16 + 8; // 8 bytes overhead for BTreeMap
530        self.samples.len() * sample_size
531            + std::mem::size_of::<TimeSeriesConfig>()
532            + self
533                .config
534                .labels
535                .iter()
536                .map(|(k, v)| k.len() + v.len() + 16)
537                .sum::<usize>()
538    }
539
540    /// Update time series configuration
541    pub fn update_config(&mut self, new_config: TimeSeriesConfig) {
542        self.config = new_config;
543        self.updated_at = SystemTime::now()
544            .duration_since(UNIX_EPOCH)
545            .unwrap()
546            .as_millis() as u64;
547
548        // Apply new retention policy
549        self.apply_retention();
550    }
