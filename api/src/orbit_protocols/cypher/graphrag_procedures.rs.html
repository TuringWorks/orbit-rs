4//! allowing users to call GraphRAG functionality through Cypher procedure syntax.

20/// GraphRAG procedure handler for Bolt/Cypher protocol
21pub struct BoltGraphRAGProcedures {

23}

25impl BoltGraphRAGProcedures {

28        Self {
29            orbit_client: Some(Arc::new(orbit_client)),

31    }

38    /// Execute a GraphRAG procedure call
39    pub async fn execute_procedure(
40        &self,
41        procedure_name: &str,
42        args: &[JsonValue],
43    ) -> ProtocolResult<QueryResult> {
44        if self.orbit_client.is_none() {
45            return Err(ProtocolError::CypherError(
46                "GraphRAG functionality requires OrbitClient integration".to_string(),
47            ));
48        }
49
50        match procedure_name.to_lowercase().as_str() {
51            "orbit.graphrag.buildknowledge" => self.execute_build_knowledge(args).await,
52            "orbit.graphrag.extractentities" => self.execute_extract_entities(args).await,
53            "orbit.graphrag.ragquery" => self.execute_rag_query(args).await,
54            "orbit.graphrag.findpaths" => self.execute_find_paths(args).await,
55            "orbit.graphrag.findsimilar" => self.execute_find_similar(args).await,
56            "orbit.graphrag.semanticsearch" => self.execute_semantic_search(args).await,
57            "orbit.graphrag.getstats" => self.execute_get_stats(args).await,
58            "orbit.graphrag.listentities" => self.execute_list_entities(args).await,
59            "orbit.graphrag.analyzetrends" => self.execute_analyze_trends(args).await,
60            "orbit.graphrag.detectcommunities" => self.execute_detect_communities(args).await,
61            "orbit.graphrag.augmentgraph" => self.execute_augment_graph(args).await,
62            "orbit.graphrag.processdocuments" => self.execute_process_documents(args).await,
63            "orbit.graphrag.streamentities" => self.execute_stream_entities(args).await,
64            _ => Err(ProtocolError::CypherError(format!(
65                "Unknown GraphRAG procedure: {procedure_name}"
66            ))),
67        }
68    }

70    /// Execute orbit.graphrag.buildKnowledge procedure
71    /// CALL orbit.graphrag.buildKnowledge(kg_name, document_id, text, metadata, config)
72    async fn execute_build_knowledge(&self, args: &[JsonValue]) -> ProtocolResult<QueryResult> {
73        if args.len() < 5 {
74            return Err(ProtocolError::CypherError(
75                "orbit.graphrag.buildKnowledge requires 5 arguments: (kg_name, document_id, text, metadata, config)".to_string()
76            ));
77        }
78
79        let orbit_client = self.orbit_client.as_ref().unwrap();
80
81        let kg_name = self.extract_string_arg(&args[0], "kg_name")?;
82        let document_id = self.extract_string_arg(&args[1], "document_id")?;
83        let text = self.extract_string_arg(&args[2], "text")?;
84        let metadata = self.parse_metadata_arg(&args[3])?;
85        let config = self.parse_config_arg(&args[4])?;
86
87        // Extract configuration
88        let extractors = config
89            .get("extractors")
90            .and_then(|v| v.as_array())
91            .map(|arr| {
92                arr.iter()
93                    .filter_map(|v| v.as_str())
94                    .map(|s| s.to_string())
95                    .collect()
96            });
97
98        let build_graph = config
99            .get("build_graph")
100            .and_then(|v| v.as_bool())
101            .unwrap_or(true);
102
103        let generate_embeddings = config
104            .get("generate_embeddings")
105            .and_then(|v| v.as_bool())
106            .unwrap_or(true);
107
108        // Get GraphRAG actor reference
109        let actor_ref = orbit_client
110            .actor_reference::<GraphRAGActor>(Key::StringKey {
111                key: kg_name.clone(),
112            })
113            .await
114            .map_err(|e| ProtocolError::CypherError(format!("Actor error: {e}")))?;
115
116        // Create document processing request
117        let request = GraphRAGDocumentRequest {
118            document_id: document_id.clone(),
119            text,
120            metadata,
121            build_knowledge_graph: build_graph,
122            generate_embeddings,
123            extractors,
124        };
125
126        // Process document
127        let result: DocumentProcessingResult = actor_ref
128            .invoke(
129                "process_document",
130                vec![serde_json::to_value(request).unwrap()],
131            )
132            .await
133            .map_err(|e| ProtocolError::CypherError(format!("Document processing failed: {e}")))?;
134
135        // Format as Cypher result
136        let columns = vec![
137            "kg_name".to_string(),
138            "document_id".to_string(),
139            "entities_extracted".to_string(),
140            "relationships_extracted".to_string(),
141            "extractors_used".to_string(),
142            "processing_time_ms".to_string(),
143            "graph_nodes_created".to_string(),
144        ];
145
146        let _rows = [vec![
147            Some(kg_name),
148            Some(document_id),
149            Some(result.entities.len().to_string()),
150            Some(result.relationships.len().to_string()),
151            Some(result.extractors_used.to_string()),
152            Some(result.processing_time_ms.to_string()),
153            Some(result.entities.len().to_string()),
154        ]];
155
156        Ok(QueryResult {
157            nodes: Vec::new(),
158            relationships: Vec::new(),
159            columns,
160        })
161    }

163    /// Execute orbit.graphrag.extractEntities procedure
164    /// CALL orbit.graphrag.extractEntities(text, config)
165    async fn execute_extract_entities(&self, args: &[JsonValue]) -> ProtocolResult<QueryResult> {
166        if args.len() < 2 {
167            return Err(ProtocolError::CypherError(
168                "orbit.graphrag.extractEntities requires 2 arguments: (text, config)".to_string(),
169            ));
170        }
171
172        let orbit_client = self.orbit_client.as_ref().unwrap();
173
174        let text = self.extract_string_arg(&args[0], "text")?;
175        let config = self.parse_config_arg(&args[1])?;
176
177        let kg_name = config
178            .get("knowledge_graph")
179            .and_then(|v| v.as_str())
180            .unwrap_or("default_kg")
181            .to_string();
182
183        // Get GraphRAG actor reference
184        let actor_ref = orbit_client
185            .actor_reference::<GraphRAGActor>(Key::StringKey {
186                key: kg_name.clone(),
187            })
188            .await
189            .map_err(|e| ProtocolError::CypherError(format!("Actor error: {e}")))?;
190
191        // Create extraction request
192        let request = GraphRAGDocumentRequest {
193            document_id: format!("extract_{}", chrono::Utc::now().timestamp()),
194            text,
195            metadata: HashMap::new(),
196            build_knowledge_graph: false,
197            generate_embeddings: false,
198            extractors: None,
199        };
200
201        // Process document for extraction only
202        let result: DocumentProcessingResult = actor_ref
203            .invoke(
204                "process_document",
205                vec![serde_json::to_value(request).unwrap()],
206            )
207            .await
208            .map_err(|e| ProtocolError::CypherError(format!("Entity extraction failed: {e}")))?;
209
210        // Format as Cypher result with multiple rows (one per entity)
211        let columns = vec![
212            "entity_text".to_string(),
213            "entity_type".to_string(),
214            "confidence".to_string(),
215            "start_pos".to_string(),
216            "end_pos".to_string(),
217        ];
218
219        let _rows: Vec<Vec<Option<String>>> = result
220            .entities
221            .iter()
222            .map(|entity| {
223                vec![
224                    Some(entity.text.clone()),
225                    Some(entity.entity_type.to_str().to_string()),
226                    Some(entity.confidence.to_string()),
227                    Some(entity.start_pos.to_string()),
228                    Some(entity.end_pos.to_string()),
229                ]
230            })
231            .collect();
232
233        Ok(QueryResult {
234            nodes: Vec::new(),
235            relationships: Vec::new(),
236            columns,
237        })
238    }

240    /// Execute orbit.graphrag.ragQuery procedure
241    /// CALL orbit.graphrag.ragQuery(kg_name, query_text, config)
242    async fn execute_rag_query(&self, args: &[JsonValue]) -> ProtocolResult<QueryResult> {
243        if args.len() < 3 {
244            return Err(ProtocolError::CypherError(
245                "orbit.graphrag.ragQuery requires 3 arguments: (kg_name, query_text, config)"
246                    .to_string(),
247            ));
248        }
249
250        let orbit_client = self.orbit_client.as_ref().unwrap();
251
252        let kg_name = self.extract_string_arg(&args[0], "kg_name")?;
253        let query_text = self.extract_string_arg(&args[1], "query_text")?;
254        let config = self.parse_config_arg(&args[2])?;
255
256        // Parse config options
257        let max_hops = config
258            .get("max_hops")
259            .and_then(|v| v.as_u64())
260            .map(|n| n as u32);
261        let context_size = config
262            .get("context_size")
263            .and_then(|v| v.as_u64())
264            .map(|n| n as usize);
265        let llm_provider = config
266            .get("llm_provider")
267            .and_then(|v| v.as_str())
268            .map(|s| s.to_string());
269        let include_explanation = config
270            .get("include_explanation")
271            .and_then(|v| v.as_bool())
272            .unwrap_or(false);
273
274        // Get GraphRAG actor reference
275        let actor_ref = orbit_client
276            .actor_reference::<GraphRAGActor>(Key::StringKey {
277                key: kg_name.clone(),
278            })
279            .await
280            .map_err(|e| ProtocolError::CypherError(format!("Actor error: {e}")))?;
281
282        // Create GraphRAG query
283        let query = GraphRAGQuery {
284            query_text: query_text.clone(),
285            max_hops,
286            context_size,
287            llm_provider,
288            search_strategy: None,
289            include_explanation,
290            max_results: None,
291        };
292
293        // Execute query
294        let result: GraphRAGQueryResult = actor_ref
295            .invoke("query_rag", vec![serde_json::to_value(query).unwrap()])
296            .await
297            .map_err(|e| ProtocolError::CypherError(format!("RAG query failed: {e}")))?;
298
299        // Format as Cypher result
300        let columns = vec![
301            "response".to_string(),
302            "confidence".to_string(),
303            "processing_time_ms".to_string(),
304            "entities_involved".to_string(),
305            "citations".to_string(),
306        ];
307
308        let entities_json = serde_json::to_string(&result.entities_involved).unwrap_or_default();
309        let citations_json = serde_json::to_string(&result.response.citations).unwrap_or_default();
310
311        let _rows = [vec![
312            Some(result.response.response),
313            Some(result.response.confidence.to_string()),
314            Some(result.processing_times.total_ms.to_string()),
315            Some(entities_json),
316            Some(citations_json),
317        ]];
318
319        Ok(QueryResult {
320            nodes: Vec::new(),
321            relationships: Vec::new(),
322            columns,
323        })
324    }

326    /// Execute orbit.graphrag.findPaths procedure
327    /// CALL orbit.graphrag.findPaths(kg_name, from_entity, to_entity, config)
328    async fn execute_find_paths(&self, args: &[JsonValue]) -> ProtocolResult<QueryResult> {
329        if args.len() < 4 {
330            return Err(ProtocolError::CypherError(
331                "orbit.graphrag.findPaths requires 4 arguments: (kg_name, from_entity, to_entity, config)".to_string()
332            ));
333        }
334
335        let orbit_client = self.orbit_client.as_ref().unwrap();
336
337        let kg_name = self.extract_string_arg(&args[0], "kg_name")?;
338        let from_entity = self.extract_string_arg(&args[1], "from_entity")?;
339        let to_entity = self.extract_string_arg(&args[2], "to_entity")?;
340        let config = self.parse_config_arg(&args[3])?;
341
342        // Parse config options
343        let max_hops = config
344            .get("max_hops")
345            .and_then(|v| v.as_u64())
346            .map(|n| n as u32);
347        let include_explanation = config
348            .get("include_explanation")
349            .and_then(|v| v.as_bool())
350            .unwrap_or(false);
351        let max_results = config
352            .get("max_results")
353            .and_then(|v| v.as_u64())
354            .map(|n| n as usize);
355
356        // Get GraphRAG actor reference
357        let actor_ref = orbit_client
358            .actor_reference::<GraphRAGActor>(Key::StringKey {
359                key: kg_name.clone(),
360            })
361            .await
362            .map_err(|e| ProtocolError::CypherError(format!("Actor error: {e}")))?;
363
364        // Create reasoning query
365        let reasoning_query = ReasoningQuery {
366            from_entity: from_entity.clone(),
367            to_entity: to_entity.clone(),
368            max_hops,
369            relationship_types: None,
370            include_explanation,
371            max_results,
372        };
373
374        // Execute reasoning query
375        let paths: Vec<ReasoningPath> = actor_ref
376            .invoke(
377                "find_connection_paths",
378                vec![serde_json::to_value(reasoning_query).unwrap()],
379            )
380            .await
381            .map_err(|e| ProtocolError::CypherError(format!("Reasoning failed: {e}")))?;
382
383        // Format as Cypher result - one row per path
384        let columns = vec![
385            "path_nodes".to_string(),
386            "relationships".to_string(),
387            "score".to_string(),
388            "length".to_string(),
389            "explanation".to_string(),
390        ];
391
392        let rows: Vec<Vec<Option<String>>> = paths
393            .iter()
394            .map(|path| {
395                let nodes_json = serde_json::to_string(&path.nodes).unwrap_or_default();
396                let relationships_json =
397                    serde_json::to_string(&path.relationships).unwrap_or_default();
398
399                vec![
400                    Some(nodes_json),
401                    Some(relationships_json),
402                    Some(path.score.to_string()),
403                    Some(path.length.to_string()),
404                    Some(if include_explanation {
405                        path.explanation.clone()
406                    } else {
407                        "".to_string()
408                    }),
409                ]
410            })
411            .collect();
412
413        // If no paths found, return empty result set with proper structure
414        let _final_rows = if rows.is_empty() {
415            vec![vec![
416                Some("[]".to_string()),
417                Some("[]".to_string()),
418                Some("0.0".to_string()),
419                Some("0".to_string()),
420                Some("No paths found".to_string()),
421            ]]
422        } else {
423            rows
424        };
425
426        Ok(QueryResult {
427            nodes: Vec::new(),
428            relationships: Vec::new(),
429            columns,
430        })
431    }
