2//!
3//! This module provides query execution capabilities for Cypher queries,
4//! integrating with the graph storage layer and translating parsed queries
5//! into graph operations.

17/// Graph query engine that executes Cypher queries
18#[derive(Debug)]
19pub struct GraphEngine<S: GraphStorage> {
20    /// Graph storage backend
21    storage: Arc<S>,
22}

24impl<S: GraphStorage> GraphEngine<S> {
25    /// Create a new graph engine with the given storage backend
26    pub fn new(storage: Arc<S>) -> Self {
27        Self { storage }

30    /// Execute a Cypher query string
31    #[instrument(skip(self, query), fields(query_length = query.len()))]
32    pub async fn execute_query(&self, query: &str) -> ProtocolResult<QueryResult> {
33        let parser = super::cypher_parser::CypherParser::new();
34        let parsed_query = parser.parse(query)?;
35
36        debug!(
37            query = query,
38            clauses_count = parsed_query.clauses.len(),
39            "Executing Cypher query"
40        );
41
42        self.execute_parsed_query(parsed_query).await
43    }
44
45    /// Execute a pre-parsed Cypher query
46    #[instrument(skip(self, query), fields(clauses_count = query.clauses.len()))]
47    pub async fn execute_parsed_query(&self, query: CypherQuery) -> ProtocolResult<QueryResult> {
48        let mut context = ExecutionContext::new();
49        let mut result_nodes = Vec::new();
50        let mut result_relationships = Vec::new();
51
52        for clause in &query.clauses {
53            match clause {
54                CypherClause::Match { pattern } => {
55                    let (nodes, rels) = self.execute_match_clause(pattern, &mut context).await?;
56                    result_nodes.extend(nodes);
57                    result_relationships.extend(rels);
58                }
59                CypherClause::Create { pattern } => {
60                    let (nodes, rels) = self.execute_create_clause(pattern, &mut context).await?;
61                    result_nodes.extend(nodes);
62                    result_relationships.extend(rels);
63                }
64                CypherClause::Return { items } => {
65                    return self
66                        .execute_return_clause(items, &context, result_nodes, result_relationships)
67                        .await;
68                }
69                CypherClause::Where {
70                    condition: _condition,
71                } => {
72                    warn!("WHERE clause not yet fully implemented");
73                    // TODO: Implement WHERE clause filtering
74                }
75            }
76        }
77
78        // If no explicit RETURN clause, return all matched/created data
79        Ok(QueryResult {
80            nodes: result_nodes,
81            relationships: result_relationships,
82            columns: vec!["nodes".to_string(), "relationships".to_string()],
83        })
84    }

86    /// Execute a MATCH clause
87    async fn execute_match_clause(
88        &self,
89        pattern: &Pattern,
90        context: &mut ExecutionContext,
91    ) -> ProtocolResult<(Vec<GraphNode>, Vec<GraphRelationship>)> {
92        let mut nodes = Vec::new();
93        let mut relationships = Vec::new();
94
95        for element in &pattern.elements {
96            match element {
97                PatternElement::Node(node_pattern) => {
98                    let matched_nodes = self.match_node_pattern(node_pattern).await?;
99                    if let Some(var) = &node_pattern.variable {
100                        context.bind_nodes(var.clone(), matched_nodes.clone());
101                    }
102                    nodes.extend(matched_nodes);
103                }
104                PatternElement::Relationship(rel_pattern) => {
105                    let matched_rels = self
106                        .match_relationship_pattern(rel_pattern, context)
107                        .await?;
108                    if let Some(var) = &rel_pattern.variable {
109                        context.bind_relationships(var.clone(), matched_rels.clone());
110                    }
111                    relationships.extend(matched_rels);
112                }
113            }
114        }
115
116        info!(
117            nodes_count = nodes.len(),
118            relationships_count = relationships.len(),
119            "Executed MATCH clause"
120        );
121        Ok((nodes, relationships))
122    }

124    /// Execute a CREATE clause
125    async fn execute_create_clause(
126        &self,
127        pattern: &Pattern,
128        context: &mut ExecutionContext,
129    ) -> ProtocolResult<(Vec<GraphNode>, Vec<GraphRelationship>)> {
130        let mut nodes = Vec::new();
131        let mut relationships = Vec::new();
132
133        for element in &pattern.elements {
134            match element {
135                PatternElement::Node(node_pattern) => {
136                    let created_node = self.create_node_from_pattern(node_pattern).await?;
137                    if let Some(var) = &node_pattern.variable {
138                        context.bind_nodes(var.clone(), vec![created_node.clone()]);
139                    }
140                    nodes.push(created_node);
141                }
142                PatternElement::Relationship(rel_pattern) => {
143                    // For CREATE relationships, we need start and end nodes from context
144                    if let Some(created_rel) = self
145                        .create_relationship_from_pattern(rel_pattern, context)
146                        .await?
147                    {
148                        if let Some(var) = &rel_pattern.variable {
149                            context.bind_relationships(var.clone(), vec![created_rel.clone()]);
150                        }
151                        relationships.push(created_rel);
152                    }
153                }
154            }
155        }
156
157        info!(
158            nodes_count = nodes.len(),
159            relationships_count = relationships.len(),
160            "Executed CREATE clause"
161        );
162        Ok((nodes, relationships))
163    }

165    /// Execute a RETURN clause
166    async fn execute_return_clause(
167        &self,
168        items: &[ReturnItem],
169        context: &ExecutionContext,
170        nodes: Vec<GraphNode>,
171        relationships: Vec<GraphRelationship>,
172    ) -> ProtocolResult<QueryResult> {
173        let mut result_nodes = Vec::new();
174        let mut result_relationships = Vec::new();
175        let mut columns = Vec::new();
176
177        for item in items {
178            columns.push(item.expression.clone());
179
180            // Check if it's a bound variable
181            if let Some(bound_nodes) = context.get_nodes(&item.expression) {
182                result_nodes.extend(bound_nodes.clone());
183            } else if let Some(bound_rels) = context.get_relationships(&item.expression) {
184                result_relationships.extend(bound_rels.clone());
185            } else {
186                // Fallback to returning all nodes/relationships
187                result_nodes.extend(nodes.clone());
188                result_relationships.extend(relationships.clone());
189            }
190        }
191
192        info!(
193            returned_nodes = result_nodes.len(),
194            returned_relationships = result_relationships.len(),
195            "Executed RETURN clause"
196        );
197
198        Ok(QueryResult {
199            nodes: result_nodes,
200            relationships: result_relationships,
201            columns,
202        })
203    }

205    /// Match nodes based on pattern
206    async fn match_node_pattern(&self, pattern: &NodePattern) -> ProtocolResult<Vec<GraphNode>> {
207        if pattern.labels.is_empty() {
208            // No labels specified - this would be expensive for large graphs
209            warn!("Pattern matching without labels is not optimized");
210            return Ok(Vec::new());
211        }
212
213        let mut matched_nodes = Vec::new();
214
215        // For each label, find matching nodes
216        for label in &pattern.labels {
217            let property_filters = if pattern.properties.is_empty() {
218                None
219            } else {
220                Some(pattern.properties.clone())
221            };
222
223            let nodes = self
224                .storage
225                .find_nodes_by_label(label, property_filters, None)
226                .await
227                .map_err(|e| ProtocolError::ActorError(e.to_string()))?;
228
229            matched_nodes.extend(nodes);
230        }
231
232        debug!(pattern = ?pattern, found_count = matched_nodes.len(), "Matched node pattern");
233        Ok(matched_nodes)
234    }
235
236    /// Match relationships based on pattern  
237    async fn match_relationship_pattern(
238        &self,
239        pattern: &RelationshipPattern,
240        context: &ExecutionContext,
241    ) -> ProtocolResult<Vec<GraphRelationship>> {
242        // For relationship matching, we need nodes from context
243        // This is a simplified implementation
244        let mut matched_relationships = Vec::new();
245
246        // Get all nodes from context to search their relationships
247        for nodes in context.bound_nodes.values() {
248            for node in nodes {
249                let direction = match pattern.direction {
250                    RelationshipDirection::Outgoing => Direction::Outgoing,
251                    RelationshipDirection::Incoming => Direction::Incoming,
252                    RelationshipDirection::Both => Direction::Both,
253                };
254
255                let rel_types = pattern.rel_type.as_ref().map(|t| vec![t.clone()]);
256
257                let relationships = self
258                    .storage
259                    .get_relationships(&node.id, direction, rel_types)
260                    .await
261                    .map_err(|e| ProtocolError::ActorError(e.to_string()))?;
262
263                matched_relationships.extend(relationships);
264            }
265        }
266
267        debug!(pattern = ?pattern, found_count = matched_relationships.len(), "Matched relationship pattern");
268        Ok(matched_relationships)
269    }

271    /// Create a node from pattern
272    async fn create_node_from_pattern(&self, pattern: &NodePattern) -> ProtocolResult<GraphNode> {
273        let node = self
274            .storage
275            .create_node(pattern.labels.clone(), pattern.properties.clone())
276            .await
277            .map_err(|e| ProtocolError::ActorError(e.to_string()))?;
278
279        info!(node_id = %node.id, labels = ?pattern.labels, "Created node from pattern");
280        Ok(node)
281    }
282
283    /// Create a relationship from pattern
284    async fn create_relationship_from_pattern(
285        &self,
286        pattern: &RelationshipPattern,
287        context: &ExecutionContext,
288    ) -> ProtocolResult<Option<GraphRelationship>> {
289        // For relationship creation, we need start and end nodes
290        // This is a simplified implementation that takes the first available nodes
291        let all_nodes: Vec<_> = context.bound_nodes.values().flatten().collect();
292
293        if all_nodes.len() < 2 {
294            warn!("Insufficient nodes in context for relationship creation");
295            return Ok(None);
296        }
297
298        let start_node = &all_nodes[0].id;
299        let end_node = &all_nodes[1].id;
300        let rel_type = pattern.rel_type.as_deref().unwrap_or("RELATED");
301
302        let relationship = self
303            .storage
304            .create_relationship(
305                start_node,
306                end_node,
307                rel_type.to_string(),
308                pattern.properties.clone(),
309            )
310            .await
311            .map_err(|e| ProtocolError::ActorError(e.to_string()))?;
312
313        info!(
314            rel_id = %relationship.id,
315            start = %start_node,
316            end = %end_node,
317            rel_type = rel_type,
318            "Created relationship from pattern"
319        );
320
321        Ok(Some(relationship))
322    }

325/// Query execution context for variable bindings
326#[derive(Debug, Default)]
327struct ExecutionContext {
328    /// Bound node variables
329    bound_nodes: HashMap<String, Vec<GraphNode>>,
330    /// Bound relationship variables
331    bound_relationships: HashMap<String, Vec<GraphRelationship>>,
332}
333
334impl ExecutionContext {
335    fn new() -> Self {
336        Self::default()
337    }
338
339    fn bind_nodes(&mut self, variable: String, nodes: Vec<GraphNode>) {
340        self.bound_nodes.insert(variable, nodes);
341    }
342
343    fn bind_relationships(&mut self, variable: String, relationships: Vec<GraphRelationship>) {
344        self.bound_relationships.insert(variable, relationships);
345    }
346
347    fn get_nodes(&self, variable: &str) -> Option<&Vec<GraphNode>> {
348        self.bound_nodes.get(variable)
349    }
350
351    fn get_relationships(&self, variable: &str) -> Option<&Vec<GraphRelationship>> {
352        self.bound_relationships.get(variable)
353    }
354}

356/// Query execution result
357#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
358pub struct QueryResult {
359    /// Result nodes
360    pub nodes: Vec<GraphNode>,
361    /// Result relationships
362    pub relationships: Vec<GraphRelationship>,
363    /// Column names for the result
364    pub columns: Vec<String>,
365}

378    #[tokio::test]
379    async fn test_create_node_query() {
380        let engine = create_test_engine().await;
381        let query = "CREATE (n:Person {name: 'Alice'}) RETURN n";
382
383        let result = engine.execute_query(query).await;
384        assert!(result.is_ok());
385
386        let query_result = result.unwrap();
387        assert_eq!(query_result.nodes.len(), 1);
388        assert!(query_result.nodes[0].has_label("Person"));
389    }

391    #[tokio::test]
392    async fn test_match_node_query() {
393        let engine = create_test_engine().await;
394
395        // First create a node
396        let create_result = engine
397            .execute_query("CREATE (n:Person {name: 'Alice'})")
398            .await;
399        assert!(create_result.is_ok());
400
401        // Then match it
402        let match_result = engine.execute_query("MATCH (n:Person) RETURN n").await;
403        assert!(match_result.is_ok());
