1//! RESP protocol server
2
3use futures::{SinkExt, StreamExt};
4use std::sync::Arc;
5use tokio::net::{TcpListener, TcpStream};
6use tokio_util::codec::Framed;
7use tracing::{debug, error, info};
8
9use super::{types::RespValue, CommandHandler, RespCodec};
10use crate::error::ProtocolResult;
11use orbit_client::OrbitClient;
12
13/// RESP protocol server
14pub struct RespServer {
15    bind_addr: String,
16    command_handler: Arc<CommandHandler>,
17}
18
19impl RespServer {
20    /// Create a new RESP server
21    pub fn new(bind_addr: impl Into<String>, orbit_client: OrbitClient) -> Self {
22        Self {
23            bind_addr: bind_addr.into(),
24            command_handler: Arc::new(CommandHandler::new(orbit_client)),
25        }
26    }
27
28    /// Start the server
29    pub async fn run(&self) -> ProtocolResult<()> {
30        let listener = TcpListener::bind(&self.bind_addr).await?;
31        info!("RESP server listening on {}", self.bind_addr);
32
33        loop {
34            match listener.accept().await {
35                Ok((socket, addr)) => {
36                    debug!("New RESP connection from {}", addr);
37                    let handler = Arc::clone(&self.command_handler);
38                    tokio::spawn(async move {
39                        if let Err(e) = Self::handle_connection(socket, handler).await {
40                            error!("Connection error: {}", e);
41                        }
42                    });
43                }
44                Err(e) => {
45                    error!("Accept error: {}", e);
46                }
47            }
48        }
49    }
50
51    async fn handle_connection(
52        socket: TcpStream,
53        handler: Arc<CommandHandler>,
54    ) -> ProtocolResult<()> {
55        let mut framed = Framed::new(socket, RespCodec::new());
56
57        while let Some(result) = framed.next().await {
58            if !Self::process_message_result(&mut framed, &handler, result).await? {
59                break;
60            }
61        }
62
63        debug!("Connection closed");
64        Ok(())
65    }
66
67    /// Process a single message result, returns false if connection should be closed
68    async fn process_message_result(
69        framed: &mut Framed<TcpStream, RespCodec>,
70        handler: &Arc<CommandHandler>,
71        result: Result<RespValue, crate::error::ProtocolError>,
72    ) -> ProtocolResult<bool> {
73        match result {
74            Ok(command) => {
75                debug!("Received command: {}", command);
76                let response = Self::handle_command_with_error_handling(handler, command).await;
77                Self::send_response(framed, response).await
78            }
79            Err(e) => {
80                error!("Protocol error: {}", e);
81                Ok(false) // Close connection on protocol error
82            }
83        }
84    }
85
86    /// Handle command with proper error handling
87    async fn handle_command_with_error_handling(
88        handler: &Arc<CommandHandler>,
89        command: RespValue,
90    ) -> RespValue {
91        handler.handle_command(command).await.unwrap_or_else(|e| {
92            error!("Command error: {}", e);
93            RespValue::error(format!("ERR {e}"))
94        })
95    }
96
97    /// Send response and handle send errors
98    async fn send_response(
99        framed: &mut Framed<TcpStream, RespCodec>,
100        response: RespValue,
101    ) -> ProtocolResult<bool> {
102        if let Err(e) = framed.send(response).await {
103            error!("Send error: {}", e);
104            Ok(false) // Close connection on send error
105        } else {
106            Ok(true) // Continue processing
107        }
108    }
109}
