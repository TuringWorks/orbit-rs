1use std::collections::HashMap;

3/// Extensions for working with metrics
4pub struct MetricsExt;
5
6impl MetricsExt {
7    /// Increment a counter with labels
8    pub fn increment_counter(name: String, labels: HashMap<String, String>) {
9        // For now, just use the basic metrics without labels to avoid lifetime issues
10        // This can be enhanced later with proper metrics recorder setup
11        metrics::counter!(name.clone()).increment(1);
12
13        // Add labels as additional separate metrics for context
14        for (key, value) in labels {
15            let metric_name = format!("{}.{}", name, key);
16            metrics::counter!(metric_name, "value" => value).increment(1);
17        }
18    }
19
20    /// Set a gauge value with labels
21    pub fn set_gauge(name: String, value: f64, labels: HashMap<String, String>) {
22        metrics::gauge!(name.clone()).set(value);
23
24        // Add context through separate labeled metrics
25        for (key, label_value) in labels {
26            let metric_name = format!("{}.{}", name, key);
27            metrics::gauge!(metric_name, "value" => label_value).set(value);
28        }
29    }
30
31    /// Record a histogram value with labels
32    pub fn record_histogram(name: String, value: f64, labels: HashMap<String, String>) {
33        metrics::histogram!(name.clone()).record(value);
34
35        // Add context through separate labeled metrics
36        for (key, label_value) in labels {
37            let metric_name = format!("{}.{}", name, key);
38            metrics::histogram!(metric_name, "value" => label_value).record(value);
39        }
40    }
41}
