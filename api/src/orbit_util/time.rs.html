1use chrono::{DateTime, Duration, Utc};
2use std::time::SystemTime;
3
4/// Time utilities for working with timestamps and durations
5pub struct TimeUtils;
6
7impl TimeUtils {
8    /// Get current UTC timestamp
9    pub fn now() -> DateTime<Utc> {
10        Utc::now()
11    }
12
13    /// Convert SystemTime to `DateTime<Utc>`
14    pub fn from_system_time(system_time: SystemTime) -> DateTime<Utc> {
15        system_time.into()
16    }
17
18    /// Get timestamp in milliseconds since Unix epoch
19    pub fn timestamp_millis() -> i64 {
20        Utc::now().timestamp_millis()
21    }
22
23    /// Create a Duration from milliseconds
24    pub fn duration_from_millis(millis: i64) -> Duration {
25        Duration::milliseconds(millis)
26    }
27
28    /// Create a Duration from seconds
29    pub fn duration_from_secs(secs: i64) -> Duration {
30        Duration::seconds(secs)
31    }
32}
