1//! SQL-style spatial functions for PostGIS compatibility.
2//!
3//! This module provides PostGIS-compatible spatial functions that can be
4//! used across multiple query protocols.
5
6use super::{Point, SpatialError, SpatialGeometry};
7
8/// PostGIS-compatible spatial functions.
9pub struct SpatialFunctions;
10
11impl SpatialFunctions {
12    /// ST_Point - Create a point geometry.
13    pub fn st_point(x: f64, y: f64) -> SpatialGeometry {
14        SpatialGeometry::Point(Point::new(x, y, None))
15    }
16
17    /// ST_MakePoint - Create a point with optional SRID.
18    pub fn st_make_point(x: f64, y: f64, srid: Option<i32>) -> SpatialGeometry {
19        SpatialGeometry::Point(Point::new(x, y, srid))
20    }
21
22    /// ST_Distance - Calculate distance between geometries.
23    pub fn st_distance(
24        geom1: &SpatialGeometry,
25        geom2: &SpatialGeometry,
26    ) -> Result<f64, SpatialError> {
27        super::operations::SpatialOperations::distance(geom1, geom2)
28    }
29
30    /// ST_Area - Calculate area of geometry.
31    pub fn st_area(geometry: &SpatialGeometry) -> Result<f64, SpatialError> {
32        super::operations::SpatialOperations::area(geometry)
33    }
34
35    /// ST_Within - Test if geometry is within another.
36    pub fn st_within(
37        _geom1: &SpatialGeometry,
38        _geom2: &SpatialGeometry,
39    ) -> Result<bool, SpatialError> {

41        Ok(false)
42    }
43}
