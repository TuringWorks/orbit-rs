61/// Batch processor for operations
62pub struct BatchProcessor<T>
63where
64    T: Clone + Send + Sync + 'static,
65{
66    config: BatchConfig,
67    queue: Arc<Mutex<VecDeque<BatchedOperation<T>>>>,
68    processor: Arc<BatchProcessorFn<T>>,
69    stats: Arc<RwLock<BatchStats>>,
70}
71
72impl<T> BatchProcessor<T>
73where
74    T: Clone + Send + Sync + 'static,
75{
76    pub fn new<F>(config: BatchConfig, processor: F) -> Self
77    where
78        F: Fn(Vec<T>) -> futures::future::BoxFuture<'static, OrbitResult<Vec<bool>>>
79            + Send
80            + Sync
81            + 'static,
82    {
83        Self {
84            config,
85            queue: Arc::new(Mutex::new(VecDeque::new())),
86            processor: Arc::new(processor),
87            stats: Arc::new(RwLock::new(BatchStats::default())),
88        }
89    }
90
91    /// Add an operation to the batch queue
92    pub async fn add(&self, operation: T) -> OrbitResult<()> {
93        let batched = BatchedOperation::new(operation);
94        let mut queue = self.queue.lock().await;
95        queue.push_back(batched);
96
97        // Update stats
98        let mut stats = self.stats.write().await;
99        stats.total_operations += 1;
100        stats.current_queue_size = queue.len();
101
102        Ok(())
103    }
104
105    /// Add a high-priority operation
106    pub async fn add_priority(&self, operation: T, priority: u8) -> OrbitResult<()> {
107        let batched = BatchedOperation::new(operation).with_priority(priority);
108        let mut queue = self.queue.lock().await;
109
110        // Insert based on priority
111        let pos = queue
112            .iter()
113            .position(|op| op.priority < priority)
114            .unwrap_or(queue.len());
115        queue.insert(pos, batched);
116
117        // Update stats
118        let mut stats = self.stats.write().await;
119        stats.total_operations += 1;
120        stats.current_queue_size = queue.len();
121
122        Ok(())
123    }
124
125    /// Process a batch
126    async fn process_batch(&self) -> OrbitResult<()> {
127        let batch_start = Instant::now();
128
129        // Collect batch
130        let operations: Vec<T> = {
131            let mut queue = self.queue.lock().await;
132            let batch_size = self.config.max_batch_size.min(queue.len());
133
134            if batch_size == 0 {
135                return Ok(());
136            }
137
138            let batch: Vec<T> = queue.drain(0..batch_size).map(|op| op.operation).collect();
139
140            batch
141        };
142
143        let batch_count = operations.len();
144
145        // Process batch
146        let results = (self.processor)(operations).await?;
147
148        let batch_duration = batch_start.elapsed();
149
150        // Update stats
151        let mut stats = self.stats.write().await;
152        stats.total_batches += 1;
153        stats.total_processed += batch_count;
154        stats.successful_operations += results.iter().filter(|&&r| r).count();
155        stats.average_batch_size = ((stats.average_batch_size * (stats.total_batches - 1) as f64)
156            + batch_count as f64)
157            / stats.total_batches as f64;
158        stats.average_processing_time_ms = ((stats.average_processing_time_ms
159            * (stats.total_batches - 1) as f64)
160            + batch_duration.as_millis() as f64)
161            / stats.total_batches as f64;
162
163        debug!(
164            "Processed batch of {} operations in {:?}",
165            batch_count, batch_duration
166        );
167
168        Ok(())
169    }
170
171    /// Start batch processing loop
172    pub async fn start(&self) {
173        let processor = self.clone();
174        tokio::spawn(async move {
175            let mut interval = tokio::time::interval(processor.config.max_wait_time);
176
177            loop {
178                interval.tick().await;
179
180                let should_process = {
181                    let queue = processor.queue.lock().await;
182                    queue.len() >= processor.config.min_batch_size
183                        || (!queue.is_empty()
184                            && queue
185                                .front()
186                                .map(|op| op.timestamp.elapsed() >= processor.config.max_wait_time)
187                                .unwrap_or(false))
188                };
189
190                if should_process {
191                    if let Err(e) = processor.process_batch().await {
192                        warn!("Batch processing failed: {}", e);
193                    }
194                }
195            }
196        });
197
198        info!("Batch processor started");
199    }
200
201    /// Get batch statistics
202    pub async fn get_stats(&self) -> BatchStats {
203        self.stats.read().await.clone()
204    }
205
206    /// Get current queue size
207    pub async fn queue_size(&self) -> usize {
208        self.queue.lock().await.len()
209    }
210}

269#[derive(Debug)]
270pub struct PooledConnection<C> {
271    connection: C,
272    created_at: Instant,
273    last_used: Instant,
274    use_count: usize,
275}

277impl<C> PooledConnection<C> {
278    pub fn new(connection: C) -> Self {
279        let now = Instant::now();
280        Self {
281            connection,
282            created_at: now,
283            last_used: now,
284            use_count: 0,
285        }
286    }
287
288    pub fn connection(&mut self) -> &mut C {
289        self.last_used = Instant::now();
290        self.use_count += 1;
291        &mut self.connection
292    }
293
294    pub fn is_expired(&self, max_lifetime: Duration, idle_timeout: Duration) -> bool {
295        self.created_at.elapsed() > max_lifetime || self.last_used.elapsed() > idle_timeout
296    }

299/// Generic connection pool
300pub struct ConnectionPool<C>
301where
302    C: Send + Sync,
303{
304    config: ConnectionPoolConfig,
305    connections: Arc<Mutex<Vec<PooledConnection<C>>>>,
306    semaphore: Arc<Semaphore>,
307    factory: Arc<dyn Fn() -> futures::future::BoxFuture<'static, OrbitResult<C>> + Send + Sync>,
308    stats: Arc<RwLock<PoolStats>>,
309}
310
311impl<C> ConnectionPool<C>
312where
313    C: Send + Sync + 'static,
314{
315    pub fn new<F>(config: ConnectionPoolConfig, factory: F) -> Self
316    where
317        F: Fn() -> futures::future::BoxFuture<'static, OrbitResult<C>> + Send + Sync + 'static,
318    {
319        let semaphore = Arc::new(Semaphore::new(config.max_size));
320
321        Self {
322            config,
323            connections: Arc::new(Mutex::new(Vec::new())),
324            semaphore,
325            factory: Arc::new(factory),
326            stats: Arc::new(RwLock::new(PoolStats::default())),
327        }
328    }
329
330    /// Get a connection from the pool
331    pub async fn get(&self) -> OrbitResult<C> {
332        // Acquire permit
333        let _permit = timeout(self.config.connect_timeout, self.semaphore.acquire())
334            .await
335            .map_err(|_| OrbitError::timeout("Connection pool timeout"))?
336            .map_err(|e| OrbitError::internal(format!("Semaphore error: {e}")))?;
337
338        // Try to get an existing connection
339        {
340            let mut connections = self.connections.lock().await;
341
342            // Remove expired connections
343            connections.retain(|conn| {
344                !conn.is_expired(self.config.max_lifetime, self.config.idle_timeout)
345            });
346
347            if let Some(mut pooled_conn) = connections.pop() {
348                let conn = std::mem::replace(
349                    pooled_conn.connection(),
350                    unsafe { std::mem::zeroed() }, // Temporary placeholder
351                );
352
353                // Update stats
354                let mut stats = self.stats.write().await;
355                stats.total_acquired += 1;
356                stats.current_active += 1;
357                stats.current_idle = connections.len();
358
359                // Note: In a real implementation, we'd need a proper way to extract the connection
360                // For now, this is a simplified version
361                return Ok(conn);
362            }
363        }
364
365        // Create new connection
366        let start = Instant::now();
367        let connection = (self.factory)().await?;
368        let create_duration = start.elapsed();
369
370        // Update stats
371        let mut stats = self.stats.write().await;
372        stats.total_created += 1;
373        stats.total_acquired += 1;
374        stats.current_active += 1;
375        stats.average_create_time_ms = ((stats.average_create_time_ms
376            * (stats.total_created - 1) as f64)
377            + create_duration.as_millis() as f64)
378            / stats.total_created as f64;
379
380        Ok(connection)
381    }
382
383    /// Return a connection to the pool
384    pub async fn return_connection(&self, connection: C) {
385        let mut connections = self.connections.lock().await;
386
387        if connections.len() < self.config.max_size {
388            connections.push(PooledConnection::new(connection));
389
390            // Update stats
391            let mut stats = self.stats.write().await;
392            stats.current_active = stats.current_active.saturating_sub(1);
393            stats.current_idle = connections.len();
394        }
395    }
396
397    /// Get pool statistics
398    pub async fn get_stats(&self) -> PoolStats {
399        self.stats.read().await.clone()
400    }
401
402    /// Start background maintenance tasks
403    pub async fn start_maintenance(&self) {
404        let pool = self.clone();
405        tokio::spawn(async move {
406            let mut interval = tokio::time::interval(pool.config.health_check_interval);
407
408            loop {
409                interval.tick().await;
410
411                // Clean up expired connections
412                let mut connections = pool.connections.lock().await;
413                let initial_count = connections.len();
414                connections.retain(|conn| {
415                    !conn.is_expired(pool.config.max_lifetime, pool.config.idle_timeout)
416                });
417                let removed = initial_count - connections.len();
418
419                if removed > 0 {
420                    debug!("Cleaned up {} expired connections", removed);
421                    let mut stats = pool.stats.write().await;
422                    stats.current_idle = connections.len();
423                }
424            }
425        });
426
427        info!("Connection pool maintenance started");
428    }

456/// Resource manager for limiting resource usage
457pub struct ResourceManager {
458    /// Maximum memory usage (in bytes)
459    max_memory: usize,
460    /// Current memory usage estimate
461    current_memory: Arc<RwLock<usize>>,
462    /// Maximum number of concurrent operations
463    max_concurrent: usize,
464    /// Current concurrency semaphore
465    concurrency_limiter: Arc<Semaphore>,
466}
467
468impl ResourceManager {
469    pub fn new(max_memory: usize, max_concurrent: usize) -> Self {
470        Self {
471            max_memory,
472            current_memory: Arc::new(RwLock::new(0)),
473            max_concurrent,
474            concurrency_limiter: Arc::new(Semaphore::new(max_concurrent)),
475        }
476    }
477
478    /// Acquire resources for an operation
479    pub async fn acquire(&self, memory_estimate: usize) -> OrbitResult<ResourceGuard> {
480        // Acquire concurrency permit
481        let permit = Arc::clone(&self.concurrency_limiter)
482            .acquire_owned()
483            .await
484            .map_err(|e| OrbitError::internal(format!("Concurrency limiter error: {e}")))?;
485
486        // Check memory availability
487        let mut current = self.current_memory.write().await;
488        if *current + memory_estimate > self.max_memory {
489            return Err(OrbitError::internal("Memory limit exceeded"));
490        }
491
492        *current += memory_estimate;
493
494        Ok(ResourceGuard {
495            memory_estimate,
496            current_memory: Arc::clone(&self.current_memory),
497            _permit: permit,
498        })
499    }
500
501    /// Get current resource usage
502    pub async fn current_usage(&self) -> (usize, usize) {
503        let memory = *self.current_memory.read().await;
504        let concurrent = self.max_concurrent - self.concurrency_limiter.available_permits();
505        (memory, concurrent)
506    }
507}

509/// RAII guard for resource usage
510pub struct ResourceGuard {
511    memory_estimate: usize,
512    current_memory: Arc<RwLock<usize>>,
513    _permit: tokio::sync::OwnedSemaphorePermit,
514}
515
516impl Drop for ResourceGuard {
517    fn drop(&mut self) {
518        let memory_estimate = self.memory_estimate;
519        let current_memory = Arc::clone(&self.current_memory);
520
521        tokio::spawn(async move {
522            let mut current = current_memory.write().await;
523            *current = current.saturating_sub(memory_estimate);
524        });
525    }
526}
