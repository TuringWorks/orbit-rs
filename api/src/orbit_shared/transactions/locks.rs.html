10/// Unique identifier for a distributed lock
11#[derive(Debug, Clone, PartialEq, Eq, Hash, Serialize, Deserialize)]
12pub struct LockId {
13    pub resource_id: String,
14    pub lock_key: String,
15}
16
17impl LockId {
18    pub fn new(resource_id: String) -> Self {
19        Self {
20            resource_id: resource_id.clone(),
21            lock_key: format!("lock:{resource_id}"),
22        }
23    }
24
25    pub fn from_key(lock_key: String) -> Self {
26        let resource_id = lock_key
27            .strip_prefix("lock:")
28            .unwrap_or(&lock_key)
29            .to_string();
30        Self {
31            resource_id,
32            lock_key,
33        }
34    }
35}

70/// Status of a lock
71#[derive(Debug, Clone, PartialEq, Eq, Serialize, Deserialize)]
72pub enum LockStatus {
73    /// Lock is available
74    Available,
75    /// Lock is held exclusively
76    HeldExclusive { owner: LockOwner },
77    /// Lock is held by one or more shared holders
78    HeldShared { owners: Vec<LockOwner> },
79    /// Lock request is waiting in queue
80    Waiting { position: usize },
81}

83/// Distributed lock with metadata
84#[derive(Debug, Clone)]
85pub struct DistributedLock {
86    pub lock_id: LockId,
87    pub mode: LockMode,
88    pub owners: Vec<LockOwner>,
89    pub acquired_at: SystemTime,
90    pub expires_at: SystemTime,
91    pub wait_queue: VecDeque<LockRequest>,
92}
93
94impl DistributedLock {
95    pub fn new(lock_id: LockId) -> Self {
96        Self {
97            lock_id,
98            mode: LockMode::Exclusive,
99            owners: Vec::new(),
100            acquired_at: SystemTime::now(),
101            expires_at: SystemTime::now(),
102            wait_queue: VecDeque::new(),
103        }
104    }
105
106    /// Check if the lock is currently held
107    pub fn is_held(&self) -> bool {
108        !self.owners.is_empty() && SystemTime::now() < self.expires_at
109    }
110
111    /// Check if the lock is available
112    pub fn is_available(&self) -> bool {
113        !self.is_held()
114    }
115
116    /// Check if a specific owner holds this lock
117    pub fn is_held_by(&self, owner: &LockOwner) -> bool {
118        self.owners.iter().any(|o| o == owner)
119    }
120
121    /// Try to acquire the lock
122    pub fn try_acquire(&mut self, owner: LockOwner, mode: LockMode, duration: Duration) -> bool {
123        if self.is_available() {
124            // Lock is available, acquire it
125            self.owners.push(owner);
126            self.mode = mode;
127            self.acquired_at = SystemTime::now();
128            self.expires_at = self.acquired_at + duration;
129            true
130        } else if matches!(mode, LockMode::Shared) && matches!(self.mode, LockMode::Shared) {
131            // Can add another shared lock holder
132            self.owners.push(owner);
133            true
134        } else {
135            // Lock is not available
136            false
137        }
138    }
139
140    /// Release the lock for a specific owner
141    pub fn release(&mut self, owner: &LockOwner) -> bool {
142        if let Some(pos) = self.owners.iter().position(|o| o == owner) {
143            self.owners.remove(pos);
144            if self.owners.is_empty() {
145                // Last owner released, process wait queue
146                return true;
147            }
148        }
149        false
150    }
151
152    /// Add a request to the wait queue
153    pub fn enqueue_request(&mut self, request: LockRequest) {
154        self.wait_queue.push_back(request);
155    }
156
157    /// Process the next request in the wait queue
158    pub fn process_wait_queue(&mut self) -> Option<LockRequest> {
159        if self.is_available() {
160            self.wait_queue.pop_front()
161        } else {
162            None
163        }
164    }
165}

175/// Deadlock detection result

183impl DeadlockCycle {
184    /// Get the victim transaction to abort (youngest transaction)

186        // Simple heuristic: select the last transaction in the cycle
187        self.transactions.last().cloned()
188    }

191/// Deadlock detector using wait-for graph
192pub struct DeadlockDetector {
193    /// Wait-for graph edges
194    edges: Arc<RwLock<HashSet<WaitForEdge>>>,
195    /// Transaction to resources mapping
196    transaction_resources: Arc<RwLock<HashMap<String, HashSet<String>>>>,
197    /// Resource to transactions mapping
198    resource_transactions: Arc<RwLock<HashMap<String, HashSet<String>>>>,
199    /// Detection interval
200    detection_interval: Duration,
201}
202
203impl DeadlockDetector {
204    pub fn new(detection_interval: Duration) -> Self {
205        Self {
206            edges: Arc::new(RwLock::new(HashSet::new())),
207            transaction_resources: Arc::new(RwLock::new(HashMap::new())),
208            resource_transactions: Arc::new(RwLock::new(HashMap::new())),
209            detection_interval,
210        }
211    }
212
213    /// Add a wait-for edge
214    pub async fn add_wait_for(&self, waiting_tx: String, holding_tx: String, resource_id: String) {
215        let edge = WaitForEdge {
216            waiting_transaction: waiting_tx.clone(),
217            holding_transaction: holding_tx,
218            resource_id: resource_id.clone(),
219        };
220
221        let mut edges = self.edges.write().await;
222        edges.insert(edge);
223
224        // Update transaction-resource mappings
225        let mut tx_resources = self.transaction_resources.write().await;
226        tx_resources
227            .entry(waiting_tx.clone())
228            .or_insert_with(HashSet::new)
229            .insert(resource_id.clone());
230
231        let mut resource_txs = self.resource_transactions.write().await;
232        resource_txs
233            .entry(resource_id)
234            .or_insert_with(HashSet::new)
235            .insert(waiting_tx);
236    }
237
238    /// Remove wait-for edges for a transaction
239    pub async fn remove_transaction(&self, transaction_id: &str) {
240        let mut edges = self.edges.write().await;
241        edges.retain(|edge| {
242            edge.waiting_transaction != transaction_id && edge.holding_transaction != transaction_id
243        });
244
245        let mut tx_resources = self.transaction_resources.write().await;
246        if let Some(resources) = tx_resources.remove(transaction_id) {
247            let mut resource_txs = self.resource_transactions.write().await;
248            for resource in resources {
249                if let Some(txs) = resource_txs.get_mut(&resource) {
250                    txs.remove(transaction_id);
251                    if txs.is_empty() {
252                        resource_txs.remove(&resource);
253                    }
254                }
255            }
256        }
257    }
258
259    /// Detect deadlock cycles in the wait-for graph
260    pub async fn detect_deadlock(&self) -> Option<DeadlockCycle> {
261        let edges = self.edges.read().await;
262
263        // Build adjacency list
264        let mut graph: HashMap<String, Vec<String>> = HashMap::new();
265        let mut all_transactions: HashSet<String> = HashSet::new();
266
267        for edge in edges.iter() {
268            all_transactions.insert(edge.waiting_transaction.clone());
269            all_transactions.insert(edge.holding_transaction.clone());
270
271            graph
272                .entry(edge.waiting_transaction.clone())
273                .or_default()
274                .push(edge.holding_transaction.clone());
275        }
276
277        // Detect cycle using DFS
278        for start_tx in all_transactions.iter() {
279            if let Some(cycle) = self.detect_cycle_dfs(&graph, start_tx) {
280                // Get resources involved in the cycle
281                let cycle_set: HashSet<_> = cycle.iter().cloned().collect();
282                let resources: Vec<String> = edges
283                    .iter()
284                    .filter(|edge| {
285                        cycle_set.contains(&edge.waiting_transaction)
286                            && cycle_set.contains(&edge.holding_transaction)
287                    })
288                    .map(|edge| edge.resource_id.clone())
289                    .collect::<HashSet<_>>()
290                    .into_iter()
291                    .collect();
292
293                return Some(DeadlockCycle {
294                    transactions: cycle,
295                    resources,
296                    detected_at: SystemTime::now(),
297                });
298            }
299        }
300
301        None
302    }
303
304    /// DFS-based cycle detection
305    fn detect_cycle_dfs(
306        &self,
307        graph: &HashMap<String, Vec<String>>,
308        start: &str,
309    ) -> Option<Vec<String>> {
310        let mut visited = HashSet::new();
311        let mut path = Vec::new();
312
313        Self::dfs_helper(graph, start, start, &mut visited, &mut path)
314    }
315
316    fn dfs_helper(
317        graph: &HashMap<String, Vec<String>>,
318        current: &str,
319        target: &str,
320        visited: &mut HashSet<String>,
321        path: &mut Vec<String>,
322    ) -> Option<Vec<String>> {
323        if !path.is_empty() && current == target {
324            // Found a cycle
325            return Some(path.clone());
326        }
327
328        if visited.contains(current) {
329            return None;
330        }
331
332        visited.insert(current.to_string());
333        path.push(current.to_string());
334
335        if let Some(neighbors) = graph.get(current) {
336            for neighbor in neighbors {
337                if let Some(cycle) = Self::dfs_helper(graph, neighbor, target, visited, path) {
338                    return Some(cycle);
339                }
340            }
341        }
342
343        path.pop();
344        None
345    }
346
347    /// Start background deadlock detection
348    pub async fn start_detection(&self) {
349        let detector = self.clone();
350        tokio::spawn(async move {
351            let mut interval = tokio::time::interval(detector.detection_interval);
352
353            loop {
354                interval.tick().await;
355
356                if let Some(deadlock) = detector.detect_deadlock().await {
357                    warn!(
358                        "Deadlock detected! Transactions: {:?}, Resources: {:?}",
359                        deadlock.transactions, deadlock.resources
360                    );
361
362                    if let Some(victim) = deadlock.select_victim() {
363                        info!("Selected victim transaction for abort: {}", victim);
364                        // In a real system, this would trigger transaction abort
365                        detector.remove_transaction(&victim).await;
366                    }
367                }
368            }
369        });
370    }
371}

406/// Distributed lock manager
407pub struct DistributedLockManager {
408    node_id: NodeId,
409    locks: Arc<RwLock<HashMap<LockId, DistributedLock>>>,
410    config: LockManagerConfig,
411    deadlock_detector: DeadlockDetector,
412    lock_waiters: Arc<Mutex<HashMap<String, tokio::sync::oneshot::Sender<bool>>>>,
413}

415impl DistributedLockManager {
416    pub fn new(node_id: NodeId, config: LockManagerConfig) -> Self {
417        let deadlock_detector = DeadlockDetector::new(config.deadlock_detection_interval);
418
419        Self {
420            node_id,
421            locks: Arc::new(RwLock::new(HashMap::new())),
422            config,
423            deadlock_detector,
424            lock_waiters: Arc::new(Mutex::new(HashMap::new())),
425        }
426    }

428    /// Acquire a lock
429    pub async fn acquire_lock(
430        &self,
431        resource_id: String,
432        transaction_id: String,
433        mode: LockMode,
434    ) -> OrbitResult<bool> {
435        let lock_id = LockId::new(resource_id.clone());
436        let owner = LockOwner {
437            node_id: self.node_id.clone(),
438            transaction_id: transaction_id.clone(),
439            acquired_at: SystemTime::now(),
440        };
441
442        let mut locks = self.locks.write().await;
443        let lock = locks
444            .entry(lock_id.clone())
445            .or_insert_with(|| DistributedLock::new(lock_id.clone()));
446
447        if lock.try_acquire(owner.clone(), mode, self.config.default_lock_timeout) {
448            debug!(
449                "Lock acquired: {} by transaction {}",
450                resource_id, transaction_id
451            );
452            Ok(true)
453        } else {
454            // Lock is not immediately available
455            debug!(
456                "Lock {} is held, adding to wait queue for transaction {}",
457                resource_id, transaction_id
458            );
459
460            // Add to wait queue
461            let request = LockRequest {
462                lock_id: lock_id.clone(),
463                owner,
464                mode,
465                timeout: self.config.max_wait_time,
466                requested_at: SystemTime::now(),
467            };
468
469            lock.enqueue_request(request);
470
471            // Add wait-for edge if deadlock detection is enabled
472            if self.config.enable_deadlock_detection {
473                for holding_owner in &lock.owners {
474                    self.deadlock_detector
475                        .add_wait_for(
476                            transaction_id.clone(),
477                            holding_owner.transaction_id.clone(),
478                            resource_id.clone(),
479                        )
480                        .await;
481                }
482            }
483
484            Ok(false)
485        }
486    }
487
488    /// Release a lock
489    pub async fn release_lock(
490        &self,
491        resource_id: String,
492        transaction_id: String,
493    ) -> OrbitResult<()> {
494        let lock_id = LockId::new(resource_id.clone());
495        let owner = LockOwner {
496            node_id: self.node_id.clone(),
497            transaction_id: transaction_id.clone(),
498            acquired_at: SystemTime::now(), // Not used for comparison
499        };
500
501        let mut locks = self.locks.write().await;
502
503        if let Some(lock) = locks.get_mut(&lock_id) {
504            if lock.release(&owner) {
505                debug!(
506                    "Lock released: {} by transaction {}",
507                    resource_id, transaction_id
508                );
509
510                self.remove_from_deadlock_detection(&transaction_id).await;
511                self.process_lock_wait_queue(lock, &resource_id).await;
512            }
513        }
514
515        Ok(())
516    }
517
518    /// Remove transaction from deadlock detection if enabled
519    async fn remove_from_deadlock_detection(&self, transaction_id: &str) {
520        if self.config.enable_deadlock_detection {
521            self.deadlock_detector
522                .remove_transaction(transaction_id)
523                .await;
524        }
525    }
526
527    /// Process the wait queue for a released lock
528    async fn process_lock_wait_queue(&self, lock: &mut DistributedLock, resource_id: &str) {
529        if let Some(next_request) = lock.process_wait_queue() {
530            if self.try_grant_lock_to_waiter(lock, &next_request, resource_id) {
531                self.notify_waiting_transaction(&next_request.owner.transaction_id)
532                    .await;
533            }
534        }
535    }
536
537    /// Try to grant a lock to a waiting transaction
538    fn try_grant_lock_to_waiter(
539        &self,
540        lock: &mut DistributedLock,
541        request: &LockRequest,
542        resource_id: &str,
543    ) -> bool {
544        if lock.try_acquire(
545            request.owner.clone(),
546            request.mode,
547            self.config.default_lock_timeout,
548        ) {
549            info!(
550                "Lock {} granted to waiting transaction {}",
551                resource_id, request.owner.transaction_id
552            );
553            true
554        } else {
555            false
556        }
557    }
558
559    /// Notify a waiting transaction that the lock is granted
560    async fn notify_waiting_transaction(&self, transaction_id: &str) {
561        let mut waiters = self.lock_waiters.lock().await;
562        if let Some(sender) = waiters.remove(transaction_id) {
563            let _ = sender.send(true);
564        }
565    }

567    /// Get lock status
568    pub async fn get_lock_status(&self, resource_id: String) -> LockStatus {
569        let lock_id = LockId::new(resource_id);
570        let locks = self.locks.read().await;
571
572        if let Some(lock) = locks.get(&lock_id) {
573            if lock.is_available() {
574                LockStatus::Available
575            } else {
576                match lock.mode {
577                    LockMode::Exclusive => LockStatus::HeldExclusive {
578                        owner: lock.owners[0].clone(),
579                    },
580                    LockMode::Shared => LockStatus::HeldShared {
581                        owners: lock.owners.clone(),
582                    },
583                }
584            }
585        } else {
586            LockStatus::Available
587        }
588    }
589
590    /// Start background tasks
591    pub async fn start(&self) {
592        if self.config.enable_deadlock_detection {
593            self.deadlock_detector.start_detection().await;
594        }
595
596        // Start lock cleanup task
597        let manager = self.clone();
598        tokio::spawn(async move {
599            manager.cleanup_expired_locks().await;
600        });
601
602        info!("Distributed lock manager started on node {}", self.node_id);
603    }
604
605    /// Clean up expired locks periodically
606    async fn cleanup_expired_locks(&self) {
607        let mut interval = tokio::time::interval(self.config.lock_cleanup_interval);
608
609        loop {
610            interval.tick().await;
611
612            let now = SystemTime::now();
613            let mut locks = self.locks.write().await;
614
615            let expired_locks: Vec<LockId> = locks
616                .iter()
617                .filter(|(_, lock)| lock.expires_at < now && lock.is_held())
618                .map(|(id, _)| id.clone())
619                .collect();
620
621            for lock_id in expired_locks {
622                if let Some(lock) = locks.get_mut(&lock_id) {
623                    warn!("Cleaning up expired lock: {}", lock_id);
624                    lock.owners.clear();
625
626                    // Process wait queue
627                    if let Some(next_request) = lock.process_wait_queue() {
628                        if lock.try_acquire(
629                            next_request.owner.clone(),
630                            next_request.mode,
631                            self.config.default_lock_timeout,
632                        ) {
633                            info!(
634                                "Lock {} granted to waiting transaction {} after cleanup",
635                                lock_id, next_request.owner.transaction_id
636                            );
637                        }
638                    }
639                }
640            }
641        }
642    }
643}
