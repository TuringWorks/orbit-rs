10/// Load balancing strategies
11#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
12pub enum LoadBalancingStrategy {
13    /// Round-robin distribution
14    RoundRobin,
15    /// Route to node with least connections
16    LeastConnections,
17    /// Random selection
18    Random,
19    /// Weighted round-robin based on node capacity
20    WeightedRoundRobin,
21    /// Route based on connection affinity
22    AffinityBased,
23}
24
25impl Default for LoadBalancingStrategy {
26    fn default() -> Self {
27        Self::LeastConnections
28    }
29}

31/// Node health information
32#[derive(Debug, Clone)]
33pub struct NodeHealth {
34    /// Node identifier
35    pub node_id: String,
36    /// Current active connections
37    pub active_connections: usize,
38    /// Maximum allowed connections
39    pub max_connections: usize,
40    /// Is the node healthy
41    pub is_healthy: bool,
42    /// Node weight (for weighted strategies)
43    pub weight: u32,
44    /// Average response time in milliseconds
45    pub avg_response_time_ms: f64,
46}
47
48impl NodeHealth {
49    pub fn new(node_id: String, max_connections: usize) -> Self {
50        Self {
51            node_id,
52            active_connections: 0,
53            max_connections,
54            is_healthy: true,
55            weight: 100,
56            avg_response_time_ms: 0.0,
57        }
58    }
59
60    pub fn utilization(&self) -> f64 {
61        if self.max_connections == 0 {
62            0.0
63        } else {
64            self.active_connections as f64 / self.max_connections as f64
65        }
66    }
67
68    pub fn has_capacity(&self) -> bool {
69        self.is_healthy && self.active_connections < self.max_connections
70    }
71}

73/// Connection load balancer
74pub struct ConnectionLoadBalancer {
75    strategy: LoadBalancingStrategy,
76    nodes: Arc<RwLock<HashMap<String, NodeHealth>>>,
77    round_robin_counter: Arc<AtomicUsize>,
78}
79
80impl ConnectionLoadBalancer {
81    /// Create a new load balancer with the specified strategy
82    pub fn new(strategy: LoadBalancingStrategy) -> Self {
83        Self {
84            strategy,
85            nodes: Arc::new(RwLock::new(HashMap::new())),
86            round_robin_counter: Arc::new(AtomicUsize::new(0)),
87        }
88    }
89
90    /// Add or update a node
91    pub async fn add_node(&self, node: NodeHealth) {
92        let mut nodes = self.nodes.write().await;
93        nodes.insert(node.node_id.clone(), node);
94    }
95
96    /// Remove a node
97    pub async fn remove_node(&self, node_id: &str) {
98        let mut nodes = self.nodes.write().await;
99        nodes.remove(node_id);
100    }
101
102    /// Update node health status
103    pub async fn update_node_health(&self, node_id: &str, is_healthy: bool) {
104        let mut nodes = self.nodes.write().await;
105        if let Some(node) = nodes.get_mut(node_id) {
106            node.is_healthy = is_healthy;
107        }
108    }
109
110    /// Update node connection count
111    pub async fn update_node_connections(&self, node_id: &str, connections: usize) {
112        let mut nodes = self.nodes.write().await;
113        if let Some(node) = nodes.get_mut(node_id) {
114            node.active_connections = connections;
115        }
116    }
117
118    /// Select a node for a new connection
119    pub async fn select_node(&self) -> OrbitResult<Option<String>> {
120        let nodes = self.nodes.read().await;
121
122        // Filter healthy nodes with capacity
123        let available_nodes: Vec<&NodeHealth> =
124            nodes.values().filter(|n| n.has_capacity()).collect();
125
126        if available_nodes.is_empty() {
127            return Ok(None);
128        }
129
130        let selected_node = match self.strategy {
131            LoadBalancingStrategy::RoundRobin => self.select_round_robin(&available_nodes),
132            LoadBalancingStrategy::LeastConnections => {
133                self.select_least_connections(&available_nodes)
134            }
135            LoadBalancingStrategy::Random => self.select_random(&available_nodes),
136            LoadBalancingStrategy::WeightedRoundRobin => self.select_weighted(&available_nodes),
137            LoadBalancingStrategy::AffinityBased => {
138                // For affinity-based, fall back to least connections
139                self.select_least_connections(&available_nodes)
140            }
141        };
142
143        Ok(Some(selected_node.node_id.clone()))
144    }
145
146    /// Select a node using affinity key
147    pub async fn select_node_with_affinity(
148        &self,
149        affinity_key: &str,
150    ) -> OrbitResult<Option<String>> {
151        let nodes = self.nodes.read().await;
152
153        let available_nodes: Vec<&NodeHealth> =
154            nodes.values().filter(|n| n.has_capacity()).collect();
155
156        if available_nodes.is_empty() {
157            return Ok(None);
158        }
159
160        // Use consistent hashing for affinity
161        let hash = self.hash_affinity_key(affinity_key);
162        let index = hash % available_nodes.len();
163
164        Ok(Some(available_nodes[index].node_id.clone()))
165    }
166
167    fn hash_affinity_key(&self, key: &str) -> usize {
168        // Simple hash function for affinity-based routing
169        key.bytes().fold(0usize, |acc, b| {
170            acc.wrapping_mul(31).wrapping_add(b as usize)
171        })
172    }
173
174    fn select_round_robin<'a>(&self, nodes: &[&'a NodeHealth]) -> &'a NodeHealth {
175        let index = self.round_robin_counter.fetch_add(1, Ordering::SeqCst) % nodes.len();
176        nodes[index]
177    }
178
179    fn select_least_connections<'a>(&self, nodes: &[&'a NodeHealth]) -> &'a NodeHealth {
180        nodes.iter().min_by_key(|n| n.active_connections).unwrap()
181    }
182
183    fn select_random<'a>(&self, nodes: &[&'a NodeHealth]) -> &'a NodeHealth {
184        let now = std::time::SystemTime::now()
185            .duration_since(std::time::UNIX_EPOCH)
186            .unwrap_or_default()
187            .as_nanos() as usize;
188        let index = now % nodes.len();
189        nodes[index]
190    }
191
192    fn select_weighted<'a>(&self, nodes: &[&'a NodeHealth]) -> &'a NodeHealth {
193        // Calculate total weight
194        let total_weight: u32 = nodes.iter().map(|n| n.weight).sum();
195
196        if total_weight == 0 {
197            return self.select_round_robin(nodes);
198        }
199
200        // Select based on weight
201        let counter = self.round_robin_counter.fetch_add(1, Ordering::SeqCst) as u32;
202        let target = counter % total_weight;
203
204        let mut cumulative = 0u32;
205        for node in nodes {
206            cumulative += node.weight;
207            if target < cumulative {
208                return node;
209            }
210        }
211
212        // Fallback
213        nodes[0]
214    }
215
216    /// Get statistics about all nodes
217    pub async fn get_stats(&self) -> LoadBalancerStats {
218        let nodes = self.nodes.read().await;
219
220        let total_nodes = nodes.len();
221        let healthy_nodes = nodes.values().filter(|n| n.is_healthy).count();
222        let total_connections: usize = nodes.values().map(|n| n.active_connections).sum();
223        let total_capacity: usize = nodes.values().map(|n| n.max_connections).sum();
224
225        LoadBalancerStats {
226            strategy: self.strategy,
227            total_nodes,
228            healthy_nodes,
229            total_connections,
230            total_capacity,
231            average_utilization: if total_capacity > 0 {
232                total_connections as f64 / total_capacity as f64
233            } else {
234                0.0
235            },
236        }
237    }
238}

266    async fn test_round_robin() {
267        let lb = ConnectionLoadBalancer::new(LoadBalancingStrategy::RoundRobin);
268
269        lb.add_node(NodeHealth::new("node1".to_string(), 10)).await;
270        lb.add_node(NodeHealth::new("node2".to_string(), 10)).await;
271        lb.add_node(NodeHealth::new("node3".to_string(), 10)).await;
272
273        // Select multiple times and verify round-robin
274        let mut selected = Vec::new();
275        for _ in 0..6 {
276            if let Some(node_id) = lb.select_node().await.unwrap() {
277                selected.push(node_id);
278            }
279        }
280
281        // Should cycle through nodes
282        assert_eq!(selected.len(), 6);
283    }
