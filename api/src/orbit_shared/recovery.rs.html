314    async fn start_background_tasks(&self) -> OrbitResult<()> {
315        let recovery_manager = self.clone();
316
317        // Health checking task
318        {
319            let manager = recovery_manager.clone();
320            tokio::spawn(async move {
321                let mut interval = tokio::time::interval(manager.config.health_check_interval);
322                loop {
323                    interval.tick().await;

325                        error!("Coordinator health check failed: {}", e);
326                    }
327                }
328            });
329        }

331        // Checkpoint management task
332        {
333            let manager = recovery_manager.clone();
334            tokio::spawn(async move {
335                let mut interval = tokio::time::interval(manager.config.checkpoint_interval);
336                loop {
337                    interval.tick().await;
338                    if let Err(e) = manager.cleanup_old_checkpoints().await {
339                        error!("Checkpoint cleanup failed: {}", e);

341                }
342            });
343        }
344
345        // Recovery monitoring task
346        {
347            let manager = recovery_manager.clone();
348            tokio::spawn(async move {
349                let mut interval = tokio::time::interval(Duration::from_secs(30));
350                loop {
351                    interval.tick().await;

353                        error!("Recovery monitoring failed: {}", e);
354                    }

362    /// Check health of known coordinators
363    async fn check_coordinator_health(&self) -> OrbitResult<()> {
364        let mut coordinators = self.coordinators.write().await;
365        let now = Instant::now();
366        let mut failed_coordinators = Vec::new();
367
368        for (node_id, health) in coordinators.iter_mut() {
369            let time_since_last_seen = now.duration_since(health.last_seen);
370
371            if time_since_last_seen > self.config.health_check_timeout {
372                health.consecutive_failures += 1;
373
374                if health.consecutive_failures >= 3 && health.is_healthy {
375                    health.is_healthy = false;
376                    failed_coordinators.push(node_id.clone());
377
378                    warn!("Coordinator failure detected: {}", node_id);
379
380                    // Update statistics
381                    {
382                        let mut stats = self.stats.write().await;
383                        stats.coordinator_failures_detected += 1;
384                    }
385                }
386            } else {
387                // Reset failure count if coordinator is responding
388                if health.consecutive_failures > 0 {
389                    health.consecutive_failures = 0;
390                    health.is_healthy = true;
391                }
392            }
393        }
394
395        // Handle failed coordinators
396        for failed_coordinator in failed_coordinators {
397            self.handle_coordinator_failure(&failed_coordinator).await?;
398        }
399
400        Ok(())
401    }

403    /// Handle coordinator failure
404    async fn handle_coordinator_failure(&self, failed_coordinator: &NodeId) -> OrbitResult<()> {
405        info!("Handling coordinator failure: {}", failed_coordinator);
406
407        // Notify event handlers
408        let handlers = self.event_handlers.read().await;
409        for handler in handlers.iter() {
410            if let Err(e) = handler.on_coordinator_failure(failed_coordinator).await {
411                error!("Recovery event handler failed: {}", e);
412            }
413        }
414
415        // Notify cluster manager
416        self.cluster_manager
417            .report_coordinator_failure(failed_coordinator)
418            .await?;
419
420        // Find transactions that need recovery
421        let transactions_to_recover = self
422            .find_transactions_needing_recovery(failed_coordinator)
423            .await?;
424
425        if !transactions_to_recover.is_empty() {
426            info!(
427                "Found {} transactions requiring recovery",
428                transactions_to_recover.len()
429            );
430
431            // Start recovery process
432            if self.config.enable_automatic_failover {
433                self.initiate_recovery_process(transactions_to_recover)
434                    .await?;
435            }
436        }
437
438        Ok(())
439    }

441    /// Find transactions that need recovery due to coordinator failure
442    pub async fn find_transactions_needing_recovery(
443        &self,
444        failed_coordinator: &NodeId,
445    ) -> OrbitResult<Vec<TransactionCheckpoint>> {
446        let checkpoints = self.checkpoints.read().await;
447        let mut transactions_to_recover = Vec::new();
448
449        for checkpoint in checkpoints.values() {
450            if &checkpoint.coordinator_node == failed_coordinator {
451                // Check if transaction is in a recoverable state
452                if matches!(
453                    checkpoint.current_state,
454                    TransactionState::Preparing
455                        | TransactionState::Prepared
456                        | TransactionState::Committing
457                        | TransactionState::Aborting
458                ) {
459                    transactions_to_recover.push(checkpoint.clone());
460                }
461            }
462        }
463
464        Ok(transactions_to_recover)
465    }

467    /// Initiate recovery process for failed transactions
468    pub async fn initiate_recovery_process(
469        &self,
470        transactions: Vec<TransactionCheckpoint>,
471    ) -> OrbitResult<()> {
472        info!(
473            "Initiating recovery process for {} transactions",
474            transactions.len()
475        );
476
477        // Check if this node should become the new coordinator
478        let should_become_coordinator = self.should_become_coordinator().await?;
479
480        if should_become_coordinator {
481            // Start election process
482            let election_won = self.cluster_manager.start_election(&self.node_id).await?;
483
484            {
485                let mut stats = self.stats.write().await;
486                stats.elections_participated += 1;
487                if election_won {
488                    stats.elections_won += 1;
489                }
490            }
491
492            if election_won {
493                info!("Won coordinator election, taking over failed transactions");
494                self.become_coordinator(transactions).await?;
495            }
496        }
497
498        Ok(())
499    }

621    /// Execute the appropriate recovery strategy based on transaction state
622    async fn execute_transaction_recovery_strategy(
623        &self,
624        checkpoint: &TransactionCheckpoint,
625    ) -> OrbitResult<bool> {
626        match checkpoint.current_state {
627            TransactionState::Preparing => {
628                // Transaction was in prepare phase - need to determine if we should commit or abort
629                self.recover_preparing_transaction(checkpoint).await
630            }
631            TransactionState::Prepared => {
632                // All participants voted yes - should commit
633                self.recover_prepared_transaction(checkpoint).await
634            }
635            TransactionState::Committing => {
636                // Transaction was committing - continue commit
637                self.recover_committing_transaction(checkpoint).await
638            }
639            TransactionState::Aborting => {
640                // Transaction was aborting - continue abort
641                self.recover_aborting_transaction(checkpoint).await
642            }
643            _ => {
644                self.handle_non_recoverable_transaction(checkpoint);
645                Ok(true)
646            }
647        }
648    }

752    /// Clean up old checkpoints
753    async fn cleanup_old_checkpoints(&self) -> OrbitResult<()> {
754        let cutoff_time =
755            chrono::Utc::now().timestamp_millis() - self.config.max_recovery_age.as_millis() as i64;
756        let mut checkpoints = self.checkpoints.write().await;
757        let mut to_remove = Vec::new();
758
759        for (transaction_id, checkpoint) in checkpoints.iter() {
760            if checkpoint.created_at < cutoff_time {
761                to_remove.push(transaction_id.clone());
762            }
763        }
764
765        for transaction_id in to_remove {
766            checkpoints.remove(&transaction_id);
767            debug!("Cleaned up old checkpoint: {}", transaction_id);
768        }
769
770        Ok(())
771    }

856    /// Reassign transactions from a failed coordinator to the current leader
857    pub async fn reassign_transactions_to_leader(
858        &self,
859        failed_coordinator: &NodeId,
860    ) -> OrbitResult<Vec<TransactionId>> {
861        let current_leader = self.get_cluster_leader().await.ok_or_else(|| {
862            OrbitError::cluster("No current leader available for transaction reassignment")
863        })?;
864
865        let mut mapping = self.transaction_coordinator_map.write().await;
866        let mut reassigned_transactions = Vec::new();
867
868        // Find all transactions owned by the failed coordinator
869        let transactions_to_reassign: Vec<TransactionId> = mapping
870            .iter()
871            .filter(|(_, coordinator)| *coordinator == failed_coordinator)
872            .map(|(tx_id, _)| tx_id.clone())
873            .collect();
874
875        // Reassign them to the current leader
876        for transaction_id in transactions_to_reassign {
877            mapping.insert(transaction_id.clone(), current_leader.clone());
878            reassigned_transactions.push(transaction_id.clone());
879
880            info!(
881                "Reassigned transaction {} from failed coordinator {} to leader {}",
882                transaction_id, failed_coordinator, current_leader
883            );
884        }
885
886        Ok(reassigned_transactions)
887    }
