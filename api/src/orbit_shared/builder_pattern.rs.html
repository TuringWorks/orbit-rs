1//! Builder pattern utilities for consistent configuration object creation
2//!
3//! This module provides macros and traits to implement the builder pattern
4//! consistently across configuration objects, reducing code duplication.

8/// Trait for objects that can be built using the builder pattern
9pub trait Buildable: Default {
10    type Builder: ConfigBuilder<Self>;
11
12    /// Create a new builder for this type
13    fn builder() -> Self::Builder {
14        Self::Builder::new()
15    }
16}
17
18/// Trait for configuration builders
19pub trait ConfigBuilder<T> {
20    /// Create a new builder
21    fn new() -> Self;
22
23    /// Build the final configuration object
24    fn build(self) -> T;
25
26    /// Validate the configuration before building
27    fn validate(&self) -> Result<(), String> {
28        Ok(())
29    }
30
31    /// Build with validation
32    fn build_validated(self) -> Result<T, String>
33    where
34        Self: Sized,
35    {
36        self.validate()?;
37        Ok(self.build())
38    }
39}

41/// Macro to generate builder pattern implementations
42#[macro_export]
43macro_rules! impl_builder {
44    (
45        $(#[$struct_meta:meta])*
46        pub struct $name:ident {
47            $(
48                $(#[$field_meta:meta])*
49                pub $field:ident: $field_type:ty = $default:expr
50            ),* $(,)?
51        }
52    ) => {
53        $(#[$struct_meta])*
54        #[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
55        pub struct $name {
56            $(
57                $(#[$field_meta])*
58                pub $field: $field_type,
59            )*
60        }
61
62        impl Default for $name {
63            fn default() -> Self {
64                Self {
65                    $($field: $default,)*
66                }
67            }
68        }
69
70        paste::paste! {
71            #[derive(Debug, Clone)]
72            pub struct [<$name Builder>] {
73                $($field: Option<$field_type>,)*
74            }
75
76            impl ConfigBuilder<$name> for [<$name Builder>] {
77                fn new() -> Self {
78                    Self {
79                        $($field: None,)*
80                    }
81                }
82
83                fn build(self) -> $name {
84                    $name {
85                        $($field: self.$field.unwrap_or_else(|| $default),)*
86                    }
87                }
88
89                // Default validation (can be overridden)
90            }
91
92            impl [<$name Builder>] {
93                /// Create a new builder
94                pub fn new() -> Self {
95                    <Self as ConfigBuilder<$name>>::new()
96                }
97
98                $(
99                    /// Set the value for the field
100                    pub fn $field(mut self, value: $field_type) -> Self {
101                        self.$field = Some(value);
102                        self
103                    }
104
105                    paste::paste! {
106                        /// Set the value for the field with validation
107                        pub fn [<$field _validated>]<F>(mut self, value: $field_type, validator: F) -> Result<Self, String>
108                        where
109                            F: FnOnce(&$field_type) -> Result<(), String>,
110                        {
111                            validator(&value)?;
112                            self.$field = Some(value);
113                            Ok(self)
114                        }
115                    }
116                )*
117
118                /// Build the final configuration
119                pub fn build(self) -> $name {
120                    <Self as ConfigBuilder<$name>>::build(self)
121                }
122
123                /// Build with validation
124                pub fn build_validated(self) -> Result<$name, String> {
125                    <Self as ConfigBuilder<$name>>::build_validated(self)
126                }
127            }
128
129            impl Default for [<$name Builder>] {
130                fn default() -> Self {
131                    Self::new()
132                }
133            }
134
135            impl Buildable for $name {
136                type Builder = [<$name Builder>];
137            }
138        }
139    };
140}

142/// Trait for configuration objects with environment variable support
143pub trait EnvConfigurable: Sized {
144    /// Load configuration from environment variables with prefix
145    fn from_env_with_prefix(prefix: &str) -> Result<Self, String>;
146
147    /// Load configuration from environment variables with default prefix
148    fn from_env() -> Result<Self, String> {
149        Self::from_env_with_prefix("ORBIT")
150    }
151}
152
153/// Macro to implement environment variable configuration loading
154#[macro_export]
155macro_rules! impl_env_config {
156    (
157        $name:ident {
158            $(
159                $field:ident: $field_type:ty = $env_name:expr => $parser:expr
160            ),* $(,)?
161        }
162    ) => {
163        impl EnvConfigurable for $name {
164            fn from_env_with_prefix(prefix: &str) -> Result<Self, String> {
165                let mut builder = Self::builder();
166
167                $(
168                    let env_var_name = if prefix.is_empty() {
169                        $env_name.to_string()
170                    } else {
171                        format!("{}_{}", prefix, $env_name)
172                    };
173
174                    if let Ok(value_str) = std::env::var(&env_var_name) {
175                        let parsed_value: $field_type = $parser(&value_str)
176                            .map_err(|e| format!("Failed to parse {} from env var {}: {}", stringify!($field), env_var_name, e))?;
177                        builder = builder.$field(parsed_value);
178                    }
179                )*
180
181                builder.build_validated()
182            }
183        }
184    };
185}

187/// Standard parsers for common types
188pub mod parsers {
189    use std::str::FromStr;
190    use std::time::Duration;
191
192    /// Parse string (identity function)
193    pub fn parse_string(s: &str) -> Result<String, String> {
194        Ok(s.to_string())
195    }
196
197    /// Parse boolean from string
198    pub fn parse_bool(s: &str) -> Result<bool, String> {
199        match s.to_lowercase().as_str() {
200            "true" | "1" | "yes" | "on" => Ok(true),
201            "false" | "0" | "no" | "off" => Ok(false),
202            _ => Err(format!("Invalid boolean value: {s}")),
203        }
204    }
205
206    /// Parse integer types
207    pub fn parse_u32(s: &str) -> Result<u32, String> {
208        s.parse().map_err(|e| format!("Invalid u32: {e}"))
209    }
210
211    pub fn parse_u64(s: &str) -> Result<u64, String> {
212        s.parse().map_err(|e| format!("Invalid u64: {e}"))
213    }
214
215    pub fn parse_usize(s: &str) -> Result<usize, String> {
216        s.parse().map_err(|e| format!("Invalid usize: {e}"))
217    }
218
219    /// Parse duration from milliseconds
220    pub fn parse_duration_ms(s: &str) -> Result<Duration, String> {
221        let ms: u64 = s.parse().map_err(|e| format!("Invalid duration ms: {e}"))?;
222        Ok(Duration::from_millis(ms))
223    }
224
225    /// Parse duration from seconds
226    pub fn parse_duration_secs(s: &str) -> Result<Duration, String> {
227        let secs: u64 = s
228            .parse()
229            .map_err(|e| format!("Invalid duration secs: {e}"))?;
230        Ok(Duration::from_secs(secs))
231    }
232
233    /// Parse optional values
234    pub fn parse_optional<T, F, E>(s: &str, parser: F) -> Result<Option<T>, String>
235    where
236        F: FnOnce(&str) -> Result<T, E>,
237        E: std::fmt::Display,
238    {
239        if s.is_empty() || s.to_lowercase() == "none" || s.to_lowercase() == "null" {
240            Ok(None)
241        } else {
242            parser(s).map(Some).map_err(|e| e.to_string())
243        }
244    }
245
246    /// Generic parser for types implementing FromStr
247    pub fn parse_from_str<T>(s: &str) -> Result<T, String>
248    where
249        T: FromStr,
250        T::Err: std::fmt::Display,
251    {
252        s.parse().map_err(|e: T::Err| e.to_string())
253    }
254}

256/// Example configuration using the builder pattern
257#[cfg(test)]
258#[allow(dead_code)]
259mod example {
260    use super::*;
261    use std::time::Duration;
262
263    impl_builder! {
264        /// Example database configuration
265        pub struct DatabaseConfig {
266            /// Database host
267            pub host: String = "localhost".to_string(),
268            /// Database port
269            pub port: u16 = 5432,
270            /// Maximum connections
271            pub max_connections: u32 = 100,
272            /// Connection timeout
273            pub timeout: Duration = Duration::from_secs(30),
274            /// Enable SSL
275            pub ssl_enabled: bool = false,
276        }
277    }
278
279    impl_env_config! {
280        DatabaseConfig {
281            host: String = "DB_HOST" => parsers::parse_string,
282            port: u16 = "DB_PORT" => parsers::parse_from_str,
283            max_connections: u32 = "DB_MAX_CONNECTIONS" => parsers::parse_u32,
284            timeout: Duration = "DB_TIMEOUT_SECS" => parsers::parse_duration_secs,
285            ssl_enabled: bool = "DB_SSL_ENABLED" => parsers::parse_bool,
286        }
287    }
288}

291mod tests {

293    use super::*;

295    #[test]
296    fn test_builder_pattern() {
297        let config = DatabaseConfig::builder()
298            .host("postgres.example.com".to_string())
299            .port(5432)
300            .max_connections(200)
301            .ssl_enabled(true)
302            .build();
303
304        assert_eq!(config.host, "postgres.example.com");
305        assert_eq!(config.port, 5432);
306        assert_eq!(config.max_connections, 200);
307        assert!(config.ssl_enabled);
308    }
309
310    #[test]
311    fn test_builder_validation() {
312        // Test custom validation using validated setters
313        let result = DatabaseConfig::builder().host_validated("".to_string(), |host| {
314            if host.is_empty() {
315                Err("Host cannot be empty".to_string())
316            } else {
317                Ok(())
318            }
319        });
320
321        assert!(result.is_err());
322        assert!(result.unwrap_err().contains("Host cannot be empty"));
323    }
324
325    #[test]
326    fn test_default_values() {
327        let config = DatabaseConfig::builder().build();
