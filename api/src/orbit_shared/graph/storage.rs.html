1//! In-memory graph storage implementation
2//!
3//! This module provides a simple in-memory implementation of the GraphStorage trait
4//! suitable for development, testing, and single-node deployments.
5
6use super::{Direction, GraphNode, GraphRelationship, GraphStorage, NodeId, RelationshipId};
7use crate::exception::{OrbitError, OrbitResult};
8use async_trait::async_trait;
9use std::collections::HashMap;
10use tokio::sync::RwLock;
11use tracing::{debug, info, instrument, warn};

13/// In-memory graph storage implementation
14#[derive(Debug)]
15pub struct InMemoryGraphStorage {
16    /// Node storage indexed by NodeId
17    nodes: RwLock<HashMap<NodeId, GraphNode>>,
18    /// Relationship storage indexed by RelationshipId
19    relationships: RwLock<HashMap<RelationshipId, GraphRelationship>>,
20    /// Index of outgoing relationships by node ID
21    outgoing_rels: RwLock<HashMap<NodeId, Vec<RelationshipId>>>,
22    /// Index of incoming relationships by node ID
23    incoming_rels: RwLock<HashMap<NodeId, Vec<RelationshipId>>>,
24    /// Label index: label -> set of node IDs
25    label_index: RwLock<HashMap<String, Vec<NodeId>>>,
26    /// Relationship type index: type -> set of relationship IDs
27    rel_type_index: RwLock<HashMap<String, Vec<RelationshipId>>>,
28}
29
30impl InMemoryGraphStorage {
31    /// Create a new in-memory graph storage
32    pub fn new() -> Self {
33        Self {
34            nodes: RwLock::new(HashMap::new()),
35            relationships: RwLock::new(HashMap::new()),
36            outgoing_rels: RwLock::new(HashMap::new()),
37            incoming_rels: RwLock::new(HashMap::new()),
38            label_index: RwLock::new(HashMap::new()),
39            rel_type_index: RwLock::new(HashMap::new()),
40        }
41    }
42
43    /// Update the label index when a node's labels change
44    async fn update_label_index(
45        &self,
46        node_id: &NodeId,
47        old_labels: &[String],
48        new_labels: &[String],
49    ) {
50        let mut label_index = self.label_index.write().await;
51
52        // Remove from old labels
53        for label in old_labels {
54            if let Some(node_ids) = label_index.get_mut(label) {
55                node_ids.retain(|id| id != node_id);
56                if node_ids.is_empty() {
57                    label_index.remove(label);
58                }
59            }
60        }
61
62        // Add to new labels
63        for label in new_labels {
64            label_index
65                .entry(label.clone())
66                .or_default()
67                .push(node_id.clone());
68        }
69    }
70
71    /// Update the relationship type index
72    async fn update_rel_type_index(&self, rel_id: &RelationshipId, rel_type: &str, add: bool) {
73        let mut rel_type_index = self.rel_type_index.write().await;
74
75        if add {
76            rel_type_index
77                .entry(rel_type.to_string())
78                .or_default()
79                .push(rel_id.clone());
80        } else if let Some(rel_ids) = rel_type_index.get_mut(rel_type) {
81            rel_ids.retain(|id| id != rel_id);
82            if rel_ids.is_empty() {
83                rel_type_index.remove(rel_type);
84            }
85        }
86    }
87
88    /// Update relationship indexes for a node
89    async fn update_relationship_indexes(
90        &self,
91        rel_id: &RelationshipId,
92        start_node: &NodeId,
93        end_node: &NodeId,
94        add: bool,
95    ) {
96        let mut outgoing = self.outgoing_rels.write().await;
97        let mut incoming = self.incoming_rels.write().await;
98
99        if add {
100            outgoing
101                .entry(start_node.clone())
102                .or_default()
103                .push(rel_id.clone());
104            incoming
105                .entry(end_node.clone())
106                .or_default()
107                .push(rel_id.clone());
108        } else {
109            if let Some(rels) = outgoing.get_mut(start_node) {
110                rels.retain(|id| id != rel_id);
111                if rels.is_empty() {
112                    outgoing.remove(start_node);
113                }
114            }
115            if let Some(rels) = incoming.get_mut(end_node) {
116                rels.retain(|id| id != rel_id);
117                if rels.is_empty() {
118                    incoming.remove(end_node);
119                }
120            }
121        }
122    }
123
124    /// Get node count for metrics
125    pub async fn node_count(&self) -> usize {
126        self.nodes.read().await.len()
127    }
128
129    /// Get relationship count for metrics
130    pub async fn relationship_count(&self) -> usize {
131        self.relationships.read().await.len()
132    }
133}

143    #[instrument(skip(self), fields(labels_count = labels.len(), props_count = properties.len()))]
144    async fn create_node(
145        &self,
146        labels: Vec<String>,
147        properties: HashMap<String, serde_json::Value>,
148    ) -> OrbitResult<GraphNode> {
149        let node = GraphNode::new(labels.clone(), properties);
150        let node_id = node.id.clone();
151
152        // Store the node
153        self.nodes
154            .write()
155            .await
156            .insert(node_id.clone(), node.clone());
157
158        // Update label index
159        self.update_label_index(&node_id, &[], &labels).await;
160
161        info!(node_id = %node_id, labels = ?labels, "Created graph node");
162        Ok(node)
163    }

165    #[instrument(skip(self))]
166    async fn get_node(&self, node_id: &NodeId) -> OrbitResult<Option<GraphNode>> {
167        let node = self.nodes.read().await.get(node_id).cloned();
168        debug!(node_id = %node_id, found = node.is_some(), "Retrieved graph node");
169        Ok(node)
170    }

172    #[instrument(skip(self, properties), fields(props_count = properties.len()))]
173    async fn update_node(
174        &self,
175        node_id: &NodeId,
176        properties: HashMap<String, serde_json::Value>,
177    ) -> OrbitResult<()> {
178        let mut nodes = self.nodes.write().await;
179
180        match nodes.get_mut(node_id) {
181            Some(node) => {
182                for (key, value) in properties {
183                    node.set_property(&key, value);
184                }
185                info!(node_id = %node_id, "Updated node properties");
186                Ok(())
187            }
188            None => {
189                warn!(node_id = %node_id, "Attempted to update non-existent node");
190                Err(OrbitError::NodeNotFound {
191                    node_id: node_id.to_string(),
192                })
193            }
194        }
195    }
196
197    #[instrument(skip(self))]
198    async fn delete_node(&self, node_id: &NodeId) -> OrbitResult<bool> {
199        // First, delete all relationships involving this node
200        let relationships_to_delete: Vec<RelationshipId> = {
201            let outgoing = self.outgoing_rels.read().await;
202            let incoming = self.incoming_rels.read().await;
203
204            let mut rels = Vec::new();
205            if let Some(out_rels) = outgoing.get(node_id) {
206                rels.extend(out_rels.clone());
207            }
208            if let Some(in_rels) = incoming.get(node_id) {
209                rels.extend(in_rels.clone());
210            }
211            rels
212        };
213
214        // Delete relationships
215        for rel_id in relationships_to_delete {
216            self.delete_relationship(&rel_id).await?;
217        }
218
219        // Get node for label cleanup
220        let node = {
221            let nodes = self.nodes.read().await;
222            nodes.get(node_id).cloned()
223        };
224
225        if let Some(node) = node {
226            // Update label index
227            self.update_label_index(node_id, &node.labels, &[]).await;
228
229            // Remove node
230            let removed = self.nodes.write().await.remove(node_id).is_some();
231
232            if removed {
233                info!(node_id = %node_id, "Deleted graph node");
234            }
235            Ok(removed)
236        } else {
237            Ok(false)
238        }
239    }

241    #[instrument(skip(self, labels), fields(labels_count = labels.len()))]
242    async fn add_labels(&self, node_id: &NodeId, labels: Vec<String>) -> OrbitResult<()> {
243        let mut nodes = self.nodes.write().await;
244
245        match nodes.get_mut(node_id) {
246            Some(node) => {
247                let old_labels = node.labels.clone();
248                node.add_labels(labels.clone());
249                let new_labels = node.labels.clone();
250
251                // Drop the write lock before updating index
252                drop(nodes);
253
254                // Update label index
255                self.update_label_index(node_id, &old_labels, &new_labels)
256                    .await;
257
258                info!(node_id = %node_id, labels = ?labels, "Added labels to node");
259                Ok(())
260            }
261            None => {
262                warn!(node_id = %node_id, "Attempted to add labels to non-existent node");
263                Err(OrbitError::NodeNotFound {
264                    node_id: node_id.to_string(),
265                })
266            }
267        }
268    }

315    #[instrument(skip(self, properties), fields(props_count = properties.len()))]
316    async fn create_relationship(
317        &self,
318        start_node: &NodeId,
319        end_node: &NodeId,
320        rel_type: String,
321        properties: HashMap<String, serde_json::Value>,
322    ) -> OrbitResult<GraphRelationship> {
323        // Verify both nodes exist
324        let nodes = self.nodes.read().await;
325        if !nodes.contains_key(start_node) {
326            return Err(OrbitError::NodeNotFound {
327                node_id: start_node.to_string(),
328            });
329        }
330        if !nodes.contains_key(end_node) {
331            return Err(OrbitError::NodeNotFound {
332                node_id: end_node.to_string(),
333            });
334        }
335        drop(nodes);
336
337        let relationship = GraphRelationship::new(
338            start_node.clone(),
339            end_node.clone(),
340            rel_type.clone(),
341            properties,
342        );
343        let rel_id = relationship.id.clone();
344
345        // Store relationship
346        self.relationships
347            .write()
348            .await
349            .insert(rel_id.clone(), relationship.clone());
350
351        // Update indexes
352        self.update_relationship_indexes(&rel_id, start_node, end_node, true)
353            .await;
354        self.update_rel_type_index(&rel_id, &rel_type, true).await;
355
356        info!(
357            rel_id = %rel_id,
358            start_node = %start_node,
359            end_node = %end_node,
360            rel_type = rel_type,
361            "Created relationship"
362        );
363
364        Ok(relationship)
365    }

367    #[instrument(skip(self))]
368    async fn get_relationships(
369        &self,
370        node_id: &NodeId,
371        direction: Direction,
372        rel_types: Option<Vec<String>>,
373    ) -> OrbitResult<Vec<GraphRelationship>> {
374        let outgoing = self.outgoing_rels.read().await;
375        let incoming = self.incoming_rels.read().await;
376        let relationships = self.relationships.read().await;
377
378        let mut rel_ids = Vec::new();
379
380        match direction {
381            Direction::Outgoing => {
382                if let Some(rels) = outgoing.get(node_id) {
383                    rel_ids.extend(rels.clone());
384                }
385            }
386            Direction::Incoming => {
387                if let Some(rels) = incoming.get(node_id) {
388                    rel_ids.extend(rels.clone());
389                }
390            }
391            Direction::Both => {
392                if let Some(rels) = outgoing.get(node_id) {
393                    rel_ids.extend(rels.clone());
394                }
395                if let Some(rels) = incoming.get(node_id) {
396                    rel_ids.extend(rels.clone());
397                }
398            }
399        }
400
401        let mut result = Vec::new();
402        for rel_id in rel_ids {
403            if let Some(relationship) = relationships.get(&rel_id) {
404                // Filter by relationship types if specified
405                if let Some(ref types) = rel_types {
406                    if !types.contains(&relationship.rel_type) {
407                        continue;
408                    }
409                }
410                result.push(relationship.clone());
411            }
412        }
413
414        debug!(
415            node_id = %node_id,
416            direction = ?direction,
417            rel_types = ?rel_types,
418            found_count = result.len(),
419            "Retrieved relationships"
420        );
421
422        Ok(result)
423    }

460    #[instrument(skip(self))]
461    async fn delete_relationship(&self, rel_id: &RelationshipId) -> OrbitResult<bool> {
462        // Get relationship details for cleanup
463        let relationship = {
464            let relationships = self.relationships.read().await;
465            relationships.get(rel_id).cloned()
466        };
467
468        if let Some(rel) = relationship {
469            // Update indexes
470            self.update_relationship_indexes(rel_id, &rel.start_node, &rel.end_node, false)
471                .await;
472            self.update_rel_type_index(rel_id, &rel.rel_type, false)
473                .await;
474
475            // Remove relationship
476            let removed = self.relationships.write().await.remove(rel_id).is_some();
477
478            if removed {
479                info!(rel_id = %rel_id, "Deleted relationship");
480            }
481            Ok(removed)
482        } else {
483            Ok(false)
484        }
485    }

487    #[instrument(skip(self, property_filters), fields(limit = limit.unwrap_or(0)))]
488    async fn find_nodes_by_label(
489        &self,
490        label: &str,
491        property_filters: Option<HashMap<String, serde_json::Value>>,
492        limit: Option<usize>,
493    ) -> OrbitResult<Vec<GraphNode>> {
494        let label_index = self.label_index.read().await;
495        let nodes = self.nodes.read().await;
496
497        let node_ids = match label_index.get(label) {
498            Some(ids) => ids,
499            None => {
500                debug!(label = label, "No nodes found with label");
501                return Ok(Vec::new());
502            }
503        };
504
505        let mut result = Vec::new();
506        let mut count = 0;
507
508        for node_id in node_ids {
509            if let Some(limit_val) = limit {
510                if count >= limit_val {
511                    break;
512                }
513            }
514
515            if let Some(node) = nodes.get(node_id) {
516                // Apply property filters if specified
517                if let Some(ref filters) = property_filters {
518                    let mut matches = true;
519                    for (key, expected_value) in filters {
520                        match node.get_property(key) {
521                            Some(actual_value) if actual_value == expected_value => continue,
522                            _ => {
523                                matches = false;
524                                break;
525                            }
526                        }
527                    }
528                    if !matches {
529                        continue;
530                    }
531                }
532
533                result.push(node.clone());
534                count += 1;
535            }
536        }
537
538        debug!(
539            label = label,
540            property_filters = ?property_filters,
541            limit = limit,
542            found_count = result.len(),
543            "Found nodes by label"
544        );
545
546        Ok(result)
547    }
