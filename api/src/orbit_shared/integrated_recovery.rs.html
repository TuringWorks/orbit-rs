13/// Integrated recovery and leader election system
14/// This combines the leader election mechanism with the transaction recovery system
15pub struct IntegratedRecoverySystem {
16    node_id: NodeId,
17    recovery_manager: Arc<TransactionRecoveryManager>,
18    election_manager: Arc<UniversalElectionManager>,
19    cluster_manager: Arc<EnhancedClusterManager>,
20    is_started: Arc<RwLock<bool>>,
21}

23impl IntegratedRecoverySystem {
24    /// Create a new integrated recovery system
25    pub async fn new(
26        node_id: NodeId,
27        cluster_nodes: Vec<NodeId>,
28        logger: Arc<dyn PersistentTransactionLogger>,
29        state_path: Option<std::path::PathBuf>,
30    ) -> OrbitResult<Self> {
31        info!(
32            "Initializing integrated recovery system for node: {}",
33            node_id
34        );
35
36        // Create cluster manager with recovery integration
37        let quorum_config = QuorumConfig {
38            min_quorum_size: cluster_nodes.len() / 2 + 1,
39            max_failures: cluster_nodes.len() / 2,
40            quorum_timeout: Duration::from_secs(30),
41            dynamic_quorum: true,
42        };
43
44        let raft_config = RaftConfig {
45            election_timeout_min: Duration::from_millis(150),
46            election_timeout_max: Duration::from_millis(300),
47            heartbeat_interval: Duration::from_millis(50),
48            ..Default::default()
49        };
50
51        let cluster_manager = Arc::new(EnhancedClusterManager::new(
52            node_id.clone(),
53            cluster_nodes,
54            quorum_config,
55            raft_config,
56        ));
57
58        // Create recovery manager using the cluster manager
59        let recovery_manager = Arc::new(TransactionRecoveryManager::new(
60            node_id.clone(),
61            crate::recovery::RecoveryConfig::default(),
62            logger,
63            cluster_manager.clone() as Arc<dyn crate::recovery::ClusterManager>,
64        ));
65
66        // Create recovery event handler and wire it to the Raft consensus
67        let recovery_event_handler =
68            Arc::new(RecoveryRaftEventHandler::new(recovery_manager.clone()));
69        cluster_manager
70            .raft_consensus
71            .add_event_handler(recovery_event_handler)
72            .await;
73
74        // Create election manager for universal deployment support
75        let deployment_mode = UniversalElectionManager::detect_deployment_mode().await;
76        let k8s_config = crate::k8s_election::K8sElectionConfig::default();
77
78        let election_manager = Arc::new(
79            UniversalElectionManager::new(node_id.clone(), deployment_mode, k8s_config, state_path)
80                .await?,
81        );
82
83        Ok(Self {
84            node_id,
85            recovery_manager,
86            election_manager,
87            cluster_manager,
88            is_started: Arc::new(RwLock::new(false)),
89        })
90    }

92    /// Start the integrated recovery system
93    pub async fn start(
94        &self,
95        transport: Option<Arc<dyn crate::consensus::RaftTransport>>,
96    ) -> OrbitResult<()> {
97        let mut started = self.is_started.write().await;
98        if *started {
99            return Err(OrbitError::internal("Recovery system is already started"));
100        }
101
102        info!(
103            "Starting integrated recovery system for node: {}",
104            self.node_id
105        );
106
107        // Start the cluster manager (includes Raft consensus)
108        if let Some(transport) = transport {
109            self.cluster_manager.start(transport).await?;
110        } else {
111            warn!("No transport provided - cluster manager running without Raft transport");
112        }
113
114        // Start the recovery manager
115        self.recovery_manager.start().await?;
116
117        // Start the election manager (for universal deployment support)
118        self.election_manager.start().await?;
119
120        // Set up leader change monitoring
121        self.start_leader_monitoring().await?;
122
123        *started = true;
124        info!("Integrated recovery system started successfully");
125
126        Ok(())
127    }

129    /// Start monitoring for leader changes and sync with recovery system
130    async fn start_leader_monitoring(&self) -> OrbitResult<()> {
131        let recovery_manager = Arc::clone(&self.recovery_manager);
132        let election_manager = Arc::clone(&self.election_manager);
133        let cluster_manager = Arc::clone(&self.cluster_manager);
134
135        tokio::spawn(async move {
136            Self::run_leader_monitoring_loop(recovery_manager, election_manager, cluster_manager)
137                .await;
138        });
139
140        Ok(())
141    }
142
143    /// Main loop for leader monitoring
144    async fn run_leader_monitoring_loop(
145        recovery_manager: Arc<TransactionRecoveryManager>,
146        election_manager: Arc<UniversalElectionManager>,
147        cluster_manager: Arc<EnhancedClusterManager>,
148    ) {
149        let mut interval = tokio::time::interval(Duration::from_secs(5));
150
151        loop {
152            interval.tick().await;
153
154            let effective_leader =
155                Self::determine_effective_leader(&cluster_manager, &election_manager).await;
156
157            if let Err(e) = recovery_manager
158                .update_cluster_leader(effective_leader.clone())
159                .await
160            {
161                error!("Failed to update cluster leader in recovery manager: {}", e);
162            }
163
164            Self::handle_leader_responsibilities(&recovery_manager, &effective_leader).await;
165        }
166    }

168    /// Determine the effective leader from both Raft and election systems
169    async fn determine_effective_leader(
170        cluster_manager: &Arc<EnhancedClusterManager>,
171        election_manager: &Arc<UniversalElectionManager>,
172    ) -> Option<NodeId> {
173        let raft_leader = cluster_manager.raft_consensus.get_leader().await;
174        let election_leader = election_manager
175            .get_current_leader()
176            .await
177            .map(|s| NodeId::from_string(&s));
178
179        raft_leader.or(election_leader)
180    }

182    /// Handle recovery responsibilities if this node is the leader
183    async fn handle_leader_responsibilities(
184        recovery_manager: &Arc<TransactionRecoveryManager>,
185        effective_leader: &Option<NodeId>,
186    ) {
187        if let Some(leader_id) = effective_leader {
188            if leader_id == &recovery_manager.node_id {
189                Self::perform_leader_recovery_duties(recovery_manager).await;
190            }
191        }
192    }
193
194    /// Perform periodic recovery duties when this node is the leader
195    async fn perform_leader_recovery_duties(recovery_manager: &Arc<TransactionRecoveryManager>) {
196        let transaction_ids = Self::get_transaction_ids_for_recovery(recovery_manager).await;
197
198        for transaction_id in transaction_ids {
199            Self::check_and_recover_transaction(recovery_manager, &transaction_id).await;
200        }
201    }
202
203    /// Get limited set of transaction IDs that may need recovery
204    async fn get_transaction_ids_for_recovery(
205        recovery_manager: &Arc<TransactionRecoveryManager>,
206    ) -> Vec<crate::TransactionId> {
207        let checkpoints = recovery_manager.get_checkpoints().await;
208        checkpoints.keys().take(10).cloned().collect()
209    }

211    /// Check a specific transaction and recover if coordinator is unhealthy
212    async fn check_and_recover_transaction(
213        recovery_manager: &Arc<TransactionRecoveryManager>,
214        transaction_id: &crate::TransactionId,
215    ) {
216        if let Some(coordinator) = recovery_manager
217            .get_transaction_coordinator(transaction_id)
218            .await
219        {
220            Self::handle_coordinator_health_check(recovery_manager, &coordinator).await;
221        }
222    }
223
224    /// Handle health check for a transaction coordinator and recover if needed
225    async fn handle_coordinator_health_check(
226        recovery_manager: &Arc<TransactionRecoveryManager>,
227        coordinator: &NodeId,
228    ) {
229        let coordinators = recovery_manager.coordinators.read().await;
230        if let Some(health) = coordinators.get(coordinator) {
231            if !health.is_healthy {
232                Self::initiate_coordinator_recovery(recovery_manager, coordinator).await;
233            }
234        }
235    }
236
237    /// Initiate recovery for an unhealthy coordinator
238    async fn initiate_coordinator_recovery(
239        recovery_manager: &Arc<TransactionRecoveryManager>,
240        coordinator: &NodeId,
241    ) {
242        info!(
243            "Found unhealthy coordinator {}, initiating recovery",
244            coordinator
245        );
246
247        if let Err(e) = recovery_manager
248            .reassign_transactions_to_leader(coordinator)
249            .await
250        {
251            error!(
252                "Failed to reassign transactions from unhealthy coordinator: {}",
253                e
254            );
255        }
256    }

268    /// Check if this node is the current leader
269    pub async fn is_leader(&self) -> bool {
270        if let Some(leader) = self.get_current_leader().await {
271            leader == self.node_id
272        } else {
273            false

277    /// Force a recovery scan (useful for testing or manual intervention)
278    pub async fn force_recovery_scan(&self) -> OrbitResult<()> {
279        self.validate_leader_for_recovery_scan().await?;
280
281        info!("Forcing comprehensive recovery scan...");
282
283        let failed_coordinators = self.identify_failed_coordinators().await;
284
285        for failed_coordinator in failed_coordinators {
286            self.recover_transactions_from_coordinator(&failed_coordinator)
287                .await?;
288        }
289
290        Ok(())
291    }

293    /// Validate that this node is the leader before initiating recovery scan
294    async fn validate_leader_for_recovery_scan(&self) -> OrbitResult<()> {
295        if !self.is_leader().await {
296            return Err(OrbitError::cluster(
297                "Only the leader can initiate recovery scans",
298            ));
299        }
300        Ok(())
301    }
302
303    /// Identify coordinators that have failed and need recovery
304    async fn identify_failed_coordinators(&self) -> Vec<NodeId> {
305        let coordinators = self.recovery_manager.coordinators.read().await;
306        coordinators
307            .iter()
308            .filter_map(|(node_id, health)| {
309                if !health.is_healthy {
310                    Some(node_id.clone())
311                } else {
312                    None
313                }
314            })
315            .collect()
316    }
317
318    /// Recover transactions from a specific failed coordinator
319    async fn recover_transactions_from_coordinator(
320        &self,
321        failed_coordinator: &NodeId,
322    ) -> OrbitResult<()> {
323        let transactions = self
324            .recovery_manager
325            .find_transactions_needing_recovery(failed_coordinator)
326            .await?;
327
328        if !transactions.is_empty() {
329            info!(
330                "Found {} transactions needing recovery from {}",
331                transactions.len(),
332                failed_coordinator
333            );
334
335            self.recovery_manager
336                .become_coordinator(transactions)
337                .await?;
338        }
339
340        Ok(())
341    }

353    /// Gracefully shutdown the recovery system
354    pub async fn shutdown(&self) -> OrbitResult<()> {
355        let mut started = self.is_started.write().await;
356        if !*started {
357            return Ok(());
358        }
359
360        info!("Shutting down integrated recovery system...");
361
362        // If this node is leader, try to transfer leadership gracefully
363        if self.is_leader().await {
364            if let Err(e) = self.election_manager.release_leadership().await {
365                warn!("Failed to gracefully release leadership: {}", e);
366            }
367        }
368
369        *started = false;
370        info!("Integrated recovery system shutdown complete");
371
372        Ok(())
373    }
