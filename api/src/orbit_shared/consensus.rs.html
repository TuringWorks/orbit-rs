185    fn random_election_timeout(config: &RaftConfig) -> Duration {
186        let min_ms = config.election_timeout_min.as_millis() as u64;
187        let max_ms = config.election_timeout_max.as_millis() as u64;
188        let random_ms = min_ms + (fastrand::u64(0..=(max_ms - min_ms)));
189        Duration::from_millis(random_ms)
190    }

238    /// Start election timer
239    async fn start_election_timer(&self, transport: Arc<dyn RaftTransport>) {
240        let consensus = self.clone();
241
242        tokio::spawn(async move {
243            loop {
244                let timeout = {
245                    let election_timeout = consensus.election_timeout.read().await;
246                    *election_timeout
247                };
248
249                tokio::time::sleep(timeout).await;
250
251                // Check if we need to start an election
252                let should_start_election = {
253                    let state = consensus.state.read().await;
254                    let last_heartbeat = consensus.last_heartbeat.read().await;
255                    let time_since_heartbeat = last_heartbeat.elapsed();
256
257                    (*state == RaftState::Follower || *state == RaftState::Candidate)
258                        && time_since_heartbeat > timeout
259                };
260
261                if should_start_election {
262                    if let Err(e) = consensus.start_election(transport.clone()).await {
263                        error!("Failed to start election: {}", e);
264                    }
265                }
266            }
267        });
268    }

272        let consensus = self.clone();
273
274        tokio::spawn(async move {
275            let mut interval = tokio::time::interval(consensus.config.heartbeat_interval);

279
280                let is_leader = consensus.is_leader().await;
281                if is_leader {
282                    if let Err(e) = consensus.send_heartbeats(transport.clone()).await {
283                        error!("Failed to send heartbeats: {}", e);
284                    }
285                }
286            }

290    /// Start election process
291    async fn start_election(&self, transport: Arc<dyn RaftTransport>) -> OrbitResult<()> {
292        let new_term = self.prepare_for_election().await;
293        info!("Starting election for term: {}", new_term);
294
295        let (last_log_index, last_log_term) = self.get_last_log_info().await;
296        let vote_results = self
297            .collect_votes(&transport, new_term, last_log_index, last_log_term)
298            .await;
299        let vote_count = self.process_vote_results(vote_results, new_term).await?;
300
301        self.finalize_election(vote_count, new_term).await
302    }

304    /// Prepare for election by transitioning to candidate and incrementing term
305    async fn prepare_for_election(&self) -> u64 {
306        // Transition to candidate
307        {
308            let mut state = self.state.write().await;
309            *state = RaftState::Candidate;
310        }
311
312        // Increment term and vote for self
313        let new_term = {
314            let mut term = self.current_term.write().await;
315            *term += 1;
316            *term
317        };
318
319        {
320            let mut voted_for = self.voted_for.write().await;
321            *voted_for = Some(self.node_id.clone());
322        }
323
324        // Reset votes
325        {
326            let mut votes = self.votes_received.write().await;
327            votes.clear();
328            votes.insert(self.node_id.clone(), true); // Vote for self
329        }
330
331        new_term
332    }

345    /// Collect votes from other nodes
346    async fn collect_votes(
347        &self,
348        transport: &Arc<dyn RaftTransport>,
349        new_term: u64,
350        last_log_index: u64,
351        last_log_term: u64,
352    ) -> Vec<(NodeId, VoteResponse)> {
353        let cluster_nodes = self.cluster_nodes.read().await;
354        let mut vote_tasks = Vec::new();
355
356        for node in cluster_nodes.iter() {
357            if node != &self.node_id {
358                let request = VoteRequest {
359                    term: new_term,
360                    candidate_id: self.node_id.clone(),
361                    last_log_index,
362                    last_log_term,
363                };
364
365                let node = node.clone();
366                let transport = transport.clone();
367
368                let task = tokio::spawn(async move {
369                    match transport.send_vote_request(&node, request).await {
370                        Ok(response) => Some((node, response)),
371                        Err(e) => {
372                            warn!("Failed to get vote from {}: {}", node, e);
373                            None
374                        }
375                    }
376                });
377
378                vote_tasks.push(task);
379            }
380        }
381
382        // Collect votes with timeout
383        let election_timeout = *self.election_timeout.read().await;
384        tokio::time::timeout(election_timeout, async {
385            let mut results = Vec::new();
386            for task in vote_tasks {
387                if let Ok(Some(result)) = task.await {
388                    results.push(result);
389                }
390            }
391            results
392        })
393        .await
394        .unwrap_or_default()
395    }

397    /// Process vote results and return vote count
398    async fn process_vote_results(
399        &self,
400        vote_results: Vec<(NodeId, VoteResponse)>,
401        new_term: u64,
402    ) -> OrbitResult<usize> {
403        let mut vote_count = 1; // Self vote
404
405        for (node_id, response) in vote_results {
406            if response.term > new_term {
407                // Higher term discovered, step down
408                self.step_down(response.term).await?;
409                return Ok(0); // Election aborted
410            }
411
412            if response.vote_granted {
413                vote_count += 1;
414                let mut votes = self.votes_received.write().await;
415                votes.insert(node_id, true);
416            }
417        }
418
419        Ok(vote_count)
420    }

422    /// Finalize election based on vote count
423    async fn finalize_election(&self, vote_count: usize, new_term: u64) -> OrbitResult<()> {
424        let cluster_nodes = self.cluster_nodes.read().await;
425        let cluster_size = cluster_nodes.len();
426        let majority = cluster_size / 2 + 1;
427
428        if vote_count >= majority {
429            self.become_leader().await?;
430        } else if vote_count > 0 {
431            // Election failed, return to follower
432            self.step_down(new_term).await?;
433        }
434        // If vote_count is 0, we already stepped down in process_vote_results
435
436        Ok(())
437    }

486    /// Step down from candidate/leader to follower
487    async fn step_down(&self, new_term: u64) -> OrbitResult<()> {
488        let old_term = {
489            let mut term = self.current_term.write().await;
490            let old = *term;
491            *term = new_term;
492            old
493        };
494
495        {
496            let mut state = self.state.write().await;
497            *state = RaftState::Follower;
498        }
499
500        {
501            let mut voted_for = self.voted_for.write().await;
502            *voted_for = None;
503        }
504
505        // Notify event handlers
506        if old_term != new_term {
507            let handlers = self.event_handlers.read().await;
508            for handler in handlers.iter() {
509                if let Err(e) = handler.on_term_changed(old_term, new_term).await {
510                    error!("Term changed event handler failed: {}", e);
511                }
512            }
513        }
514
515        Ok(())
516    }

518    /// Send heartbeats to all followers
519    async fn send_heartbeats(&self, transport: Arc<dyn RaftTransport>) -> OrbitResult<()> {
520        let cluster_nodes = self.cluster_nodes.read().await;
521        let current_term = self.get_current_term().await;
522        let commit_index = *self.commit_index.read().await;
523
524        // Get previous log entry info
525        let (prev_log_index, prev_log_term) = {
526            let log = self.log.read().await;
527            if log.is_empty() {
528                (0, 0)
529            } else {
530                let last_entry = log.last().unwrap();
531                (last_entry.index, last_entry.term)
532            }
533        };
534
535        let request = AppendEntriesRequest {
536            term: current_term,
537            leader_id: self.node_id.clone(),
538            prev_log_index,
539            prev_log_term,
540            entries: vec![], // Heartbeat - no entries
541            leader_commit: commit_index,
542        };
543
544        let other_nodes: Vec<NodeId> = cluster_nodes
545            .iter()
546            .filter(|node| *node != &self.node_id)
547            .cloned()
548            .collect();
549
550        let responses = transport.broadcast_heartbeat(&other_nodes, request).await?;
551
552        // Process heartbeat responses
553        for response in responses {
554            if response.term > current_term {
555                self.step_down(response.term).await?;
556                return Ok(());
557            }
558        }
559
560        Ok(())
561    }

563    /// Handle incoming vote request
564    pub async fn handle_vote_request(&self, request: VoteRequest) -> OrbitResult<VoteResponse> {
565        let current_term = self.get_current_term().await;
566        let voted_for = self.voted_for.read().await.clone();
567
568        // Reply false if term < currentTerm
569        if request.term < current_term {
570            return Ok(VoteResponse {
571                term: current_term,
572                vote_granted: false,
573                voter_id: self.node_id.clone(),
574            });
575        }
576
577        // If term > currentTerm, update term and step down
578        if request.term > current_term {
579            self.step_down(request.term).await?;
580        }
581
582        // Check if we can grant vote
583        let can_vote = voted_for.is_none() || voted_for == Some(request.candidate_id.clone());
584
585        let log_up_to_date = {
586            let log = self.log.read().await;
587            if log.is_empty() {
588                true
589            } else {
590                let last_entry = log.last().unwrap();
591                request.last_log_term > last_entry.term
592                    || (request.last_log_term == last_entry.term
593                        && request.last_log_index >= last_entry.index)
594            }
595        };
596
597        let vote_granted = can_vote && log_up_to_date;
598
599        if vote_granted {
600            let mut voted_for_write = self.voted_for.write().await;
601            *voted_for_write = Some(request.candidate_id.clone());
602        }
603
604        Ok(VoteResponse {
605            term: request.term,
606            vote_granted,
607            voter_id: self.node_id.clone(),
608        })
609    }

611    /// Handle incoming append entries request
612    pub async fn handle_append_entries(
613        &self,
614        request: AppendEntriesRequest,
615    ) -> OrbitResult<AppendEntriesResponse> {
616        let current_term = self.get_current_term().await;
617
618        // Reply false if term < currentTerm
619        if request.term < current_term {
620            return Ok(AppendEntriesResponse {
621                term: current_term,
622                success: false,
623                follower_id: self.node_id.clone(),
624                last_log_index: 0,
625            });
626        }
627
628        // Update heartbeat timestamp
629        {
630            let mut last_heartbeat = self.last_heartbeat.write().await;
631            *last_heartbeat = Instant::now();
632        }
633
634        // If term >= currentTerm, convert to follower
635        if request.term >= current_term {
636            self.step_down(request.term).await?;
637        }
638
639        // Reset election timeout
640        {
641            let mut election_timeout = self.election_timeout.write().await;
642            *election_timeout = Self::random_election_timeout(&self.config);
643        }
644
645        // Handle log consistency check and append entries
646        // (Simplified implementation - full Raft log handling would go here)
647
648        Ok(AppendEntriesResponse {
649            term: request.term,
650            success: true,
651            follower_id: self.node_id.clone(),
652            last_log_index: request.prev_log_index + request.entries.len() as u64,
653        })
654    }
