105    /// Create a recovery checkpoint for a saga execution
106    pub async fn create_saga_checkpoint(&self, execution: &SagaExecution) -> OrbitResult<()> {
107        let checkpoint = SagaRecoveryCheckpoint::from(execution);
108
109        // Store checkpoint in memory
110        {
111            let mut checkpoints = self.checkpoints.write().await;
112            checkpoints.insert(checkpoint.saga_id.clone(), checkpoint.clone());
113        }
114
115        // Persist checkpoint to log
116        self.persist_saga_checkpoint(&checkpoint).await?;
117
118        debug!(
119            "Created saga recovery checkpoint for: {}",
120            checkpoint.saga_id
121        );
122        Ok(())
123    }

125    /// Update an existing saga checkpoint
126    pub async fn update_saga_checkpoint(
127        &self,
128        saga_id: &SagaId,
129        execution: &SagaExecution,
130    ) -> OrbitResult<()> {
131        let mut checkpoints = self.checkpoints.write().await;
132        if let Some(checkpoint) = checkpoints.get_mut(saga_id) {
133            // Update checkpoint with current execution state
134            checkpoint.saga_state = execution.state.clone();
135            checkpoint.completed_steps = execution.completed_steps.clone();
136            checkpoint.failed_steps = execution.failed_steps.clone();
137            checkpoint.compensated_steps = execution.compensated_steps.clone();
138            checkpoint.current_step = execution.current_step.clone();
139            checkpoint.context_data = execution.context.data.clone();
140            checkpoint.last_updated = chrono::Utc::now().timestamp_millis();
141
142            // Persist updated checkpoint
143            self.persist_saga_checkpoint(checkpoint).await?;
144
145            debug!("Updated saga recovery checkpoint for: {}", saga_id);
146        } else {
147            return Err(OrbitError::internal("Saga checkpoint not found"));
148        }
149
150        Ok(())
151    }

161    /// Recover sagas from a failed coordinator
162    pub async fn recover_sagas_from_coordinator(
163        &self,
164        failed_coordinator: &NodeId,
165    ) -> OrbitResult<()> {
166        info!(
167            "Recovering sagas from failed coordinator: {}",
168            failed_coordinator
169        );
170
171        let sagas_to_recover = {
172            let checkpoints = self.checkpoints.read().await;
173            checkpoints
174                .values()
175                .filter(|checkpoint| &checkpoint.coordinator_node == failed_coordinator)
176                .filter(|checkpoint| self.is_recoverable_state(&checkpoint.saga_state))
177                .cloned()
178                .collect::<Vec<_>>()
179        };
180
181        if sagas_to_recover.is_empty() {
182            info!(
183                "No sagas to recover from coordinator: {}",
184                failed_coordinator
185            );
186            return Ok(());
187        }
188
189        info!(
190            "Found {} sagas to recover from coordinator: {}",
191            sagas_to_recover.len(),
192            failed_coordinator
193        );
194
195        for checkpoint in sagas_to_recover {
196            if let Err(e) = self.recover_saga_execution(&checkpoint).await {
197                error!("Failed to recover saga {}: {}", checkpoint.saga_id, e);
198            }
199        }
200
201        Ok(())
202    }

204    /// Check if saga state is recoverable
205    fn is_recoverable_state(&self, state: &SagaState) -> bool {
206        matches!(
207            state,
208            SagaState::Initializing
209                | SagaState::Executing
210                | SagaState::Compensating
211                | SagaState::Paused
212        )
213    }

215    /// Recover a specific saga execution
216    async fn recover_saga_execution(&self, checkpoint: &SagaRecoveryCheckpoint) -> OrbitResult<()> {
217        info!("Recovering saga execution: {}", checkpoint.saga_id);
218
219        // Validate recovery can proceed
220        self.validate_recovery_attempts(checkpoint)?;
221
222        // Reconstruct saga execution from checkpoint
223        let recovered_execution = self.reconstruct_saga_execution(checkpoint).await?;
224
225        // Execute recovery strategy based on saga state
226        self.execute_recovery_strategy(checkpoint, &recovered_execution)
227            .await?;
228
229        // Update recovery attempts counter
230        self.update_recovery_attempts(&checkpoint.saga_id).await;
231
232        info!("Saga recovery completed: {}", checkpoint.saga_id);
233        Ok(())
234    }

236    /// Validate that recovery attempts are within limits
237    fn validate_recovery_attempts(&self, checkpoint: &SagaRecoveryCheckpoint) -> OrbitResult<()> {
238        if checkpoint.recovery_attempts >= self.config.max_recovery_attempts {
239            error!(
240                "Max recovery attempts exceeded for saga: {}",
241                checkpoint.saga_id
242            );
243            return Err(OrbitError::internal("Max recovery attempts exceeded"));
244        }
245        Ok(())
246    }

248    /// Execute the appropriate recovery strategy based on saga state
249    async fn execute_recovery_strategy(
250        &self,
251        checkpoint: &SagaRecoveryCheckpoint,
252        recovered_execution: &SagaExecution,
253    ) -> OrbitResult<()> {
254        match checkpoint.saga_state {
255            SagaState::Initializing => {
256                self.handle_initializing_saga_recovery(checkpoint, recovered_execution)
257                    .await
258            }
259            SagaState::Executing => {
260                self.handle_executing_saga_recovery(checkpoint, recovered_execution)
261                    .await
262            }
263            SagaState::Compensating => {
264                self.handle_compensating_saga_recovery(checkpoint, recovered_execution)
265                    .await
266            }
267            SagaState::Paused => {
268                self.handle_paused_saga_recovery(checkpoint, recovered_execution)
269                    .await
270            }
271            _ => {
272                self.handle_non_recoverable_saga(checkpoint);
273                Ok(())
274            }
275        }
276    }

338    /// Reconstruct saga execution from checkpoint
339    async fn reconstruct_saga_execution(
340        &self,
341        checkpoint: &SagaRecoveryCheckpoint,
342    ) -> OrbitResult<SagaExecution> {
343        use crate::saga::{SagaContext, SagaExecution};
344
345        // Reconstruct saga context
346        let mut context = SagaContext::new(checkpoint.saga_id.clone(), self.node_id.clone());
347        context.data = checkpoint.context_data.clone();
348        context.current_step = checkpoint.current_step.clone();
349
350        // Reconstruct execution state
351        let mut execution = SagaExecution::new(
352            checkpoint.saga_id.clone(),
353            checkpoint.definition_id.clone(),
354            context,
355        );
356
357        execution.state = checkpoint.saga_state.clone();
358        execution.completed_steps = checkpoint.completed_steps.clone();
359        execution.failed_steps = checkpoint.failed_steps.clone();
360        execution.compensated_steps = checkpoint.compensated_steps.clone();
361        execution.current_step = checkpoint.current_step.clone();
362        execution.started_at = checkpoint.started_at;
363
364        Ok(execution)
365    }

435    /// Start background recovery tasks
436    async fn start_background_tasks(&self) -> OrbitResult<()> {
437        let recovery_manager = self.clone();
438
439        // Checkpoint management task
440        tokio::spawn(async move {
441            let mut interval = tokio::time::interval(recovery_manager.config.checkpoint_interval);
442            loop {
443                interval.tick().await;
444                if let Err(e) = recovery_manager.cleanup_old_checkpoints().await {
445                    error!("Saga checkpoint cleanup failed: {}", e);
446                }
447            }
448        });
449
450        Ok(())
451    }

453    /// Clean up old saga checkpoints
454    async fn cleanup_old_checkpoints(&self) -> OrbitResult<()> {
455        let cutoff_time = chrono::Utc::now().timestamp_millis() - (24 * 60 * 60 * 1000); // 24 hours
456
457        let mut checkpoints = self.checkpoints.write().await;
458        let mut to_remove = Vec::new();
459
460        for (saga_id, checkpoint) in checkpoints.iter() {
461            // Remove checkpoints for completed sagas or very old ones
462            if matches!(
463                checkpoint.saga_state,
464                SagaState::Completed | SagaState::Aborted | SagaState::Failed
465            ) || checkpoint.last_updated < cutoff_time
466            {
467                to_remove.push(saga_id.clone());
468            }
469        }
470
471        for saga_id in to_remove {
472            checkpoints.remove(&saga_id);
473            debug!("Cleaned up old saga checkpoint: {}", saga_id);
474        }
475
476        Ok(())
477    }

483        let mut stats = SagaRecoveryStats {

488        };
489
490        for checkpoint in checkpoints.values() {

492                SagaState::Initializing | SagaState::Executing | SagaState::Compensating => {

495                SagaState::Completed => {

497                }
498                SagaState::Failed | SagaState::Aborted => {
499                    stats.failed_recoveries += 1;
500                }
501                _ => {}
502            }
503        }
504
505        stats
