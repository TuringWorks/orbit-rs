584    /// Detect performance bottlenecks
585    fn detect_bottlenecks(
586        &self,
587        detector: &BottleneckDetector,
588        total_duration: Duration,
589    ) -> Vec<PerformanceBottleneck> {
590        let mut bottlenecks = Vec::new();
591
592        // Check for CPU bottlenecks
593        let avg_cpu = detector.cpu_usage_samples.iter().sum::<f64>()
594            / detector.cpu_usage_samples.len() as f64;
595        if avg_cpu > 80.0 {
596            bottlenecks.push(PerformanceBottleneck {
597                bottleneck_type: BottleneckType::CPU,
598                severity: 8,
599                description: format!("High CPU usage detected (avg: {avg_cpu:.1}%)"),
600                affected_operation: "Query execution".to_string(),
601                time_impact: Duration::from_millis(
602                    (total_duration.as_millis() as f64 * 0.6) as u64,
603                ),
604                percentage_impact: 60.0,
605            });
606        }
607
608        // Check for memory pressure
609        if detector.memory_pressure_events > 0 {
610            bottlenecks.push(PerformanceBottleneck {
611                bottleneck_type: BottleneckType::Memory,
612                severity: 7,
613                description: format!(
614                    "{} memory pressure events detected",
615                    detector.memory_pressure_events
616                ),
617                affected_operation: "Memory allocation".to_string(),
618                time_impact: Duration::from_millis(100),
619                percentage_impact: 10.0,
620            });
621        }
622
623        // Check for I/O wait
624        if detector.io_wait_time > Duration::from_millis(100) {
625            let percentage = (detector.io_wait_time.as_millis() as f64
626                / total_duration.as_millis() as f64)
627                * 100.0;
628            bottlenecks.push(PerformanceBottleneck {
629                bottleneck_type: BottleneckType::DiskIO,
630                severity: 6,
631                description: format!("I/O wait time: {:.2}s", detector.io_wait_time.as_secs_f64()),
632                affected_operation: "Data access".to_string(),
633                time_impact: detector.io_wait_time,
634                percentage_impact: percentage,
635            });
636        }
637
638        bottlenecks
639    }

683    /// Generate an EXPLAIN ANALYZE report
684    pub fn explain_analyze(&self, profile: &QueryProfile) -> String {
685        let mut report = String::new();
686
687        report.push_str(&format!("Query Profile ID: {}\n", profile.profile_id));
688        report.push_str(&format!("Query: {}\n", profile.query));
689        report.push_str(&format!(
690            "Total execution time: {:.2}s\n",
691            profile.overall_stats.total_duration.as_secs_f64()
692        ));
693        report.push_str(&format!(
694            "Rows returned: {}\n",
695            profile.overall_stats.rows_returned
696        ));
697        report.push_str(&format!(
698            "Peak memory usage: {:.1} MB\n",
699            profile.overall_stats.peak_memory_mb
700        ));
701        report.push_str(&format!(
702            "Cache hit ratio: {:.1}%\n",
703            profile.overall_stats.cache_hit_ratio * 100.0
704        ));
705        report.push('\n');
706
707        // Execution phases breakdown
708        report.push_str("Execution Phases:\n");
709        for phase in &profile.phases {
710            report.push_str(&format!(
711                "  {}: {:.2}s ({:.1}%)\n",
712                phase.name,
713                phase.duration.as_secs_f64(),
714                phase.percentage
715            ));
716        }
717        report.push('\n');
718
719        // Bottlenecks
720        if !profile.bottlenecks.is_empty() {
721            report.push_str("Performance Bottlenecks:\n");
722            for bottleneck in &profile.bottlenecks {
723                report.push_str(&format!(
724                    "  {:?} (Severity {}): {}\n",
725                    bottleneck.bottleneck_type, bottleneck.severity, bottleneck.description
726                ));
727            }
728            report.push('\n');
729        }
730
731        // Optimization suggestions
732        if !profile.suggestions.is_empty() {
733            report.push_str("Optimization Suggestions:\n");
734            for suggestion in &profile.suggestions {
735                report.push_str(&format!(
736                    "  {} (Priority {}): {}\n",
737                    suggestion.title, suggestion.priority, suggestion.estimated_improvement
738                ));
739            }
740        }
741
742        report
743    }
