283/// Activation functions for neural network
284#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Default)]
285pub enum ActivationFunction {
286    /// Rectified Linear Unit: f(x) = max(0, x)
287    #[default]
288    ReLU,
289    /// Sigmoid: f(x) = 1 / (1 + e^(-x))
290    Sigmoid,
291    /// Hyperbolic Tangent: f(x) = tanh(x)
292    Tanh,
293    /// Linear: f(x) = x
294    Linear,
295    /// Leaky ReLU: f(x) = max(αx, x) where α is typically 0.01
296    LeakyReLU { alpha: f64 },
297    /// Parametric ReLU: f(x) = max(αx, x) where α is learned
298    PReLU { alpha: f64 },
299    /// Exponential Linear Unit: f(x) = x if x > 0, else α(e^x - 1)
300    ELU { alpha: f64 },
301    /// Scaled Exponential Linear Unit: f(x) = λ * ELU(x, α)
302    SELU,
303    /// Gaussian Error Linear Unit: f(x) = x * Φ(x)
304    GELU,
305    /// Swish (SiLU): f(x) = x * sigmoid(βx)
306    Swish { beta: f64 },
307    /// Mish: f(x) = x * tanh(softplus(x))
308    Mish,
309    /// Softmax: f(x_i) = e^(x_i) / Σ(e^(x_j))
310    Softmax,
311    /// Maxout: f(x) = max(w1^T x + b1, w2^T x + b2, ...)
312    Maxout { num_pieces: usize },

1272    fn linear_regression(&self, points: &VecDeque<TimeSeriesPoint>) -> (f64, f64) {
1273        // Simple linear regression implementation (mock)
1274        let n = points.len() as f64;
1275        let sum_x: f64 = (0..points.len()).map(|i| i as f64).sum();
1276        let sum_y: f64 = points.iter().map(|p| p.value).sum();
1277        let sum_xy: f64 = points
1278            .iter()
1279            .enumerate()
1280            .map(|(i, p)| i as f64 * p.value)
1281            .sum();
1282        let sum_x2: f64 = (0..points.len()).map(|i| (i as f64).powi(2)).sum();
1283
1284        let slope = (n * sum_xy - sum_x * sum_y) / (n * sum_x2 - sum_x.powi(2));
1285        let intercept = (sum_y - slope * sum_x) / n;
1286
1287        (slope, intercept)
1288    }

1698impl ActivationFunction {
1699    /// Apply the activation function to a single value
1700    pub fn apply(&self, x: f64) -> f64 {
1701        match self {
1702            ActivationFunction::ReLU => x.max(0.0),
1703            ActivationFunction::Sigmoid => 1.0 / (1.0 + (-x).exp()),
1704            ActivationFunction::Tanh => x.tanh(),
1705            ActivationFunction::Linear => x,
1706            ActivationFunction::LeakyReLU { alpha } => {
1707                if x > 0.0 {
1708                    x
1709                } else {
1710                    alpha * x
1711                }
1712            }
1713            ActivationFunction::PReLU { alpha } => {
1714                if x > 0.0 {
1715                    x
1716                } else {
1717                    alpha * x
1718                }
1719            }
1720            ActivationFunction::ELU { alpha } => {
1721                if x > 0.0 {
1722                    x
1723                } else {
1724                    alpha * (x.exp() - 1.0)
1725                }
1726            }
1727            ActivationFunction::SELU => {
1728                // SELU constants: λ ≈ 1.0507, α ≈ 1.6733
1729                let lambda = 1.050_700_987_355_480_5;
1730                let alpha = 1.673_263_242_354_377_2;
1731                if x > 0.0 {
1732                    lambda * x
1733                } else {
1734                    lambda * alpha * (x.exp() - 1.0)
1735                }
1736            }
1737            ActivationFunction::GELU => {
1738                // GELU approximation: 0.5 * x * (1 + tanh(sqrt(2/π) * (x + 0.044715 * x^3)))
1739                let sqrt_2_over_pi = (2.0 / std::f64::consts::PI).sqrt();
1740                0.5 * x * (1.0 + (sqrt_2_over_pi * (x + 0.044715 * x.powi(3))).tanh())
1741            }
1742            ActivationFunction::Swish { beta } => x * (1.0 / (1.0 + (-beta * x).exp())),
1743            ActivationFunction::Mish => {
1744                // Mish: x * tanh(softplus(x)) = x * tanh(ln(1 + e^x))
1745                x * (x.exp().ln_1p()).tanh()
1746            }
1747            ActivationFunction::Softmax => {
1748                // Softmax is typically applied to vectors, not scalars
1749                // This returns the input for scalar case
1750                x.exp()
1751            }
1752            ActivationFunction::Maxout { .. } => {
1753                // Maxout requires multiple linear transformations
1754                // This is a simplified implementation
1755                x.max(0.0)
1756            }
1757        }
1758    }
1759
1760    /// Apply softmax to a vector of values
1761    pub fn apply_softmax(values: &[f64]) -> Vec<f64> {
1762        if values.is_empty() {
1763            return vec![];
1764        }
1765
1766        // Find max for numerical stability
1767        let max_val = values.iter().fold(f64::NEG_INFINITY, |a, &b| a.max(b));
1768
1769        // Compute exponentials
1770        let exp_values: Vec<f64> = values.iter().map(|&x| (x - max_val).exp()).collect();
1771
1772        // Compute sum
1773        let sum: f64 = exp_values.iter().sum();
1774
1775        // Normalize
1776        exp_values.iter().map(|&x| x / sum).collect()
1777    }
1778
1779    /// Apply activation function to a vector of values
1780    pub fn apply_vector(&self, values: &[f64]) -> Vec<f64> {
1781        match self {
1782            ActivationFunction::Softmax => Self::apply_softmax(values),
1783            _ => values.iter().map(|&x| self.apply(x)).collect(),
1784        }
1785    }

1787    /// Compute the derivative of the activation function
1788    pub fn derivative(&self, x: f64) -> f64 {
1789        match self {
1790            ActivationFunction::ReLU => {
1791                if x > 0.0 {
1792                    1.0
1793                } else {
1794                    0.0
1795                }
1796            }
1797            ActivationFunction::Sigmoid => {
1798                let s = self.apply(x);
1799                s * (1.0 - s)
1800            }
1801            ActivationFunction::Tanh => {
1802                let t = x.tanh();
1803                1.0 - t * t
1804            }
1805            ActivationFunction::Linear => 1.0,
1806            ActivationFunction::LeakyReLU { alpha } => {
1807                if x > 0.0 {
1808                    1.0
1809                } else {
1810                    *alpha
1811                }
1812            }
1813            ActivationFunction::PReLU { alpha } => {
1814                if x > 0.0 {
1815                    1.0
1816                } else {
1817                    *alpha
1818                }
1819            }
1820            ActivationFunction::ELU { alpha } => {
1821                if x > 0.0 {
1822                    1.0
1823                } else {
1824                    alpha * x.exp()
1825                }
1826            }
1827            ActivationFunction::SELU => {
1828                let lambda = 1.050_700_987_355_480_5;
1829                let alpha = 1.673_263_242_354_377_2;
1830                if x > 0.0 {
1831                    lambda
1832                } else {
1833                    lambda * alpha * x.exp()
1834                }
1835            }
1836            ActivationFunction::GELU => {
1837                // Approximation of GELU derivative
1838                let sqrt_2_over_pi = (2.0 / std::f64::consts::PI).sqrt();
1839                let inner = sqrt_2_over_pi * (x + 0.044715 * x.powi(3));
1840                let tanh_val = inner.tanh();
1841                let sech2 = 1.0 - tanh_val.powi(2);
1842                0.5 * (1.0 + tanh_val)
1843                    + 0.5 * x * sech2 * sqrt_2_over_pi * (1.0 + 3.0 * 0.044715 * x.powi(2))
1844            }
1845            ActivationFunction::Swish { beta } => {
1846                let sigmoid = 1.0 / (1.0 + (-beta * x).exp());
1847                sigmoid + x * sigmoid * (1.0 - sigmoid) * beta
1848            }
1849            ActivationFunction::Mish => {
1850                // Derivative of Mish is complex, this is a simplified approximation
1851                let softplus = x.exp().ln_1p();
1852                let tanh_val = softplus.tanh();
1853                let sech2 = 1.0 - tanh_val.powi(2);
1854                tanh_val + x * sech2 * x.exp() / (1.0 + x.exp())
1855            }
1856            ActivationFunction::Softmax => {
1857                // Softmax derivative is matrix-valued for vectors
1858                // This is simplified for scalar case
1859                let s = self.apply(x);
1860                s * (1.0 - s)
1861            }
1862            ActivationFunction::Maxout { .. } => {
1863                // Maxout derivative depends on which piece is active
1864                if x > 0.0 {
1865                    1.0
1866                } else {
1867                    0.0
1868                }
1869            }
1870        }
1871    }

2799    fn initialize_training(&mut self, data: &[TrainingExample]) -> Vec<f64> {
2800        let mean_target: f64 =
2801            data.iter().map(|ex| ex.actual_cost).sum::<f64>() / data.len() as f64;
2802        let residuals: Vec<f64> = data.iter().map(|ex| ex.actual_cost - mean_target).collect();

2811    fn train_iteration(
2812        &mut self,
2813        data: &[TrainingExample],
2814        residuals: &mut [f64],
2815    ) -> Result<bool, AnalyticsError> {
2816        // Create and train weak learner
2817        let mut weak_learner = WeakLearner::new();
2818        weak_learner.train_on_residuals(data, residuals)?;
2819
2820        // Update residuals
2821        self.update_residuals(data, &weak_learner, residuals);
2822

2824        weak_learner.weight = self.learning_rate;
2825        self.estimators.push(weak_learner);
2826
2827        // Check for early stopping
2828        let loss = self.calculate_and_record_loss(residuals);
2829        Ok(loss < 1e-6)

2838    ) {
2839        for (i, example) in data.iter().enumerate() {
2840            let prediction = weak_learner.predict(&example.features);
2841            residuals[i] -= self.learning_rate * prediction;
2842        }

2848        self.loss_history.push(loss);
2849        loss
