27/// Multi-model query cost estimation
28#[derive(Debug, Clone, PartialEq, Serialize, Deserialize)]
29pub struct MultiModelQueryCost {
30    /// Base SQL query cost
31    pub sql_cost: f64,
32    /// Graph traversal cost
33    pub graph_cost: f64,
34    /// Document scanning cost
35    pub document_cost: f64,
36    /// Time-series aggregation cost
37    pub timeseries_cost: f64,
38    /// Cross-model join cost
39    pub cross_model_join_cost: f64,
40    /// Network I/O cost for distributed queries
41    pub network_cost: f64,
42}
43
44impl MultiModelQueryCost {
45    pub fn new() -> Self {
46        Self {
47            sql_cost: 0.0,
48            graph_cost: 0.0,
49            document_cost: 0.0,
50            timeseries_cost: 0.0,
51            cross_model_join_cost: 0.0,
52            network_cost: 0.0,
53        }
54    }
55
56    pub fn total_cost(&self) -> f64 {
57        self.sql_cost
58            + self.graph_cost
59            + self.document_cost
60            + self.timeseries_cost
61            + self.cross_model_join_cost
62            + self.network_cost
63    }
64
65    pub fn combine(&self, other: &MultiModelQueryCost) -> MultiModelQueryCost {
66        MultiModelQueryCost {
67            sql_cost: self.sql_cost + other.sql_cost,
68            graph_cost: self.graph_cost + other.graph_cost,
69            document_cost: self.document_cost + other.document_cost,
70            timeseries_cost: self.timeseries_cost + other.timeseries_cost,
71            cross_model_join_cost: self.cross_model_join_cost + other.cross_model_join_cost,
72            network_cost: self.network_cost + other.network_cost,
73        }
74    }
75}
76
77impl Default for MultiModelQueryCost {
78    fn default() -> Self {
79        Self::new()
80    }
81}

162}

164impl fmt::Display for OptimizationError {

166        match self {
167            OptimizationError::InvalidTransformation(msg) => {
168                write!(f, "Invalid transformation: {msg}")
169            }
170            OptimizationError::UnsupportedOperation(msg) => {
171                write!(f, "Unsupported operation: {msg}")
172            }
173            OptimizationError::InternalError(msg) => write!(f, "Internal error: {msg}"),
174        }
175    }

186/// Predicate pushdown optimization
187#[derive(Debug, Clone)]
188pub struct PredicatePushdown;
189
190impl OptimizationRule for PredicatePushdown {
191    fn name(&self) -> &'static str {
192        "PredicatePushdown"
193    }
194
195    fn apply(&self, stmt: &Statement) -> Result<Statement, OptimizationError> {
196        match stmt {
197            Statement::Select(select) => {
198                let optimized_select = self.push_predicates_down(select.clone())?;
199                Ok(Statement::Select(optimized_select))
200            }
201            _ => Ok(stmt.clone()),
202        }
203    }
204}
205
206impl PredicatePushdown {
207    fn push_predicates_down(
208        &self,
209        mut select: SelectStatement,
210    ) -> Result<SelectStatement, OptimizationError> {
211        if let Some(where_clause) = &select.where_clause {
212            // Extract predicates that can be pushed down
213            let (pushable, remaining) = self.separate_predicates(where_clause);
214
215            // Apply pushable predicates to FROM clauses
216            for from_clause in &mut select.from {
217                if let FromClause::Table { name, .. } = from_clause {
218                    // Find predicates that reference this table
219                    let table_predicates = self.extract_table_predicates(name, &pushable);
220                    if !table_predicates.is_empty() {
221                        // In a real implementation, these would be pushed to the storage layer
222                        // For now, we'll leave them in the WHERE clause but mark them as optimized
223                    }
224                }
225            }
226
227            // Update WHERE clause with remaining predicates
228            if remaining.is_empty() {
229                select.where_clause = None;
230            } else {
231                select.where_clause = Some(self.combine_predicates(remaining));
232            }
233        }
234
235        Ok(select)
236    }
237
238    fn separate_predicates(&self, expr: &Expression) -> (Vec<Expression>, Vec<Expression>) {
239        let mut pushable = Vec::new();
240        let mut remaining = Vec::new();
241
242        match expr {
243            Expression::Binary {
244                left,
245                operator: BinaryOperator::And,
246                right,
247            } => {
248                let (left_push, left_remain) = self.separate_predicates(left);
249                let (right_push, right_remain) = self.separate_predicates(right);
250
251                pushable.extend(left_push);
252                pushable.extend(right_push);
253                remaining.extend(left_remain);
254                remaining.extend(right_remain);
255            }
256            _ => {
257                if self.can_push_predicate(expr) {
258                    pushable.push(expr.clone());
259                } else {
260                    remaining.push(expr.clone());
261                }
262            }
263        }
264
265        (pushable, remaining)
266    }
267
268    fn can_push_predicate(&self, expr: &Expression) -> bool {
269        match expr {
270            Expression::Binary {
271                left,
272                operator,
273                right,
274            } => {
275                matches!(
276                    operator,
277                    BinaryOperator::Equal
278                        | BinaryOperator::GreaterThan
279                        | BinaryOperator::LessThan
280                        | BinaryOperator::GreaterThanOrEqual
281                        | BinaryOperator::LessThanOrEqual
282                        | BinaryOperator::NotEqual
283                ) && self.is_simple_column_reference(left)
284                    && self.is_literal_or_parameter(right)
285            }
286            _ => false,
287        }
288    }
289
290    fn is_simple_column_reference(&self, expr: &Expression) -> bool {
291        matches!(expr, Expression::Identifier(_))
292    }
293
294    fn is_literal_or_parameter(&self, expr: &Expression) -> bool {
295        matches!(expr, Expression::Literal(_) | Expression::Parameter(_))
296    }
297
298    fn extract_table_predicates(
299        &self,
300        table_name: &str,
301        predicates: &[Expression],
302    ) -> Vec<Expression> {
303        predicates
304            .iter()
305            .filter(|pred| self.references_table(pred, table_name))
306            .cloned()
307            .collect()
308    }
309
310    #[allow(clippy::only_used_in_recursion)]
311    fn references_table(&self, expr: &Expression, table_name: &str) -> bool {
312        match expr {
313            Expression::Binary { left, right, .. } => {
314                self.references_table(left, table_name) || self.references_table(right, table_name)
315            }
316            Expression::Identifier(name) => {
317                // Simple heuristic: assume unqualified column references belong to the current table
318                !name.contains('.')
319            }
320            _ => false,
321        }
322    }
323
324    fn combine_predicates(&self, predicates: Vec<Expression>) -> Expression {
325        if predicates.is_empty() {
326            return Expression::Literal(crate::orbitql::QueryValue::Boolean(true));
327        }
328
329        if predicates.len() == 1 {
330            return predicates[0].clone();
331        }
332
333        // Combine with AND
334        predicates
335            .into_iter()
336            .reduce(|acc, pred| Expression::Binary {
337                left: Box::new(acc),
338                operator: BinaryOperator::And,
339                right: Box::new(pred),
340            })
341            .unwrap()
342    }
343}

345/// Projection pushdown optimization
346#[derive(Debug, Clone)]
347pub struct ProjectionPushdown;
348
349impl OptimizationRule for ProjectionPushdown {
350    fn name(&self) -> &'static str {
351        "ProjectionPushdown"
352    }
353
354    fn apply(&self, stmt: &Statement) -> Result<Statement, OptimizationError> {
355        match stmt {
356            Statement::Select(select) => {
357                let optimized_select = self.push_projections_down(select.clone())?;
358                Ok(Statement::Select(optimized_select))
359            }
360            _ => Ok(stmt.clone()),
361        }
362    }
363}
364
365impl ProjectionPushdown {
366    fn push_projections_down(
367        &self,
368        mut select: SelectStatement,
369    ) -> Result<SelectStatement, OptimizationError> {
370        let required_columns = self.analyze_required_columns(&select);
371
372        // Update FROM clauses to only select required columns
373        for from_clause in &mut select.from {
374            match from_clause {
375                FromClause::Table { name, .. } => {
376                    if let Some(table_columns) = required_columns.get(name) {
377                        // In a real implementation, this would inform the storage layer
378                        // to only read the required columns
379                        self.optimize_table_projection(name, table_columns);
380                    }
381                }
382                FromClause::Subquery { query, .. } => {
383                    // Recursively optimize subqueries
384                    **query = self.push_projections_down(*query.clone())?;
385                }
386                _ => {}
387            }
388        }
389
390        Ok(select)
391    }
392
393    fn analyze_required_columns(&self, select: &SelectStatement) -> HashMap<String, Vec<String>> {
394        let mut required_columns: HashMap<String, Vec<String>> = HashMap::new();
395
396        // Analyze SELECT fields
397        for field in &select.fields {
398            match field {
399                SelectField::All => {
400                    // SELECT * requires all columns from all tables
401                    return HashMap::new(); // Return empty to indicate all columns needed
402                }
403                SelectField::Expression { expr, .. } => {
404                    self.extract_column_references(expr, &mut required_columns);
405                }
406                SelectField::AllFrom(table_name) => {
407                    // table.* requires all columns from specific table
408                    required_columns.insert(table_name.clone(), vec!["*".to_string()]);
409                }
410                SelectField::Graph { alias, .. } => {
411                    // Graph queries need special handling
412                    if let Some(alias) = alias {
413                        required_columns.insert(alias.clone(), vec!["*".to_string()]);
414                    }
415                }
416                SelectField::TimeSeries { alias, .. } => {
417                    // Time series queries need special handling
418                    if let Some(alias) = alias {
419                        required_columns.insert(alias.clone(), vec!["*".to_string()]);
420                    }
421                }
422            }
423        }
424
425        // Analyze WHERE clause
426        if let Some(where_expr) = &select.where_clause {
427            self.extract_column_references(where_expr, &mut required_columns);
428        }
429
430        // Analyze JOIN conditions
431        for join in &select.join_clauses {
432            self.extract_column_references(&join.condition, &mut required_columns);
433        }
434
435        // Analyze GROUP BY
436        for group_expr in &select.group_by {
437            self.extract_column_references(group_expr, &mut required_columns);
438        }
439
440        // Analyze HAVING
441        if let Some(having_expr) = &select.having {
442            self.extract_column_references(having_expr, &mut required_columns);
443        }
444
445        // Analyze ORDER BY
446        for order in &select.order_by {
447            self.extract_column_references(&order.expression, &mut required_columns);
448        }
449
450        required_columns
451    }
452
453    #[allow(clippy::only_used_in_recursion)]
454    fn extract_column_references(
455        &self,
456        expr: &Expression,
457        columns: &mut HashMap<String, Vec<String>>,
458    ) {
459        match expr {
460            Expression::Identifier(name) => {
461                if name.contains('.') {
462                    let parts: Vec<&str> = name.split('.').collect();
463                    if parts.len() == 2 {
464                        let table = parts[0].to_string();
465                        let column = parts[1].to_string();
466                        columns.entry(table).or_default().push(column);
467                    }
468                } else {
469                    // Unqualified column - add to default table
470                    columns
471                        .entry("_default".to_string())
472                        .or_default()
473                        .push(name.clone());
474                }
475            }
476            Expression::Binary { left, right, .. } => {
477                self.extract_column_references(left, columns);
478                self.extract_column_references(right, columns);
479            }
480            Expression::Unary { operand, .. } => {
481                self.extract_column_references(operand, columns);
482            }
483            Expression::Function { args, .. } => {
484                for arg in args {
485                    self.extract_column_references(arg, columns);
486                }
487            }
488            Expression::FieldAccess { object, .. } => {
489                self.extract_column_references(object, columns);
490            }
491            Expression::IndexAccess { object, index, .. } => {
492                self.extract_column_references(object, columns);
493                self.extract_column_references(index, columns);
494            }
495            _ => {} // Other expressions don't contain column references
496        }
497    }
498
499    fn optimize_table_projection(&self, table_name: &str, columns: &[String]) {
500        // In a real implementation, this would communicate with the storage layer
501        // to inform it about the required columns for this table
502        println!("Optimizing table {table_name} to only fetch columns: {columns:?}");
503    }
504}

506/// Join reordering optimization
507#[derive(Debug, Clone)]
508pub struct JoinReordering;
509
510impl OptimizationRule for JoinReordering {
511    fn name(&self) -> &'static str {
512        "JoinReordering"
513    }
514
515    fn apply(&self, stmt: &Statement) -> Result<Statement, OptimizationError> {
516        match stmt {
517            Statement::Select(select) => {
518                let optimized_select = self.reorder_joins(select.clone())?;
519                Ok(Statement::Select(optimized_select))
520            }
521            _ => Ok(stmt.clone()),
522        }
523    }
524}
525
526impl JoinReordering {
527    fn reorder_joins(
528        &self,
529        mut select: SelectStatement,
530    ) -> Result<SelectStatement, OptimizationError> {
531        if select.join_clauses.len() <= 1 {
532            // No reordering needed for 0 or 1 joins
533            return Ok(select);
534        }
535
536        // Estimate costs for different join orders
537        let join_permutations = self.generate_join_permutations(&select.join_clauses);
538        let mut best_cost = f64::INFINITY;
539        let mut best_order = select.join_clauses.clone();
540
541        for permutation in join_permutations {
542            let estimated_cost = self.estimate_join_sequence_cost(&select.from, &permutation);
543            if estimated_cost < best_cost {
544                best_cost = estimated_cost;
545                best_order = permutation;
546            }
547        }
548
549        select.join_clauses = best_order;
550        Ok(select)
551    }
552
553    fn generate_join_permutations(&self, joins: &[JoinClause]) -> Vec<Vec<JoinClause>> {
554        if joins.len() <= 1 {
555            return vec![joins.to_vec()];
556        }
557
558        // For simplicity, only generate a few heuristic orderings
559        // In a real implementation, this would use dynamic programming
560        let mut permutations = Vec::new();
561
562        // Original order
563        permutations.push(joins.to_vec());
564
565        // Reverse order
566        let mut reversed = joins.to_vec();
567        reversed.reverse();
568        permutations.push(reversed);
569
570        // Order by estimated selectivity (smaller tables first)
571        let mut by_selectivity = joins.to_vec();
572        by_selectivity.sort_by(|a, b| {
573            self.estimate_table_selectivity(&a.target)
574                .partial_cmp(&self.estimate_table_selectivity(&b.target))
575                .unwrap_or(std::cmp::Ordering::Equal)
576        });
577        permutations.push(by_selectivity);
578
579        permutations
580    }
581
582    fn estimate_join_sequence_cost(&self, from: &[FromClause], joins: &[JoinClause]) -> f64 {
583        let mut total_cost = 0.0;
584        let mut current_cardinality = self.estimate_from_cardinality(from);
585
586        for join in joins {
587            let right_cardinality = self.estimate_table_cardinality(&join.target);
588            let join_cost = current_cardinality * right_cardinality.log2();
589            total_cost += join_cost;
590
591            // Update cardinality after join (rough estimate)
592            current_cardinality *= right_cardinality * self.estimate_join_selectivity(join);
593        }
594
595        total_cost
596    }
597
598    fn estimate_from_cardinality(&self, from: &[FromClause]) -> f64 {
599        from.iter()
600            .map(|clause| self.estimate_table_cardinality(clause))
601            .product()
602    }
603
604    fn estimate_table_cardinality(&self, from_clause: &FromClause) -> f64 {
605        match from_clause {
606            FromClause::Table { name, .. } => {
607                // In a real implementation, this would use actual table statistics
608                self.get_table_row_count(name)
609            }
610            FromClause::Subquery { .. } => 1000.0, // Default estimate for subqueries
611            FromClause::Graph { .. } => 10000.0,   // Default estimate for graph traversals
612            FromClause::TimeSeries { .. } => 50000.0, // Default estimate for time series
613        }
614    }
615
616    fn estimate_table_selectivity(&self, from_clause: &FromClause) -> f64 {
617        match from_clause {
618            FromClause::Table { name, .. } => {
619                // Selectivity is inverse of cardinality (smaller tables are more selective)
620                1.0 / self.get_table_row_count(name)
621            }
622            _ => 0.1, // Default selectivity
623        }
624    }
625
626    fn estimate_join_selectivity(&self, _join: &JoinClause) -> f64 {
627        // Default join selectivity estimate
628        // In a real implementation, this would analyze the join condition
629        0.1
630    }
631
632    fn get_table_row_count(&self, table_name: &str) -> f64 {
633        // Mock table statistics - in a real implementation,
634        // this would query the statistics manager
635        match table_name {
636            "users" => 10000.0,
637            "orders" => 100000.0,
638            "products" => 5000.0,
639            "order_items" => 500000.0,
640            _ => 1000.0, // Default estimate
641        }
642    }

645/// Constant folding optimization
646#[derive(Debug, Clone)]
647pub struct ConstantFolding;
648
649impl OptimizationRule for ConstantFolding {
650    fn name(&self) -> &'static str {
651        "ConstantFolding"
652    }
653
654    fn apply(&self, stmt: &Statement) -> Result<Statement, OptimizationError> {
655        // TODO: Implement constant folding for expressions
656        match stmt {
657            Statement::Select(select) => {
658                let mut new_select = select.clone();
659                // Apply constant folding to expressions
660                if let Some(ref where_clause) = new_select.where_clause {
661                    new_select.where_clause = Some(self.fold_expression(where_clause.clone())?);
662                }
663                Ok(Statement::Select(new_select))
664            }
665            _ => Ok(stmt.clone()),
666        }
667    }
668}
669
670impl ConstantFolding {
671    fn fold_expression(&self, expr: Expression) -> Result<Expression, OptimizationError> {
672        match expr {
673            Expression::Binary {
674                left,
675                operator,
676                right,
677            } => {
678                let left = self.fold_expression(*left)?;
679                let right = self.fold_expression(*right)?;
680
681                // Try to evaluate constant expressions
682                if let (Expression::Literal(l_val), Expression::Literal(r_val)) = (&left, &right) {
683                    if let Some(result) = self.evaluate_binary_op(l_val, &operator, r_val) {
684                        return Ok(Expression::Literal(result));
685                    }
686                }
687
688                Ok(Expression::Binary {
689                    left: Box::new(left),
690                    operator,
691                    right: Box::new(right),
692                })
693            }
694            Expression::Unary { operator, operand } => {
695                let operand = self.fold_expression(*operand)?;
696
697                if let Expression::Literal(val) = &operand {
698                    if let Some(result) = self.evaluate_unary_op(&operator, val) {
699                        return Ok(Expression::Literal(result));
700                    }
701                }
702
703                Ok(Expression::Unary {
704                    operator,
705                    operand: Box::new(operand),
706                })
707            }
708            _ => Ok(expr),
709        }
710    }
711
712    fn evaluate_binary_op(
713        &self,
714        left: &crate::orbitql::QueryValue,
715        op: &BinaryOperator,
716        right: &crate::orbitql::QueryValue,
717    ) -> Option<crate::orbitql::QueryValue> {
718        use crate::orbitql::QueryValue;
719
720        match (left, op, right) {
721            (QueryValue::Integer(a), BinaryOperator::Add, QueryValue::Integer(b)) => {
722                Some(QueryValue::Integer(a + b))
723            }
724            (QueryValue::Integer(a), BinaryOperator::Subtract, QueryValue::Integer(b)) => {
725                Some(QueryValue::Integer(a - b))
726            }
727            (QueryValue::Integer(a), BinaryOperator::Multiply, QueryValue::Integer(b)) => {
728                Some(QueryValue::Integer(a * b))
729            }

733            (QueryValue::Float(a), BinaryOperator::Add, QueryValue::Float(b)) => {

739            (QueryValue::Float(a), BinaryOperator::Multiply, QueryValue::Float(b)) => {
740                Some(QueryValue::Float(a * b))
741            }
742            (QueryValue::Float(a), BinaryOperator::Divide, QueryValue::Float(b)) if *b != 0.0 => {
743                Some(QueryValue::Float(a / b))
744            }
745            (QueryValue::Boolean(a), BinaryOperator::And, QueryValue::Boolean(b)) => {
746                Some(QueryValue::Boolean(*a && *b))
747            }
748            (QueryValue::Boolean(a), BinaryOperator::Or, QueryValue::Boolean(b)) => {
749                Some(QueryValue::Boolean(*a || *b))
750            }
751            _ => None,
752        }
753    }

755    fn evaluate_unary_op(
756        &self,
757        op: &UnaryOperator,
758        operand: &crate::orbitql::QueryValue,
759    ) -> Option<crate::orbitql::QueryValue> {
760        use crate::orbitql::QueryValue;
761
762        match (op, operand) {
763            (UnaryOperator::Not, QueryValue::Boolean(b)) => Some(QueryValue::Boolean(!b)),
764            (UnaryOperator::Minus, QueryValue::Integer(n)) => Some(QueryValue::Integer(-n)),
765            (UnaryOperator::Minus, QueryValue::Float(f)) => Some(QueryValue::Float(-f)),
766            (UnaryOperator::Plus, QueryValue::Integer(n)) => Some(QueryValue::Integer(*n)),
767            (UnaryOperator::Plus, QueryValue::Float(f)) => Some(QueryValue::Float(*f)),
768            _ => None,
769        }
770    }

1015    /// Optimize a statement by applying all rules iteratively
1016    pub fn optimize(&self, stmt: Statement) -> Result<Statement, OptimizationError> {
1017        let mut current = stmt;
1018        let mut changed = true;
1019        let mut iterations = 0;
1020
1021        // Use advanced cost model to guide optimization if enabled
1022        let initial_cost = if self.cost_based_enabled {
1023            Some(self.advanced_cost_model.estimate_query_cost(&current))
1024        } else {
1025            None
1026        };
1027
1028        while changed && iterations < self.max_iterations {
1029            changed = false;
1030            iterations += 1;
1031
1032            for rule in &self.rules {
1033                let optimized = rule.apply(&current)?;
1034
1035                // Use advanced cost model to validate optimizations
1036                let should_apply = if let Some(ref initial) = initial_cost {
1037                    let new_cost = self.advanced_cost_model.estimate_query_cost(&optimized);
1038                    new_cost.total_cost() <= initial.total_cost()
1039                } else {
1040                    !statements_equal(&current, &optimized)
1041                };
1042
1043                if should_apply && !statements_equal(&current, &optimized) {
1044                    current = optimized;
1045                    changed = true;
1046                }
1047            }
1048        }
1049
1050        Ok(current)
1051    }

1241        // Generate multiple optimization alternatives

1243
1244        // Cost each alternative
1245        let mut best_stmt = alternatives[0].clone();
1246        let mut best_cost = self.estimate_query_cost(&best_stmt);
1247
1248        for alternative in alternatives.iter().skip(1) {
1249            let alt_cost = self.estimate_query_cost(alternative);
1250            if alt_cost.total_cost() < best_cost.total_cost() {
1251                best_stmt = alternative.clone();
1252                best_cost = alt_cost;
1253            }
1254        }
1255
1256        Ok((best_stmt, best_cost))
1257    }
