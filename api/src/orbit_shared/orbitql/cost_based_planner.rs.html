105    /// Generate optimal execution plan for a statement
106    pub async fn generate_plan(&self, stmt: Statement) -> Result<ExecutionPlan, PlanningError> {
107        // Check cache first if enabled
108        if self.config.enable_plan_cache {
109            let cache_key = self.generate_cache_key(&stmt);
110            if let Some(cached_plan) = self.get_cached_plan(&cache_key).await {
111                return Ok(cached_plan.plan);
112            }
113        }
114
115        // Apply rule-based optimizations first
116        let optimized_stmt = self
117            .optimizer
118            .optimize(stmt.clone())
119            .map_err(PlanningError::OptimizationError)?;
120
121        // Generate alternative plans
122        let alternatives = self.generate_alternatives(optimized_stmt).await?;
123
124        // Select the best plan based on cost
125        let best_plan = self.select_best_plan(alternatives).await?;
126
127        // Cache the plan if enabled
128        if self.config.enable_plan_cache {
129            let cache_key = self.generate_cache_key(&stmt);
130            self.cache_plan(&cache_key, &best_plan).await;
131        }
132
133        Ok(best_plan.plan)
134    }

136    /// Generate multiple alternative execution plans
137    pub async fn generate_alternatives(
138        &self,
139        stmt: Statement,
140    ) -> Result<Vec<PlanAlternative>, PlanningError> {
141        let mut alternatives = Vec::new();
142
143        match stmt {
144            Statement::Select(select) => {
145                // Generate different scan alternatives
146                alternatives.extend(self.generate_scan_alternatives(&select).await?);
147
148                // Generate different join alternatives if applicable
149                if !select.join_clauses.is_empty() && self.config.enable_join_enumeration {
150                    alternatives.extend(self.generate_join_alternatives(&select).await?);
151                }
152
153                // Generate different aggregation alternatives
154                if !select.group_by.is_empty() {
155                    alternatives.extend(self.generate_aggregation_alternatives(&select).await?);
156                }
157
158                // Generate different sorting alternatives
159                if !select.order_by.is_empty() {
160                    alternatives.extend(self.generate_sort_alternatives(&select).await?);
161                }
162            }
163            _ => {
164                // For non-SELECT statements, generate a single basic plan
165                let basic_plan = self.generate_basic_plan(stmt).await?;
166                alternatives.push(basic_plan);
167            }
168        }
169
170        // Limit number of alternatives
171        alternatives.truncate(self.config.max_alternatives);
172
173        Ok(alternatives)
174    }

176    /// Generate different scan alternatives (sequential, index, etc.)
177    async fn generate_scan_alternatives(
178        &self,
179        select: &SelectStatement,
180    ) -> Result<Vec<PlanAlternative>, PlanningError> {
181        let mut alternatives = Vec::new();
182
183        for from_clause in &select.from {
184            if let FromClause::Table { name, .. } = from_clause {
185                // Sequential scan alternative
186                let seq_scan_plan = self.generate_sequential_scan_plan(name, select).await?;
187                alternatives.push(seq_scan_plan);
188
189                // Index scan alternatives
190                let index_alternatives = self.generate_index_scan_plans(name, select).await?;
191                alternatives.extend(index_alternatives);
192            }
193        }
194
195        Ok(alternatives)
196    }

198    /// Generate sequential scan plan
199    async fn generate_sequential_scan_plan(
200        &self,
201        table_name: &str,
202        select: &SelectStatement,
203    ) -> Result<PlanAlternative, PlanningError> {
204        let stats_manager = self.statistics_manager.read().await;
205        let table_stats = stats_manager
206            .get_table_statistics(table_name)
207            .await
208            .unwrap_or_else(|| self.create_default_table_stats(table_name));
209
210        // Estimate selectivity from WHERE clause
211        let selectivity = if let Some(where_clause) = &select.where_clause {
212            self.estimate_where_selectivity(table_name, where_clause, &stats_manager)
213                .await
214        } else {
215            1.0
216        };
217
218        let (cost, cardinality) = self
219            .cost_model
220            .calculate_scan_cost(&table_stats, selectivity);
221
222        let plan = ExecutionPlan {
223            root: PlanNode::TableScan {
224                table: table_name.to_string(),
225                columns: vec![], // TODO: Extract required columns
226                filter: select.where_clause.clone(),
227            },
228            estimated_cost: cost.total_cost(),
229            estimated_rows: cardinality.rows,
230        };
231
232        Ok(PlanAlternative {
233            plan,
234            cost,
235            description: format!("Sequential scan on {table_name}"),
236        })
237    }

468    /// Select the best plan from alternatives based on cost
469    async fn select_best_plan(
470        &self,
471        alternatives: Vec<PlanAlternative>,
472    ) -> Result<PlanAlternative, PlanningError> {
473        if alternatives.is_empty() {
474            return Err(PlanningError::InvalidPlan(
475                "No alternatives generated".to_string(),
476            ));
477        }
478
479        let best_plan = alternatives
480            .into_iter()
481            .min_by(|a, b| {
482                a.cost
483                    .total_cost()
484                    .partial_cmp(&b.cost.total_cost())
485                    .unwrap_or(std::cmp::Ordering::Equal)
486            })
487            .unwrap();
488
489        Ok(best_plan)
490    }

519    async fn cache_plan(&self, cache_key: &str, plan: &PlanAlternative) {
520        let mut cache = self.plan_cache.write().await;
521
522        // Remove expired entries if cache is full
523        if cache.len() >= self.config.max_cache_size {
524            let now = std::time::Instant::now();
525            let timeout = std::time::Duration::from_secs(self.config.cache_timeout_seconds);
526            cache.retain(|_, cached| now.duration_since(cached.timestamp) < timeout);
527        }
528
529        cache.insert(
530            cache_key.to_string(),
531            CachedPlan {
532                plan: plan.plan.clone(),
533                cost: plan.cost.clone(),
534                timestamp: std::time::Instant::now(),
535                hit_count: 0,
536            },
537        );
538    }

564    async fn estimate_where_selectivity(
565        &self,
566        table_name: &str,
567        where_clause: &Expression,
568        stats_manager: &StatisticsManager,
569    ) -> f64 {
570        if let Expression::Binary {
571            left,
572            operator,
573            right,
574        } = where_clause
575        {
576            if let (Expression::Identifier(column), Expression::Literal(value)) =
577                (left.as_ref(), right.as_ref())
578            {
579                let op_str = match operator {
580                    BinaryOperator::Equal => "=",
581                    BinaryOperator::GreaterThan => ">",
582                    BinaryOperator::LessThan => "<",
583                    _ => "=",
584                };
585                return stats_manager
586                    .estimate_selectivity(table_name, column, op_str, value)
587                    .await;
588            }
589        }
590        0.1 // Default selectivity
591    }
