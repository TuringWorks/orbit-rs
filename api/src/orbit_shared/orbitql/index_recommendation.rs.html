34/// Configuration for index recommendation system
35#[derive(Debug, Clone, Serialize, Deserialize)]
36pub struct RecommendationConfig {
37    /// Minimum query frequency to consider for indexing
38    pub min_query_frequency: u32,
39    /// Minimum performance improvement threshold (as percentage)
40    pub min_improvement_threshold: f64,
41    /// Maximum number of recommendations to generate
42    pub max_recommendations: usize,
43    /// Analysis window in days
44    pub analysis_window_days: u32,
45    /// Minimum table size to consider for indexing (rows)
46    pub min_table_size: u64,
47    /// Maximum index overhead tolerance (as percentage of table size)
48    pub max_index_overhead: f64,
49}

313    /// Generate index recommendations based on query patterns
314    pub async fn generate_recommendations(
315        &self,
316    ) -> Result<Vec<IndexRecommendation>, Box<dyn std::error::Error>> {
317        let mut recommendations = Vec::new();
318
319        // Analyze query patterns
320        let query_patterns = self
321            .query_analyzer
322            .get_significant_patterns(&self.config)
323            .await;
324
325        for pattern in query_patterns {
326            // Generate recommendations for this pattern
327            let pattern_recommendations = self.recommend_indexes_for_pattern(&pattern).await?;
328            recommendations.extend(pattern_recommendations);
329        }
330
331        // Deduplicate and prioritize recommendations
332        let optimized_recommendations = self.optimize_recommendations(recommendations).await;
333
334        // Limit to max recommendations
335        Ok(optimized_recommendations
336            .into_iter()
337            .take(self.config.max_recommendations)
338            .collect())
339    }

341    /// Analyze existing indexes for redundancy and unused indexes
342    pub async fn analyze_index_redundancy(
343        &self,
344    ) -> Result<RedundantIndexAnalysis, Box<dyn std::error::Error>> {
345        let mut redundant_groups = Vec::new();
346        let mut unused_indexes = Vec::new();
347        let mut reclaimable_storage = 0u64;
348
349        // Get all tables with indexes
350        let tables = self.get_all_indexed_tables().await;
351
352        for table in tables {
353            let table_indexes = self.index_selector.get_table_indexes(&table).await;
354
355            // Find redundant indexes
356            let table_redundant = self.find_redundant_indexes(&table_indexes).await;
357            redundant_groups.extend(table_redundant);
358
359            // Find unused indexes
360            let table_unused = self.find_unused_indexes(&table_indexes).await;
361            unused_indexes.extend(table_unused);
362        }
363
364        // Calculate reclaimable storage
365        for group in &redundant_groups {
366            for index in &group.redundant_indexes {
367                reclaimable_storage += index.size_bytes;
368            }
369        }
370
371        for unused in &unused_indexes {
372            reclaimable_storage += unused.index.size_bytes;
373        }
374
375        Ok(RedundantIndexAnalysis {
376            redundant_indexes: redundant_groups,
377            unused_indexes,
378            reclaimable_storage_bytes: reclaimable_storage,
379        })
380    }

398    async fn recommend_indexes_for_pattern(
399        &self,
400        pattern: &QueryPattern,
401    ) -> Result<Vec<IndexRecommendation>, Box<dyn std::error::Error>> {
402        let mut recommendations = Vec::new();
403
404        // Check if pattern meets minimum criteria
405        if pattern.frequency < self.config.min_query_frequency {
406            return Ok(recommendations);
407        }
408
409        // Analyze WHERE clause columns for single-column indexes
410        for col_ref in &pattern.where_columns {
411            if let Some(recommendation) =
412                self.recommend_single_column_index(pattern, col_ref).await?
413            {
414                recommendations.push(recommendation);
415            }
416        }
417
418        // Analyze multi-column indexes for WHERE clauses
419        if pattern.where_columns.len() > 1 {
420            if let Some(recommendation) = self.recommend_multi_column_index(pattern).await? {
421                recommendations.push(recommendation);
422            }
423        }
424
425        // Analyze covering indexes for frequent SELECT patterns
426        if let Some(recommendation) = self.recommend_covering_index(pattern).await? {
427            recommendations.push(recommendation);
428        }
429
430        // Analyze ORDER BY indexes
431        for col_ref in &pattern.order_columns {
432            if let Some(recommendation) = self.recommend_order_index(pattern, col_ref).await? {
433                recommendations.push(recommendation);
434            }
435        }
436
437        Ok(recommendations)
438    }

440    async fn recommend_single_column_index(
441        &self,
442        pattern: &QueryPattern,
443        col_ref: &ColumnReference,
444    ) -> Result<Option<IndexRecommendation>, Box<dyn std::error::Error>> {
445        // Check if table is large enough
446        let stats_manager = self.statistics_manager.read().await;
447        if let Some(table_stats) = stats_manager.get_table_statistics(&col_ref.table).await {
448            if table_stats.row_count < self.config.min_table_size {
449                return Ok(None);
450            }
451
452            // Estimate improvement
453            let current_cost = self
454                .estimate_current_query_cost(pattern, &table_stats)
455                .await;
456            let index_cost = self
457                .estimate_index_query_cost(
458                    pattern,
459                    &col_ref.table,
460                    std::slice::from_ref(&col_ref.column),
461                )
462                .await;
463
464            let speedup = current_cost.total_cost() / index_cost.total_cost();
465
466            if speedup < (1.0 + self.config.min_improvement_threshold / 100.0) {
467                return Ok(None);
468            }
469
470            // Create recommendation
471            let index_spec = RecommendedIndex {
472                index_name: format!("idx_{}_{}", col_ref.table, col_ref.column),
473                index_type: self.determine_optimal_index_type(&col_ref.operator),
474                columns: vec![col_ref.column.clone()],
475                included_columns: vec![],
476                filter_condition: None,
477                estimated_size_bytes: self
478                    .estimate_index_size(&table_stats, std::slice::from_ref(&col_ref.column)),
479            };
480
481            let improvement = PerformanceImprovement {
482                speedup_factor: speedup,
483                time_savings_ms: pattern.avg_execution_time_ms * (speedup - 1.0) / speedup,
484                io_reduction_percentage: 80.0, // Estimate
485                cpu_savings_percentage: 50.0,  // Estimate
486                affected_query_count: pattern.frequency,
487            };
488
489            let impact = self.analyze_index_impact(&table_stats, &index_spec).await;
490
491            let recommendation = IndexRecommendation {
492                id: format!(
493                    "rec_{}_{}_{}",
494                    Utc::now().timestamp(),
495                    col_ref.table,
496                    col_ref.column
497                ),
498                table_name: col_ref.table.clone(),
499                index_spec,
500                expected_improvement: improvement,
501                supporting_queries: vec![pattern.pattern_hash.clone()],
502                impact_analysis: impact,
503                priority_score: self.calculate_priority_score(speedup, pattern.frequency),
504                confidence: 0.8, // High confidence for single-column indexes
505                created_at: Utc::now(),
506            };
507
508            Ok(Some(recommendation))
509        } else {
510            Ok(None)
511        }
512    }

514    async fn recommend_multi_column_index(
515        &self,
516        pattern: &QueryPattern,
517    ) -> Result<Option<IndexRecommendation>, Box<dyn std::error::Error>> {
518        if pattern.where_columns.len() < 2 {
519            return Ok(None);
520        }
521
522        // Group columns by table
523        let mut table_columns: HashMap<String, Vec<String>> = HashMap::new();
524        for col_ref in &pattern.where_columns {
525            table_columns
526                .entry(col_ref.table.clone())
527                .or_default()
528                .push(col_ref.column.clone());
529        }
530
531        // Create multi-column index recommendation for the table with most columns
532        if let Some((table, columns)) = table_columns.iter().max_by_key(|(_, cols)| cols.len()) {
533            if columns.len() >= 2 {
534                let stats_manager = self.statistics_manager.read().await;
535                if let Some(table_stats) = stats_manager.get_table_statistics(table).await {
536                    let index_spec = RecommendedIndex {
537                        index_name: format!("idx_{table}_composite"),
538                        index_type: IndexType::BTree, // Multi-column indexes are typically B-Tree
539                        columns: columns.clone(),
540                        included_columns: vec![],
541                        filter_condition: None,
542                        estimated_size_bytes: self.estimate_index_size(&table_stats, columns),
543                    };
544
545                    let improvement = PerformanceImprovement {
546                        speedup_factor: 5.0, // Composite indexes can provide significant improvement
547                        time_savings_ms: pattern.avg_execution_time_ms * 0.8,
548                        io_reduction_percentage: 90.0,
549                        cpu_savings_percentage: 70.0,
550                        affected_query_count: pattern.frequency,
551                    };
552
553                    let impact = self.analyze_index_impact(&table_stats, &index_spec).await;
554
555                    let recommendation = IndexRecommendation {
556                        id: format!("rec_composite_{}_{}", Utc::now().timestamp(), table),
557                        table_name: table.clone(),
558                        index_spec,
559                        expected_improvement: improvement,
560                        supporting_queries: vec![pattern.pattern_hash.clone()],
561                        impact_analysis: impact,
562                        priority_score: self.calculate_priority_score(5.0, pattern.frequency),
563                        confidence: 0.9, // High confidence for composite indexes
564                        created_at: Utc::now(),
565                    };
566
567                    return Ok(Some(recommendation));
568                }
569            }
570        }
571
572        Ok(None)
573    }

585        &self,
586        pattern: &QueryPattern,
587        col_ref: &ColumnReference,

590        let stats_manager = self.statistics_manager.read().await;
591        if let Some(table_stats) = stats_manager.get_table_statistics(&col_ref.table).await {
592            if table_stats.row_count >= self.config.min_table_size {
593                let index_spec = RecommendedIndex {
594                    index_name: format!("idx_{}_order_{}", col_ref.table, col_ref.column),
595                    index_type: IndexType::BTree, // B-Tree is optimal for ORDER BY
596                    columns: vec![col_ref.column.clone()],
597                    included_columns: vec![],
598                    filter_condition: None,
599                    estimated_size_bytes: self
600                        .estimate_index_size(&table_stats, std::slice::from_ref(&col_ref.column)),
601                };
602
603                let improvement = PerformanceImprovement {
604                    speedup_factor: 3.0, // ORDER BY can benefit significantly from indexes
605                    time_savings_ms: pattern.avg_execution_time_ms * 0.67,
606                    io_reduction_percentage: 60.0,
607                    cpu_savings_percentage: 80.0, // Eliminates sorting
608                    affected_query_count: pattern.frequency,
609                };
610
611                let impact = self.analyze_index_impact(&table_stats, &index_spec).await;
612
613                let recommendation = IndexRecommendation {
614                    id: format!("rec_order_{}_{}", Utc::now().timestamp(), col_ref.column),
615                    table_name: col_ref.table.clone(),
616                    index_spec,
617                    expected_improvement: improvement,
618                    supporting_queries: vec![pattern.pattern_hash.clone()],

620                    priority_score: self.calculate_priority_score(3.0, pattern.frequency),
621                    confidence: 0.7,
622                    created_at: Utc::now(),
623                };

625                return Ok(Some(recommendation));

632    async fn optimize_recommendations(
633        &self,
634        mut recommendations: Vec<IndexRecommendation>,
635    ) -> Vec<IndexRecommendation> {
636        // Remove duplicates and merge similar recommendations
637        recommendations.sort_by(|a, b| {
638            b.priority_score
639                .partial_cmp(&a.priority_score)
640                .unwrap_or(std::cmp::Ordering::Equal)
641        });
642
643        // Simple deduplication by table + columns
644        let mut seen = HashSet::new();
645        recommendations.retain(|rec| {
646            let key = format!("{}_{}", rec.table_name, rec.index_spec.columns.join("_"));
647            seen.insert(key)
648        });
649
650        recommendations
651    }

653    async fn find_redundant_indexes(&self, indexes: &[IndexMetadata]) -> Vec<RedundantIndexGroup> {
654        let mut redundant_groups = Vec::new();
655
656        // Simple redundancy detection: if one index is a prefix of another
657        for (i, index1) in indexes.iter().enumerate() {
658            for (j, index2) in indexes.iter().enumerate() {
659                if i != j && self.is_redundant(index1, index2) {
660                    // Create redundant group
661                    redundant_groups.push(RedundantIndexGroup {
662                        primary_index: index2.clone(), // Keep the more comprehensive index
663                        redundant_indexes: vec![index1.clone()],
664                        reason: "Index is a prefix of another index".to_string(),
665                        recommended_action: "Drop the redundant index".to_string(),
666                    });
667                }
668            }
669        }
670
671        redundant_groups
672    }

674    async fn find_unused_indexes(&self, indexes: &[IndexMetadata]) -> Vec<UnusedIndex> {
675        let mut unused_indexes = Vec::new();
676
677        for index in indexes {
678            if let Some(usage_stats) = self
679                .index_selector
680                .get_index_usage_stats(&index.index_id)
681                .await
682            {
683                // Consider unused if not used in the last 30 days or very low usage
684                let is_unused = usage_stats
685                    .last_used
686                    .is_none_or(|last_used| Utc::now() - last_used > Duration::days(30))
687                    || usage_stats.usage_count < 5; // Very low usage
688
689                if is_unused {
690                    unused_indexes.push(UnusedIndex {
691                        index: index.clone(),
692                        last_used: usage_stats.last_used,
693                        usage_stats,
694                        recommendation: "Consider dropping this unused index".to_string(),
695                    });
696                }
697            } else {

721    fn is_redundant(&self, index1: &IndexMetadata, index2: &IndexMetadata) -> bool {
722        // Check if index1 is a prefix of index2
723        if index1.columns.len() >= index2.columns.len() {
724            return false;
725        }
726
727        index1
728            .columns
729            .iter()
730            .zip(index2.columns.iter())
731            .all(|(col1, col2)| col1 == col2)
732    }

734    fn determine_optimal_index_type(&self, operator: &str) -> IndexType {
735        match operator {
736            "=" => IndexType::Hash,                      // Hash is optimal for equality
737            ">" | "<" | ">=" | "<=" => IndexType::BTree, // B-Tree for range queries
738            "LIKE" => IndexType::BTree,                  // B-Tree for pattern matching
739            _ => IndexType::BTree,                       // Default to B-Tree
740        }

744        // Rough estimate: 64 bytes per row per column + 20% overhead
745        let row_size_estimate = columns.len() as u64 * 64;
746        let total_size = table_stats.row_count * row_size_estimate;
747        (total_size as f64 * 1.2) as u64 // 20% overhead

750    async fn analyze_index_impact(
751        &self,
752        table_stats: &TableStatistics,
753        index_spec: &RecommendedIndex,
754    ) -> IndexImpactAnalysis {
755        let storage_overhead = index_spec.estimated_size_bytes;
756        let table_size = table_stats.row_count * table_stats.avg_row_size as u64;
757        let overhead_percentage = (storage_overhead as f64 / table_size as f64) * 100.0;
758
759        let risk_level = if overhead_percentage > 50.0 {
760            RiskLevel::High
761        } else if overhead_percentage > 20.0 {
762            RiskLevel::Medium
763        } else {
764            RiskLevel::Low
765        };
766
767        IndexImpactAnalysis {
768            storage_overhead_bytes: storage_overhead,
769            maintenance_cost_factor: 1.05 + (index_spec.columns.len() as f64 * 0.02), // More columns = higher maintenance
770            write_impact_percentage: index_spec.columns.len() as f64 * 2.0, // Rough estimate
771            memory_usage_increase_mb: (storage_overhead as f64 / (1024.0 * 1024.0)) * 0.1, // Assume 10% of index size in memory
772            risk_level,
773        }
774    }

843    pub async fn record_query(
844        &self,
845        query: &Statement,
846        execution_time_ms: f64,
847        rows_examined: u64,
848        rows_returned: u64,
849        used_indexes: &[String],
850    ) -> Result<(), Box<dyn std::error::Error>> {
851        // Extract pattern from query
852        let pattern_hash = self.generate_pattern_hash(query);
853        let normalized_query = self.normalize_query(query);
854
855        let mut patterns = self.query_patterns.write().await;
856        let pattern = patterns
857            .entry(pattern_hash.clone())
858            .or_insert_with(|| QueryPattern {
859                pattern_hash: pattern_hash.clone(),
860                query_template: normalized_query,
861                tables: self.extract_tables(query),
862                where_columns: self.extract_where_columns(query),
863                order_columns: self.extract_order_columns(query),
864                group_columns: self.extract_group_columns(query),
865                joins: self.extract_joins(query),
866                frequency: 0,
867                avg_execution_time_ms: 0.0,
868                first_seen: Utc::now(),
869                last_seen: Utc::now(),
870                performance_stats: QueryPerformanceStats::default(),
871            });
872
873        // Update pattern statistics
874        pattern.frequency += 1;
875        pattern.avg_execution_time_ms =
876            (pattern.avg_execution_time_ms * (pattern.frequency - 1) as f64 + execution_time_ms)
877                / pattern.frequency as f64;
878        pattern.last_seen = Utc::now();
879
880        // Update performance stats
881        let stats = &mut pattern.performance_stats;
882        if execution_time_ms < stats.min_execution_time_ms || stats.min_execution_time_ms == 0.0 {
883            stats.min_execution_time_ms = execution_time_ms;
884        }
885        if execution_time_ms > stats.max_execution_time_ms {
886            stats.max_execution_time_ms = execution_time_ms;
887        }
888        stats.rows_examined_avg = (stats.rows_examined_avg * (pattern.frequency - 1) as u64
889            + rows_examined)
890            / pattern.frequency as u64;
891        stats.rows_returned_avg = (stats.rows_returned_avg * (pattern.frequency - 1) as u64
892            + rows_returned)
893            / pattern.frequency as u64;
894
895        // Update index usage stats
896        if !used_indexes.is_empty() {
897            stats.index_usage_count += 1;
898        } else {
899            stats.full_table_scan_count += 1;
900        }
901
902        // Update column access patterns
903        self.update_column_access_patterns(query).await;
904
905        // Update join patterns
906        self.update_join_patterns(query).await;
907
908        Ok(())
909    }

945    fn extract_tables(&self, query: &Statement) -> Vec<String> {
946        match query {
947            Statement::Select(select) => select
948                .from
949                .iter()
950                .filter_map(|from| match from {
951                    FromClause::Table { name, .. } => Some(name.clone()),
952                    _ => None,
953                })
954                .collect(),
955            _ => vec![],
956        }

960        match query {
961            Statement::Select(select) => {
962                if let Some(where_clause) = &select.where_clause {
963                    self.extract_column_references_from_expression(where_clause, "WHERE")
964                } else {
965                    vec![]
966                }
967            }
968            _ => vec![],
969        }

973        match query {
974            Statement::Select(select) => {
975                select
976                    .order_by
977                    .iter()
978                    .filter_map(|order| {
979                        if let Expression::Identifier(col) = &order.expression {
980                            Some(ColumnReference {
981                                table: "unknown".to_string(), // Would need table resolution
982                                column: col.clone(),
983                                operator: "ORDER".to_string(),
984                                selectivity_estimate: 1.0,
985                            })
986                        } else {
987                            None
988                        }
989                    })
990                    .collect()
991            }
992            _ => vec![],
993        }

996    fn extract_group_columns(&self, query: &Statement) -> Vec<ColumnReference> {
997        match query {
998            Statement::Select(select) => {
999                select
1000                    .group_by
1001                    .iter()
1002                    .filter_map(|expr| {
1003                        if let Expression::Identifier(col) = expr {
1004                            Some(ColumnReference {
1005                                table: "unknown".to_string(),
1006                                column: col.clone(),
1007                                operator: "GROUP".to_string(),
1008                                selectivity_estimate: 0.1, // Groups typically reduce cardinality
1009                            })
1010                        } else {
1011                            None
1012                        }
1013                    })
1014                    .collect()
1015            }
1016            _ => vec![],
1017        }

1020    fn extract_joins(&self, query: &Statement) -> Vec<JoinCondition> {
1021        match query {
1022            Statement::Select(select) => {
1023                select
1024                    .join_clauses
1025                    .iter()
1026                    .filter_map(|join| {
1027                        if let Expression::Binary {
1028                            left,
1029                            operator: BinaryOperator::Equal,
1030                            right,
1031                        } = &join.condition
1032                        {
1033                            if let (
1034                                Expression::Identifier(left_col),
1035                                Expression::Identifier(right_col),
1036                            ) = (left.as_ref(), right.as_ref())
1037                            {
1038                                return Some(JoinCondition {
1039                                    left_table: "unknown".to_string(), // Would need resolution
1040                                    left_column: left_col.clone(),
1041                                    right_table: "unknown".to_string(),
1042                                    right_column: right_col.clone(),
1043                                    join_type: format!("{:?}", join.join_type),
1044                                });
1045                            }
1046                        }
1047                        None
1048                    })
1049                    .collect()
1050            }
1051            _ => vec![],
1052        }
1053    }

1055    fn extract_column_references_from_expression(
1056        &self,
1057        expr: &Expression,
1058        _context: &str,
1059    ) -> Vec<ColumnReference> {
1060        match expr {
1061            Expression::Binary {
1062                left,
1063                operator,
1064                right,
1065            } => {
1066                let mut refs = Vec::new();
1067                if let (Expression::Identifier(col), Expression::Literal(_)) =
1068                    (left.as_ref(), right.as_ref())
1069                {
1070                    refs.push(ColumnReference {
1071                        table: "unknown".to_string(),
1072                        column: col.clone(),
1073                        operator: format!("{operator:?}"),
1074                        selectivity_estimate: self.estimate_operator_selectivity(operator),
1075                    });
1076                }
1077                refs
1078            }
1079            _ => vec![],
1080        }
1081    }

1083    fn estimate_operator_selectivity(&self, operator: &BinaryOperator) -> f64 {
1084        match operator {
1085            BinaryOperator::Equal => 0.01,
1086            BinaryOperator::GreaterThan | BinaryOperator::LessThan => 0.33,
1087            BinaryOperator::GreaterThanOrEqual | BinaryOperator::LessThanOrEqual => 0.33,
1088            BinaryOperator::Like => 0.1,
1089            _ => 0.1,
1090        }
1091    }

1114            });
1115
1116            pattern.access_count += 1;
1117            pattern.filter_usage_count += 1;
1118            pattern
1119                .selectivity_history
1120                .push(col_ref.selectivity_estimate);
1121            pattern.last_updated = Utc::now();
1122        }
1123
1124        // Track ORDER BY column usage
1125        for col_ref in order_cols {
1126            let key = format!("{}.{}", col_ref.table, col_ref.column);

1128                table: col_ref.table.clone(),

1131                filter_usage_count: 0,

1165    /// Update join patterns based on query analysis
1166    async fn update_join_patterns(&self, query: &Statement) {
1167        let mut patterns = self.join_patterns.write().await;
1168
1169        let joins = self.extract_joins(query);
1170
1171        for join in joins {
1172            let key = format!(
1173                "{}.{}:{}.{}",
1174                join.left_table, join.left_column, join.right_table, join.right_column
1175            );
1176
1177            let pattern = patterns.entry(key).or_insert_with(|| JoinPattern {
1178                left_table: join.left_table.clone(),
1179                right_table: join.right_table.clone(),
1180                join_columns: vec![(join.left_column.clone(), join.right_column.clone())],
1181                frequency: 0,
1182                avg_selectivity: 0.5, // Default estimate
1183                last_seen: Utc::now(),
1184            });
1185
1186            pattern.frequency += 1;
1187            pattern.last_seen = Utc::now();
1188
1189            // Update join columns if not already present
1190            let join_col_pair = (join.left_column, join.right_column);
1191            if !pattern.join_columns.contains(&join_col_pair) {
1192                pattern.join_columns.push(join_col_pair);
1193            }
1194        }
1195    }
