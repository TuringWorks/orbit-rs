83    /// Check if current cluster configuration might cause split-brain
84    pub async fn check_split_brain_risk(
85        &self,
86        cluster_nodes: &[NodeId],
87        unreachable_nodes: &[NodeId],
88    ) -> bool {
89        let reachable_count = cluster_nodes.len() - unreachable_nodes.len();
90        let total_nodes = cluster_nodes.len();
91
92        // Classic split-brain condition: can't reach majority
93        let majority_threshold = total_nodes / 2 + 1;
94
95        if reachable_count < majority_threshold {
96            warn!(
97                "Split-brain risk detected: {} reachable out of {} total (need {})",
98                reachable_count, total_nodes, majority_threshold
99            );
100            return true;
101        }
102
103        // Additional check: ensure minimum cluster size
104        if reachable_count < self.min_cluster_size {
105            warn!(
106                "Cluster too small: {} reachable (minimum {})",
107                reachable_count, self.min_cluster_size
108            );
109            return true;
110        }
111
112        false
113    }

115    /// Detect network partitions
116    pub async fn detect_partitions(
117        &self,
118        node_health: &HashMap<NodeId, NodeHealthStatus>,
119    ) -> HashMap<String, Vec<NodeId>> {
120        let mut partitions = HashMap::new();
121
122        // Group nodes by connectivity patterns
123        let mut reachable_nodes = Vec::new();
124        let mut unreachable_nodes = Vec::new();
125
126        for (node_id, health) in node_health.iter() {
127            if health.is_reachable {
128                reachable_nodes.push(node_id.clone());
129            } else {
130                unreachable_nodes.push(node_id.clone());
131            }
132        }
133
134        if !reachable_nodes.is_empty() {
135            partitions.insert("reachable".to_string(), reachable_nodes);
136        }
137
138        if !unreachable_nodes.is_empty() {
139            partitions.insert("unreachable".to_string(), unreachable_nodes);
140        }
141
142        // Store detected partitions
143        {
144            let mut stored_partitions = self.partitions.write().await;
145            *stored_partitions = partitions.clone();
146        }
147
148        partitions
149    }

304    /// Update health status of all cluster nodes
305    async fn update_node_health(&self) -> OrbitResult<()> {
306        let cluster_nodes = self.get_cluster_nodes().await?;
307        let connectivity = self
308            .partition_detector
309            .check_cluster_connectivity(&cluster_nodes, &self.node_id)
310            .await;
311
312        let mut health_map = self.node_health.write().await;
313        let now = Instant::now();
314
315        for node in &cluster_nodes {
316            if node != &self.node_id {
317                let is_reachable = connectivity.get(node).copied().unwrap_or(false);
318
319                let health = health_map
320                    .entry(node.clone())
321                    .or_insert_with(|| NodeHealthStatus {
322                        node_id: node.clone(),
323                        last_seen: now,
324                        consecutive_failures: 0,
325                        network_latency: None,
326                        is_reachable: true,
327                        partition_group: None,
328                    });
329
330                if is_reachable {
331                    health.last_seen = now;
332                    health.consecutive_failures = 0;
333                    health.is_reachable = true;
334                } else {
335                    health.consecutive_failures += 1;
336                    health.is_reachable = false;
337                }
338            }
339        }
340
341        Ok(())
342    }

344    /// Check overall cluster health and split-brain risk
345    async fn check_cluster_health(&self) -> OrbitResult<()> {
346        let cluster_nodes = self.get_cluster_nodes().await?;
347        let health_map = self.node_health.read().await;
348
349        let unreachable_nodes: Vec<NodeId> = health_map
350            .values()
351            .filter(|health| !health.is_reachable)
352            .map(|health| health.node_id.clone())
353            .collect();
354
355        let split_brain_risk = self
356            .split_brain_detector
357            .check_split_brain_risk(&cluster_nodes, &unreachable_nodes)
358            .await;
359
360        if split_brain_risk {
361            error!(
362                "Split-brain risk detected! Unreachable nodes: {:?}",
363                unreachable_nodes
364            );
365
366            // In a production system, you might want to:
367            // 1. Stop accepting new coordinator roles
368            // 2. Increase election timeouts
369            // 3. Alert monitoring systems
370            // 4. Attempt network partition healing
371        }
372
373        // Detect partitions
374        let partitions = self
375            .split_brain_detector
376            .detect_partitions(&health_map)
377            .await;
378
379        if partitions.len() > 1 {
380            warn!("Network partitions detected: {:?}", partitions);
381        }
382
383        Ok(())
384    }

386    /// Check if current node has quorum
387    pub async fn has_quorum(&self) -> OrbitResult<bool> {
388        let cluster_nodes = self.get_cluster_nodes().await?;
389        let health_map = self.node_health.read().await;
390
391        let reachable_count = health_map
392            .values()
393            .filter(|health| health.is_reachable)
394            .count()
395            + 1; // +1 for self
396
397        let total_nodes = cluster_nodes.len();
398        let majority = total_nodes / 2 + 1;
399
400        Ok(reachable_count >= majority && reachable_count >= self.quorum_config.min_quorum_size)
401    }

432#[async_trait]
433impl ClusterManager for EnhancedClusterManager {
434    async fn get_cluster_nodes(&self) -> OrbitResult<Vec<NodeId>> {
435        // Get nodes from Raft consensus
436        let cluster_nodes = self.raft_consensus.get_cluster_nodes().await;
437        Ok(cluster_nodes.clone())
438    }
439
440    async fn is_leader(&self, node_id: &NodeId) -> OrbitResult<bool> {
441        if node_id == &self.node_id {
442            Ok(self.raft_consensus.is_leader().await)
443        } else {
444            // Check if the given node is the current leader according to Raft
445            let current_leader = self.raft_consensus.get_leader().await;
446            Ok(current_leader.as_ref() == Some(node_id))
447        }
448    }
449
450    async fn start_election(&self, candidate: &NodeId) -> OrbitResult<bool> {
451        // Only allow election if we have quorum
452        if !self.has_quorum().await? {
453            warn!("Cannot start election: no quorum available");
454            return Ok(false);
455        }
456
457        // Only this node can start its own election
458        if candidate != &self.node_id {
459            return Err(OrbitError::configuration(
460                "Can only start election for self",
461            ));
462        }
463
464        // Check if already leader
465        if self.raft_consensus.is_leader().await {
466            return Ok(true);
467        }
468
469        // The election is handled by the Raft consensus algorithm
470        // Return true to indicate election attempt was successful
471        Ok(true)
472    }
473
474    async fn report_coordinator_failure(&self, failed_coordinator: &NodeId) -> OrbitResult<()> {
475        info!("Reporting coordinator failure: {}", failed_coordinator);
476
477        // Update health status
478        {
479            let mut health_map = self.node_health.write().await;
480            if let Some(health) = health_map.get_mut(failed_coordinator) {
481                health.is_reachable = false;
482                health.consecutive_failures += 1;
483            }
484        }
485
486        // Check if we still have quorum after this failure
487        if !self.has_quorum().await? {
488            warn!(
489                "Lost quorum after coordinator failure: {}",
490                failed_coordinator
491            );
492        }
493
494        Ok(())
495    }
496
497    async fn get_cluster_config(&self) -> OrbitResult<ClusterConfig> {
498        let cluster_nodes = self.get_cluster_nodes().await?;
499        let current_leader = self.raft_consensus.get_leader().await;
500
501        let total_nodes = cluster_nodes.len();
502        let majority_threshold = total_nodes / 2 + 1;
503
504        Ok(ClusterConfig {
505            total_nodes,
506            majority_threshold,
507            current_leader,
508        })
509    }

512/// Event handler that integrates with recovery system
513pub struct RecoveryRaftEventHandler {
514    recovery_manager: Arc<crate::recovery::TransactionRecoveryManager>,
515}
516
517impl RecoveryRaftEventHandler {
518    pub fn new(recovery_manager: Arc<crate::recovery::TransactionRecoveryManager>) -> Self {
519        Self { recovery_manager }
520    }
521
522    /// Initiate a comprehensive recovery scan when this node becomes leader
523    async fn initiate_leader_recovery_scan(&self) -> OrbitResult<()> {
524        info!("Starting leader recovery scan for orphaned transactions");
525
526        let failed_coordinators = self.identify_failed_coordinators().await;
527        let transactions_to_recover = self
528            .collect_transactions_for_recovery(&failed_coordinators)
529            .await?;
530        self.process_recovery_transactions(transactions_to_recover, failed_coordinators.len())
531            .await?;
532
533        Ok(())
534    }
535
536    /// Identify coordinators that have failed
537    async fn identify_failed_coordinators(&self) -> Vec<crate::mesh::NodeId> {
538        let coordinators = self.recovery_manager.coordinators.read().await;
539        let mut failed_coordinators = Vec::new();
540
541        for (node_id, health) in coordinators.iter() {
542            if !health.is_healthy {
543                failed_coordinators.push(node_id.clone());
544            }
545        }
546
547        failed_coordinators
548    }
549
550    /// Collect all transactions that need recovery from failed coordinators
551    async fn collect_transactions_for_recovery(
552        &self,
553        failed_coordinators: &[crate::mesh::NodeId],
554    ) -> OrbitResult<Vec<crate::recovery::TransactionCheckpoint>> {
555        let mut all_transactions_to_recover = Vec::new();
556
557        for failed_coordinator in failed_coordinators {
558            let transactions = self
559                .recovery_manager
560                .find_transactions_needing_recovery(failed_coordinator)
561                .await?;
562            all_transactions_to_recover.extend(transactions);
563        }
564
565        Ok(all_transactions_to_recover)
566    }
567
568    /// Process the collected transactions for recovery
569    async fn process_recovery_transactions(
570        &self,
571        transactions_to_recover: Vec<crate::recovery::TransactionCheckpoint>,
572        failed_coordinators_count: usize,
573    ) -> OrbitResult<()> {
574        if !transactions_to_recover.is_empty() {
575            info!(
576                "Found {} total transactions needing recovery from {} failed coordinators",
577                transactions_to_recover.len(),
578                failed_coordinators_count
579            );
580
581            // Since this node just became leader, it should take over these transactions
582            self.recovery_manager
583                .become_coordinator(transactions_to_recover)
584                .await?;
585        } else {
586            info!("No orphaned transactions found during leader recovery scan");
587        }
588
589        Ok(())
590    }
591}

594impl RaftEventHandler for RecoveryRaftEventHandler {
595    async fn on_leader_elected(&self, leader_id: &NodeId, term: u64) -> OrbitResult<()> {
596        info!("New leader elected: {} (term: {})", leader_id, term);
597
598        // Notify recovery manager about the new leader
599        let handlers = self.recovery_manager.event_handlers.read().await;
600        for handler in handlers.iter() {
601            if let Err(e) = handler.on_coordinator_elected(leader_id).await {
602                error!("Recovery event handler failed: {}", e);
603            }
604        }
605
606        // If this node became leader, initiate recovery scan for orphaned transactions
607        if let Ok(cluster_config) = self
608            .recovery_manager
609            .cluster_manager
610            .get_cluster_config()
611            .await
612        {
613            if cluster_config.current_leader.as_ref() == Some(leader_id) {
614                // This node is the new leader - scan for transactions that need recovery
615                info!("This node became leader, scanning for orphaned transactions...");
616
617                // Trigger a comprehensive recovery check
618                if let Err(e) = self.initiate_leader_recovery_scan().await {
619                    error!("Failed to initiate leader recovery scan: {}", e);
620                }
621            }
622        }
623
624        Ok(())
625    }
626
627    async fn on_leader_lost(&self, former_leader_id: &NodeId, term: u64) -> OrbitResult<()> {
628        warn!("Leader lost: {} (term: {})", former_leader_id, term);
629
630        // Report coordinator failure to trigger recovery process
631        if let Err(e) = self
632            .recovery_manager
633            .cluster_manager
634            .report_coordinator_failure(former_leader_id)
635            .await
636        {
637            error!("Failed to report coordinator failure: {}", e);
638        }
639
640        // Find and handle transactions coordinated by the failed leader
641        let transactions_needing_recovery = self
642            .recovery_manager
643            .find_transactions_needing_recovery(former_leader_id)
644            .await?;
645
646        if !transactions_needing_recovery.is_empty() {
647            warn!(
648                "Found {} transactions needing recovery from failed leader {}",
649                transactions_needing_recovery.len(),
650                former_leader_id
651            );
652
653            // The recovery manager will handle these through normal coordinator failure process
654            self.recovery_manager
655                .initiate_recovery_process(transactions_needing_recovery)
656                .await?;
657        }
658
659        Ok(())
660    }
