46/// Persistent transaction log entry with additional metadata
47#[derive(Debug, Clone, Serialize, Deserialize)]
48pub struct PersistentLogEntry {
49    pub id: Uuid,
50    pub timestamp: i64,
51    pub transaction_id: TransactionId,
52    pub event: TransactionEvent,
53    pub details: Option<serde_json::Value>,
54    pub node_id: String,
55    pub checksum: String,
56    pub archived: bool,
57}
58
59impl From<TransactionLogEntry> for PersistentLogEntry {
60    fn from(entry: TransactionLogEntry) -> Self {
61        let serialized = serde_json::to_string(&entry).unwrap_or_default();
62        let checksum = format!("{:x}", md5::compute(serialized.as_bytes()));
63
64        Self {
65            id: Uuid::new_v4(),
66            timestamp: entry.timestamp,
67            transaction_id: entry.transaction_id,
68            event: entry.event,
69            details: entry.details,
70            node_id: "unknown".to_string(), // Would be populated from context
71            checksum,
72            archived: false,
73        }
74    }
75}

134impl SqliteTransactionLogger {
135    /// Create a new SQLite transaction logger
136    pub async fn new(config: PersistentLogConfig) -> OrbitResult<Self> {
137        // Ensure database directory exists
138        if let Some(parent) = config.database_path.parent() {
139            tokio::fs::create_dir_all(parent).await.map_err(|e| {
140                OrbitError::internal(format!("Failed to create database directory: {e}"))
141            })?;
142        }
143
144        // Build connection string
145        let database_url = format!("sqlite:{}?mode=rwc", config.database_path.display());
146
147        // Configure connection pool
148        let pool = SqlitePool::connect(&database_url)
149            .await
150            .map_err(|e| OrbitError::internal(format!("Failed to connect to database: {e}")))?;
151
152        let logger = Self {
153            pool,
154            config,
155            stats_cache: Arc::new(RwLock::new(None)),
156            write_buffer: Arc::new(Mutex::new(Vec::new())),
157        };
158
159        logger.initialize_schema().await?;
160        logger.configure_database().await?;
161
162        info!(
163            "SQLite transaction logger initialized: {}",
164            logger.config.database_path.display()
165        );
166        Ok(logger)
167    }

169    /// Initialize database schema
170    async fn initialize_schema(&self) -> OrbitResult<()> {
171        let schema_sql = r#"
172            CREATE TABLE IF NOT EXISTS transaction_log (
173                id TEXT PRIMARY KEY,
174                timestamp INTEGER NOT NULL,
175                transaction_id TEXT NOT NULL,
176                coordinator_node TEXT NOT NULL,
177                event_type TEXT NOT NULL,
178                event_data TEXT NOT NULL,
179                details TEXT,
180                node_id TEXT NOT NULL,
181                checksum TEXT NOT NULL,
182                archived BOOLEAN NOT NULL DEFAULT FALSE,
183                created_at INTEGER NOT NULL DEFAULT (strftime('%s', 'now'))
184            );
185
186            CREATE INDEX IF NOT EXISTS idx_transaction_log_transaction_id 
187            ON transaction_log(transaction_id);
188            
189            CREATE INDEX IF NOT EXISTS idx_transaction_log_timestamp 
190            ON transaction_log(timestamp);
191            
192            CREATE INDEX IF NOT EXISTS idx_transaction_log_node_id 
193            ON transaction_log(node_id);
194            
195            CREATE INDEX IF NOT EXISTS idx_transaction_log_archived 
196            ON transaction_log(archived);
197
198            -- Transaction recovery table for coordinator state
199            CREATE TABLE IF NOT EXISTS transaction_recovery (
200                transaction_id TEXT PRIMARY KEY,
201                coordinator_node TEXT NOT NULL,
202                state TEXT NOT NULL,
203                participants TEXT NOT NULL,
204                operations TEXT NOT NULL,
205                timeout_seconds INTEGER NOT NULL,
206                created_at INTEGER NOT NULL,
207                updated_at INTEGER NOT NULL
208            );
209
210            CREATE INDEX IF NOT EXISTS idx_recovery_coordinator_node 
211            ON transaction_recovery(coordinator_node);
212        "#;
213
214        sqlx::query(schema_sql)
215            .execute(&self.pool)
216            .await
217            .map_err(|e| OrbitError::internal(format!("Failed to initialize schema: {e}")))?;
218
219        Ok(())
220    }

243            .map_err(|e| OrbitError::internal(format!("Failed to set temp store: {e}")))?;

248    /// Flush buffered entries to database
249    async fn flush_buffer(&self) -> OrbitResult<()> {
250        let entries = {
251            let mut buffer = self.write_buffer.lock().await;
252            if buffer.is_empty() {
253                return Ok(());
254            }
255            let entries = buffer.clone();
256            buffer.clear();
257            entries
258        };
259
260        if !entries.is_empty() {
261            self.write_batch_internal(&entries).await?;
262            debug!("Flushed {} buffered log entries", entries.len());
263        }
264
265        Ok(())
266    }
267
268    /// Internal batch write implementation
269    async fn write_batch_internal(&self, entries: &[TransactionLogEntry]) -> OrbitResult<()> {
270        let mut tx = self
271            .pool
272            .begin()
273            .await
274            .map_err(|e| OrbitError::internal(format!("Failed to begin transaction: {e}")))?;
275
276        for entry in entries {
277            let persistent_entry = PersistentLogEntry::from(entry.clone());
278
279            let event_data = serde_json::to_string(&entry.event)
280                .map_err(|e| OrbitError::internal(format!("Failed to serialize event: {e}")))?;
281
282            let details_json = entry
283                .details
284                .as_ref()
285                .map(|d| serde_json::to_string(d).unwrap_or_default());
286
287            sqlx::query(
288                r#"INSERT INTO transaction_log 
289                   (id, timestamp, transaction_id, coordinator_node, event_type, event_data, details, node_id, checksum, archived) 
290                   VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)"#
291            )
292            .bind(persistent_entry.id.to_string())
293            .bind(persistent_entry.timestamp)
294            .bind(entry.transaction_id.id.clone())
295            .bind(entry.transaction_id.coordinator_node.to_string())
296            .bind(discriminant_name(&entry.event))
297            .bind(event_data)
298            .bind(details_json)
299            .bind(persistent_entry.node_id)
300            .bind(persistent_entry.checksum)
301            .bind(persistent_entry.archived)
302            .execute(&mut *tx).await
303            .map_err(|e| OrbitError::internal(format!("Failed to insert log entry: {e}")))?;
304        }
305
306        tx.commit()
307            .await
308            .map_err(|e| OrbitError::internal(format!("Failed to commit batch: {e}")))?;
309
310        Ok(())
311    }

313    /// Start background maintenance tasks
314    pub async fn start_background_tasks(&self) -> OrbitResult<()> {
315        let logger = Arc::new(self.clone());
316
317        // Buffer flush task
318        {
319            let logger_clone = Arc::clone(&logger);
320            tokio::spawn(async move {
321                let mut interval = tokio::time::interval(Duration::from_secs(5));
322                loop {
323                    interval.tick().await;
324                    if let Err(e) = logger_clone.flush_buffer().await {
325                        error!("Failed to flush log buffer: {}", e);
326                    }
327                }
328            });
329        }
330
331        // Maintenance task
332        {
333            let logger_clone = Arc::clone(&logger);
334            tokio::spawn(async move {
335                let mut interval = tokio::time::interval(Duration::from_secs(300)); // 5 minutes
336                loop {
337                    interval.tick().await;
338                    if let Err(e) = logger_clone.maintenance().await {
339                        error!("Log maintenance failed: {}", e);
340                    }
341                }
342            });
343        }
344
345        info!("Transaction logger background tasks started");
346        Ok(())
347    }

362impl PersistentTransactionLogger for SqliteTransactionLogger {
363    async fn write_entry(&self, entry: &TransactionLogEntry) -> OrbitResult<()> {
364        // Add to buffer for batch processing
365        {
366            let mut buffer = self.write_buffer.lock().await;
367            buffer.push(entry.clone());
368
369            // Flush if buffer is full
370            if buffer.len() >= self.config.batch_size {
371                let entries = buffer.clone();
372                buffer.clear();
373                drop(buffer); // Release lock before async call
374
375                self.write_batch_internal(&entries).await?;
376            }
377        }
378
379        Ok(())
380    }
381
382    async fn write_batch(&self, entries: &[TransactionLogEntry]) -> OrbitResult<()> {
383        self.write_batch_internal(entries).await
384    }

386    async fn get_transaction_log(
387        &self,
388        transaction_id: &TransactionId,
389    ) -> OrbitResult<Vec<PersistentLogEntry>> {
390        let rows = sqlx::query(
391            "SELECT * FROM transaction_log WHERE transaction_id = ? ORDER BY timestamp ASC",
392        )
393        .bind(&transaction_id.id)
394        .fetch_all(&self.pool)
395        .await
396        .map_err(|e| OrbitError::internal(format!("Failed to query transaction log: {e}")))?;
397
398        let mut entries = Vec::new();
399        for row in rows {
400            entries.push(self.row_to_persistent_entry(row)?);
401        }
402
403        Ok(entries)
404    }

428    async fn archive_old_entries(&self, before_timestamp: i64) -> OrbitResult<u64> {
429        let result = sqlx::query(
430            "UPDATE transaction_log SET archived = TRUE WHERE timestamp < ? AND archived = FALSE",
431        )
432        .bind(before_timestamp)
433        .execute(&self.pool)
434        .await
435        .map_err(|e| OrbitError::internal(format!("Failed to archive entries: {e}")))?;
436
437        let archived_count = result.rows_affected();
438        if archived_count > 0 {
439            info!("Archived {} old transaction log entries", archived_count);
440        }
441
442        Ok(archived_count)
443    }

445    async fn get_stats(&self) -> OrbitResult<LogStats> {
446        // Check cache first
447        {
448            let cache = self.stats_cache.read().await;
449            if let Some((stats, cached_at)) = cache.as_ref() {
450                if cached_at.elapsed() < Duration::from_secs(30) {
451                    return Ok(stats.clone());
452                }
453            }
454        }
455
456        // Query fresh stats
457        let stats_row = sqlx::query(
458            r#"SELECT 
459                COUNT(*) as total,
460                COUNT(CASE WHEN archived = FALSE THEN 1 END) as active,
461                COUNT(CASE WHEN archived = TRUE THEN 1 END) as archived,
462                MIN(timestamp) as oldest,
463                MAX(timestamp) as newest
464               FROM transaction_log"#,
465        )
466        .fetch_one(&self.pool)
467        .await
468        .map_err(|e| OrbitError::internal(format!("Failed to get stats: {e}")))?;
469
470        let database_size = tokio::fs::metadata(&self.config.database_path)
471            .await
472            .map(|m| m.len())
473            .unwrap_or(0);
474
475        let stats = LogStats {
476            total_entries: stats_row.get::<i64, _>("total") as u64,
477            active_entries: stats_row.get::<i64, _>("active") as u64,
478            archived_entries: stats_row.get::<i64, _>("archived") as u64,
479            database_size_bytes: database_size,
480            oldest_entry_timestamp: stats_row.get::<Option<i64>, _>("oldest"),
481            newest_entry_timestamp: stats_row.get::<Option<i64>, _>("newest"),
482        };
483
484        // Update cache
485        {
486            let mut cache = self.stats_cache.write().await;
487            *cache = Some((stats.clone(), Instant::now()));
488        }
489
490        Ok(stats)
491    }

493    async fn maintenance(&self) -> OrbitResult<()> {
494        // Flush any pending writes
495        self.flush_buffer().await?;
496
497        // Archive old entries if auto-rotation is enabled
498        if self.config.auto_rotate {
499            let cutoff_time = chrono::Utc::now().timestamp() - self.config.archive_after_seconds;
500            self.archive_old_entries(cutoff_time).await?;
501        }
502
503        // Vacuum database to reclaim space
504        sqlx::query("VACUUM")
505            .execute(&self.pool)
506            .await
507            .map_err(|e| OrbitError::internal(format!("Failed to vacuum database: {e}")))?;
508
509        // Update statistics
510        sqlx::query("ANALYZE")
511            .execute(&self.pool)
512            .await
513            .map_err(|e| OrbitError::internal(format!("Failed to analyze database: {e}")))?;
514
515        debug!("Transaction log maintenance completed");
516        Ok(())
517    }

519    async fn recover_transaction_state(
520        &self,
521        transaction_id: &TransactionId,
522    ) -> OrbitResult<Option<TransactionEvent>> {
523        let row = sqlx::query(
524            r#"SELECT event_data FROM transaction_log 
525               WHERE transaction_id = ? 
526               ORDER BY timestamp DESC 
527               LIMIT 1"#,
528        )
529        .bind(&transaction_id.id)
530        .fetch_optional(&self.pool)
531        .await
532        .map_err(|e| OrbitError::internal(format!("Failed to recover transaction state: {e}")))?;
533
534        if let Some(row) = row {
535            let event_data: String = row.get("event_data");
536            let event: TransactionEvent = serde_json::from_str(&event_data)
537                .map_err(|e| OrbitError::internal(format!("Failed to deserialize event: {e}")))?;
538            Ok(Some(event))
539        } else {
540            Ok(None)
541        }
542    }

545impl SqliteTransactionLogger {
546    /// Convert database row to persistent log entry
547    fn row_to_persistent_entry(
548        &self,
549        row: sqlx::sqlite::SqliteRow,
550    ) -> OrbitResult<PersistentLogEntry> {
551        let id_str: String = row.get("id");
552        let transaction_id_str: String = row.get("transaction_id");
553        let coordinator_node_str: String = row.get("coordinator_node");
554        let event_data: String = row.get("event_data");
555        let details_str: Option<String> = row.get("details");
556
557        let id = Uuid::parse_str(&id_str)
558            .map_err(|e| OrbitError::internal(format!("Invalid UUID: {e}")))?;
559
560        let transaction_id = TransactionId {
561            id: transaction_id_str,
562            coordinator_node: parse_node_id_from_string(&coordinator_node_str),
563            created_at: row.get("timestamp"),
564        };
565
566        let event: TransactionEvent = serde_json::from_str(&event_data)
567            .map_err(|e| OrbitError::internal(format!("Failed to deserialize event: {e}")))?;
568
569        let details =
570            if let Some(details_str) = details_str {
571                Some(serde_json::from_str(&details_str).map_err(|e| {
572                    OrbitError::internal(format!("Failed to deserialize details: {e}"))
573                })?)
574            } else {
575                None
576            };
577
578        Ok(PersistentLogEntry {
579            id,
580            timestamp: row.get("timestamp"),
581            transaction_id,
582            event,
583            details,
584            node_id: row.get("node_id"),
585            checksum: row.get("checksum"),
586            archived: row.get("archived"),
587        })
588    }
589}

591/// Get discriminant name for transaction event (for indexing)
592fn discriminant_name(event: &TransactionEvent) -> &'static str {
593    match event {
594        TransactionEvent::Started => "Started",
595        TransactionEvent::PrepareRequested => "PrepareRequested",
596        TransactionEvent::VoteReceived { .. } => "VoteReceived",
597        TransactionEvent::CommitRequested => "CommitRequested",
598        TransactionEvent::AbortRequested { .. } => "AbortRequested",
599        TransactionEvent::Committed => "Committed",
600        TransactionEvent::Aborted { .. } => "Aborted",
601        TransactionEvent::TimedOut => "TimedOut",
602        TransactionEvent::Failed { .. } => "Failed",
603    }
604}
