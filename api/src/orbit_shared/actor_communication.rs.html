18/// Actor message types for inter-actor communication
19#[derive(Debug, Clone, Serialize, Deserialize)]
20pub enum ActorMessage {
21    /// Direct method invocation on another actor
22    Invoke {
23        target: AddressableReference,
24        invocation: AddressableInvocation,
25        reply_to: Option<ActorMessageId>,
26    },
27    /// Response to a previous invocation
28    Response {
29        request_id: ActorMessageId,
30        result: Result<serde_json::Value, String>,
31    },
32    /// Fire-and-forget message
33    Tell {
34        target: AddressableReference,
35        message: serde_json::Value,
36        message_type: String,
37    },
38    /// Actor lifecycle events
39    LifecycleEvent {
40        actor: AddressableReference,
41        event: ActorLifecycleEvent,
42    },
43    /// Heartbeat for actor health monitoring
44    Heartbeat {
45        actor: AddressableReference,
46        timestamp: i64,
47        metrics: ActorMetrics,
48    },
49}

195    /// Discover the location of an actor
196    pub async fn discover_actor(
197        &self,
198        reference: &AddressableReference,
199    ) -> OrbitResult<Option<NodeId>> {
200        // Check if we know the location
201        {
202            let locations = self.actor_locations.read().await;
203            if let Some(node_id) = locations.get(reference) {
204                debug!("Found actor {} on node {}", reference, node_id);
205                return Ok(Some(node_id.clone()));
206            }
207        }
208
209        // Check if it's a local actor
210        {
211            let local_actors = self.local_actors.read().await;
212            if local_actors.contains_key(reference) {
213                // Return local node ID (would be provided by the system)
214                debug!("Actor {} is local", reference);
215                return Ok(None); // None indicates local
216            }
217        }
218
219        // Actor not found locally or in cache
220        debug!("Actor {} not found, initiating discovery", reference);
221        self.initiate_actor_discovery(reference).await
222    }

224    /// Initiate discovery process for an actor
225    async fn initiate_actor_discovery(
226        &self,
227        reference: &AddressableReference,
228    ) -> OrbitResult<Option<NodeId>> {
229        let (sender, receiver) = oneshot::channel();
230
231        // Add to pending discoveries
232        {
233            let mut pending = self.pending_discoveries.write().await;
234            pending
235                .entry(reference.clone())
236                .or_insert_with(Vec::new)
237                .push(sender);
238        }
239
240        // In a real implementation, this would send discovery requests to other nodes
241        // For now, we'll simulate with a timeout
242        match tokio::time::timeout(self.config.message_timeout, receiver).await {
243            Ok(result) => result.map_err(|_| OrbitError::internal("Discovery channel closed")),
244            Err(_) => {
245                // Remove from pending discoveries on timeout
246                let mut pending = self.pending_discoveries.write().await;
247                pending.remove(reference);
248                Ok(None)
249            }
250        }
251    }

253    /// Update actor location information
254    pub async fn update_actor_location(
255        &self,
256        reference: AddressableReference,
257        node_id: NodeId,
258    ) -> OrbitResult<()> {
259        // Update location mapping
260        {
261            let mut locations = self.actor_locations.write().await;
262            locations.insert(reference.clone(), node_id.clone());
263        }
264
265        // Update node actors mapping
266        {
267            let mut node_actors = self.node_actors.write().await;
268            node_actors
269                .entry(node_id.clone())
270                .or_insert_with(Vec::new)
271                .push(reference.clone());
272        }
273
274        // Notify any pending discoveries
275        {
276            let mut pending = self.pending_discoveries.write().await;
277            if let Some(waiters) = pending.remove(&reference) {
278                for waiter in waiters {
279                    let _ = waiter.send(Some(node_id.clone()));
280                }
281            }
282        }
283
284        debug!("Updated actor {} location to node {}", reference, node_id);
285        Ok(())
286    }

289    pub async fn remove_actor_location(&self, reference: &AddressableReference) -> OrbitResult<()> {
290        let node_id = {
291            let mut locations = self.actor_locations.write().await;
292            locations.remove(reference)
293        };
294
295        if let Some(node_id) = node_id {
296            let mut node_actors = self.node_actors.write().await;
297            if let Some(actors) = node_actors.get_mut(&node_id) {
298                actors.retain(|actor| actor != reference);
299                if actors.is_empty() {
300                    node_actors.remove(&node_id);
301                }
302            }
303        }
304
305        debug!("Removed actor {} from location tracking", reference);
306        Ok(())
307    }

321    /// Start background services for actor discovery and health monitoring
322    pub async fn start_background_services(&self) -> OrbitResult<()> {
323        let discovery_service = Arc::new(self.clone());
324
325        // Start heartbeat service
326        let heartbeat_service = discovery_service.clone();
327        tokio::spawn(async move {
328            let mut interval = tokio::time::interval(heartbeat_service.config.heartbeat_interval);
329            loop {
330                interval.tick().await;
331                if let Err(e) = heartbeat_service.send_heartbeats().await {
332                    error!("Heartbeat service error: {}", e);
333                }
334            }
335        });
336
337        // Start cleanup service
338        let cleanup_service = discovery_service.clone();
339        tokio::spawn(async move {
340            let mut interval = tokio::time::interval(cleanup_service.config.discovery_interval);
341            loop {
342                interval.tick().await;
343                if let Err(e) = cleanup_service.cleanup_stale_actors().await {
344                    error!("Cleanup service error: {}", e);
345                }
346            }
347        });
348
349        info!("Actor discovery background services started");
350        Ok(())
351    }

372    /// Clean up actors that haven't been accessed recently
373    async fn cleanup_stale_actors(&self) -> OrbitResult<()> {
374        let stale_threshold = Instant::now() - Duration::from_secs(300); // 5 minutes
375        let mut to_remove = Vec::new();
376
377        {
378            let local_actors = self.local_actors.read().await;
379            for (reference, handle) in local_actors.iter() {
380                let last_accessed = *handle.last_accessed.read().await;
381                if last_accessed < stale_threshold {
382                    to_remove.push(reference.clone());
383                }
384            }
385        }
386
387        for reference in to_remove {
388            warn!("Removing stale actor: {}", reference);
389            self.unregister_local_actor(&reference).await?;
390        }
391
392        Ok(())
393    }

429    /// Send a message to another actor
430    pub async fn send_message(&self, message: ActorMessage) -> OrbitResult<()> {
431        match &message {
432            ActorMessage::Invoke { target, .. } | ActorMessage::Tell { target, .. } => {
433                self.route_to_actor(target, message.clone()).await
434            }
435            ActorMessage::Response { request_id, .. } => {
436                self.handle_response(request_id.clone(), message).await
437            }
438            ActorMessage::LifecycleEvent { actor, .. } => {
439                self.broadcast_lifecycle_event(actor, message.clone()).await
440            }
441            ActorMessage::Heartbeat { .. } => {
442                // Heartbeats are handled by the discovery service
443                Ok(())
444            }
445        }
446    }

448    /// Send a message and wait for a response
449    pub async fn send_and_wait(
450        &self,
451        target: AddressableReference,
452        invocation: AddressableInvocation,
453        timeout: Duration,
454    ) -> OrbitResult<serde_json::Value> {
455        let message_id = ActorMessageId::new(invocation.reference.clone());
456        let (sender, receiver) = oneshot::channel();
457
458        // Store pending response
459        {
460            let mut pending = self.pending_responses.write().await;
461            pending.insert(message_id.clone(), sender);
462        }
463
464        // Send the message
465        let message = ActorMessage::Invoke {
466            target: target.clone(),
467            invocation,
468            reply_to: Some(message_id.clone()),
469        };
470
471        self.send_message(message).await?;
472
473        // Use provided timeout or fall back to config default
474        let effective_timeout = if timeout == Duration::ZERO {
475            self.config.message_timeout
476        } else {
477            timeout
478        };
479
480        // Wait for response with timeout
481        match tokio::time::timeout(effective_timeout, receiver).await {
482            Ok(Ok(result)) => result.map_err(OrbitError::internal),
483            Ok(Err(_)) => Err(OrbitError::internal("Response channel closed")),
484            Err(_) => {
485                // Clean up pending response on timeout
486                let mut pending = self.pending_responses.write().await;
487                pending.remove(&message_id);
488                Err(OrbitError::timeout("send_and_wait"))
489            }
490        }
491    }

493    /// Route a message to the appropriate actor
494    async fn route_to_actor(
495        &self,
496        target: &AddressableReference,
497        message: ActorMessage,
498    ) -> OrbitResult<()> {
499        // Try local routing first
500        if self.try_route_locally(target, message.clone()).await? {
501            return Ok(());
502        }
503
504        // Route to remote actor
505        self.route_to_remote_actor(target, message).await
506    }

508    /// Try to route message to local actor
509    async fn try_route_locally(
510        &self,
511        target: &AddressableReference,
512        message: ActorMessage,
513    ) -> OrbitResult<bool> {
514        let local_actors = self.discovery_service.local_actors.read().await;
515        if let Some(handle) = local_actors.get(target) {
516            self.send_to_local_actor(handle, message).await?;
517            return Ok(true);
518        }
519        Ok(false)
520    }

523    async fn send_to_local_actor(
524        &self,
525        handle: &ActorHandle,
526        message: ActorMessage,
527    ) -> OrbitResult<()> {
528        if let Err(e) = handle.sender.send(message) {
529            error!("Failed to send message to local actor: {}", e);
530            return Err(OrbitError::internal(format!(
531                "Local actor send failed: {e}"
532            )));
533        }
534
535        // Update last accessed time
536        let mut last_accessed = handle.last_accessed.write().await;
537        *last_accessed = Instant::now();
538        Ok(())
539    }

541    /// Route message to remote actor
542    async fn route_to_remote_actor(
543        &self,
544        target: &AddressableReference,
545        message: ActorMessage,
546    ) -> OrbitResult<()> {
547        match self.discovery_service.discover_actor(target).await? {
548            Some(node_id) => {
549                info!("Forwarding message to actor {} on node {}", target, node_id);
550                self.forward_to_node(&node_id, message).await
551            }
552            None => {
553                warn!("Actor {} not found in cluster", target);
554                Err(OrbitError::AddressableNotFound {
555                    reference: target.to_string(),
556                })
557            }
558        }
559    }

573    async fn handle_response(
574        &self,
575        request_id: ActorMessageId,
576        message: ActorMessage,
577    ) -> OrbitResult<()> {
578        if let ActorMessage::Response { result, .. } = message {
579            let mut pending = self.pending_responses.write().await;
580            if let Some(sender) = pending.remove(&request_id) {
581                if sender.send(result).is_err() {
582                    warn!("Failed to deliver response for request: {:?}", request_id);
583                }
584            } else {
585                warn!("Received response for unknown request: {:?}", request_id);
586            }
587        }
588        Ok(())
589    }
