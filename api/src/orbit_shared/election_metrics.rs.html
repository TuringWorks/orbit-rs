113    /// Start recording an election
114    pub async fn start_election(&self, term: u64) -> ElectionTracker {
115        let now = Instant::now();
116
117        // Update timing metrics
118        {
119            let mut timing = self.timing.write().await;
120            if let Some(last_time) = timing.last_election_time {
121                let time_between = now.duration_since(last_time);
122
123                // Update average time between elections
124                let current_avg = timing.avg_time_between_elections;
125                timing.avg_time_between_elections = if current_avg == Duration::ZERO {
126                    time_between
127                } else {
128                    Duration::from_millis(
129                        (current_avg.as_millis() as u64 + time_between.as_millis() as u64) / 2,
130                    )
131                };
132            }
133
134            timing.last_election_time = Some(now);
135        }
136
137        // Update outcome metrics
138        {
139            let mut outcomes = self.outcomes.write().await;
140            outcomes.total_elections += 1;
141        }
142
143        ElectionTracker {
144            start_time: now,
145            term,
146            metrics: self.clone(),
147        }
148    }

150    /// Record vote request latency
151    pub async fn record_vote_latency(&self, target_node: &NodeId, latency: Duration) {
152        let mut network = self.network.write().await;
153        network
154            .vote_request_latencies
155            .entry(target_node.clone())
156            .or_insert_with(Vec::new)
157            .push(latency);
158
159        // Keep only last 100 measurements per node
160        if let Some(latencies) = network.vote_request_latencies.get_mut(target_node) {
161            if latencies.len() > 100 {
162                latencies.remove(0);
163            }
164        }
165    }
166
167    /// Record message loss
168    pub async fn record_message_loss(
169        &self,
170        target_node: &NodeId,
171        total_sent: u64,
172        total_received: u64,
173    ) {
174        let loss_rate = if total_sent > 0 {
175            1.0 - (total_received as f64 / total_sent as f64)
176        } else {
177            0.0
178        };
179
180        let mut network = self.network.write().await;
181        network
182            .message_loss_rates
183            .insert(target_node.clone(), loss_rate);
184        network.total_messages_sent += total_sent;
185        network.total_messages_received += total_received;
186    }

188    /// Record network partition
189    pub async fn record_partition(&self, affected_nodes: Vec<NodeId>) {
190        let mut network = self.network.write().await;
191        network.partition_events.push(PartitionEvent {
192            timestamp: Instant::now(),
193            affected_nodes,
194            duration: None,
195            resolved: false,
196        });
197    }

199    /// Record partition resolution
200    pub async fn record_partition_resolved(&self, partition_index: usize) {
201        let mut network = self.network.write().await;
202        if let Some(partition) = network.partition_events.get_mut(partition_index) {
203            if !partition.resolved {
204                partition.duration = Some(partition.timestamp.elapsed());
205                partition.resolved = true;
206            }
207        }
208    }

210    /// Record performance metrics during election
211    pub async fn record_performance(&self, cpu_usage: f64, memory_usage: u64) {
212        let mut performance = self.performance.write().await;
213        performance.cpu_usage_during_election.push(cpu_usage);
214        performance.memory_usage_during_election.push(memory_usage);
215
216        // Keep only last 50 measurements
217        if performance.cpu_usage_during_election.len() > 50 {
218            performance.cpu_usage_during_election.remove(0);
219        }
220        if performance.memory_usage_during_election.len() > 50 {
221            performance.memory_usage_during_election.remove(0);
222        }
223    }

225    /// Record heartbeat performance
226    pub async fn record_heartbeat(
227        &self,
228        interval: Duration,
229        success: bool,
230        network_delay: Option<Duration>,
231    ) {
232        let mut performance = self.performance.write().await;
233        let heartbeat = &mut performance.heartbeat_performance;
234
235        // Update average interval
236        let current_avg = heartbeat.avg_heartbeat_interval;
237        heartbeat.avg_heartbeat_interval = if current_avg == Duration::ZERO {
238            interval
239        } else {
240            Duration::from_millis(
241                (current_avg.as_millis() as u64 + interval.as_millis() as u64) / 2,
242            )
243        };
244
245        // Update success rate
246        if !success {
247            heartbeat.missed_heartbeats += 1;
248        }
249
250        // Record network delay
251        if let Some(delay) = network_delay {
252            heartbeat.heartbeat_network_delays.push(delay);
253
254            // Keep only last 100 measurements
255            if heartbeat.heartbeat_network_delays.len() > 100 {
256                heartbeat.heartbeat_network_delays.remove(0);
257            }
258        }
259    }

263        let timing = self.timing.read().await;
264        let network = self.network.read().await;
265        let outcomes = self.outcomes.read().await;
266        let performance = self.performance.read().await;
267

271            fastest_election: timing.fastest_election,
272            slowest_election: timing.slowest_election,
273            avg_time_between_elections: timing.avg_time_between_elections,
274
275            // Network metrics
276            avg_vote_latency: calculate_avg_latency(&network.vote_request_latencies),
277            avg_message_loss_rate: calculate_avg_loss_rate(&network.message_loss_rates),
278            partition_count: network.partition_events.len(),
279            unresolved_partitions: network
280                .partition_events
281                .iter()
282                .filter(|p| !p.resolved)
283                .count(),
284
285            // Outcome metrics
286            total_elections: outcomes.total_elections,
287            win_rate: if outcomes.total_elections > 0 {
288                outcomes.elections_won as f64 / outcomes.total_elections as f64
289            } else {
290                0.0
291            },
292            timeout_rate: if outcomes.total_elections > 0 {
293                outcomes.elections_timeout as f64 / outcomes.total_elections as f64
294            } else {
295                0.0
296            },
297            avg_vote_percentage: if !outcomes.vote_percentages.is_empty() {
298                outcomes.vote_percentages.iter().sum::<f64>()
299                    / outcomes.vote_percentages.len() as f64
300            } else {
301                0.0
302            },
303            avg_leadership_duration: if !outcomes.leadership_durations.is_empty() {
304                let total_millis: u64 = outcomes
305                    .leadership_durations
306                    .iter()
307                    .map(|d| d.as_millis() as u64)
308                    .sum();
309                Duration::from_millis(total_millis / outcomes.leadership_durations.len() as u64)
310            } else {
311                Duration::ZERO
312            },
313
314            // Performance metrics
315            avg_cpu_during_election: if !performance.cpu_usage_during_election.is_empty() {
316                performance.cpu_usage_during_election.iter().sum::<f64>()
317                    / performance.cpu_usage_during_election.len() as f64
318            } else {
319                0.0
320            },
321            avg_memory_during_election: if !performance.memory_usage_during_election.is_empty() {
322                performance.memory_usage_during_election.iter().sum::<u64>()
323                    / performance.memory_usage_during_election.len() as u64
324            } else {
325                0
326            },
327            heartbeat_success_rate: performance.heartbeat_performance.heartbeat_success_rate,
328            missed_heartbeats: performance.heartbeat_performance.missed_heartbeats,
329        }
330    }

386/// Tracks a single election from start to finish
387pub struct ElectionTracker {
388    start_time: Instant,
389    term: u64,
390    metrics: ElectionMetrics,
391}
392
393impl ElectionTracker {
394    /// Complete the election with outcome
395    pub async fn complete(self, outcome: ElectionOutcome, vote_count: u32, total_nodes: u32) {
396        let duration = self.start_time.elapsed();
397
398        // Update timing metrics
399        {
400            let mut timing = self.metrics.timing.write().await;
401
402            // Update average duration
403            let current_avg = timing.avg_election_duration;
404            timing.avg_election_duration = if current_avg == Duration::ZERO {
405                duration
406            } else {
407                Duration::from_millis(
408                    (current_avg.as_millis() as u64 + duration.as_millis() as u64) / 2,
409                )
410            };
411
412            // Update fastest/slowest
413            if timing.fastest_election.is_none() || duration < timing.fastest_election.unwrap() {
414                timing.fastest_election = Some(duration);
415            }
416            if timing.slowest_election.is_none() || duration > timing.slowest_election.unwrap() {
417                timing.slowest_election = Some(duration);
418            }
419
420            // Add to recent durations
421            timing.recent_durations.push(duration);
422            if timing.recent_durations.len() > 50 {
423                timing.recent_durations.remove(0);
424            }
425        }
426
427        // Update outcome metrics
428        {
429            let mut outcomes = self.metrics.outcomes.write().await;
430
431            match outcome {
432                ElectionOutcome::Won => {
433                    outcomes.elections_won += 1;
434                    let vote_percentage = vote_count as f64 / total_nodes as f64;
435                    outcomes.vote_percentages.push(vote_percentage);
436                }
437                ElectionOutcome::Lost => {
438                    outcomes.elections_lost += 1;
439                }
440                ElectionOutcome::Timeout => {
441                    outcomes.elections_timeout += 1;
442                }
443                ElectionOutcome::SplitVote => {
444                    outcomes.split_votes += 1;
445                }
446            }
447        }
448
449        info!(
450            "Election completed: term={}, duration={:?}, outcome={:?}, votes={}/{}",
451            self.term, duration, outcome, vote_count, total_nodes
452        );
453    }
454}

493// Helper functions
494fn calculate_avg_latency(latencies: &HashMap<NodeId, Vec<Duration>>) -> Duration {
495    let all_latencies: Vec<Duration> = latencies.values().flat_map(|v| v.iter()).cloned().collect();
496
497    if all_latencies.is_empty() {
498        return Duration::ZERO;
499    }
500
501    let total_millis: u64 = all_latencies.iter().map(|d| d.as_millis() as u64).sum();
502
503    Duration::from_millis(total_millis / all_latencies.len() as u64)
504}
505
506fn calculate_avg_loss_rate(loss_rates: &HashMap<NodeId, f64>) -> f64 {
507    if loss_rates.is_empty() {
508        return 0.0;
509    }
510
511    loss_rates.values().sum::<f64>() / loss_rates.len() as f64
512}
