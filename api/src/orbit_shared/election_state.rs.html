9/// Persistent election state that survives node restarts
10#[derive(Debug, Clone, Serialize, Deserialize)]
11pub struct ElectionState {
12    /// Current Raft term
13    pub current_term: u64,
14    /// Node voted for in current term
15    pub voted_for: Option<NodeId>,
16    /// Last known leader
17    pub last_known_leader: Option<NodeId>,
18    /// Election history
19    pub election_history: Vec<ElectionRecord>,
20    /// Last update timestamp
21    pub last_updated: i64,
22}
23
24/// Record of an election attempt
25#[derive(Debug, Clone, Serialize, Deserialize)]
26pub struct ElectionRecord {
27    pub term: u64,
28    pub candidate: NodeId,
29    pub timestamp: i64,
30    pub successful: bool,
31    pub vote_count: u32,
32    pub total_nodes: u32,
33}
34
35impl Default for ElectionState {
36    fn default() -> Self {
37        Self {
38            current_term: 0,
39            voted_for: None,
40            last_known_leader: None,
41            election_history: Vec::new(),
42            last_updated: chrono::Utc::now().timestamp_millis(),
43        }
44    }
45}
46
47impl ElectionState {
48    /// Update the current term and reset vote
49    pub fn update_term(&mut self, new_term: u64) {
50        if new_term > self.current_term {
51            self.current_term = new_term;
52            self.voted_for = None; // Reset vote for new term
53            self.last_updated = chrono::Utc::now().timestamp_millis();
54        }
55    }
56
57    /// Cast vote for a candidate in current term
58    pub fn vote_for(&mut self, candidate: NodeId, term: u64) -> Result<(), String> {
59        if term != self.current_term {
60            return Err(format!(
61                "Term mismatch: expected {}, got {}",
62                self.current_term, term
63            ));
64        }
65
66        if let Some(existing_vote) = &self.voted_for {
67            if existing_vote != &candidate {
68                return Err(format!("Already voted for {existing_vote} in term {term}"));
69            }
70        }
71
72        self.voted_for = Some(candidate);
73        self.last_updated = chrono::Utc::now().timestamp_millis();
74        Ok(())
75    }
76
77    /// Record an election attempt
78    pub fn record_election(&mut self, record: ElectionRecord) {
79        self.election_history.push(record);
80        self.last_updated = chrono::Utc::now().timestamp_millis();
81
82        // Keep only last 100 election records
83        if self.election_history.len() > 100 {
84            self.election_history.remove(0);
85        }
86    }
87
88    /// Update the last known leader
89    pub fn update_leader(&mut self, leader: Option<NodeId>) {
90        self.last_known_leader = leader;
91        self.last_updated = chrono::Utc::now().timestamp_millis();
92    }
93
94    /// Check if we can vote for a candidate
95    pub fn can_vote_for(&self, candidate: &NodeId, term: u64) -> bool {
96        // Can't vote for old terms
97        if term < self.current_term {
98            return false;
99        }
100
101        // If it's a new term, we can vote
102        if term > self.current_term {
103            return true;
104        }
105
106        // For current term, check if we haven't voted or voted for same candidate
107        match &self.voted_for {
108            None => true,
109            Some(voted_candidate) => voted_candidate == candidate,
110        }
111    }
112
113    /// Get election statistics
114    pub fn get_election_stats(&self) -> ElectionStats {
115        let total_elections = self.election_history.len();
116        let successful_elections = self
117            .election_history
118            .iter()
119            .filter(|r| r.successful)
120            .count();
121
122        let recent_elections = self
123            .election_history
124            .iter()
125            .rev()
126            .take(10)
127            .cloned()
128            .collect();
129
130        let avg_vote_percentage = if !self.election_history.is_empty() {
131            let total_percentage: f64 = self
132                .election_history
133                .iter()
134                .map(|r| r.vote_count as f64 / r.total_nodes as f64)
135                .sum();
136            total_percentage / self.election_history.len() as f64
137        } else {
138            0.0
139        };
140
141        ElectionStats {
142            total_elections,
143            successful_elections,
144            success_rate: if total_elections > 0 {
145                successful_elections as f64 / total_elections as f64
146            } else {
147                0.0
148            },
149            current_term: self.current_term,
150            last_known_leader: self.last_known_leader.clone(),
151            recent_elections,
152            avg_vote_percentage,
153        }
154    }
155}

169/// Persistent storage manager for election state
170pub struct ElectionStateManager {
171    state_path: std::path::PathBuf,
172    state: RwLock<ElectionState>,
173    node_id: NodeId,
174}
175
176impl ElectionStateManager {
177    /// Create a new election state manager
178    pub fn new<P: AsRef<Path>>(state_path: P, node_id: NodeId) -> Self {
179        Self {
180            state_path: state_path.as_ref().to_path_buf(),
181            state: RwLock::new(ElectionState::default()),
182            node_id,
183        }
184    }
185
186    /// Load election state from disk
187    pub async fn load(&self) -> OrbitResult<()> {
188        if !self.state_path.exists() {
189            info!("No existing election state found, starting with defaults");
190            return Ok(());
191        }
192
193        let content = fs::read_to_string(&self.state_path)
194            .await
195            .map_err(|e| OrbitError::internal(format!("Failed to read election state: {e}")))?;
196
197        let loaded_state: ElectionState = serde_json::from_str(&content)
198            .map_err(|e| OrbitError::internal(format!("Failed to parse election state: {e}")))?;
199
200        {
201            let mut state = self.state.write().await;
202            *state = loaded_state;
203        }
204
205        info!(
206            "Loaded election state from disk: term {}",
207            self.get_current_term().await
208        );
209        Ok(())
210    }
211
212    /// Save election state to disk
213    pub async fn save(&self) -> OrbitResult<()> {
214        let state = self.state.read().await;
215        let content = serde_json::to_string_pretty(&*state).map_err(|e| {
216            OrbitError::internal(format!("Failed to serialize election state: {e}"))
217        })?;
218
219        // Create parent directory if it doesn't exist
220        if let Some(parent) = self.state_path.parent() {
221            fs::create_dir_all(parent).await.map_err(|e| {
222                OrbitError::internal(format!("Failed to create state directory: {e}"))
223            })?;
224        }
225
226        // Write to temporary file first, then rename for atomicity
227        let temp_path = self.state_path.with_extension("tmp");
228        fs::write(&temp_path, content)
229            .await
230            .map_err(|e| OrbitError::internal(format!("Failed to write election state: {e}")))?;
231
232        fs::rename(&temp_path, &self.state_path)
233            .await
234            .map_err(|e| {
235                OrbitError::internal(format!("Failed to update election state file: {e}"))
236            })?;
237
238        Ok(())
239    }
240

329    /// Start periodic state saving
330    pub async fn start_periodic_save(&self) -> OrbitResult<()> {
331        let state_manager = Self {
332            state_path: self.state_path.clone(),
333            state: RwLock::new(ElectionState::default()),
334            node_id: self.node_id.clone(),
335        };
336
337        tokio::spawn(async move {
338            let mut interval = tokio::time::interval(std::time::Duration::from_secs(30));
339
340            loop {
341                interval.tick().await;
342
343                if let Err(e) = state_manager.save().await {
344                    error!("Failed to save election state: {}", e);
345                }
346            }
347        });
348
349        Ok(())
350    }
351
352    /// Cleanup old election records
353    pub async fn cleanup_old_records(&self, keep_count: usize) -> OrbitResult<()> {
354        {
355            let mut state = self.state.write().await;
356            if state.election_history.len() > keep_count {
357                let remove_count = state.election_history.len() - keep_count;
358                state.election_history.drain(0..remove_count);
359                state.last_updated = chrono::Utc::now().timestamp_millis();
360            }
361        }
362
363        self.save().await?;
364        Ok(())
365    }

372    }

374    /// Import election state (for recovery)
375    pub async fn import_state(&self, state_json: &str) -> OrbitResult<()> {
376        let imported_state: ElectionState = serde_json::from_str(state_json)
377            .map_err(|e| OrbitError::internal(format!("Failed to parse imported state: {e}")))?;

382        }
383
384        self.save().await?;
385        info!("Imported election state successfully");
386        Ok(())

396    async fn test_election_state_persistence() {
397        let temp_dir = tempdir().unwrap();
398        let state_path = temp_dir.path().join("election_state.json");
399        let node_id = NodeId::new("test-node".to_string(), "default".to_string());
400
401        let state_manager = ElectionStateManager::new(&state_path, node_id.clone());
402
403        // Update term and vote
404        state_manager.update_term(5).await.unwrap();
405        state_manager.vote_for(node_id.clone(), 5).await.unwrap();
406
407        // Verify persistence
408        assert!(state_path.exists());
409
410        // Create new manager and load
411        let state_manager2 = ElectionStateManager::new(&state_path, node_id.clone());
412        state_manager2.load().await.unwrap();
413
414        assert_eq!(state_manager2.get_current_term().await, 5);
415        assert_eq!(state_manager2.get_voted_for().await, Some(node_id));
416    }
