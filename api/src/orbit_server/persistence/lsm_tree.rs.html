158impl MemTable {
159    fn new() -> Self {
160        Self {
161            data: BTreeMap::new(),
162            size_bytes: 0,
163            created_at: SystemTime::now(),
164        }
165    }
166
167    fn insert(&mut self, key: String, value: Vec<u8>) {
168        let old_size = self.data.get(&key).map(|v| v.len()).unwrap_or(0);
169        self.data.insert(key, value.clone());
170        self.size_bytes = self.size_bytes - old_size + value.len();
171    }
172
173    fn get(&self, key: &str) -> Option<Vec<u8>> {
174        self.data.get(key).cloned()
175    }
176
177    fn remove(&mut self, key: &str) -> bool {
178        if let Some(old_value) = self.data.remove(key) {
179            self.size_bytes -= old_value.len();
180            // Insert tombstone marker
181            self.data.insert(format!("__TOMBSTONE__{}", key), vec![]);
182            true
183        } else {
184            false
185        }
186    }
187
188    fn is_full(&self, limit: usize) -> bool {
189        self.size_bytes >= limit
190    }
191}

193impl WriteAheadLog {
194    async fn new(path: &PathBuf) -> OrbitResult<Self> {
195        let file = OpenOptions::new()
196            .create(true)
197            .append(true)
198            .open(path)
199            .await
200            .map_err(|e| OrbitError::internal(format!("Failed to open WAL: {}", e)))?;
201
202        Ok(Self { file })
203    }
204
205    async fn append(&mut self, key: &str, value: &[u8], operation: &str) -> OrbitResult<()> {
206        let entry = format!(
207            "{}|{}|{}|{}\n",
208            SystemTime::now()
209                .duration_since(std::time::UNIX_EPOCH)
210                .unwrap()
211                .as_millis(),
212            operation,
213            key,
214            general_purpose::STANDARD.encode(value)
215        );
216
217        self.file
218            .write_all(entry.as_bytes())
219            .await
220            .map_err(|e| OrbitError::internal(format!("WAL write error: {}", e)))?;
221        self.file
222            .sync_all()
223            .await
224            .map_err(|e| OrbitError::internal(format!("WAL sync error: {}", e)))?;
225
226        Ok(())
227    }
228}

248    async fn update_metrics(&self, operation: &str, duration: std::time::Duration, success: bool) {
249        let mut metrics = self.metrics.write().await;
250
251        match operation {
252            "read" => {
253                metrics.read_operations += 1;
254                metrics.read_latency_avg =
255                    (metrics.read_latency_avg + duration.as_secs_f64()) / 2.0;
256            }
257            "write" => {
258                metrics.write_operations += 1;
259                metrics.write_latency_avg =
260                    (metrics.write_latency_avg + duration.as_secs_f64()) / 2.0;
261            }
262            "delete" => {
263                metrics.delete_operations += 1;
264                metrics.delete_latency_avg =
265                    (metrics.delete_latency_avg + duration.as_secs_f64()) / 2.0;
266            }
267            _ => {}
268        }
269
270        if !success {
271            metrics.error_count += 1;
272        }
273    }

275    fn reference_to_key(reference: &AddressableReference) -> String {
276        format!("{}:{}", reference.addressable_type, reference.key)
277    }

279    async fn maybe_trigger_compaction(&self) -> OrbitResult<()> {
280        if !self.config.enable_compaction {
281            return Ok(());
282        }
283
284        let mut compaction_guard = self.compaction_running.lock().await;
285        if *compaction_guard {
286            return Ok(()); // Compaction already running
287        }
288
289        let sstable_count = {
290            let sstables = self.sstables.read().unwrap();
291            sstables.len()
292        };
293
294        if sstable_count >= self.config.compaction_threshold {
295            *compaction_guard = true;
296            drop(compaction_guard);
297
298            // Run compaction in background
299            let _sstables = self.sstables.clone();
300            let _data_dir = PathBuf::from(&self.config.data_dir);
301            let compaction_running = self.compaction_running.clone();
302
303            tokio::spawn(async move {
304                // Simplified compaction - merge two smallest SSTables
305                tracing::info!("Starting background compaction");
306
307                // Actual compaction would merge SSTables, remove duplicates,
308                // and create new merged SSTable
309                tokio::time::sleep(tokio::time::Duration::from_millis(100)).await;
310
311                let mut guard = compaction_running.lock().await;
312                *guard = false;
313
314                tracing::info!("Background compaction completed");
315            });
316        }
317
318        Ok(())
319    }
320
321    async fn search_sstables(&self, key: &str) -> OrbitResult<Option<Vec<u8>>> {
322        let sstables = {
323            let sstables_guard = self.sstables.read().unwrap();
324            sstables_guard.clone()
325        };
326
327        // Search SSTables in reverse chronological order (newest first)
328        for sstable in sstables.iter().rev() {
329            // Check bloom filter first
330            if !sstable.bloom_filter.contains(key.as_bytes()) {
331                continue;
332            }
333
334            // Check key range
335            if key < sstable.min_key.as_str() || key > sstable.max_key.as_str() {
336                continue;
337            }
338
339            // Search in SSTable (simplified - would use binary search on index)
340            if let Ok(data) = tokio::fs::read(&sstable.file_path).await {
341                // Simplified search - in production this would use the index
342                let content = String::from_utf8_lossy(&data);
343                for line in content.lines() {
344                    if line.starts_with(key) {
345                        let parts: Vec<&str> = line.splitn(2, '|').collect();
346                        if parts.len() == 2 {
347                            return Ok(Some(
348                                general_purpose::STANDARD
349                                    .decode(parts[1])
350                                    .unwrap_or_default(),
351                            ));
352                        }
353                    }
354                }
355            }
356        }
357
358        Ok(None)
359    }

420#[async_trait]
421impl AddressableDirectoryProvider for LsmTreeAddressableProvider {
422    async fn store_lease(&self, lease: &AddressableLease) -> OrbitResult<()> {
423        let start = Instant::now();
424        let key = Self::reference_to_key(&lease.reference);
425        let value = serde_json::to_vec(lease)
426            .map_err(|e| OrbitError::internal(format!("Serialization error: {}", e)))?;
427
428        // Write to WAL
429        {
430            let mut wal = self.wal.lock().await;
431            wal.append(&key, &value, "INSERT").await?;
432        }
433
434        // Write to active memtable
435        let is_full = {
436            let mut memtable = self.active_memtable.write().unwrap();
437            memtable.insert(key, value);
438            memtable.is_full(self.config.memtable_size_limit)
439        };
440
441        // Check if memtable is full
442        if is_full {
443            // Move to immutable and create new active
444            let old_memtable = {
445                let mut memtable = self.active_memtable.write().unwrap();
446                std::mem::replace(&mut *memtable, MemTable::new())
447            };
448
449            {
450                let mut immutable = self.immutable_memtables.write().unwrap();
451                immutable.push(old_memtable);
452            }
453
454            // Trigger flush and compaction if needed
455            self.maybe_trigger_compaction().await?;
456        }
457
458        self.update_metrics("write", start.elapsed(), true).await;
459        Ok(())
460    }
461
462    async fn get_lease(
463        &self,
464        reference: &AddressableReference,
465    ) -> OrbitResult<Option<AddressableLease>> {
466        let start = Instant::now();
467        let key = Self::reference_to_key(reference);
468
469        // Search in active memtable
470        let active_result = {
471            let active = self.active_memtable.read().unwrap();
472            active.get(&key)
473        };
474
475        if let Some(data) = active_result {
476            let lease = serde_json::from_slice(&data)
477                .map_err(|e| OrbitError::internal(format!("Deserialization error: {}", e)))?;
478            self.update_metrics("read", start.elapsed(), true).await;
479            return Ok(Some(lease));
480        }
481
482        // Search in immutable memtables
483        let immutable_result = {
484            let immutable = self.immutable_memtables.read().unwrap();
485            let mut result = None;
486            for memtable in immutable.iter().rev() {
487                if let Some(data) = memtable.get(&key) {
488                    result = Some(data);
489                    break;
490                }
491            }
492            result
493        };
494
495        if let Some(data) = immutable_result {
496            let lease = serde_json::from_slice(&data)
497                .map_err(|e| OrbitError::internal(format!("Deserialization error: {}", e)))?;
498            self.update_metrics("read", start.elapsed(), true).await;
499            return Ok(Some(lease));
500        }
501
502        // Search in SSTables
503        if let Some(data) = self.search_sstables(&key).await? {
504            let lease = serde_json::from_slice(&data)
505                .map_err(|e| OrbitError::internal(format!("Deserialization error: {}", e)))?;
506            self.update_metrics("read", start.elapsed(), true).await;
507            return Ok(Some(lease));
508        }
509
510        self.update_metrics("read", start.elapsed(), true).await;
511        Ok(None)
512    }
513
514    async fn update_lease(&self, lease: &AddressableLease) -> OrbitResult<()> {
515        // LSM-Trees handle updates as inserts (newer values override older ones)
516        self.store_lease(lease).await
517    }
518
519    async fn remove_lease(&self, reference: &AddressableReference) -> OrbitResult<bool> {
520        let start = Instant::now();
521        let key = Self::reference_to_key(reference);
522
523        // Write tombstone to WAL and memtable
524        {
525            let mut wal = self.wal.lock().await;
526            wal.append(&key, b"", "DELETE").await?;
527        }
528
529        let removed = {
530            let mut memtable = self.active_memtable.write().unwrap();
531            memtable.remove(&key)
532        };
533
534        self.update_metrics("delete", start.elapsed(), removed)
535            .await;
536        Ok(removed)
537    }

539    async fn list_node_leases(&self, node_id: &NodeId) -> OrbitResult<Vec<AddressableLease>> {
540        let start = Instant::now();
541        let mut leases = Vec::new();
542
543        // This is simplified - a real implementation would need efficient range queries
544        // For now, we'll scan all data (not efficient for large datasets)
545
546        // Scan active memtable
547        {
548            let active = self.active_memtable.read().unwrap();
549            for (key, data) in &active.data {
550                if !key.starts_with("__TOMBSTONE__") {
551                    if let Ok(lease) = serde_json::from_slice::<AddressableLease>(data) {
552                        if &lease.node_id == node_id {
553                            leases.push(lease);
554                        }
555                    }
556                }
557            }
558        }
559
560        self.update_metrics("read", start.elapsed(), true).await;
561        Ok(leases)
562    }

687impl ClusterNodeProvider for LsmTreeClusterProvider {
688    async fn store_node(&self, node: &NodeInfo) -> OrbitResult<()> {
689        let key = node.id.to_string();
690        let value = serde_json::to_vec(node)
691            .map_err(|e| OrbitError::internal(format!("Serialization error: {}", e)))?;
692
693        self.wal.lock().await.append(&key, &value, "INSERT").await?;
694
695        let mut memtable = self.active_memtable.write().unwrap();
696        memtable.insert(key, value);
697
698        Ok(())
699    }
700
701    async fn get_node(&self, node_id: &NodeId) -> OrbitResult<Option<NodeInfo>> {
702        let key = node_id.to_string();
703
704        if let Some(data) = self.active_memtable.read().unwrap().get(&key) {
705            let node = serde_json::from_slice(&data)
706                .map_err(|e| OrbitError::internal(format!("Deserialization error: {}", e)))?;
707            return Ok(Some(node));
708        }
709
710        Ok(None)
711    }
712
713    async fn update_node(&self, node: &NodeInfo) -> OrbitResult<()> {
714        self.store_node(node).await
715    }
716
717    async fn remove_node(&self, node_id: &NodeId) -> OrbitResult<bool> {
718        let key = node_id.to_string();
719
720        self.wal.lock().await.append(&key, b"", "DELETE").await?;
721
722        let mut memtable = self.active_memtable.write().unwrap();
723        let removed = memtable.remove(&key);
724
725        Ok(removed)
726    }

728    async fn list_active_nodes(&self) -> OrbitResult<Vec<NodeInfo>> {
729        let mut nodes = Vec::new();
730
731        let active = self.active_memtable.read().unwrap();
732        for (key, data) in &active.data {
733            if !key.starts_with("__TOMBSTONE__") {
734                if let Ok(node) = serde_json::from_slice::<NodeInfo>(data) {
735                    if node.status == NodeStatus::Active {
736                        nodes.push(node);
737                    }
738                }
739            }
740        }
741
742        Ok(nodes)
743    }

754            }
