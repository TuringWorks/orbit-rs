10/// Create an addressable directory persistence provider based on configuration
11pub async fn create_addressable_provider(
12    config: &PersistenceConfig,
13) -> OrbitResult<Arc<dyn AddressableDirectoryProvider>> {
14    match config {
15        PersistenceConfig::Memory(memory_config) => {
16            let provider = memory::MemoryAddressableDirectoryProvider::new(memory_config.clone());
17            provider.initialize().await?;
18            Ok(Arc::new(provider))
19        }
20        PersistenceConfig::CowBTree(cow_config) => {
21            let provider = CowBTreeAddressableProvider::new(cow_config.clone());
22            provider.initialize().await?;
23            Ok(Arc::new(provider))
24        }
25        PersistenceConfig::LsmTree(lsm_config) => {
26            let provider = LsmTreeAddressableProvider::new(lsm_config.clone()).await?;
27            provider.initialize().await?;
28            Ok(Arc::new(provider))
29        }
30        PersistenceConfig::RocksDB(rocks_config) => {
31            let provider = RocksDbAddressableProvider::new(rocks_config.clone())?;
32            provider.initialize().await?;
33            Ok(Arc::new(provider))
34        }
35        // TODO: Implement other providers as needed
36        _ => {
37            tracing::warn!(
38                "Unsupported persistence config for addressable provider, falling back to memory"
39            );
40            let provider = memory::MemoryAddressableDirectoryProvider::new(MemoryConfig::default());
41            provider.initialize().await?;
42            Ok(Arc::new(provider))
43        }
44    }
45}

84/// Load persistence configuration from environment variables
85pub fn load_config_from_env() -> OrbitResult<PersistenceConfig> {
86    let backend_type =
87        std::env::var("ORBIT_PERSISTENCE_BACKEND").unwrap_or_else(|_| "memory".to_string());
88
89    match backend_type.as_str() {
90        "memory" => {
91            let max_entries = std::env::var("ORBIT_MEMORY_MAX_ENTRIES")
92                .ok()
93                .and_then(|s| s.parse().ok());
94
95            let disk_backup = if std::env::var("ORBIT_MEMORY_DISK_BACKUP").is_ok() {
96                let path = std::env::var("ORBIT_MEMORY_BACKUP_PATH")
97                    .unwrap_or_else(|_| "./orbit_backup.json".to_string());
98                let sync_interval = std::env::var("ORBIT_MEMORY_BACKUP_INTERVAL")
99                    .ok()
100                    .and_then(|s| s.parse().ok())
101                    .unwrap_or(300); // 5 minutes
102
103                Some(DiskBackupConfig {
104                    path,
105                    sync_interval,
106                    compression: CompressionType::Gzip,
107                })
108            } else {
109                None
110            };
111
112            Ok(PersistenceConfig::Memory(MemoryConfig {
113                max_entries,
114                disk_backup,
115            }))
116        }
117        "cow_btree" => {
118            let config = CowBTreeConfig {
119                data_dir: std::env::var("ORBIT_COW_DATA_DIR")
120                    .unwrap_or_else(|_| "./orbit_cow_data".to_string()),
121                max_keys_per_node: std::env::var("ORBIT_COW_MAX_KEYS_PER_NODE")
122                    .ok()
123                    .and_then(|s| s.parse().ok())
124                    .unwrap_or(16),
125                wal_buffer_size: std::env::var("ORBIT_COW_WAL_BUFFER_SIZE")
126                    .ok()
127                    .and_then(|s| s.parse().ok())
128                    .unwrap_or(1024 * 1024), // 1MB
129                enable_compression: std::env::var("ORBIT_COW_ENABLE_COMPRESSION")
130                    .map(|s| s.parse().unwrap_or(true))
131                    .unwrap_or(true),
132                wal_sync_interval: std::env::var("ORBIT_COW_WAL_SYNC_INTERVAL")
133                    .ok()
134                    .and_then(|s| s.parse().ok())
135                    .unwrap_or(5),
136            };
137            Ok(PersistenceConfig::CowBTree(config))
138        }
139        "lsm_tree" => {
140            let config = LsmTreeConfig {
141                data_dir: std::env::var("ORBIT_LSM_DATA_DIR")
142                    .unwrap_or_else(|_| "./orbit_lsm_data".to_string()),
143                memtable_size_limit: std::env::var("ORBIT_LSM_MEMTABLE_SIZE")
144                    .ok()
145                    .and_then(|s| s.parse().ok())
146                    .unwrap_or(64 * 1024 * 1024), // 64MB
147                max_memtables: std::env::var("ORBIT_LSM_MAX_MEMTABLES")
148                    .ok()
149                    .and_then(|s| s.parse().ok())
150                    .unwrap_or(10),
151                bloom_filter_fp_rate: std::env::var("ORBIT_LSM_BLOOM_FP_RATE")
152                    .ok()
153                    .and_then(|s| s.parse().ok())
154                    .unwrap_or(0.01),
155                enable_compaction: std::env::var("ORBIT_LSM_ENABLE_COMPACTION")
156                    .map(|s| s.parse().unwrap_or(true))
157                    .unwrap_or(true),
158                compaction_threshold: std::env::var("ORBIT_LSM_COMPACTION_THRESHOLD")
159                    .ok()
160                    .and_then(|s| s.parse().ok())
161                    .unwrap_or(4),
162            };
163            Ok(PersistenceConfig::LsmTree(config))
164        }
165        "rocksdb" => {
166            let config = RocksDbConfig {
167                data_dir: std::env::var("ORBIT_ROCKSDB_DATA_DIR")
168                    .unwrap_or_else(|_| "./orbit_rocksdb_data".to_string()),
169                enable_wal: std::env::var("ORBIT_ROCKSDB_ENABLE_WAL")
170                    .map(|s| s.parse().unwrap_or(true))
171                    .unwrap_or(true),
172                max_background_jobs: std::env::var("ORBIT_ROCKSDB_MAX_BACKGROUND_JOBS")
173                    .ok()
174                    .and_then(|s| s.parse().ok())
175                    .unwrap_or(4),
176                write_buffer_size: std::env::var("ORBIT_ROCKSDB_WRITE_BUFFER_SIZE")
177                    .ok()
178                    .and_then(|s| s.parse().ok())
179                    .unwrap_or(128 * 1024 * 1024), // 128MB
180                max_write_buffer_number: std::env::var("ORBIT_ROCKSDB_MAX_WRITE_BUFFER_NUMBER")
181                    .ok()
182                    .and_then(|s| s.parse().ok())
183                    .unwrap_or(3),
184                target_file_size_base: std::env::var("ORBIT_ROCKSDB_TARGET_FILE_SIZE_BASE")
185                    .ok()
186                    .and_then(|s| s.parse().ok())
187                    .unwrap_or(64 * 1024 * 1024), // 64MB
188                enable_statistics: std::env::var("ORBIT_ROCKSDB_ENABLE_STATISTICS")
189                    .map(|s| s.parse().unwrap_or(true))
190                    .unwrap_or(true),
191                block_cache_size: std::env::var("ORBIT_ROCKSDB_BLOCK_CACHE_SIZE")
192                    .ok()
193                    .and_then(|s| s.parse().ok())
194                    .unwrap_or(256 * 1024 * 1024), // 256MB
195            };
196            Ok(PersistenceConfig::RocksDB(config))
197        }
198        _ => Err(OrbitError::configuration(format!(
199            "Unknown persistence backend: {}",
200            backend_type
201        ))),
202    }
203}

205/// Load persistence configuration from a TOML file
206pub async fn load_config_from_file(path: &str) -> OrbitResult<PersistenceConfig> {
207    let content = tokio::fs::read_to_string(path).await.map_err(|e| {
208        OrbitError::configuration(format!("Failed to read config file {}: {}", path, e))
209    })?;
210
211    let config: PersistenceConfig = toml::from_str(&content).map_err(|e| {
212        OrbitError::configuration(format!("Failed to parse config file {}: {}", path, e))
213    })?;
214
215    Ok(config)
216}

218/// Configuration builder for programmatic setup
219pub struct PersistenceConfigBuilder {
220    backend_type: Option<String>,
221    data_dir: Option<String>,
222}
223
224impl PersistenceConfigBuilder {
225    pub fn new() -> Self {
226        Self {
227            backend_type: None,
228            data_dir: None,
229        }
230    }
231
232    pub fn backend(mut self, backend_type: &str) -> Self {
233        self.backend_type = Some(backend_type.to_string());
234        self
235    }
236
237    pub fn data_dir(mut self, data_dir: &str) -> Self {
238        self.data_dir = Some(data_dir.to_string());
239        self
240    }
241
242    pub fn build(self) -> OrbitResult<PersistenceConfig> {
243        let backend_type = self.backend_type.unwrap_or_else(|| "memory".to_string());
244        let data_dir = self
245            .data_dir
246            .unwrap_or_else(|| format!("./orbit_{}_data", backend_type));
247
248        match backend_type.as_str() {
249            "memory" => Ok(PersistenceConfig::Memory(MemoryConfig::default())),
250            "cow_btree" => Ok(PersistenceConfig::CowBTree(CowBTreeConfig {
251                data_dir,
252                ..Default::default()
253            })),
254            "lsm_tree" => Ok(PersistenceConfig::LsmTree(LsmTreeConfig {
255                data_dir,
256                ..Default::default()
257            })),
258            "rocksdb" => Ok(PersistenceConfig::RocksDB(RocksDbConfig {
259                data_dir,
260                ..Default::default()
261            })),
262            _ => Err(OrbitError::configuration(format!(
263                "Unknown persistence backend: {}",
264                backend_type
265            ))),
266        }
267    }
268}
269
270impl Default for PersistenceConfigBuilder {
271    fn default() -> Self {
272        Self::new()
273    }
274}

276/// Initialize a complete persistence registry with the configured providers
277pub async fn initialize_registry(
278    addressable_config: &PersistenceConfig,
279    cluster_config: &PersistenceConfig,
280) -> OrbitResult<PersistenceProviderRegistry> {
281    let registry = PersistenceProviderRegistry::new();
282
283    // Create and register addressable provider
284    let addressable_provider = create_addressable_provider(addressable_config).await?;
285    registry
286        .register_addressable_provider("default".to_string(), addressable_provider, true)
287        .await?;
288
289    // Create and register cluster provider
290    let cluster_provider = create_cluster_provider(cluster_config).await?;
291    registry
292        .register_cluster_provider("default".to_string(), cluster_provider, true)
293        .await?;
294
295    tracing::info!("Persistence registry initialized with providers");
296    Ok(registry)
