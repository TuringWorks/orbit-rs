140impl BTreeNode {
141    fn new_leaf(version: u64) -> Self {
142        Self {
143            keys: Vec::new(),
144            addressable_values: Vec::new(),
145            node_values: Vec::new(),
146            children: Vec::new(),
147            is_leaf: true,
148            version,
149            parent: None,
150        }
151    }
152
153    #[allow(dead_code)] // Reserved for future B+ tree internal node functionality
154    fn new_internal(version: u64) -> Self {
155        Self {
156            keys: Vec::new(),
157            addressable_values: Vec::new(),
158            node_values: Vec::new(),
159            children: Vec::new(),
160            is_leaf: false,
161            version,
162            parent: None,
163        }
164    }
165
166    #[allow(dead_code)] // Reserved for future B+ tree split functionality
167    fn is_full(&self, max_keys: usize) -> bool {
168        self.keys.len() >= max_keys
169    }
170
171    /// Copy-on-write clone of this node
172    #[allow(dead_code)] // Reserved for future COW functionality
173    fn cow_clone(&self, new_version: u64) -> Self {
174        Self {
175            keys: self.keys.clone(),
176            addressable_values: self.addressable_values.clone(),
177            node_values: self.node_values.clone(),
178            children: self.children.clone(), // Arc references are cheap to clone
179            is_leaf: self.is_leaf,
180            version: new_version,
181            parent: None, // Will be set by parent
182        }
183    }
184
185    /// Find the index where a key should be inserted
186    fn find_key_index(&self, key: &str) -> Result<usize, usize> {
187        self.keys.binary_search(&key.to_string())
188    }
189
190    /// Insert an addressable lease in a leaf node
191    fn insert_addressable_lease(&mut self, key: String, lease: AddressableLease) -> bool {
192        match self.find_key_index(&key) {
193            Ok(idx) => {
194                // Key exists, update value
195                self.addressable_values[idx] = Some(lease);
196                false // Not a new insertion
197            }
198            Err(idx) => {
199                // Key doesn't exist, insert new
200                self.keys.insert(idx, key);
201                self.addressable_values.insert(idx, Some(lease));
202                if self.node_values.len() == self.keys.len() - 1 {
203                    self.node_values.insert(idx, None);
204                }
205                true // New insertion
206            }
207        }
208    }
209
210    /// Insert a node info in a leaf node
211    fn insert_node_info(&mut self, key: String, node: NodeInfo) -> bool {
212        match self.find_key_index(&key) {
213            Ok(idx) => {
214                // Key exists, update value
215                self.node_values[idx] = Some(node);
216                false // Not a new insertion
217            }
218            Err(idx) => {
219                // Key doesn't exist, insert new
220                self.keys.insert(idx, key);
221                self.node_values.insert(idx, Some(node));
222                if self.addressable_values.len() == self.keys.len() - 1 {
223                    self.addressable_values.insert(idx, None);
224                }
225                true // New insertion
226            }
227        }
228    }
229}

231impl WriteAheadLog {
232    async fn new(path: &std::path::Path) -> OrbitResult<Self> {
233        let file = OpenOptions::new()
234            .create(true)
235            .append(true)
236            .open(path)
237            .await
238            .map_err(|e| OrbitError::internal(format!("Failed to open WAL file: {}", e)))?;
239
240        // Determine the next sequence number by reading existing entries
241        let sequence = Self::get_last_sequence(path).await.unwrap_or(0) + 1;
242
243        Ok(Self {
244            file,
245            sequence,
246            buffer: Vec::with_capacity(WAL_BUFFER_SIZE),
247        })
248    }
249
250    /// Get the last sequence number from the WAL file
251    async fn get_last_sequence(path: &std::path::Path) -> OrbitResult<u64> {
252        if !path.exists() {
253            return Ok(0);
254        }
255
256        let content = tokio::fs::read(path)
257            .await
258            .map_err(|e| OrbitError::internal(format!("Failed to read WAL file: {}", e)))?;
259        let mut last_sequence = 0u64;
260        let mut offset = 0;
261
262        while offset + 4 <= content.len() {
263            // Read entry length
264            let len_bytes = &content[offset..offset + 4];
265            let entry_len =
266                u32::from_le_bytes([len_bytes[0], len_bytes[1], len_bytes[2], len_bytes[3]])
267                    as usize;
268            offset += 4;
269
270            if offset + entry_len <= content.len() {
271                let entry_data = &content[offset..offset + entry_len];
272
273                // Try to parse the entry to get sequence number
274                if let Ok(entry) = serde_json::from_slice::<WALEntry>(entry_data) {
275                    last_sequence = entry.sequence;
276                }
277
278                offset += entry_len;
279            } else {
280                break;
281            }
282        }
283
284        Ok(last_sequence)
285    }
286
287    /// Append an entry to the WAL
288    async fn append_entry(&mut self, operation: WALOperation) -> OrbitResult<()> {
289        let entry = WALEntry {
290            sequence: self.sequence,
291            timestamp: SystemTime::now(),
292            operation,
293        };
294
295        let serialized = serde_json::to_vec(&entry)
296            .map_err(|e| OrbitError::internal(format!("WAL serialization error: {}", e)))?;
297
298        // Write entry length followed by entry data
299        let len_bytes = (serialized.len() as u32).to_le_bytes();
300        self.buffer.extend_from_slice(&len_bytes);
301        self.buffer.extend_from_slice(&serialized);
302
303        // Flush if buffer is getting full
304        if self.buffer.len() >= WAL_BUFFER_SIZE / 2 {
305            self.flush().await?;
306        }
307
308        self.sequence += 1;
309        tracing::debug!(
310            "[WAL] Appending entry {} with {} bytes",
311            entry.sequence,
312            serialized.len()
313        );
314
315        Ok(())
316    }
317
318    /// Flush buffered entries to disk
319    async fn flush(&mut self) -> OrbitResult<()> {
320        if !self.buffer.is_empty() {
321            self.file
322                .write_all(&self.buffer)
323                .await
324                .map_err(|e| OrbitError::internal(format!("WAL flush error: {}", e)))?;
325            self.file
326                .sync_all()
327                .await
328                .map_err(|e| OrbitError::internal(format!("WAL sync error: {}", e)))?;
329            self.buffer.clear();
330        }
331        Ok(())
332    }
333}

335impl CowBTreeAddressableProvider {
336    pub fn new(config: CowBTreeConfig) -> Self {
337        Self {
338            root: Arc::new(StdRwLock::new(BTreeNode::new_leaf(0))),
339            wal: Arc::new(Mutex::new(None)), // Will be initialized in initialize()
340            snapshots: Arc::new(StdRwLock::new(HashMap::new())),
341            version_counter: Arc::new(Mutex::new(0)),
342            metrics: Arc::new(RwLock::new(PersistenceMetrics::default())),
343            transactions: Arc::new(RwLock::new(HashMap::new())),
344            config,
345        }
346    }
347
348    async fn update_metrics(&self, operation: &str, duration: std::time::Duration, success: bool) {
349        let mut metrics = self.metrics.write().await;
350
351        match operation {
352            "read" => {
353                metrics.read_operations += 1;
354                metrics.read_latency_avg =
355                    (metrics.read_latency_avg + duration.as_secs_f64()) / 2.0;
356            }
357            "write" => {
358                metrics.write_operations += 1;
359                metrics.write_latency_avg =
360                    (metrics.write_latency_avg + duration.as_secs_f64()) / 2.0;
361            }
362            "delete" => {
363                metrics.delete_operations += 1;
364                metrics.delete_latency_avg =
365                    (metrics.delete_latency_avg + duration.as_secs_f64()) / 2.0;
366            }
367            _ => {}
368        }
369
370        if !success {
371            metrics.error_count += 1;
372        }
373    }
374
375    fn reference_to_key(reference: &AddressableReference) -> String {
376        format!("{}:{}", reference.addressable_type, reference.key)
377    }
378}

380#[async_trait]
381impl PersistenceProvider for CowBTreeAddressableProvider {
382    async fn initialize(&self) -> OrbitResult<()> {
383        // Create data directory
384        let data_dir = PathBuf::from(&self.config.data_dir);
385        tokio::fs::create_dir_all(&data_dir)
386            .await
387            .map_err(|e| OrbitError::internal(format!("Failed to create data directory: {}", e)))?;
388
389        // Initialize WAL
390        let wal_path = data_dir.join("orbit.wal");
391        let wal = WriteAheadLog::new(&wal_path).await?;
392        *self.wal.lock().await = Some(wal);
393
394        tracing::info!(
395            "COW B+ Tree addressable provider initialized at {}",
396            data_dir.display()
397        );
398        Ok(())
399    }
400
401    async fn shutdown(&self) -> OrbitResult<()> {
402        // Flush WAL
403        if let Some(ref mut wal) = *self.wal.lock().await {
404            wal.flush().await?;
405        }
406
407        tracing::info!("COW B+ Tree addressable provider shutdown");
408        Ok(())
409    }
410
411    async fn health_check(&self) -> ProviderHealth {
412        // Check if WAL file is accessible
413        let wal_path = PathBuf::from(&self.config.data_dir).join("orbit.wal");
414        if !wal_path.exists() {
415            return ProviderHealth::Unhealthy {
416                reason: "WAL file not found".to_string(),
417            };
418        }
419
420        // Check memory usage (simplified)
421        let metrics = self.metrics.read().await;
422        if metrics.error_count > 100 {
423            return ProviderHealth::Degraded {
424                reason: format!("High error count: {}", metrics.error_count),
425            };
426        }
427
428        ProviderHealth::Healthy
429    }
430
431    async fn metrics(&self) -> PersistenceMetrics {
432        self.metrics.read().await.clone()
433    }
434
435    async fn begin_transaction(&self, context: TransactionContext) -> OrbitResult<String> {
436        let mut transactions = self.transactions.write().await;
437        transactions.insert(context.id.clone(), context.clone());
438        Ok(context.id)
439    }
440
441    async fn commit_transaction(&self, transaction_id: &str) -> OrbitResult<()> {
442        let mut transactions = self.transactions.write().await;
443        transactions.remove(transaction_id);
444        Ok(())
445    }
446
447    async fn rollback_transaction(&self, transaction_id: &str) -> OrbitResult<()> {
448        let mut transactions = self.transactions.write().await;
449        transactions.remove(transaction_id);
450        // Note: Full rollback would require more complex implementation
451        tracing::warn!("Transaction rollback is simplified in COW B+ Tree implementation");
452        Ok(())
453    }
454}

456#[async_trait]
457impl AddressableDirectoryProvider for CowBTreeAddressableProvider {
458    async fn store_lease(&self, lease: &AddressableLease) -> OrbitResult<()> {
459        let start = Instant::now();
460        let key = Self::reference_to_key(&lease.reference);
461
462        // Log to WAL
463        {
464            let mut wal_guard = self.wal.lock().await;
465            if let Some(ref mut wal) = *wal_guard {
466                wal.append_entry(WALOperation::InsertLease {
467                    key: key.clone(),
468                    lease: lease.clone(),
469                })
470                .await?;
471            }
472        }
473
474        // Insert into B+ tree
475        let success = {
476            let mut root = self.root.write().unwrap();
477            root.insert_addressable_lease(key, lease.clone())
478        };
479
480        self.update_metrics("write", start.elapsed(), success).await;
481        Ok(())
482    }
483
484    async fn get_lease(
485        &self,
486        reference: &AddressableReference,
487    ) -> OrbitResult<Option<AddressableLease>> {
488        let start = Instant::now();
489        let key = Self::reference_to_key(reference);
490
491        let result = {
492            let root = self.root.read().unwrap();
493            match root.find_key_index(&key) {
494                Ok(idx) => root.addressable_values[idx].clone(),
495                Err(_) => None,
496            }
497        };
498
499        self.update_metrics("read", start.elapsed(), true).await;
500        Ok(result)
501    }
502
503    async fn update_lease(&self, lease: &AddressableLease) -> OrbitResult<()> {
504        let start = Instant::now();
505        let key = Self::reference_to_key(&lease.reference);
506
507        // Log to WAL
508        {
509            let mut wal_guard = self.wal.lock().await;
510            if let Some(ref mut wal) = *wal_guard {
511                wal.append_entry(WALOperation::UpdateLease {
512                    key: key.clone(),
513                    lease: lease.clone(),
514                })
515                .await?;
516            }
517        }
518
519        // Update in B+ tree
520        {
521            let mut root = self.root.write().unwrap();
522            root.insert_addressable_lease(key, lease.clone());
523        }
524
525        self.update_metrics("write", start.elapsed(), true).await;
526        Ok(())
527    }
528
529    async fn remove_lease(&self, reference: &AddressableReference) -> OrbitResult<bool> {
530        let start = Instant::now();
531        let key = Self::reference_to_key(reference);
532
533        // Log to WAL
534        {
535            let mut wal_guard = self.wal.lock().await;
536            if let Some(ref mut wal) = *wal_guard {
537                wal.append_entry(WALOperation::DeleteLease { key: key.clone() })
538                    .await?;
539            }
540        }
541
542        // Remove from B+ tree
543        let removed = {
544            let mut root = self.root.write().unwrap();
545            match root.find_key_index(&key) {
546                Ok(idx) => {
547                    root.keys.remove(idx);
548                    root.addressable_values.remove(idx);
549                    true
550                }
551                Err(_) => false,
552            }
553        };
554
555        self.update_metrics("delete", start.elapsed(), removed)
556            .await;
557        Ok(removed)
558    }
559
560    async fn list_node_leases(&self, node_id: &NodeId) -> OrbitResult<Vec<AddressableLease>> {
561        let start = Instant::now();
562
563        let leases = {
564            let root = self.root.read().unwrap();
565            let mut leases = Vec::new();
566            for lease in root.addressable_values.iter().flatten() {
567                if &lease.node_id == node_id {
568                    leases.push(lease.clone());
569                }
570            }
571            leases
572        };
573
574        self.update_metrics("read", start.elapsed(), true).await;
575        Ok(leases)
576    }
577
578    async fn list_all_leases(&self) -> OrbitResult<Vec<AddressableLease>> {
579        let start = Instant::now();
580
581        let leases = {
582            let root = self.root.read().unwrap();
583            root.addressable_values
584                .iter()
585                .filter_map(|v| v.clone())
586                .collect::<Vec<AddressableLease>>()
587        };
588
589        self.update_metrics("read", start.elapsed(), true).await;
590        Ok(leases)
591    }
592
593    async fn cleanup_expired_leases(&self) -> OrbitResult<u64> {
594        let start = Instant::now();
595        let now = chrono::Utc::now();
596
597        let count = {
598            let mut root = self.root.write().unwrap();
599            let mut indices_to_remove = Vec::new();
600            let mut count = 0;
601
602            // Find expired leases
603            for (i, value) in root.addressable_values.iter().enumerate() {
604                if let Some(lease) = value {
605                    if lease.expires_at < now {
606                        indices_to_remove.push(i);
607                    }
608                }
609            }
610
611            // Remove expired leases (in reverse order to maintain indices)
612            indices_to_remove.reverse();
613            for idx in indices_to_remove {
614                root.keys.remove(idx);
615                root.addressable_values.remove(idx);
616                count += 1;
617            }
618            count
619        };
620
621        self.update_metrics("delete", start.elapsed(), true).await;
622        Ok(count)
623    }
624
625    async fn store_leases_bulk(&self, leases: &[AddressableLease]) -> OrbitResult<()> {
626        let start = Instant::now();
627
628        for lease in leases {
629            self.store_lease(lease).await?;
630        }
631
632        self.update_metrics("write", start.elapsed(), true).await;
633        Ok(())
634    }
635
636    async fn remove_leases_bulk(&self, references: &[AddressableReference]) -> OrbitResult<u64> {
637        let start = Instant::now();
638        let mut count = 0;
639
640        for reference in references {
641            if self.remove_lease(reference).await? {
642                count += 1;
643            }
644        }
645
646        self.update_metrics("delete", start.elapsed(), true).await;
647        Ok(count)
648    }
649}

721impl ClusterNodeProvider for CowBTreeClusterProvider {
722    async fn store_node(&self, node: &NodeInfo) -> OrbitResult<()> {
723        let key = node.id.to_string();
724        let mut root = self.root.write().unwrap();
725        root.insert_node_info(key, node.clone());
726        Ok(())
727    }
728
729    async fn get_node(&self, node_id: &NodeId) -> OrbitResult<Option<NodeInfo>> {
730        let key = node_id.to_string();
731        let root = self.root.read().unwrap();
732        let result = match root.find_key_index(&key) {
733            Ok(idx) => root.node_values[idx].clone(),
734            Err(_) => None,
735        };
736        Ok(result)
737    }
738
739    async fn update_node(&self, node: &NodeInfo) -> OrbitResult<()> {
740        self.store_node(node).await
741    }
742
743    async fn remove_node(&self, node_id: &NodeId) -> OrbitResult<bool> {
744        let key = node_id.to_string();
745        let mut root = self.root.write().unwrap();
746        let removed = match root.find_key_index(&key) {
747            Ok(idx) => {
748                root.keys.remove(idx);
749                root.node_values.remove(idx);
750                true
751            }
752            Err(_) => false,
753        };
754        Ok(removed)
755    }
756
757    async fn list_active_nodes(&self) -> OrbitResult<Vec<NodeInfo>> {
758        let root = self.root.read().unwrap();
759        let nodes: Vec<NodeInfo> = root
760            .node_values
761            .iter()
762            .filter_map(|v| v.clone())
763            .filter(|node| node.status == NodeStatus::Active)
764            .collect();
765        Ok(nodes)
766    }
767
768    async fn list_all_nodes(&self) -> OrbitResult<Vec<NodeInfo>> {
769        let root = self.root.read().unwrap();
770        let nodes: Vec<NodeInfo> = root.node_values.iter().filter_map(|v| v.clone()).collect();
771        Ok(nodes)
772    }
773
774    async fn cleanup_expired_nodes(&self) -> OrbitResult<u64> {
775        let mut count = 0;
776        let mut root = self.root.write().unwrap();
777        let mut indices_to_remove = Vec::new();
778
779        for (i, value) in root.node_values.iter().enumerate() {
780            if let Some(node) = value {
781                if let Some(lease) = &node.lease {
782                    if lease.is_expired() {
783                        indices_to_remove.push(i);
784                    }
785                }
786            }
787        }
788
789        for &idx in indices_to_remove.iter().rev() {
790            root.keys.remove(idx);
791            root.node_values.remove(idx);
792            count += 1;
793        }
794
795        Ok(count)
796    }
797
798    async fn renew_node_lease(&self, node_id: &NodeId, lease: &NodeLease) -> OrbitResult<()> {
799        let key = node_id.to_string();
800        let mut root = self.root.write().unwrap();
801
802        if let Ok(idx) = root.find_key_index(&key) {
803            if let Some(node) = &mut root.node_values[idx] {
804                node.lease = Some(lease.clone());
805            }
806        }
807
808        Ok(())
809    }
810}
