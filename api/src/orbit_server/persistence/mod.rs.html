622/// Registry for managing multiple persistence providers
623pub struct PersistenceProviderRegistry {
624    addressable_providers: Arc<RwLock<HashMap<String, Arc<dyn AddressableDirectoryProvider>>>>,
625    cluster_providers: Arc<RwLock<HashMap<String, Arc<dyn ClusterNodeProvider>>>>,
626    default_addressable: Arc<RwLock<Option<String>>>,
627    default_cluster: Arc<RwLock<Option<String>>>,
628}
629
630impl PersistenceProviderRegistry {
631    pub fn new() -> Self {
632        Self {
633            addressable_providers: Arc::new(RwLock::new(HashMap::new())),
634            cluster_providers: Arc::new(RwLock::new(HashMap::new())),
635            default_addressable: Arc::new(RwLock::new(None)),
636            default_cluster: Arc::new(RwLock::new(None)),
637        }
638    }
639
640    /// Register an addressable directory provider
641    pub async fn register_addressable_provider(
642        &self,
643        name: String,
644        provider: Arc<dyn AddressableDirectoryProvider>,
645        is_default: bool,
646    ) -> OrbitResult<()> {
647        let mut providers = self.addressable_providers.write().await;
648        providers.insert(name.clone(), provider);
649
650        if is_default {
651            let mut default = self.default_addressable.write().await;
652            *default = Some(name);
653        }
654
655        Ok(())
656    }
657
658    /// Register a cluster node provider
659    pub async fn register_cluster_provider(
660        &self,
661        name: String,
662        provider: Arc<dyn ClusterNodeProvider>,
663        is_default: bool,
664    ) -> OrbitResult<()> {
665        let mut providers = self.cluster_providers.write().await;
666        providers.insert(name.clone(), provider);
667
668        if is_default {
669            let mut default = self.default_cluster.write().await;
670            *default = Some(name);
671        }
672
673        Ok(())
674    }
675
676    /// Get the default addressable directory provider
677    pub async fn get_default_addressable_provider(
678        &self,
679    ) -> OrbitResult<Arc<dyn AddressableDirectoryProvider>> {
680        let default_name = {
681            let default = self.default_addressable.read().await;
682            default.clone().ok_or_else(|| {
683                OrbitError::configuration("No default addressable provider configured")
684            })?
685        };
686
687        let providers = self.addressable_providers.read().await;
688        providers.get(&default_name).cloned().ok_or_else(|| {
689            OrbitError::configuration(format!(
690                "Default addressable provider '{}' not found",
691                default_name
692            ))
693        })
694    }
695
696    /// Get the default cluster node provider
697    pub async fn get_default_cluster_provider(&self) -> OrbitResult<Arc<dyn ClusterNodeProvider>> {
698        let default_name = {
699            let default = self.default_cluster.read().await;
700            default.clone().ok_or_else(|| {
701                OrbitError::configuration("No default cluster provider configured")
702            })?
703        };
704
705        let providers = self.cluster_providers.read().await;
706        providers.get(&default_name).cloned().ok_or_else(|| {
707            OrbitError::configuration(format!(
708                "Default cluster provider '{}' not found",
709                default_name
710            ))
711        })
712    }
713
714    /// Get a specific addressable provider by name
715    pub async fn get_addressable_provider(
716        &self,
717        name: &str,
718    ) -> OrbitResult<Arc<dyn AddressableDirectoryProvider>> {
719        let providers = self.addressable_providers.read().await;
720        providers.get(name).cloned().ok_or_else(|| {
721            OrbitError::configuration(format!("Addressable provider '{}' not found", name))
722        })
723    }
724
725    /// Get a specific cluster provider by name
726    pub async fn get_cluster_provider(
727        &self,
728        name: &str,
729    ) -> OrbitResult<Arc<dyn ClusterNodeProvider>> {
730        let providers = self.cluster_providers.read().await;
731        providers.get(name).cloned().ok_or_else(|| {
732            OrbitError::configuration(format!("Cluster provider '{}' not found", name))
733        })
734    }
735
736    /// List all registered provider names
737    pub async fn list_providers(&self) -> (Vec<String>, Vec<String>) {
738        let addressable = self.addressable_providers.read().await;
739        let cluster = self.cluster_providers.read().await;
740
741        (
742            addressable.keys().cloned().collect(),
743            cluster.keys().cloned().collect(),
744        )
745    }
