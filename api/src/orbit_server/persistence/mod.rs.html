26/// Configuration for different storage providers
27#[derive(Debug, Clone, Serialize, Deserialize)]
28pub enum PersistenceConfig {
29    /// In-memory storage (default, non-persistent)
30    Memory(MemoryConfig),
31    /// Amazon S3 compatible storage
32    S3(S3Config),
33    /// Digital Ocean Spaces object storage
34    DigitalOceanSpaces(DigitalOceanSpacesConfig),
35    /// Azure Blob Storage
36    Azure(AzureConfig),
37    /// Google Cloud Storage
38    GoogleCloud(GoogleCloudConfig),
39    /// etcd distributed key-value store
40    Etcd(EtcdConfig),
41    /// Redis in-memory data structure store
42    Redis(RedisConfig),
43    /// Kubernetes-based storage
44    Kubernetes(KubernetesConfig),
45    /// MinIO object storage
46    MinIO(MinIOConfig),
47    /// Flash-optimized storage with multipathing
48    Flash(FlashConfig),
49    /// Composite provider with primary/backup configuration
50    Composite(CompositeConfig),
51    /// Copy-on-Write B+ Tree with WAL
52    CowBTree(crate::persistence::cow_btree::CowBTreeConfig),
53    /// LSM-Tree with compaction
54    LsmTree(crate::persistence::lsm_tree::LsmTreeConfig),
55    /// RocksDB embedded database
56    RocksDB(crate::persistence::rocksdb::RocksDbConfig),
57}

270/// Core persistence provider trait
271#[async_trait]
272pub trait PersistenceProvider: Send + Sync {
273    /// Initialize the provider
274    async fn initialize(&self) -> OrbitResult<()>;
275
276    /// Shutdown the provider gracefully
277    async fn shutdown(&self) -> OrbitResult<()>;
278
279    /// Check health status
280    async fn health_check(&self) -> ProviderHealth;
281
282    /// Get current metrics
283    async fn metrics(&self) -> PersistenceMetrics;
284
285    /// Begin a transaction
286    async fn begin_transaction(&self, context: TransactionContext) -> OrbitResult<String>;
287
288    /// Commit a transaction
289    async fn commit_transaction(&self, transaction_id: &str) -> OrbitResult<()>;
290
291    /// Rollback a transaction
292    async fn rollback_transaction(&self, transaction_id: &str) -> OrbitResult<()>;
293}

355/// Registry for managing multiple persistence providers
356pub struct PersistenceProviderRegistry {
357    addressable_providers: Arc<RwLock<HashMap<String, Arc<dyn AddressableDirectoryProvider>>>>,
358    cluster_providers: Arc<RwLock<HashMap<String, Arc<dyn ClusterNodeProvider>>>>,
359    default_addressable: Arc<RwLock<Option<String>>>,
360    default_cluster: Arc<RwLock<Option<String>>>,
361}
362
363impl PersistenceProviderRegistry {
364    pub fn new() -> Self {
365        Self {
366            addressable_providers: Arc::new(RwLock::new(HashMap::new())),
367            cluster_providers: Arc::new(RwLock::new(HashMap::new())),
368            default_addressable: Arc::new(RwLock::new(None)),
369            default_cluster: Arc::new(RwLock::new(None)),
370        }
371    }
372
373    /// Register an addressable directory provider
374    pub async fn register_addressable_provider(
375        &self,
376        name: String,
377        provider: Arc<dyn AddressableDirectoryProvider>,
378        is_default: bool,
379    ) -> OrbitResult<()> {
380        let mut providers = self.addressable_providers.write().await;
381        providers.insert(name.clone(), provider);
382
383        if is_default {
384            let mut default = self.default_addressable.write().await;
385            *default = Some(name);
386        }
387
388        Ok(())
389    }
390
391    /// Register a cluster node provider
392    pub async fn register_cluster_provider(
393        &self,
394        name: String,
395        provider: Arc<dyn ClusterNodeProvider>,
396        is_default: bool,
397    ) -> OrbitResult<()> {
398        let mut providers = self.cluster_providers.write().await;
399        providers.insert(name.clone(), provider);
400
401        if is_default {
402            let mut default = self.default_cluster.write().await;
403            *default = Some(name);
404        }
405
406        Ok(())
407    }
408
409    /// Get the default addressable directory provider
410    pub async fn get_default_addressable_provider(
411        &self,
412    ) -> OrbitResult<Arc<dyn AddressableDirectoryProvider>> {
413        let default_name = {
414            let default = self.default_addressable.read().await;
415            default.clone().ok_or_else(|| {
416                OrbitError::configuration("No default addressable provider configured")
417            })?
418        };
419
420        let providers = self.addressable_providers.read().await;
421        providers.get(&default_name).cloned().ok_or_else(|| {
422            OrbitError::configuration(format!(
423                "Default addressable provider '{}' not found",
424                default_name
425            ))
426        })
427    }
428
429    /// Get the default cluster node provider
430    pub async fn get_default_cluster_provider(&self) -> OrbitResult<Arc<dyn ClusterNodeProvider>> {
431        let default_name = {
432            let default = self.default_cluster.read().await;
433            default.clone().ok_or_else(|| {
434                OrbitError::configuration("No default cluster provider configured")
435            })?
436        };
437
438        let providers = self.cluster_providers.read().await;
439        providers.get(&default_name).cloned().ok_or_else(|| {
440            OrbitError::configuration(format!(
441                "Default cluster provider '{}' not found",
442                default_name
443            ))
444        })
445    }
446
447    /// Get a specific addressable provider by name
448    pub async fn get_addressable_provider(
449        &self,
450        name: &str,
451    ) -> OrbitResult<Arc<dyn AddressableDirectoryProvider>> {
452        let providers = self.addressable_providers.read().await;
453        providers.get(name).cloned().ok_or_else(|| {
454            OrbitError::configuration(format!("Addressable provider '{}' not found", name))
455        })
456    }
457
458    /// Get a specific cluster provider by name
459    pub async fn get_cluster_provider(
460        &self,
461        name: &str,
462    ) -> OrbitResult<Arc<dyn ClusterNodeProvider>> {
463        let providers = self.cluster_providers.read().await;
464        providers.get(name).cloned().ok_or_else(|| {
465            OrbitError::configuration(format!("Cluster provider '{}' not found", name))
466        })
467    }
468
469    /// List all registered provider names
470    pub async fn list_providers(&self) -> (Vec<String>, Vec<String>) {
471        let addressable = self.addressable_providers.read().await;
472        let cluster = self.cluster_providers.read().await;
473
474        (
475            addressable.keys().cloned().collect(),
476            cluster.keys().cloned().collect(),
477        )
478    }
479}
480
481impl Default for PersistenceProviderRegistry {
482    fn default() -> Self {
483        Self::new()
484    }
