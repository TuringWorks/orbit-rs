178/// Main Orbit server for hosting actors and managing cluster membership
179pub struct OrbitServer {
180    config: OrbitServerConfig,
181    node_info: NodeInfo,
182    cluster_manager: ClusterManager,
183    addressable_directory: AddressableDirectory,
184    #[allow(dead_code)]
185    load_balancer: LoadBalancer,
186    connection_service: OrbitConnectionService,
187    health_service: OrbitHealthService,
188    #[allow(dead_code)]
189    persistence_registry: Arc<PersistenceProviderRegistry>,
190    /// PostgreSQL wire protocol server (initialized at startup)
191    postgres_server: Option<PostgresServer>,
192}

194impl OrbitServer {
195    /// Create a new Orbit server with the given configuration
196    pub async fn new(config: OrbitServerConfig) -> OrbitResult<Self> {
197        let node_id = NodeId::generate(config.namespace.clone());
198
199        let capabilities = NodeCapabilities {
200            addressable_types: vec![], // Will be populated as actors are registered
201            max_addressables: config.max_addressables,
202            tags: config.tags.clone(),
203        };
204
205        let node_info = NodeInfo {
206            id: node_id,
207            url: config.bind_address.clone(),
208            port: config.port,
209            capabilities,
210            status: NodeStatus::Active,
211            lease: None,
212        };
213
214        // Initialize persistence provider registry
215        let persistence_registry = Arc::new(config.persistence.create_registry().await?);
216
217        let cluster_manager = ClusterManager::new(config.lease_duration);
218        let addressable_directory = AddressableDirectory::new();
219        let load_balancer = LoadBalancer::new();
220        let connection_service = OrbitConnectionService::new();
221        let health_service = OrbitHealthService::new();
222
223        // PostgreSQL server can be initialized with persistent storage if available
224        let postgres_server = if config.protocols.postgres_enabled {
225            let postgres_addr = format!(
226                "{}:{}",
227                config.protocols.postgres_bind_address, config.protocols.postgres_port
228            );
229
230            // Try to create QueryEngine with persistent storage if RocksDB is available
231            let query_engine = {
232                // Check if we have a RocksDB provider configured
233                // The addressable provider is registered as "rocksdb_addressable"
234                match persistence_registry
235                    .get_addressable_provider("rocksdb_addressable")
236                    .await
237                {
238                    Ok(_provider) => {
239                        // Create a separate RocksDB instance for PostgreSQL table storage
240                        // Use the same data directory structure as the server but with postgresql subdirectory
241                        let pg_data_path = "./orbit_integrated_data/postgresql";
242
243                        match RocksDbTableStorage::new(&pg_data_path) {
244                            Ok(storage) => {
245                                tracing::info!(
246                                    "PostgreSQL using persistent RocksDB storage at: {}",
247                                    pg_data_path
248                                );
249                                QueryEngine::new_with_persistent_storage(Arc::new(storage))
250                            }
251                            Err(e) => {
252                                tracing::warn!("Failed to create PostgreSQL persistent storage, falling back to memory: {}", e);
253                                QueryEngine::new()
254                            }
255                        }
256                    }
257                    Err(_) => {
258                        tracing::info!("PostgreSQL using in-memory storage (no RocksDB persistence configured)");
259                        QueryEngine::new()
260                    }
261                }
262            };
263
264            Some(PostgresServer::new_with_query_engine(
265                postgres_addr,
266                query_engine,
267            ))
268        } else {
269            None
270        };
271
272        // RESP server will be created after gRPC server starts
273
274        let server = Self {
275            config,
276            node_info,
277            cluster_manager,
278            addressable_directory,
279            load_balancer,
280            connection_service,
281            health_service,
282            persistence_registry,
283            postgres_server,
284        };
285
286        Ok(server)
287    }

294    /// Start the server and begin accepting connections
295    pub async fn start(&mut self) -> OrbitResult<()> {
296        // Register this node in the cluster
297        self.cluster_manager
298            .register_node(self.node_info.clone())
299            .await?;
300
301        // Start background tasks
302        self.start_background_tasks().await;
303
304        // Create futures for all servers
305        let mut server_tasks = Vec::new();
306
307        // Start gRPC server (non-blocking)
308        let grpc_addr = format!("{}:{}", self.config.bind_address, self.config.port)
309            .parse()
310            .map_err(|e| OrbitError::configuration(format!("Invalid bind address: {}", e)))?;
311
312        tracing::info!("Starting Orbit gRPC server on {}", grpc_addr);
313
314        let grpc_server = Server::builder()
315            .add_service(connection_service_server::ConnectionServiceServer::new(
316                self.connection_service.clone(),
317            ))
318            .add_service(health_service_server::HealthServiceServer::new(
319                self.health_service.clone(),
320            ))
321            .serve(grpc_addr);
322
323        server_tasks.push(tokio::spawn(async move {
324            if let Err(e) = grpc_server.await {
325                tracing::error!("gRPC server failed: {}", e);
326            }
327        }));
328
329        // Give the gRPC server a moment to start
330        tokio::time::sleep(Duration::from_millis(100)).await;
331
332        // Start RESP server if enabled (after gRPC server is ready)
333        if self.config.protocols.redis_enabled {
334            let redis_addr = format!(
335                "{}:{}",
336                self.config.protocols.redis_bind_address, self.config.protocols.redis_port
337            );
338            tracing::info!("Starting RESP (Redis) server on {}", redis_addr);
339
340            // Create OrbitClient that connects to this server
341            let server_url = format!("http://{}:{}", self.config.bind_address, self.config.port);
342            match OrbitClient::builder()
343                .with_namespace(self.config.namespace.clone())
344                .with_server_urls(vec![server_url])
345                .build()
346                .await
347            {
348                Ok(client) => {
349                    let resp_server = RespServer::new(redis_addr, client);
350                    server_tasks.push(tokio::spawn(async move {
351                        if let Err(e) = resp_server.run().await {
352                            tracing::error!("RESP server failed: {}", e);
353                        }
354                    }));
355                }
356                Err(e) => {
357                    tracing::error!("Failed to create OrbitClient for RESP server: {}", e);
358                }
359            }
360        }
361
362        // Start PostgreSQL server if enabled
363        if let Some(postgres_server) = self.postgres_server.take() {
364            let postgres_addr = format!(
365                "{}:{}",
366                self.config.protocols.postgres_bind_address, self.config.protocols.postgres_port
367            );
368            tracing::info!("Starting PostgreSQL server on {}", postgres_addr);
369
370            server_tasks.push(tokio::spawn(async move {
371                if let Err(e) = postgres_server.run().await {
372                    tracing::error!("PostgreSQL server failed: {}", e);
373                }
374            }));
375        }
376
377        tracing::info!("All Orbit servers started successfully");
378        tracing::info!(
379            "  - gRPC: {}:{}",
380            self.config.bind_address,
381            self.config.port
382        );
383        if self.config.protocols.redis_enabled {
384            tracing::info!(
385                "  - Redis: {}:{}",
386                self.config.protocols.redis_bind_address,
387                self.config.protocols.redis_port
388            );
389        }
390        if self.config.protocols.postgres_enabled {
391            tracing::info!(
392                "  - PostgreSQL: {}:{}",
393                self.config.protocols.postgres_bind_address,
394                self.config.protocols.postgres_port
395            );
396        }
397
398        // Wait for any server to finish (which likely means an error occurred)
399        if !server_tasks.is_empty() {
400            let (result, _index, _remaining) = futures::future::select_all(server_tasks).await;
401            if let Err(e) = result {
402                tracing::error!("One of the servers failed: {}", e);
403                return Err(OrbitError::network(format!("Server task failed: {}", e)));
404            }
405        }
406
407        Ok(())
408    }

415    /// Register an addressable type that this server can host
416    pub async fn register_addressable_type(&mut self, addressable_type: String) -> OrbitResult<()> {
417        if !self
418            .node_info
419            .capabilities
420            .addressable_types
421            .contains(&addressable_type)
422        {
423            self.node_info
424                .capabilities
425                .addressable_types
426                .push(addressable_type);
427
428            // Update cluster with new capabilities
429            self.cluster_manager
430                .update_node(self.node_info.clone())
431                .await?;
432        }
433        Ok(())
434    }

436    /// Get server statistics
437    pub async fn stats(&self) -> OrbitResult<ServerStats> {
438        let cluster_stats = self.cluster_manager.stats().await;
439        let directory_stats = self.addressable_directory.stats().await;
440
441        let protocol_stats = ProtocolStats {
442            redis_enabled: self.config.protocols.redis_enabled,
443            redis_address: if self.config.protocols.redis_enabled {
444                Some(format!(
445                    "{}:{}",
446                    self.config.protocols.redis_bind_address, self.config.protocols.redis_port
447                ))
448            } else {
449                None
450            },
451            postgres_enabled: self.config.protocols.postgres_enabled,
452            postgres_address: if self.config.protocols.postgres_enabled {
453                Some(format!(
454                    "{}:{}",
455                    self.config.protocols.postgres_bind_address,
456                    self.config.protocols.postgres_port
457                ))
458            } else {
459                None
460            },
461        };
462
463        Ok(ServerStats {
464            node_id: self.node_info.id.clone(),
465            cluster_stats,
466            directory_stats,
467            supported_types: self.node_info.capabilities.addressable_types.clone(),
468            protocol_stats,
469        })
470    }

472    /// Shutdown the server gracefully
473    pub async fn shutdown(&self) -> OrbitResult<()> {
474        // Set health status to not serving
475        self.health_service
476            .set_serving_status(health_check_response::ServingStatus::NotServing)
477            .await;
478
479        // Remove from cluster
480        self.cluster_manager.remove_node(&self.node_info.id).await?;
481
482        tracing::info!("Orbit server shutdown completed");
483        Ok(())
484    }
485
486    async fn start_background_tasks(&self) {
487        let cluster_manager = self.cluster_manager.clone();
488        let addressable_directory = self.addressable_directory.clone();
489        let cleanup_interval = self.config.cleanup_interval;
490
491        // Cluster and directory cleanup task
492        tokio::spawn(async move {
493            let mut interval = interval(cleanup_interval);
494            loop {
495                interval.tick().await;
496
497                if let Err(e) = cluster_manager.cleanup_expired_nodes().await {
498                    tracing::warn!("Failed to cleanup expired nodes: {}", e);
499                }
500
501                if let Err(e) = addressable_directory.cleanup_expired_leases().await {
502                    tracing::warn!("Failed to cleanup expired leases: {}", e);
503                }
504            }
505        });
506
507        // Start cluster manager background tasks
508        self.cluster_manager.start_background_tasks().await;
509    }
