Using Kubernetes Operator


# orbit-cluster.yaml
apiVersion: orbit.turingworks.com/v1alpha1
kind: OrbitCluster
metadata:
  name: production-cluster
spec:
  size: 5
  version: "0.1.0"
  storage:
    size: 500Gi
    storageClass: fast-ssd
  monitoring:
    prometheus: true
    grafana: true
  protocols:
    redis: 
      enabled: true
      port: 6379
    postgresql:
      enabled: true  
      port: 5432

kubectl apply -f orbit-cluster.yaml
