1//! Vector Store Example
2//! 
3//! Demonstrates using Orbit-RS as a vector database with Redis protocol compatibility.
4//! This example shows how to:
5//! - Start a Redis-compatible vector store server
6//! - Add high-dimensional vectors with metadata
7//! - Perform similarity searches and k-nearest neighbors queries
8//! - Use both Redis FT.* commands and simplified VECTOR.* commands
9
10use std::collections::HashMap;
11use tokio::time::{sleep, Duration};
12use tracing::info;
13use orbit_client::OrbitClientBuilder;
14use orbit_protocols::{
15    resp::server::RespServer,
16    vector_store::{Vector, VectorIndexConfig, SimilarityMetric, VectorActor, VectorStats}
17};
18use orbit_shared::Key;
19
20#[tokio::main]
21async fn main() -> Result<(), Box<dyn std::error::Error>> {
22    // Initialize tracing
23    tracing_subscriber::fmt()
24        .with_env_filter("vector_store=debug,orbit=info")
25        .init();
26
27    info!("🚀 Starting Orbit Vector Store Example");
28
29    // Create OrbitClient
30    let orbit_client = OrbitClientBuilder::new()
31        .with_server_urls(vec!["http://127.0.0.1:50051".to_string()])
32        .build()
33        .await
34        .map_err(|e| format!("Failed to create OrbitClient: {}", e))?;
35
36    info!("✅ Connected to Orbit cluster");
37
38    // Set up vector store with sample data
39    setup_vector_data(&orbit_client).await?;
40
41    // Start RESP server for Redis compatibility
42    let addr = "127.0.0.1:6381"; // Different port to avoid conflict
43    info!("🎯 Vector Store server listening on {}", addr);
44    info!("🔗 Connect with: redis-cli -h 127.0.0.1 -p 6381");
45
46    // Create and start server
47    let server = RespServer::new(addr, orbit_client);
48
49    // Print usage examples
50    print_usage_examples();
51
52    // Start server
53    server.run().await?;
54
55    Ok(())
56}

58async fn setup_vector_data(orbit_client: &orbit_client::OrbitClient) -> Result<(), Box<dyn std::error::Error>> {
59    info!("📊 Setting up sample vector data...");
60    
61    // Get a VectorActor for our examples
62    let vector_actor_ref = orbit_client.actor_reference::<VectorActor>(
63        Key::StringKey { 
64            key: "ml-embeddings".to_string() 
65        }
66    ).await.map_err(|e| format!("Failed to get actor reference: {}", e))?;
67
68    // Create a vector index for ML embeddings (384 dimensions - typical for sentence transformers)
69    let index_config = VectorIndexConfig::new(
70        "ml-embeddings".to_string(),
71        384,
72        SimilarityMetric::Cosine
73    );
74    
75    let config_value = serde_json::to_value(index_config)?;
76    vector_actor_ref.invoke("create_index", vec![config_value])
77        .await.map_err(|e| format!("Failed to create index: {}", e))?;
78
79    // Add some sample vectors (simulated embeddings)
80    let sample_vectors = vec![
81        (
82            "doc1",
83            create_sample_vector(384, 0.1),
84            HashMap::from([
85                ("title".to_string(), "Introduction to Machine Learning".to_string()),
86                ("category".to_string(), "AI".to_string()),
87                ("author".to_string(), "John Doe".to_string()),
88            ])
89        ),
90        (
91            "doc2", 
92            create_sample_vector(384, 0.2),
93            HashMap::from([
94                ("title".to_string(), "Deep Learning Fundamentals".to_string()),
95                ("category".to_string(), "AI".to_string()),
96                ("author".to_string(), "Jane Smith".to_string()),
97            ])
98        ),
99        (
100            "doc3",
101            create_sample_vector(384, 0.3),
102            HashMap::from([
103                ("title".to_string(), "Neural Network Architectures".to_string()),
104                ("category".to_string(), "AI".to_string()),
105                ("author".to_string(), "Bob Johnson".to_string()),
106            ])
107        ),
108        (
109            "doc4",
110            create_sample_vector(384, 0.4),
111            HashMap::from([
112                ("title".to_string(), "Computer Vision Applications".to_string()),
113                ("category".to_string(), "CV".to_string()),
114                ("author".to_string(), "Alice Brown".to_string()),
115            ])
116        ),
117        (
118            "doc5",
119            create_sample_vector(384, 0.5),
120            HashMap::from([
121                ("title".to_string(), "Natural Language Processing".to_string()),
122                ("category".to_string(), "NLP".to_string()),
123                ("author".to_string(), "Charlie Wilson".to_string()),
124            ])
125        ),
126    ];
127
128    // Add vectors to the store
129    for (id, data, metadata) in sample_vectors {
130        let vector = Vector::with_metadata(id.to_string(), data, metadata);
131        let vector_value = serde_json::to_value(vector)?;
132        
133        vector_actor_ref.invoke("add_vector", vec![vector_value])
134            .await.map_err(|e| format!("Failed to add vector {}: {}", id, e))?;
135        
136        info!("📝 Added vector: {}", id);
137    }
138
139    // Sleep briefly to ensure vectors are stored
140    sleep(Duration::from_millis(100)).await;
141
142    // Get vector statistics
143    let stats: VectorStats = vector_actor_ref.invoke("get_stats", vec![])
144        .await.map_err(|e| format!("Failed to get stats: {}", e))?;
145
146    info!("📈 Vector store statistics:");
147    info!("   • Vector count: {}", stats.vector_count);
148    info!("   • Index count: {}", stats.index_count);
149    info!("   • Average dimension: {:.1}", stats.avg_dimension);
150    info!("   • Dimension range: {} - {}", stats.min_dimension, stats.max_dimension);
151
152    Ok(())
153}

155fn create_sample_vector(dimension: usize, seed: f32) -> Vec<f32> {
156    // Create a simple deterministic vector for demonstration
157    (0..dimension)
158        .map(|i| (seed + (i as f32 * 0.01)).sin() * 0.5)
159        .collect()
160}

162fn print_usage_examples() {
163    info!("🎯 Vector Store Ready! Try these Redis commands:");
164    info!("");
165    info!("📋 Basic Vector Operations:");
166    info!("   VECTOR.ADD my-vectors doc6 \"0.1,0.2,0.3,...\"");
167    info!("   VECTOR.GET my-vectors doc1");
168    info!("   VECTOR.DEL my-vectors doc1");
169    info!("   VECTOR.STATS my-vectors");
170    info!("");
171    info!("🔍 Vector Search:");
172    info!("   VECTOR.SEARCH ml-embeddings \"0.1,0.2,0.3,...\" 5");
173    info!("   VECTOR.KNN ml-embeddings \"0.1,0.2,0.3,...\" 3");
174    info!("");
175    info!("🚀 Redis FT Commands (Advanced):");
176    info!("   FT.CREATE my-index DIM 384 DISTANCE_METRIC COSINE");
177    info!("   FT.ADD my-index vec1 \"0.1,0.2,0.3,...\" title \"My Document\"");
178    info!("   FT.SEARCH my-index \"0.1,0.2,0.3,...\" 10 DISTANCE_METRIC COSINE");
179    info!("   FT.INFO my-index");
180    info!("   FT.DEL my-index vec1");
181    info!("");
182    info!("💡 Sample query with pre-loaded data:");
183    info!("   VECTOR.SEARCH ml-embeddings \"{}\" 3", 
184          create_sample_vector(384, 0.15).iter()
185              .take(10)
186              .map(|f| format!("{:.3}", f))
187              .collect::<Vec<_>>()
188              .join(","));
189    info!("");
190    info!("📊 Distance Metrics:");
191    info!("   • COSINE: Cosine similarity (default)");
192    info!("   • L2: Euclidean distance");
193    info!("   • IP: Inner product (dot product)");
194    info!("   • L1: Manhattan distance");
195    info!("");
196}
