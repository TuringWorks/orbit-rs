1//! Cost-Based Query Optimization

16/// Represents the estimated cost of a query operation
17#[derive(Debug, Clone, PartialEq)]
18pub struct QueryCost {
19    /// CPU cost (operations)
20    pub cpu_cost: f64,
21    /// I/O cost (page reads)
22    pub io_cost: f64,
23    /// Memory cost (bytes)
24    pub memory_cost: f64,
25    /// Network cost (for distributed queries)
26    pub network_cost: f64,
27    /// Estimated result cardinality
28    pub cardinality: usize,
29}
30
31impl QueryCost {
32    /// Create a new query cost
33    pub fn new(
34        cpu_cost: f64,
35        io_cost: f64,
36        memory_cost: f64,
37        network_cost: f64,
38        cardinality: usize,
39    ) -> Self {
40        Self {
41            cpu_cost,
42            io_cost,
43            memory_cost,
44            network_cost,
45            cardinality,
46        }
47    }
48
49    /// Calculate total cost (weighted sum)
50    pub fn total_cost(&self) -> f64 {
51        // Default weights - could be configurable
52        const CPU_WEIGHT: f64 = 1.0;
53        const IO_WEIGHT: f64 = 10.0;
54        const MEMORY_WEIGHT: f64 = 0.1;
55        const NETWORK_WEIGHT: f64 = 100.0;
56
57        self.cpu_cost * CPU_WEIGHT
58            + self.io_cost * IO_WEIGHT
59            + self.memory_cost * MEMORY_WEIGHT
60            + self.network_cost * NETWORK_WEIGHT
61    }
62
63    /// Combine two costs (e.g., for joins)
64    pub fn combine(&self, other: &QueryCost) -> QueryCost {
65        QueryCost {
66            cpu_cost: self.cpu_cost + other.cpu_cost,
67            io_cost: self.io_cost + other.io_cost,
68            memory_cost: self.memory_cost.max(other.memory_cost), // Max for memory
69            network_cost: self.network_cost + other.network_cost,
70            cardinality: self.cardinality * other.cardinality, // Estimated join result
71        }
72    }
73}

83    /// Apply cost-based optimizations to a statement
84    pub fn optimize(
85        &mut self,
86        statement: Statement,
87        stats: &StatisticsCollector,
88    ) -> ProtocolResult<Statement> {
89        match statement {
90            Statement::Select(select) => {
91                let optimized_select = self.optimize_select(select, stats)?;
92                Ok(Statement::Select(optimized_select))
93            }
94            // For other statement types, return as-is for now
95            other => Ok(other),
96        }
97    }

99    /// Apply cost-based optimizations with step tracking
100    pub fn optimize_with_steps(
101        &mut self,
102        statement: Statement,
103        stats: &StatisticsCollector,
104    ) -> ProtocolResult<(Statement, Vec<String>)> {
105        let mut steps = Vec::new();
106
107        match statement {
108            Statement::Select(select) => {
109                let (optimized_select, select_steps) =
110                    self.optimize_select_with_steps(select, stats)?;
111                steps.extend(select_steps);
112                Ok((Statement::Select(optimized_select), steps))
113            }
114            other => Ok((other, steps)),
115        }
116    }

118    /// Optimize SELECT statement using cost-based techniques
119    fn optimize_select(
120        &mut self,
121        select: SelectStatement,
122        stats: &StatisticsCollector,
123    ) -> ProtocolResult<SelectStatement> {
124        let mut optimized = select;
125
126        // 1. Join reordering based on estimated costs
127        if self.config.enable_join_reorder {
128            optimized = self.optimize_join_order(optimized, stats)?;
129        }
130
131        // 2. Access method selection (table scan vs index scan)
132        optimized = self.optimize_access_methods(optimized, stats)?;
133
134        // 3. Join algorithm selection (nested loop, hash join, merge join)
135        optimized = self.optimize_join_algorithms(optimized, stats)?;
136
137        Ok(optimized)
138    }

205    /// Estimate the cost of a table scan
206    pub fn estimate_table_scan_cost(
207        &self,
208        table_name: &str,
209        stats: &StatisticsCollector,
210    ) -> QueryCost {
211        if let Some(table_stats) = stats.get_table_stats(table_name) {
212            let pages = (table_stats.row_count as f64 / table_stats.rows_per_page as f64).ceil();
213            QueryCost::new(
214                table_stats.row_count as f64, // CPU cost proportional to rows
215                pages,                        // I/O cost proportional to pages
216                0.0,                          // Memory cost minimal for scan
217                0.0,                          // No network cost for local scan
218                table_stats.row_count,
219            )
220        } else {
221            // Default estimates for unknown tables
222            QueryCost::new(1000.0, 100.0, 0.0, 0.0, 1000)
223        }
224    }

226    /// Estimate the cost of an index scan
227    pub fn estimate_index_scan_cost(
228        &self,
229        table_name: &str,
230        index_name: &str,
231        selectivity: f64,
232        stats: &StatisticsCollector,
233    ) -> QueryCost {
234        if let Some(table_stats) = stats.get_table_stats(table_name) {
235            let estimated_rows = (table_stats.row_count as f64 * selectivity) as usize;
236            let index_pages = (estimated_rows as f64 / 100.0).ceil(); // Assume 100 rows per index page
237
238            QueryCost::new(
239                estimated_rows as f64 * 2.0, // Slightly higher CPU cost for index traversal
240                index_pages + (estimated_rows as f64 / table_stats.rows_per_page as f64), // Index pages + data pages
241                0.0,
242                0.0,
243                estimated_rows,
244            )
245        } else {
246            // Default estimates
247            let estimated_rows = (1000.0 * selectivity) as usize;
248            QueryCost::new(
249                estimated_rows as f64 * 2.0,
250                estimated_rows as f64 / 10.0,
251                0.0,
252                0.0,
253                estimated_rows,
254            )
255        }
256    }

258    /// Estimate the cost of a nested loop join
259    pub fn estimate_nested_loop_join_cost(
260        &self,
261        left_cost: &QueryCost,
262        right_cost: &QueryCost,
263    ) -> QueryCost {
264        QueryCost::new(
265            left_cost.cpu_cost + (left_cost.cardinality as f64 * right_cost.cpu_cost),
266            left_cost.io_cost + (left_cost.cardinality as f64 * right_cost.io_cost),
267            left_cost.memory_cost.max(right_cost.memory_cost),
268            left_cost.network_cost + right_cost.network_cost,
269            (left_cost.cardinality as f64 * right_cost.cardinality as f64 * 0.1) as usize, // Assume 10% join selectivity
270        )
271    }

273    /// Estimate the cost of a hash join
274    pub fn estimate_hash_join_cost(
275        &self,
276        left_cost: &QueryCost,
277        right_cost: &QueryCost,
278    ) -> QueryCost {
279        // Hash join: build hash table from smaller relation, probe with larger
280        let (build_cost, probe_cost) = if left_cost.cardinality <= right_cost.cardinality {
281            (left_cost, right_cost)
282        } else {
283            (right_cost, left_cost)
284        };
285
286        QueryCost::new(
287            build_cost.cpu_cost
288                + probe_cost.cpu_cost
289                + (build_cost.cardinality + probe_cost.cardinality) as f64,
290            build_cost.io_cost + probe_cost.io_cost,
291            build_cost.cardinality as f64 * 8.0, // Memory for hash table (8 bytes per row estimate)
292            build_cost.network_cost + probe_cost.network_cost,
293            (build_cost.cardinality as f64 * probe_cost.cardinality as f64 * 0.1) as usize,
294        )
295    }

315    /// Estimate selectivity of a predicate
316    pub fn estimate_selectivity(&self, expr: &Expression, stats: &StatisticsCollector) -> f64 {
317        match expr {
318            Expression::Binary {
319                left: _,
320                operator,
321                right: _,
322            } => {
323                match operator {
324                    BinaryOperator::Equal => 0.1, // Equality typically has good selectivity
325                    BinaryOperator::NotEqual => 0.9,
326                    BinaryOperator::LessThan
327                    | BinaryOperator::LessThanOrEqual
328                    | BinaryOperator::GreaterThan
329                    | BinaryOperator::GreaterThanOrEqual => 0.33,
330                    BinaryOperator::And => 0.5, // Conservative estimate
331                    BinaryOperator::Or => 0.75,
332                    _ => 0.5, // Default
333                }
334            }
335            Expression::Unary {
336                operator: UnaryOperator::Not,
337                ..
338            } => 0.9, // NOT is usually high selectivity
339            _ => 0.5, // Default selectivity
340        }
341    }
