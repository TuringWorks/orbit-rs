16/// Represents the estimated cost of a query operation
17#[derive(Debug, Clone, PartialEq)]
18pub struct QueryCost {
19    /// CPU cost (operations)
20    pub cpu_cost: f64,
21    /// I/O cost (page reads)
22    pub io_cost: f64,
23    /// Memory cost (bytes)
24    pub memory_cost: f64,
25    /// Network cost (for distributed queries)
26    pub network_cost: f64,
27    /// Estimated result cardinality
28    pub cardinality: usize,
29}
30
31impl QueryCost {
32    /// Create a new query cost
33    pub fn new(cpu_cost: f64, io_cost: f64, memory_cost: f64, network_cost: f64, cardinality: usize) -> Self {
34        Self {
35            cpu_cost,
36            io_cost,
37            memory_cost,
38            network_cost,
39            cardinality,
40        }
41    }
42
43    /// Calculate total cost (weighted sum)
44    pub fn total_cost(&self) -> f64 {
45        // Default weights - could be configurable
46        const CPU_WEIGHT: f64 = 1.0;
47        const IO_WEIGHT: f64 = 10.0;
48        const MEMORY_WEIGHT: f64 = 0.1;
49        const NETWORK_WEIGHT: f64 = 100.0;
50
51        self.cpu_cost * CPU_WEIGHT +
52        self.io_cost * IO_WEIGHT +
53        self.memory_cost * MEMORY_WEIGHT +
54        self.network_cost * NETWORK_WEIGHT
55    }
56
57    /// Combine two costs (e.g., for joins)
58    pub fn combine(&self, other: &QueryCost) -> QueryCost {
59        QueryCost {
60            cpu_cost: self.cpu_cost + other.cpu_cost,
61            io_cost: self.io_cost + other.io_cost,
62            memory_cost: self.memory_cost.max(other.memory_cost), // Max for memory
63            network_cost: self.network_cost + other.network_cost,
64            cardinality: self.cardinality * other.cardinality, // Estimated join result
65        }
66    }
67}

79        match statement {
80            Statement::Select(select) => {
81                let optimized_select = self.optimize_select(select, stats)?;
82                Ok(Statement::Select(optimized_select))
83            }
84            // For other statement types, return as-is for now
85            other => Ok(other),
86        }
87    }

89    /// Apply cost-based optimizations with step tracking
90    pub fn optimize_with_steps(&mut self, statement: Statement, stats: &StatisticsCollector) -> ProtocolResult<(Statement, Vec<String>)> {
91        let mut steps = Vec::new();
92
93        match statement {
94            Statement::Select(select) => {
95                let (optimized_select, select_steps) = self.optimize_select_with_steps(select, stats)?;
96                steps.extend(select_steps);
97                Ok((Statement::Select(optimized_select), steps))
98            }
99            other => Ok((other, steps)),
100        }

103    /// Optimize SELECT statement using cost-based techniques
104    fn optimize_select(&mut self, select: SelectStatement, stats: &StatisticsCollector) -> ProtocolResult<SelectStatement> {
105        let mut optimized = select;
106
107        // 1. Join reordering based on estimated costs
108        if self.config.enable_join_reorder {
109            optimized = self.optimize_join_order(optimized, stats)?;
110        }
111
112        // 2. Access method selection (table scan vs index scan)
113        optimized = self.optimize_access_methods(optimized, stats)?;
114
115        // 3. Join algorithm selection (nested loop, hash join, merge join)
116        optimized = self.optimize_join_algorithms(optimized, stats)?;
117
118        Ok(optimized)
119    }

126        // 1. Join reordering

134
135        // 2. Access method optimization
136        let before = optimized.clone();

139            steps.push("Optimized access methods".to_string());
140        }
141
142        Ok((optimized, steps))

170    /// Estimate the cost of a table scan
171    pub fn estimate_table_scan_cost(&self, table_name: &str, stats: &StatisticsCollector) -> QueryCost {
172        if let Some(table_stats) = stats.get_table_stats(table_name) {
173            let pages = (table_stats.row_count as f64 / table_stats.rows_per_page as f64).ceil();
174            QueryCost::new(
175                table_stats.row_count as f64, // CPU cost proportional to rows
176                pages, // I/O cost proportional to pages
177                0.0, // Memory cost minimal for scan
178                0.0, // No network cost for local scan
179                table_stats.row_count,
180            )
181        } else {
182            // Default estimates for unknown tables
183            QueryCost::new(1000.0, 100.0, 0.0, 0.0, 1000)
184        }
185    }
186
187    /// Estimate the cost of an index scan
188    pub fn estimate_index_scan_cost(&self, table_name: &str, index_name: &str, selectivity: f64, stats: &StatisticsCollector) -> QueryCost {
189        if let Some(table_stats) = stats.get_table_stats(table_name) {
190            let estimated_rows = (table_stats.row_count as f64 * selectivity) as usize;
191            let index_pages = (estimated_rows as f64 / 100.0).ceil(); // Assume 100 rows per index page
192            
193            QueryCost::new(
194                estimated_rows as f64 * 2.0, // Slightly higher CPU cost for index traversal
195                index_pages + (estimated_rows as f64 / table_stats.rows_per_page as f64), // Index pages + data pages
196                0.0,
197                0.0,
198                estimated_rows,
199            )
200        } else {
201            // Default estimates
202            let estimated_rows = (1000.0 * selectivity) as usize;
203            QueryCost::new(
204                estimated_rows as f64 * 2.0,
205                estimated_rows as f64 / 10.0,
206                0.0,
207                0.0,
208                estimated_rows,
209            )
210        }
211    }

213    /// Estimate the cost of a nested loop join
214    pub fn estimate_nested_loop_join_cost(&self, left_cost: &QueryCost, right_cost: &QueryCost) -> QueryCost {
215        QueryCost::new(
216            left_cost.cpu_cost + (left_cost.cardinality as f64 * right_cost.cpu_cost),
217            left_cost.io_cost + (left_cost.cardinality as f64 * right_cost.io_cost),
218            left_cost.memory_cost.max(right_cost.memory_cost),
219            left_cost.network_cost + right_cost.network_cost,
220            (left_cost.cardinality as f64 * right_cost.cardinality as f64 * 0.1) as usize, // Assume 10% join selectivity
221        )
222    }
223
224    /// Estimate the cost of a hash join
225    pub fn estimate_hash_join_cost(&self, left_cost: &QueryCost, right_cost: &QueryCost) -> QueryCost {
226        // Hash join: build hash table from smaller relation, probe with larger
227        let (build_cost, probe_cost) = if left_cost.cardinality <= right_cost.cardinality {
228            (left_cost, right_cost)
229        } else {
230            (right_cost, left_cost)
231        };
232
233        QueryCost::new(
234            build_cost.cpu_cost + probe_cost.cpu_cost + (build_cost.cardinality + probe_cost.cardinality) as f64,
235            build_cost.io_cost + probe_cost.io_cost,
236            build_cost.cardinality as f64 * 8.0, // Memory for hash table (8 bytes per row estimate)
237            build_cost.network_cost + probe_cost.network_cost,
238            (build_cost.cardinality as f64 * probe_cost.cardinality as f64 * 0.1) as usize,
239        )
240    }

242    /// Estimate the cost of a merge join
243    pub fn estimate_merge_join_cost(&self, left_cost: &QueryCost, right_cost: &QueryCost) -> QueryCost {
244        // Assume inputs are already sorted, or add sort cost if needed
245        QueryCost::new(
246            left_cost.cpu_cost + right_cost.cpu_cost + (left_cost.cardinality + right_cost.cardinality) as f64,
247            left_cost.io_cost + right_cost.io_cost,
248            (left_cost.cardinality.max(right_cost.cardinality)) as f64 * 4.0, // Memory for merge
249            left_cost.network_cost + right_cost.network_cost,
250            (left_cost.cardinality as f64 * right_cost.cardinality as f64 * 0.1) as usize,
251        )
252    }

254    /// Estimate selectivity of a predicate
255    pub fn estimate_selectivity(&self, expr: &Expression, stats: &StatisticsCollector) -> f64 {
256        match expr {
257            Expression::Binary { left: _, operator, right: _ } => {
258                match operator {
259                    BinaryOperator::Equal => 0.1, // Equality typically has good selectivity
260                    BinaryOperator::NotEqual => 0.9,
261                    BinaryOperator::LessThan | BinaryOperator::LessThanOrEqual |
262                    BinaryOperator::GreaterThan | BinaryOperator::GreaterThanOrEqual => 0.33,
263                    BinaryOperator::And => 0.5, // Conservative estimate
264                    BinaryOperator::Or => 0.75,
265                    _ => 0.5, // Default
266                }
267            }
268            Expression::Unary { operator: UnaryOperator::Not, .. } => 0.9, // NOT is usually high selectivity
269            _ => 0.5, // Default selectivity
270        }
271    }
