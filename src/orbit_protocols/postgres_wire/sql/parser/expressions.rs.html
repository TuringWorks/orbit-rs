21    /// Parse a complete expression with proper precedence
22    pub fn parse_expression(
23        &mut self,
24        tokens: &[Token],
25        pos: &mut usize,
26    ) -> ProtocolResult<Expression> {
27        self.parse_or_expression(tokens, pos)
28    }

30    /// Parse OR expressions (lowest precedence)
31    fn parse_or_expression(
32        &mut self,
33        tokens: &[Token],
34        pos: &mut usize,
35    ) -> ProtocolResult<Expression> {
36        let mut left = self.parse_and_expression(tokens, pos)?;
37
38        while *pos < tokens.len() {
39            if matches!(tokens[*pos], Token::Or) {
40                *pos += 1;
41                let right = self.parse_and_expression(tokens, pos)?;
42                left = Expression::Binary {
43                    left: Box::new(left),
44                    operator: BinaryOperator::Or,
45                    right: Box::new(right),
46                };
47            } else {
48                break;
49            }
50        }
51
52        Ok(left)
53    }

55    /// Parse AND expressions
56    fn parse_and_expression(
57        &mut self,
58        tokens: &[Token],
59        pos: &mut usize,
60    ) -> ProtocolResult<Expression> {
61        let mut left = self.parse_equality_expression(tokens, pos)?;
62
63        while *pos < tokens.len() {
64            if matches!(tokens[*pos], Token::And) {
65                *pos += 1;
66                let right = self.parse_equality_expression(tokens, pos)?;
67                left = Expression::Binary {
68                    left: Box::new(left),
69                    operator: BinaryOperator::And,
70                    right: Box::new(right),
71                };
72            } else {
73                break;
74            }
75        }
76
77        Ok(left)
78    }

80    /// Parse equality expressions (=, !=, <>, IS, IS NOT)
81    fn parse_equality_expression(
82        &mut self,
83        tokens: &[Token],
84        pos: &mut usize,
85    ) -> ProtocolResult<Expression> {
86        let mut left = self.parse_comparison_expression(tokens, pos)?;
87
88        while *pos < tokens.len() {
89            let operator = match &tokens[*pos] {
90                Token::Equal => BinaryOperator::Equal,
91                Token::NotEqual => BinaryOperator::NotEqual,
92                Token::Is => {
93                    *pos += 1;
94                    if *pos < tokens.len() && matches!(tokens[*pos], Token::Not) {
95                        *pos += 1;
96                        BinaryOperator::IsNot
97                    } else {
98                        BinaryOperator::Is
99                    }
100                }
101                _ => break,
102            };
103
104            if !matches!(operator, BinaryOperator::Is | BinaryOperator::IsNot) {
105                *pos += 1;
106            }
107
108            let right = self.parse_comparison_expression(tokens, pos)?;
109            left = Expression::Binary {
110                left: Box::new(left),
111                operator,
112                right: Box::new(right),
113            };
114        }
115
116        Ok(left)
117    }

119    /// Parse comparison expressions (<, <=, >, >=, LIKE, IN, BETWEEN)
120    fn parse_comparison_expression(
121        &mut self,
122        tokens: &[Token],
123        pos: &mut usize,
124    ) -> ProtocolResult<Expression> {
125        let mut left = self.parse_additive_expression(tokens, pos)?;
126
127        while *pos < tokens.len() {
128            let operator = match &tokens[*pos] {
129                Token::LessThan => BinaryOperator::LessThan,
130                Token::LessThanOrEqual => BinaryOperator::LessThanOrEqual,
131                Token::GreaterThan => BinaryOperator::GreaterThan,
132                Token::GreaterThanOrEqual => BinaryOperator::GreaterThanOrEqual,
133                Token::Like => BinaryOperator::Like,
134                Token::ILike => BinaryOperator::ILike,
135                Token::In => BinaryOperator::In,
136                Token::VectorDistance => BinaryOperator::VectorDistance,
137                Token::VectorInnerProduct => BinaryOperator::VectorInnerProduct,
138                Token::VectorCosineDistance => BinaryOperator::VectorCosineDistance,
139                _ => break,
140            };
141
142            *pos += 1;
143            let right = self.parse_additive_expression(tokens, pos)?;
144            left = Expression::Binary {
145                left: Box::new(left),
146                operator,
147                right: Box::new(right),
148            };
149        }
150
151        Ok(left)
152    }

154    /// Parse additive expressions (+, -, ||)
155    fn parse_additive_expression(
156        &mut self,
157        tokens: &[Token],
158        pos: &mut usize,
159    ) -> ProtocolResult<Expression> {
160        let mut left = self.parse_multiplicative_expression(tokens, pos)?;
161
162        while *pos < tokens.len() {
163            let operator = match &tokens[*pos] {
164                Token::Plus => BinaryOperator::Plus,
165                Token::Minus => BinaryOperator::Minus,
166                Token::Concat => BinaryOperator::Concat,
167                _ => break,
168            };
169
170            *pos += 1;
171            let right = self.parse_multiplicative_expression(tokens, pos)?;
172            left = Expression::Binary {
173                left: Box::new(left),
174                operator,
175                right: Box::new(right),
176            };
177        }
178
179        Ok(left)
180    }

182    /// Parse multiplicative expressions (*, /, %)
183    fn parse_multiplicative_expression(
184        &mut self,
185        tokens: &[Token],
186        pos: &mut usize,
187    ) -> ProtocolResult<Expression> {
188        let mut left = self.parse_unary_expression(tokens, pos)?;
189
190        while *pos < tokens.len() {
191            let operator = match &tokens[*pos] {
192                Token::Multiply => BinaryOperator::Multiply,
193                Token::Divide => BinaryOperator::Divide,
194                Token::Modulo => BinaryOperator::Modulo,
195                _ => break,
196            };
197
198            *pos += 1;
199            let right = self.parse_unary_expression(tokens, pos)?;
200            left = Expression::Binary {
201                left: Box::new(left),
202                operator,
203                right: Box::new(right),
204            };
205        }
206
207        Ok(left)
208    }

210    /// Parse unary expressions (NOT, -, +)
211    fn parse_unary_expression(
212        &mut self,
213        tokens: &[Token],
214        pos: &mut usize,
215    ) -> ProtocolResult<Expression> {
216        if *pos >= tokens.len() {
217            return Err(crate::error::ProtocolError::ParseError(
218                "Unexpected end of tokens".to_string(),
219            ));
220        }
221
222        match &tokens[*pos] {
223            Token::Not => {
224                *pos += 1;
225                let expr = self.parse_unary_expression(tokens, pos)?;
226                Ok(Expression::Unary {
227                    operator: UnaryOperator::Not,
228                    operand: Box::new(expr),
229                })
230            }
231            Token::Minus => {
232                *pos += 1;
233                let expr = self.parse_unary_expression(tokens, pos)?;
234                Ok(Expression::Unary {
235                    operator: UnaryOperator::Minus,
236                    operand: Box::new(expr),
237                })
238            }
239            Token::Plus => {
240                *pos += 1;
241                let expr = self.parse_unary_expression(tokens, pos)?;
242                Ok(Expression::Unary {
243                    operator: UnaryOperator::Plus,
244                    operand: Box::new(expr),
245                })
246            }
247            _ => self.parse_primary_expression(tokens, pos),
248        }
249    }

251    /// Parse primary expressions (literals, identifiers, function calls, parenthesized expressions)
252    fn parse_primary_expression(
253        &mut self,
254        tokens: &[Token],
255        pos: &mut usize,
256    ) -> ProtocolResult<Expression> {
257        if *pos >= tokens.len() {
258            return Err(crate::error::ProtocolError::ParseError(
259                "Unexpected end of tokens".to_string(),
260            ));
261        }
262
263        match &tokens[*pos] {
264            Token::StringLiteral(s) => {
265                *pos += 1;
266                Ok(Expression::Literal(
267                    crate::postgres_wire::sql::types::SqlValue::Text(s.clone()),
268                ))
269            }
270            Token::NumericLiteral(n) => {
271                *pos += 1;
272                // Try to parse as integer first, then as decimal
273                if let Ok(i) = n.parse::<i32>() {
274                    Ok(Expression::Literal(
275                        crate::postgres_wire::sql::types::SqlValue::Integer(i),
276                    ))
277                } else if let Ok(f) = n.parse::<f64>() {
278                    Ok(Expression::Literal(
279                        crate::postgres_wire::sql::types::SqlValue::DoublePrecision(f),
280                    ))
281                } else {
282                    Ok(Expression::Literal(
283                        crate::postgres_wire::sql::types::SqlValue::Text(n.clone()),
284                    ))
285                }
286            }
287            Token::BooleanLiteral(b) => {
288                *pos += 1;
289                Ok(Expression::Literal(
290                    crate::postgres_wire::sql::types::SqlValue::Boolean(*b),
291                ))
292            }
293            Token::Null => {
294                *pos += 1;
295                Ok(Expression::Literal(
296                    crate::postgres_wire::sql::types::SqlValue::Null,
297                ))
298            }
299            Token::Identifier(name) => {
300                // Check for function call
301                if *pos + 1 < tokens.len() && matches!(tokens[*pos + 1], Token::LeftParen) {
302                    self.parse_function_call(tokens, pos, name.clone())
303                } else {
304                    *pos += 1;
305                    // Regular column reference
306                    Ok(Expression::Column(
307                        crate::postgres_wire::sql::ast::ColumnRef {
308                            table: None,
309                            name: name.clone(),
310                        },
311                    ))
312                }
313            }
314
315            // Handle aggregate function keywords
316            Token::Count | Token::Sum | Token::Avg | Token::Min | Token::Max => {
317                let func_name = match &tokens[*pos] {
318                    Token::Count => "COUNT".to_string(),
319                    Token::Sum => "SUM".to_string(),
320                    Token::Avg => "AVG".to_string(),
321                    Token::Min => "MIN".to_string(),
322                    Token::Max => "MAX".to_string(),
323                    _ => unreachable!(),
324                };
325                self.parse_function_call(tokens, pos, func_name)
326            }
327
328            // Handle window function keywords
329            Token::RowNumber
330            | Token::Rank
331            | Token::DenseRank
332            | Token::PercentRank
333            | Token::CumeDist
334            | Token::Ntile
335            | Token::Lag
336            | Token::Lead
337            | Token::FirstValue
338            | Token::LastValue
339            | Token::NthValue => self.parse_window_function(tokens, pos),
340            Token::LeftParen => {
341                *pos += 1; // consume '('
342                let expr = self.parse_expression(tokens, pos)?;
343
344                if *pos >= tokens.len() || !matches!(tokens[*pos], Token::RightParen) {
345                    return Err(crate::error::ProtocolError::ParseError(
346                        "Expected ')' after expression".to_string(),
347                    ));
348                }
349                *pos += 1; // consume ')'
350
351                Ok(expr)
352            }
353            _ => Err(crate::error::ProtocolError::ParseError(format!(
354                "Unexpected token in expression: {:?}",
355                tokens[*pos]
356            ))),
357        }
358    }

360    /// Parse a function call with proper DISTINCT and FILTER support
361    fn parse_function_call(
362        &mut self,
363        tokens: &[Token],
364        pos: &mut usize,
365        func_name: String,
366    ) -> ProtocolResult<Expression> {
367        *pos += 1; // consume function name
368
369        if *pos >= tokens.len() || !matches!(tokens[*pos], Token::LeftParen) {
370            return Err(crate::error::ProtocolError::ParseError(
371                "Expected '(' after function name".to_string(),
372            )
373            .into());
374        }
375        *pos += 1; // consume '('
376
377        // Check for DISTINCT/ALL
378        let distinct = if *pos < tokens.len() && matches!(tokens[*pos], Token::Distinct) {
379            *pos += 1;
380            true
381        } else if *pos < tokens.len() && matches!(tokens[*pos], Token::All) {
382            *pos += 1;
383            false
384        } else {
385            false
386        };
387
388        let mut args = Vec::new();
389
390        // Parse arguments
391        if *pos < tokens.len() && !matches!(tokens[*pos], Token::RightParen) {
392            loop {
393                // Handle special case for COUNT(*)
394                if func_name.to_uppercase() == "COUNT" && matches!(tokens[*pos], Token::Multiply) {
395                    *pos += 1;
396                    args.push(Expression::Column(ColumnRef {
397                        table: None,
398                        name: "*".to_string(),
399                    }));
400                } else {
401                    args.push(self.parse_expression(tokens, pos)?);
402                }
403
404                if *pos < tokens.len() && matches!(tokens[*pos], Token::Comma) {
405                    *pos += 1; // consume ','
406                } else {
407                    break;
408                }
409            }
410        }
411
412        if *pos >= tokens.len() || !matches!(tokens[*pos], Token::RightParen) {
413            return Err(crate::error::ProtocolError::ParseError(
414                "Expected ')' after function arguments".to_string(),
415            )
416            .into());
417        }
418        *pos += 1; // consume ')'
419
420        // Parse optional ORDER BY clause for aggregate functions
421        let order_by = if *pos < tokens.len() && matches!(tokens[*pos], Token::Order) {
422            *pos += 1;
423            if *pos < tokens.len() && matches!(tokens[*pos], Token::By) {
424                *pos += 1;
425                Some(self.parse_order_by_list(tokens, pos)?)
426            } else {
427                None
428            }
429        } else {
430            None
431        };
432
433        // Parse optional FILTER clause
434        let filter = if *pos < tokens.len() {
435            if let Token::Identifier(s) = &tokens[*pos] {
436                if s.to_uppercase() == "FILTER" {
437                    *pos += 1;
438                    if *pos < tokens.len() && matches!(tokens[*pos], Token::LeftParen) {
439                        *pos += 1;
440                        if *pos < tokens.len() && matches!(tokens[*pos], Token::Where) {
441                            *pos += 1;
442                            let filter_expr = self.parse_expression(tokens, pos)?;
443                            if *pos < tokens.len() && matches!(tokens[*pos], Token::RightParen) {
444                                *pos += 1;
445                                Some(Box::new(filter_expr))
446                            } else {
447                                None
448                            }
449                        } else {
450                            None
451                        }
452                    } else {
453                        None
454                    }
455                } else {
456                    None
457                }
458            } else {
459                None
460            }
461        } else {
462            None
463        };
464
465        // Check if this is followed by an OVER clause (window function)
466        if *pos < tokens.len() && matches!(tokens[*pos], Token::Over) {
467            self.parse_window_over_clause(tokens, pos, func_name, args, distinct, order_by, filter)
468        } else {
469            Ok(Expression::Function(Box::new(FunctionCall {
470                name: FunctionName::Simple(func_name),
471                args,
472                distinct,
473                order_by,
474                filter,
475            })))
476        }
477    }

479    /// Parse window function expressions
480    fn parse_window_function(
481        &mut self,
482        tokens: &[Token],
483        pos: &mut usize,
484    ) -> ProtocolResult<Expression> {
485        let window_func = match &tokens[*pos] {
486            Token::RowNumber => {
487                *pos += 1;
488                self.expect_token(tokens, pos, &Token::LeftParen)?;
489                self.expect_token(tokens, pos, &Token::RightParen)?;
490                WindowFunctionType::RowNumber
491            }
492            Token::Rank => {
493                *pos += 1;
494                self.expect_token(tokens, pos, &Token::LeftParen)?;
495                self.expect_token(tokens, pos, &Token::RightParen)?;
496                WindowFunctionType::Rank
497            }
498            Token::DenseRank => {
499                *pos += 1;
500                self.expect_token(tokens, pos, &Token::LeftParen)?;
501                self.expect_token(tokens, pos, &Token::RightParen)?;
502                WindowFunctionType::DenseRank
503            }
504            Token::PercentRank => {
505                *pos += 1;
506                self.expect_token(tokens, pos, &Token::LeftParen)?;
507                self.expect_token(tokens, pos, &Token::RightParen)?;
508                WindowFunctionType::PercentRank
509            }
510            Token::CumeDist => {
511                *pos += 1;
512                self.expect_token(tokens, pos, &Token::LeftParen)?;
513                self.expect_token(tokens, pos, &Token::RightParen)?;
514                WindowFunctionType::CumeDist
515            }
516            Token::Ntile => {
517                *pos += 1;
518                self.expect_token(tokens, pos, &Token::LeftParen)?;
519                let n = Box::new(self.parse_expression(tokens, pos)?);
520                self.expect_token(tokens, pos, &Token::RightParen)?;
521                WindowFunctionType::Ntile(n)
522            }
523            Token::Lag => {
524                *pos += 1;
525                self.expect_token(tokens, pos, &Token::LeftParen)?;
526                let expr = Box::new(self.parse_expression(tokens, pos)?);
527
528                let offset = if self.matches_at(tokens, *pos, &Token::Comma) {
529                    *pos += 1;
530                    Some(Box::new(self.parse_expression(tokens, pos)?))
531                } else {
532                    None
533                };
534
535                let default = if self.matches_at(tokens, *pos, &Token::Comma) {
536                    *pos += 1;
537                    Some(Box::new(self.parse_expression(tokens, pos)?))
538                } else {
539                    None
540                };
541
542                self.expect_token(tokens, pos, &Token::RightParen)?;
543                WindowFunctionType::Lag {
544                    expr,
545                    offset,
546                    default,
547                }
548            }
549            Token::Lead => {
550                *pos += 1;
551                self.expect_token(tokens, pos, &Token::LeftParen)?;
552                let expr = Box::new(self.parse_expression(tokens, pos)?);
553
554                let offset = if self.matches_at(tokens, *pos, &Token::Comma) {
555                    *pos += 1;
556                    Some(Box::new(self.parse_expression(tokens, pos)?))
557                } else {
558                    None
559                };
560
561                let default = if self.matches_at(tokens, *pos, &Token::Comma) {
562                    *pos += 1;
563                    Some(Box::new(self.parse_expression(tokens, pos)?))
564                } else {
565                    None
566                };
567
568                self.expect_token(tokens, pos, &Token::RightParen)?;
569                WindowFunctionType::Lead {
570                    expr,
571                    offset,
572                    default,
573                }
574            }
575            Token::FirstValue => {
576                *pos += 1;
577                self.expect_token(tokens, pos, &Token::LeftParen)?;
578                let expr = Box::new(self.parse_expression(tokens, pos)?);
579                self.expect_token(tokens, pos, &Token::RightParen)?;
580                WindowFunctionType::FirstValue(expr)
581            }
582            Token::LastValue => {
583                *pos += 1;
584                self.expect_token(tokens, pos, &Token::LeftParen)?;
585                let expr = Box::new(self.parse_expression(tokens, pos)?);
586                self.expect_token(tokens, pos, &Token::RightParen)?;
587                WindowFunctionType::LastValue(expr)
588            }
589            Token::NthValue => {
590                *pos += 1;
591                self.expect_token(tokens, pos, &Token::LeftParen)?;
592                let expr = Box::new(self.parse_expression(tokens, pos)?);
593                self.expect_token(tokens, pos, &Token::Comma)?;
594                let n = Box::new(self.parse_expression(tokens, pos)?);
595                self.expect_token(tokens, pos, &Token::RightParen)?;
596                WindowFunctionType::NthValue { expr, n }
597            }
598            _ => {
599                return Err(crate::error::ProtocolError::ParseError(
600                    "Invalid window function".to_string(),
601                )
602                .into())
603            }
604        };
605
606        // Parse OVER clause
607        self.expect_token(tokens, pos, &Token::Over)?;
608        let (partition_by, order_by, frame) = self.parse_over_clause(tokens, pos)?;
609
610        Ok(Expression::WindowFunction {
611            function: window_func,
612            partition_by,
613            order_by,
614            frame,
615        })
616    }

618    /// Parse window OVER clause for aggregate functions used as window functions
619    fn parse_window_over_clause(
620        &mut self,
621        tokens: &[Token],
622        pos: &mut usize,
623        func_name: String,
624        args: Vec<Expression>,
625        distinct: bool,
626        order_by: Option<Vec<OrderByItem>>,
627        filter: Option<Box<Expression>>,
628    ) -> ProtocolResult<Expression> {
629        *pos += 1; // consume OVER
630        let (partition_by, window_order_by, frame) = self.parse_over_clause(tokens, pos)?;
631
632        let aggregate_func = FunctionCall {
633            name: FunctionName::Simple(func_name),
634            args,
635            distinct,
636            order_by,
637            filter,
638        };
639
640        Ok(Expression::WindowFunction {
641            function: WindowFunctionType::Aggregate(Box::new(aggregate_func)),
642            partition_by,
643            order_by: window_order_by,
644            frame,
645        })
646    }

648    /// Parse the contents of an OVER clause
649    fn parse_over_clause(
650        &mut self,
651        tokens: &[Token],
652        pos: &mut usize,
653    ) -> ProtocolResult<(Vec<Expression>, Vec<OrderByItem>, Option<WindowFrame>)> {
654        self.expect_token(tokens, pos, &Token::LeftParen)?;
655
656        // Parse PARTITION BY
657        let partition_by = if self.matches_at(tokens, *pos, &Token::Partition) {
658            *pos += 1;
659            self.expect_token(tokens, pos, &Token::By)?;
660            self.parse_expression_list(tokens, pos)?
661        } else {
662            Vec::new()
663        };
664
665        // Parse ORDER BY
666        let order_by = if self.matches_at(tokens, *pos, &Token::Order) {
667            *pos += 1;
668            self.expect_token(tokens, pos, &Token::By)?;
669            self.parse_order_by_list(tokens, pos)?
670        } else {
671            Vec::new()
672        };
673
674        // Parse optional window frame
675        let frame = if self.matches_at(tokens, *pos, &Token::Rows)
676            || self.matches_at(tokens, *pos, &Token::Range)
677        {
678            Some(self.parse_window_frame(tokens, pos)?)
679        } else {
680            None
681        };
682
683        self.expect_token(tokens, pos, &Token::RightParen)?;
684
685        Ok((partition_by, order_by, frame))
686    }

688    /// Parse window frame specification
689    fn parse_window_frame(
690        &mut self,
691        tokens: &[Token],
692        pos: &mut usize,
693    ) -> ProtocolResult<WindowFrame> {
694        // Skip ROWS or RANGE for now, we'll just parse the bounds
695        *pos += 1;
696
697        let start_bound = self.parse_frame_bound(tokens, pos)?;
698
699        let end_bound = if self.matches_at(tokens, *pos, &Token::And) {
700            *pos += 1;
701            Some(self.parse_frame_bound(tokens, pos)?)
702        } else {
703            None
704        };
705
706        Ok(WindowFrame {
707            start_bound,
708            end_bound,
709        })
710    }

712    /// Parse window frame bound
713    fn parse_frame_bound(
714        &mut self,
715        tokens: &[Token],
716        pos: &mut usize,
717    ) -> ProtocolResult<FrameBound> {
718        if self.matches_at(tokens, *pos, &Token::Unbounded) {
719            *pos += 1;
720            if self.matches_at(tokens, *pos, &Token::Preceding) {
721                *pos += 1;
722                Ok(FrameBound::UnboundedPreceding)
723            } else if self.matches_at(tokens, *pos, &Token::Following) {
724                *pos += 1;
725                Ok(FrameBound::UnboundedFollowing)
726            } else {
727                Err(crate::error::ProtocolError::ParseError(
728                    "Expected PRECEDING or FOLLOWING after UNBOUNDED".to_string(),
729                )
730                .into())
731            }
732        } else if self.matches_at(tokens, *pos, &Token::CurrentRow) {
733            *pos += 1;
734            Ok(FrameBound::CurrentRow)
735        } else {
736            let expr = Box::new(self.parse_expression(tokens, pos)?);
737            if self.matches_at(tokens, *pos, &Token::Preceding) {
738                *pos += 1;
739                Ok(FrameBound::Preceding(expr))
740            } else if self.matches_at(tokens, *pos, &Token::Following) {
741                *pos += 1;
742                Ok(FrameBound::Following(expr))
743            } else {
744                Err(crate::error::ProtocolError::ParseError(
745                    "Expected PRECEDING or FOLLOWING".to_string(),
746                )
747                .into())
748            }
749        }
750    }
751
752    // Helper methods
753    fn matches_at(&self, tokens: &[Token], pos: usize, expected: &Token) -> bool {
754        tokens.get(pos).map_or(false, |token| {
755            std::mem::discriminant(token) == std::mem::discriminant(expected)
756        })
757    }
758
759    fn expect_token(
760        &self,
761        tokens: &[Token],
762        pos: &mut usize,
763        expected: &Token,
764    ) -> ProtocolResult<()> {
765        if self.matches_at(tokens, *pos, expected) {
766            *pos += 1;
767            Ok(())
768        } else {
769            Err(crate::error::ProtocolError::ParseError(format!(
770                "Expected {:?}, found {:?}",
771                expected,
772                tokens.get(*pos)
773            ))
774            .into())
775        }
776    }
777
778    fn parse_expression_list(
779        &mut self,
780        tokens: &[Token],
781        pos: &mut usize,
782    ) -> ProtocolResult<Vec<Expression>> {
783        let mut expressions = Vec::new();
784
785        loop {
786            expressions.push(self.parse_expression(tokens, pos)?);
787
788            if self.matches_at(tokens, *pos, &Token::Comma) {
789                *pos += 1;
790            } else {
791                break;
792            }
793        }
794
795        Ok(expressions)
796    }
797
798    fn parse_order_by_list(
799        &mut self,
800        tokens: &[Token],
801        pos: &mut usize,
802    ) -> ProtocolResult<Vec<OrderByItem>> {
803        let mut items = Vec::new();
804
805        loop {
806            let expression = self.parse_expression(tokens, pos)?;
807
808            let direction = if self.matches_at(tokens, *pos, &Token::Asc) {
809                *pos += 1;
810                Some(SortDirection::Ascending)
811            } else if self.matches_at(tokens, *pos, &Token::Desc) {
812                *pos += 1;
813                Some(SortDirection::Descending)
814            } else {
815                None
816            };
817
818            let nulls = if self.matches_at(tokens, *pos, &Token::Nulls) {
819                *pos += 1;
820                if self.matches_at(tokens, *pos, &Token::First) {
821                    *pos += 1;
822                    Some(NullsOrder::First)
823                } else if self.matches_at(tokens, *pos, &Token::Last) {
824                    *pos += 1;
825                    Some(NullsOrder::Last)
826                } else {
827                    None
828                }
829            } else {
830                None
831            };
832
833            items.push(OrderByItem {
834                expression,
835                direction,
836                nulls,
837            });
838
839            if self.matches_at(tokens, *pos, &Token::Comma) {
840                *pos += 1;
841            } else {
842                break;
843            }
844        }
845
846        Ok(items)
847    }
848}
